50 void Clusterizer::getClusterFromNeighbours(std::vector<Digit*>& clusterDigits, int row, int column)
51 {
52  // Recursion 0, add seed digit to cluster
53  if (!clusterDigits.size()) {
54  clusterDigits.emplace_back(mDigitMap[row][column]);
55  }
56 
57  // Mark the current cell as clustered
58  mCellMask[row][column] = kTRUE;
59 
60  // Now go recursively to the next 4 neighbours and add them to the cluster if they fulfill the conditions
61  constexpr int rowDiffs[4] = {-1, 0, 0, 1};
62  constexpr int colDiffs[4] = {0, -1, 1, 0};
63  for (int dir = 0; dir < 4; dir++) {
64  if ((row + rowDiffs[dir] < 0) || (row + rowDiffs[dir] >= NROWS)) {
65  continue;
66  }
67  if ((column + colDiffs[dir] < 0) || (column + colDiffs[dir] >= NCOLS)) {
68  continue;
69  }
70 
71  if (mDigitMap[row + rowDiffs[dir]][column + colDiffs[dir]]) {
72  if (!mCellMask[row + rowDiffs[dir]][column + colDiffs[dir]]) {
73  if (mDoEnergyGradientCut && not(mDigitMap[row + rowDiffs[dir]][column + colDiffs[dir]]->getEnergy() > mDigitMap[row][column]->getEnergy() + mGradientCut)) {
74  if (not(TMath::Abs(mDigitMap[row + rowDiffs[dir]][column + colDiffs[dir]]->getTimeStamp() - mDigitMap[row][column]->getTimeStamp()) > mTimeCut)) {
75  getClusterFromNeighbours(clusterDigits, row + rowDiffs[dir], column + colDiffs[dir]);
76  // Add the digit to the current cluster -- if we end up here, the selected cluster fulfills the condition
77  clusterDigits.emplace_back(mDigitMap[row + rowDiffs[dir]][column + colDiffs[dir]]);
78  }
79  }
80  }
81  }
82  }
83 }

89 void Clusterizer::getTopologicalRowColumn(const Digit& digit, int& row, int& column)
90 {
91  // Get SM number and relative row/column for SM
92  auto cellIndex = mEMCALGeometry->GetCellIndex(digit.getTower());
93  int nSupMod = std::get<0>(cellIndex);
94 
95  auto phiEtaIndex = mEMCALGeometry->GetCellPhiEtaIndexInSModule(nSupMod, std::get<1>(cellIndex), std::get<2>(cellIndex), std::get<3>(cellIndex));
96  row = std::get<0>(phiEtaIndex);
97  column = std::get<1>(phiEtaIndex);
98 
99  // Add shifts wrt. supermodule and type of calorimeter
100  // NOTE:
101  // * Rows (phi) are arranged that one space is left empty between supermodules in phi
102  // This is due to the physical gap that forbids clustering
103  // * For DCAL, there is an additional empty column between two supermodules in eta
104  // Again, this is to account for the gap in DCAL
105 
106  row += nSupMod / 2 * (24 + 1);
107  // In DCAL, leave a gap between two SMs with same phi
108  if (!mEMCALGeometry->IsDCALSM(nSupMod)) { // EMCAL
109  column += nSupMod % 2 * 48;
110  } else {
111  column += nSupMod % 2 * (48 + 1);
112  }
113 }

118 void Clusterizer::findClusters(const std::vector<Digit>& digitArray)
119 {
120  mFoundClusters.clear();
121  mDigitIndices.clear();
122 
123  // Algorithm
124  // - Fill digits in 2D topological map
125  // - Fill struct arrays (energy,x,y) (to get mapping energy -> (x,y))
126  // - Create 2D bitmap (digit is already clustered or not)
127  // - Sort struct arrays with descending energy
128  //
129  // - Loop over arrays:
130  // --> Check 2D bitmap (don't use digit which are already clustered)
131  // --> Take valid digit with highest energy as seed (they are already sorted)
132  // --> Recursive to neighboughs and create cluster
133  // --> Seed cell and all neighbours belonging to cluster will be put in 2D bitmap
134 
135  // Reset digit maps and cell masks
136  // Loop over one array dim, then reset each array
137  for (auto iArr = 0; iArr < NROWS; iArr++) {
138  mCellMask[iArr].fill(kFALSE);
139  mDigitMap[iArr].fill(nullptr);
140  }
141 
142  // Calibrate digits and fill the maps/arrays
143  int nCells = 0;
144  double ehs = 0.0;
145  for (auto dig : digitArray) {
146  Float_t digitEnergy = dig.getEnergy();
147  Float_t time = dig.getTimeStamp();
148 
149  if (digitEnergy < mThresholdCellEnergy || time > mTimeMax || time < mTimeMin) {
150  continue;
151  }
152  if (!mEMCALGeometry->CheckAbsCellId(dig.getTower())) {
153  continue;
154  }
155  ehs += digitEnergy;
156 
157  // Put digit to 2D map
158  int row = 0, column = 0;
159  getTopologicalRowColumn(dig, row, column);
160  mDigitMap[row][column] = &dig; // mDigitMap saves pointers to digits, therefore use addr operator here
161  mSeedList[nCells].energy = digitEnergy;
162  mSeedList[nCells].row = row;
163  mSeedList[nCells].column = column;
164  nCells++;
165  }
166 
167  // Sort struct arrays with ascending energy
168  std::sort(mSeedList.begin(), std::next(std::begin(mSeedList), nCells));
169  //std::sort(mSeedList, mSeedList+nCells);
170 
171  // Take next valid digit in calorimeter as seed (in descending energy order)
172  for (int i = nCells; i--;) {
173  int row = mSeedList[i].row, column = mSeedList[i].column;
174  // Continue if the cell is already masked (i.e. was already clustered)
175  if (mCellMask[row][column]) {
176  continue;
177  }
178  // Continue if energy constraints are not fulfilled
179  if (mSeedList[i].energy <= mThresholdSeedEnergy) {
180  continue;
181  }
182 
183  // Seed is found, form cluster recursively
184  std::vector<Digit*> clusterDigits;
185  getClusterFromNeighbours(clusterDigits, row, column);
186 
187  // Add digits for current cluster to digit index vector
188  int digitIndexStart = mDigitIndices.size();
189  for (auto dig : clusterDigits) {
190  mDigitIndices.emplace_back(dig->getTower());
191  }
192  int digitIndexSize = mDigitIndices.size() - digitIndexStart;
193 
194  // Now form cluster object from digits
195  mFoundClusters.emplace_back(mDigitMap[row][column]->getTimeStamp(), digitIndexStart, digitIndexSize); // Cluster object initialized w/ time of seed cell, start + size of associated cells
196  }
197  LOG(DEBUG) << mFoundClusters.size() << "clusters found from " << nCells << " digits (total=" << digitArray.size() << ")-> ehs " << ehs << " (minE " << mThresholdCellEnergy << ")";
198 }
