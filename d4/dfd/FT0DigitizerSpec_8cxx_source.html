44  public:
45  FT0DPLDigitizerTask() : mDigitizer(DigitizationParameters{}) {}
46  explicit FT0DPLDigitizerTask(o2::ft0::DigitizationParameters const& parameters)
47  : mDigitizer(parameters){};
48  ~FT0DPLDigitizerTask() = default;
49  static constexpr o2::detectors::DetID::ID DETID = o2::detectors::DetID::FT0;
50  static constexpr o2::header::DataOrigin DETOR = o2::header::gDataOriginFT0;
51  void init(framework::InitContext& ic)
52  {
53  // setup the input chain for the hits
54  mSimChains.emplace_back(new TChain("o2sim"));
55 
56  // add the main (background) file
57  mSimChains.back()->AddFile(ic.options().get<std::string>("simFile").c_str());
58  // maybe add a particular signal file
59  auto signalfilename = ic.options().get<std::string>("simFileS");
60  if (signalfilename.size() > 0) {
61  mSimChains.emplace_back(new TChain("o2sim"));
62  mSimChains.back()->AddFile(signalfilename.c_str());
63  }
64  mDigitizer.init();
65  const bool isContinuous = ic.options().get<int>("pileup");
66  }

68  void run(framework::ProcessingContext& pc)
69  {
70 
71  static bool finished = false;
72  if (finished) {
73  return;
74  }
75 
76  // read collision context from input
77  auto context = pc.inputs().get<o2::steer::RunContext*>("collisioncontext");
78  auto& timesview = context->getEventRecords();
79 
80  // if there is nothing to do ... return
81  if (timesview.size() == 0) {
82  return;
83  }
84 
85  TStopwatch timer;
86  timer.Start();
87 
88  LOG(INFO) << "CALLING FT0 DIGITIZATION";
89 
90  static std::vector<o2::ft0::HitType> hits;
91  o2::dataformats::MCTruthContainer<o2::ft0::MCLabel> labelAccum;
92  o2::dataformats::MCTruthContainer<o2::ft0::MCLabel> labels;
93  o2::ft0::Digit digit;
94  std::vector<o2::ft0::Digit> digitAccum; // digit accumulator
95  mDigitizer.setMCLabels(&labels);
96  auto& eventParts = context->getEventParts();
97  // loop over all composite collisions given from context
98  // (aka loop over all the interaction records)
99  for (int collID = 0; collID < timesview.size(); ++collID) {
100  mDigitizer.setEventTime(timesview[collID].timeNS);
101  mDigitizer.setInteractionRecord(timesview[collID]);
102  digit.cleardigits();
103  std::vector<std::vector<double>> channel_times;
104  // for each collision, loop over the constituents event and source IDs
105  // (background signal merging is basically taking place here)
106  for (auto& part : eventParts[collID]) {
107  // get the hits for this event and this source
108  hits.clear();
109  retrieveHits(mSimChains, part.sourceID, part.entryID, &hits);
110  LOG(INFO) << "For collision " << collID << " eventID " << part.entryID << " source ID " << part.sourceID << " found " << hits.size() << " hits ";
111 
112  // call actual digitization procedure
113  labels.clear();
114  mDigitizer.setEventID(collID);
115  mDigitizer.setSrcID(part.sourceID);
116 
117  mDigitizer.process(&hits, &digit, channel_times);
118  const auto& data = digit.getChDgData();
119  LOG(INFO) << "Have " << data.size() << " fired channels ";
120  // copy digits into accumulator
121  labelAccum.mergeAtBack(labels);
122  }
123  // mDigitizer.computeAverage(digit);
124  mDigitizer.smearCFDtime(&digit, channel_times);
125  mDigitizer.setTriggers(&digit);
126  digitAccum.push_back(digit); // we should move it there actually
127  LOG(INFO) << "Have " << digitAccum.back().getChDgData().size() << " fired channels ";
128  digit.printStream(std::cout);
129  }
130 
131  // here we have all digits and we can send them to consumer (aka snapshot it onto output)
132  pc.outputs().snapshot(Output{DETOR, "DIGITS", 0, Lifetime::Timeframe}, digitAccum);
133  pc.outputs().snapshot(Output{DETOR, "DIGITSMCTR", 0, Lifetime::Timeframe}, labelAccum);
134 
135  LOG(INFO) << "FT0: Sending ROMode= " << mROMode << " to GRPUpdater";
136  pc.outputs().snapshot(Output{DETOR, "ROMode", 0, Lifetime::Timeframe}, mROMode);
137  timer.Stop();
138  LOG(INFO) << "Digitization took " << timer.CpuTime() << "s";
139 
140  // we should be only called once; tell DPL that this process is ready to exit
141  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
142  finished = true;
143  }

158  void retrieveHits(std::vector<TChain*> const& chains,
159  int sourceID,
160  int entryID,
161  std::vector<o2::ft0::HitType>* hits)
162  {
163  std::string detStr = mID.getName();
164  auto br = mSimChains[sourceID]->GetBranch((detStr + "Hit").c_str());
165  if (!br) {
166  LOG(ERROR) << "No branch found";
167  return;
168  }
169  br->SetAddress(&hits);
170  br->GetEntry(entryID);
171  }

193  {"pileup", VariantType::Int, 1, {"whether to run in continuous time mode"}}}
