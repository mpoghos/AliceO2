24 // TableToTree allows to save the contents of a given arrow::Table to a TTree
25 // branchIterator is used by TableToTree
26 //
27 // To write the contents of a table ta to a tree tr on file f do:
28 // . TableToTree t2t(ta,f,treename);
29 // . t2t.AddBranch(branchname1); t2t.AddBranch(branchname2); ...
30 // OR
31 // t2t.AddAllBranches();
32 // . t2t.Process();
33 //
34 // .............................................................................
35 class branchIterator
36 {
37 
38  private:
39  std::string brn; // branch name
40  arrow::ArrayVector chs; // chunks
41  Int_t nchs; // number of chunks
42  Int_t ich; // chunk counter
43  Int_t nr; // number of rows
44  Int_t ir; // row counter
45 
46  // data buffers for each data type
47  bool status = false;
48  arrow::Type::type dt;
49  std::string leaflist;
50 
51  TBranch* br = nullptr;
52 
53  char* dbuf = nullptr;
54  void* v = nullptr;
55 
56  Float_t* var_f = nullptr;
57  Double_t* var_d = nullptr;
58  UShort_t* vs = nullptr;
59  UInt_t* vi = nullptr;
60  ULong64_t* vl = nullptr;
61  Short_t* var_s = nullptr;
62  Int_t* var_i = nullptr;
63  Long64_t* var_l = nullptr;
64 
65  // initialize a branch
66  bool branchini(TTree* tree)
67  {
68 
69  // try to find branch in tree
70  br = tree->GetBranch(brn.c_str());
71 
72  if (!br) {
73 
74  // create new branch of given data type
75  switch (dt) {
76  case arrow::Type::type::FLOAT:
77  leaflist = brn + "/F";
78  break;
79  case arrow::Type::type::DOUBLE:
80  leaflist = brn + "/D";
81  break;
82  case arrow::Type::type::UINT16:
83  leaflist = brn + "/s";
84  break;
85  case arrow::Type::type::UINT32:
86  leaflist = brn + "/i";
87  break;
88  case arrow::Type::type::UINT64:
89  leaflist = brn + "/l";
90  break;
91  case arrow::Type::type::INT16:
92  leaflist = brn + "/S";
93  break;
94  case arrow::Type::type::INT32:
95  leaflist = brn + "/I";
96  break;
97  case arrow::Type::type::INT64:
98  leaflist = brn + "/L";
99  break;
100  default:
101  LOG(FATAL) << "Type not handled: " << dt << std::endl;
102  break;
103  }
104 
105  br = tree->Branch(brn.c_str(), dbuf, leaflist.c_str());
106  }
107  if (!br)
108  return false;
109 
110  return true;
111  };
112 
113  // initialize chunk ib
114  bool dbufini(Int_t ib)
115  {
116  // get next chunk of given data type dt
117  switch (dt) {
118  case arrow::Type::type::FLOAT:
119  var_f = (Float_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::FloatType>>(chs.at(ib))->raw_values();
120  v = (void*)var_f;
121  break;
122  case arrow::Type::type::DOUBLE:
123  var_d = (Double_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::DoubleType>>(chs.at(ib))->raw_values();
124  v = (void*)var_d;
125  break;
126  case arrow::Type::type::UINT16:
127  vs = (UShort_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::UInt16Type>>(chs.at(ib))->raw_values();
128  v = (void*)vs;
129  break;
130  case arrow::Type::type::UINT32:
131  vi = (UInt_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::UInt32Type>>(chs.at(ib))->raw_values();
132  v = (void*)vi;
133  break;
134  case arrow::Type::type::UINT64:
135  vl = (ULong64_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::UInt64Type>>(chs.at(ib))->raw_values();
136  v = (void*)vl;
137  break;
138  case arrow::Type::type::INT16:
139  var_s = (Short_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::Int16Type>>(chs.at(ib))->raw_values();
140  v = (void*)var_s;
141  break;
142  case arrow::Type::type::INT32:
143  var_i = (Int_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::Int32Type>>(chs.at(ib))->raw_values();
144  v = (void*)var_i;
145  break;
146  case arrow::Type::type::INT64:
147  var_l = (Long64_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::Int64Type>>(chs.at(ib))->raw_values();
148  v = (void*)var_l;
149  break;
150  default:
151  LOG(FATAL) << "Type not handled: " << dt << std::endl;
152  break;
153  }
154  br->SetAddress(v);
155 
156  // reset number of rows nr and row counter ir
157  nr = chs.at(ib)->length();
158  ir = 0;
159 
160  return true;
161  };
162 
163  public:
164  branchIterator(TTree* tree, std::shared_ptr<arrow::Column> col)
165  {
166  brn = col->name().c_str();
167  dt = col->type()->id();
168  chs = col->data()->chunks();
169  nchs = chs.size();
170 
171  // initialize the branch
172  status = branchini(tree);
173 
174  ich = 0;
175  status &= dbufini(ich);
176  LOG(DEBUG) << "branchIterator: " << col->type()->ToString() << " with " << nchs << " chunks" << std::endl;
177  };
178 
179  ~branchIterator() = default;
180 
181  // has the iterator been properly initialized
182  bool Status()
183  {
184  return status;
185  };
186 
187  // fills buffer with next value
188  // returns false if end of buffer reached
189  bool push()
190  {
191 
192  // ich and ir contain the current chunk and row
193  // return the next element if available
194  if (ir >= nr) {
195  ich++;
196  if (ich < nchs) {
197  dbufini(ich);
198  } else {
199  // end of data buffer reached
200  return false;
201  }
202  } else {
203  switch (dt) {
204  case arrow::Type::type::FLOAT:
205  v = (void*)var_f++;
206  break;
207  case arrow::Type::type::DOUBLE:
208  v = (void*)var_d++;
209  break;
210  case arrow::Type::type::UINT16:
211  v = (void*)vs++;
212  break;
213  case arrow::Type::type::UINT32:
214  v = (void*)vi++;
215  break;
216  case arrow::Type::type::UINT64:
217  v = (void*)vl++;
218  break;
219  case arrow::Type::type::INT16:
220  v = (void*)var_s++;
221  break;
222  case arrow::Type::type::INT32:
223  v = (void*)var_i++;
224  break;
225  case arrow::Type::type::INT64:
226  v = (void*)var_l++;
227  break;
228  default:
229  LOG(FATAL) << "Type not handled: " << dt << std::endl;
230  break;
231  }
232  }
233  br->SetAddress(v);
234  ir++;
235 
236  return true;
237  };
238 };

241 class TableToTree
242 {
243 
244  private:
245  TTree* tr;
246 
247  // a list of branchIterator
248  std::vector<branchIterator*> brits;
249 
250  // table to convert
251  std::shared_ptr<arrow::Table> ta;
252 
253  public:
254  TableToTree(std::shared_ptr<arrow::Table> table,
255  TFile* file,
256  const char* treename)
257  {
258 
259  ta = table;
260 
261  // try to get the tree
262  tr = (TTree*)file->Get(treename);
263 
264  // create the tree if it does not exist already
265  if (!tr)
266  tr = new TTree(treename, treename);
267  }
268 
269  ~TableToTree()
270  {
271 
272  // clean up branch iterators
273  brits.clear();
274  }
275 
276  bool AddBranch(std::shared_ptr<arrow::Column> col)
277  {
278  branchIterator* brit = new branchIterator(tr, col);
279  if (brit->Status())
280  brits.push_back(brit);
281 
282  return brit->Status();
283  }
284 
285  bool AddAllBranches()
286  {
287 
288  bool status = ta->num_columns() > 0;
289  for (auto ii = 0; ii < ta->num_columns(); ii++) {
290  branchIterator* brit =
291  new branchIterator(tr, ta->column(ii));
292  if (brit->Status()) {
293  brits.push_back(brit);
294  } else {
295  status = false;
296  }
297  }
298 
299  return status;
300  }
301 
302  TTree* Process()
303  {
304 
305  bool togo = true;
306  while (togo) {
307  // fill the tree
308  tr->Fill();
309 
310  // update the branches
311  for (auto ii = 0; ii < ta->num_columns(); ii++)
312  togo &= brits.at(ii)->push();
313  }
314  tr->Write();
315 
316  return tr;
317  }

575 class TreeToTable
576 {
577 
578  private:
579  // the TTreeReader allows to efficiently loop over
580  // the rows of a TTree
581  TTreeReader* reader;
582 
583  // a list of columnIterator*
584  std::vector<std::shared_ptr<columnIterator>> colits;
585 
586  // Append next set of branch values to the
587  // corresponding table columns
588  void push()
589  {
590  for (auto colit : colits)
591  colit->push();
592  }
593 
594  public:
595  TreeToTable(TTree* tree)
596  {
597  // initialize the TTreeReader
598  reader = new TTreeReader(tree);
599  }
600 
601  ~TreeToTable()
602  {
603  if (reader)
604  delete reader;
605  };
606 
607  // add a column to be included in the arrow::table
608  bool AddColumn(const char* colname)
609  {
610  auto colit = std::make_shared<columnIterator>(reader, colname);
611  auto stat = colit->Status();
612  if (stat)
613  colits.push_back(std::move(colit));
614 
615  return stat;
616  }
617 
618  // add all columns
619  bool AddAllColumns()
620  {
621  // get a list of column names
622  auto tree = reader->GetTree();
623  if (!tree) {
624  LOG(INFO) << "Tree not found!";
625  return false;
626  }
627  auto branchList = tree->GetListOfBranches();
628 
629  // loop over branches
630  bool status = !branchList->IsEmpty();
631  for (Int_t ii = 0; ii < branchList->GetEntries(); ii++) {
632  auto br = (TBranch*)branchList->At(ii);
633 
634  // IMPROVE: make sure that a column is not added more than one time
635  auto colit = std::make_shared<columnIterator>(reader, br->GetName());
636  if (colit->Status()) {
637  colits.push_back(std::move(colit));
638  } else {
639  status = false;
640  }
641  }
642 
643  return status;
644  }
645 
646  // do the looping with the TTreeReader and fill the table
647  std::shared_ptr<arrow::Table> Process()
648  {
649 
650  // copy all values from the tree to the table builders
651  reader->Restart();
652  while (reader->Next())
653  push();
654 
655  // prepare the elements needed to create the final table
656  std::vector<std::shared_ptr<arrow::Array>> array_vector;
657  std::vector<std::shared_ptr<arrow::Field>> schema_vector;
658  for (auto colit : colits) {
659  colit->finish();
660  array_vector.push_back(colit->Array());
661  schema_vector.push_back(colit->Schema());
662  }
663  auto fields = std::make_shared<arrow::Schema>(schema_vector);
664 
665  // create the final table
666  // ta is of type std::shared_ptr<arrow::Table>
667  auto ta = (arrow::Table::Make(fields, array_vector));
668 
669  return ta;
670  }
671 };
