92 inline int phase(int i, bool forceNoPhase)
93 {
94  // generate the phase for the i-th ElinkEncoder
95  // the default value of -1 means it will be random and decided
96  // by the ElinkEncoder ctor
97  //
98  // if > 0 it will set a fixed phase at the beginning of the life
99  // of the ElinkEncoder
100  //
101  // returning zero will simply disable the phase
102 
103  if (forceNoPhase) {
104  return 0;
105  }
106  return -1;
107 }
