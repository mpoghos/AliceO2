245 GPUdi() void TPCFastTransform::convPadTimeToUV(int slice, int row, float pad, float time, float& u, float& v, float vertexTime) const
246 {
247  bool sideC = (slice >= getGeometry().getNumberOfSlicesA());
248 
249  const TPCFastTransformGeo::RowInfo& rowInfo = getGeometry().getRowInfo(row);
250  const TPCFastTransformGeo::SliceInfo& sliceInfo = getGeometry().getSliceInfo(slice);
251 
252  float x = rowInfo.x;
253  u = (pad - 0.5 * rowInfo.maxPad) * rowInfo.padWidth;
254 
255  float y = sideC ? -u : u; // pads are mirrorred on C-side
256  float yLab = y * sliceInfo.cosAlpha + x * sliceInfo.sinAlpha;
257 
258  v = (time - mT0 - vertexTime) * (mVdrift + mVdriftCorrY * yLab) + mLdriftCorr; // drift length cm
259 }

261 GPUdi() void TPCFastTransform::convPadTimeToUVinTimeFrame(int slice, int row, float pad, float time, float& u, float& v, float maxTimeBin) const
262 {
263  const TPCFastTransformGeo::RowInfo& rowInfo = getGeometry().getRowInfo(row);
264  u = (pad - 0.5 * rowInfo.maxPad) * rowInfo.padWidth;
265  v = (time - mT0 - maxTimeBin) * mVdrift + mLdriftCorr; // drift length cm
266  if (slice < getGeometry().getNumberOfSlicesA()) {
267  v += getGeometry().getTPCzLengthA();
268  } else {
269  v += getGeometry().getTPCzLengthC();
270  }
271 }

273 GPUdi() void TPCFastTransform::convUVtoPadTime(int slice, int row, float u, float v, float& pad, float& time, float vertexTime) const
274 {
275  bool sideC = (slice >= getGeometry().getNumberOfSlicesA());
276 
277  const TPCFastTransformGeo::RowInfo& rowInfo = getGeometry().getRowInfo(row);
278  const TPCFastTransformGeo::SliceInfo& sliceInfo = getGeometry().getSliceInfo(slice);
279 
280  pad = u / rowInfo.padWidth + 0.5 * rowInfo.maxPad;
281 
282  float x = rowInfo.x;
283  float y = sideC ? -u : u; // pads are mirrorred on C-side
284  float yLab = y * sliceInfo.cosAlpha + x * sliceInfo.sinAlpha;
285  time = mT0 + vertexTime + (v - mLdriftCorr) / (mVdrift + mVdriftCorrY * yLab);
286 }

288 GPUdi() void TPCFastTransform::convUVtoPadTimeInTimeFrame(int slice, int row, float u, float v, float& pad, float& time, float maxTimeBin) const
289 {
290  if (slice < getGeometry().getNumberOfSlicesA()) {
291  v -= getGeometry().getTPCzLengthA();
292  } else {
293  v -= getGeometry().getTPCzLengthC();
294  }
295  const TPCFastTransformGeo::RowInfo& rowInfo = getGeometry().getRowInfo(row);
296  pad = u / rowInfo.padWidth + 0.5 * rowInfo.maxPad;
297  time = mT0 + maxTimeBin + (v - mLdriftCorr) / mVdrift;
298 }

300 GPUdi() void TPCFastTransform::getTOFcorrection(int slice, int /*row*/, float x, float y, float z, float& dz) const
301 {
302  // calculate time of flight correction for z coordinate
303 
304  bool sideC = (slice >= getGeometry().getNumberOfSlicesA());
305  float distZ = z - mPrimVtxZ;
306  float dv = -GPUCommonMath::Sqrt(x * x + y * y + distZ * distZ) * mTOFcorr;
307  dz = sideC ? dv : -dv;
308 }

310 GPUdi() void TPCFastTransform::Transform(int slice, int row, float pad, float time, float& x, float& y, float& z, float vertexTime) const
311 {
317 
318  const TPCFastTransformGeo::RowInfo& rowInfo = getGeometry().getRowInfo(row);
319 
320  // const SliceInfo &sliceInfo = getSliceInfo( slice );
321  // bool sideC = ( slice >= NumberOfSlices / 2 );
322 
323  x = rowInfo.x;
324  float u = 0, v = 0;
325  convPadTimeToUV(slice, row, pad, time, u, v, vertexTime);
326 
327  if (mApplyDistortion) {
328  float dx, du, dv;
329  mDistortion.getDistortion(slice, row, u, v, dx, du, dv);
330  x += dx;
331  u += du;
332  v += dv;
333  }
334 
335  getGeometry().convUVtoLocal(slice, u, v, y, z);
336 
337  float dzTOF = 0;
338  getTOFcorrection(slice, row, x, y, z, dzTOF);
339  z += dzTOF;
340 }

383 GPUdi() float TPCFastTransform::convTimeToZinTimeFrame(int slice, float time, float maxTimeBin) const
384 {
391 
392  float v = (time - mT0 - maxTimeBin) * mVdrift + mLdriftCorr; // drift length cm
393  float z = getGeometry().getTPCalignmentZ(); // global TPC alignment
394  if (slice < getGeometry().getNumberOfSlicesA()) {
395  z -= v;
396  } else {
397  z += v;
398  }
399  return z;
400 }

402 GPUdi() float TPCFastTransform::convZtoTimeInTimeFrame(float z, float maxTimeBin) const
403 {
405  z = z - getGeometry().getTPCalignmentZ(); // global TPC alignment
406  float v = fabs(z);
407  return mT0 + maxTimeBin + (v - mLdriftCorr) / mVdrift;
408 }
