43 class FileStream : public arrow::io::InputStream
44 {
45  public:
46  FileStream(FILE* stream) : mStream(stream)
47  {
48  set_mode(arrow::io::FileMode::READ);
49  }
50  ~FileStream() override = default;
51 
52  // FIXME: handle return code
53  arrow::Status Close() override { return arrow::Status::OK(); }
54  bool closed() const override { return false; }
55 
56  arrow::Status Tell(int64_t* position) const override
57  {
58  *position = mPos;
59  return arrow::Status::OK();
60  }
61 
62  arrow::Status Read(int64_t nbytes, int64_t* bytes_read, void* out) override
63  {
64  auto count = fread(out, nbytes, 1, mStream);
65  if (ferror(mStream) == 0) {
66  *bytes_read = nbytes;
67  mPos += nbytes;
68  } else {
69  *bytes_read = 0;
70  }
71  return arrow::Status::OK();
72  }
73 
74  arrow::Status Read(int64_t nbytes, std::shared_ptr<arrow::Buffer>* out) override
75  {
76  std::shared_ptr<arrow::ResizableBuffer> buffer;
77  ARROW_RETURN_NOT_OK(AllocateResizableBuffer(nbytes, &buffer));
78  int64_t bytes_read;
79  ARROW_RETURN_NOT_OK(Read(nbytes, &bytes_read, buffer->mutable_data()));
80  ARROW_RETURN_NOT_OK(buffer->Resize(bytes_read, false));
81  buffer->ZeroPadding();
82  *out = buffer;
83  return arrow::Status::OK();
84  }
85 
86  private:
87  FILE* mStream = nullptr;
88  int64_t mPos = 0;
89 };

106 uint64_t calculateReadMask(std::vector<OutputRoute> const& routes, header::DataOrigin const& origin)
107 {
108  uint64_t readMask = None;
109  for (auto& route : routes) {
110  auto concrete = DataSpecUtils::asConcreteDataTypeMatcher(route.matcher);
111  auto description = concrete.description;
112  if (description == header::DataDescription{"TRACKPAR"}) {
113  readMask |= AODTypeMask::Tracks;
114  } else if (description == header::DataDescription{"TRACKPARCOV"}) {
115  readMask |= AODTypeMask::TracksCov;
116  } else if (description == header::DataDescription{"TRACKEXTRA"}) {
117  readMask |= AODTypeMask::TracksExtra;
118  } else if (description == header::DataDescription{"CALO"}) {
119  readMask |= AODTypeMask::Calo;
120  } else if (description == header::DataDescription{"MUON"}) {
121  readMask |= AODTypeMask::Muon;
122  } else if (description == header::DataDescription{"VZERO"}) {
123  readMask |= AODTypeMask::VZero;
124  } else if (description == header::DataDescription{"COLLISION"}) {
125  readMask |= AODTypeMask::Collisions;
126  } else if (description == header::DataDescription{"TIMEFRAME"}) {
127  readMask |= AODTypeMask::Timeframe;
128  } else if (description == header::DataDescription{"DZEROFLAGGED"}) {
129  readMask |= AODTypeMask::DZeroFlagged;
130  } else {
131  throw std::runtime_error(std::string("Unknown AOD type: ") + description.str);
132  }
133  }
134  return readMask;
135 }

137 AlgorithmSpec AODReaderHelpers::run2ESDConverterCallback()
138 {
139  auto callback = AlgorithmSpec{adaptStateful([](ConfigParamRegistry const& options,
140  ControlService& control,
141  DeviceSpec const& spec) {
142  std::vector<std::string> filenames;
143  auto filename = options.get<std::string>("esd-file");
144  auto nEvents = options.get<int>("events");
145 
146  if (filename.empty()) {
147  LOG(error) << "Option --esd-file did not provide a filename";
148  control.readyToQuit(QuitRequest::All);
149  return adaptStateless([](RawDeviceService& service) {
150  service.device()->WaitFor(std::chrono::milliseconds(1000));
151  });
152  }
153 
154  // If option starts with a @, we consider the file as text which contains a list of
155  // files.
156  if (filename.size() && filename[0] == '@') {
157  try {
158  filename.erase(0, 1);
159  std::ifstream filelist(filename);
160  while (std::getline(filelist, filename)) {
161  filenames.push_back(filename);
162  }
163  } catch (...) {
164  LOG(error) << "Unable to process file list: " << filename;
165  }
166  } else {
167  filenames.push_back(filename);
168  }
169 
170  uint64_t readMask = calculateReadMask(spec.outputs, header::DataOrigin{"AOD"});
171  auto counter = std::make_shared<unsigned int>(0);
172  return adaptStateless([readMask,
173  counter,
174  filenames,
175  spec, nEvents](DataAllocator& outputs, ControlService& ctrl, RawDeviceService& service) {
176  if (*counter >= filenames.size()) {
177  LOG(info) << "All input files processed";
178  ctrl.endOfStream();
179  ctrl.readyToQuit(QuitRequest::Me);
180  return;
181  }
182  auto f = filenames[*counter];
183  setenv("O2RUN2CONVERTER", "run2ESD2Run3AOD", 0);
184  auto command = std::string(getenv("O2RUN2CONVERTER"));
185  if (nEvents > 0) {
186  command += fmt::format(" -n {} ", nEvents);
187  }
188  FILE* pipe = popen((command + " " + f).c_str(), "r");
189  if (pipe == nullptr) {
190  LOG(ERROR) << "Unable to run converter: " << (command + " " + f).c_str() << f;
191  ctrl.endOfStream();
192  ctrl.readyToQuit(QuitRequest::All);
193  return;
194  }
195  *counter += 1;
196 
198  while ((feof(pipe) == false) && (ferror(pipe) == 0)) {
199  // Skip extra 0-padding...
200  int c = fgetc(pipe);
201  if (c == 0 || c == EOF) {
202  continue;
203  }
204  ungetc(c, pipe);
205 
206  std::shared_ptr<arrow::RecordBatchReader> reader;
207  auto input = std::make_shared<FileStream>(pipe);
208  auto readerStatus = arrow::ipc::RecordBatchStreamReader::Open(input, &reader);
209  if (readerStatus.ok() == false) {
210  LOG(ERROR) << "Reader status not ok: " << readerStatus.message();
211  break;
212  }
213 
214  while (true) {
215  std::shared_ptr<arrow::RecordBatch> batch;
216  auto status = reader->ReadNext(&batch);
217  if (batch.get() == nullptr) {
218  break;
219  }
220  std::unordered_map<std::string, std::string> meta;
221  batch->schema()->metadata()->ToUnorderedMap(&meta);
222  std::shared_ptr<arrow::ipc::RecordBatchWriter> writer;
223  if (meta["description"] == "TRACKPAR" && (readMask & AODTypeMask::Tracks)) {
224  writer = outputs.make<arrow::ipc::RecordBatchWriter>(Output{"AOD", "TRACKPAR"}, batch->schema());
225  } else if (meta["description"] == "TRACKPARCOV" && (readMask & AODTypeMask::TracksCov)) {
226  writer = outputs.make<arrow::ipc::RecordBatchWriter>(Output{"AOD", "TRACKPARCOV"}, batch->schema());
227  } else if (meta["description"] == "TRACKEXTRA" && (readMask & AODTypeMask::TracksExtra)) {
228  writer = outputs.make<arrow::ipc::RecordBatchWriter>(Output{"AOD", "TRACKEXTRA"}, batch->schema());
229  } else if (meta["description"] == "CALO" && (readMask & AODTypeMask::Calo)) {
230  writer = outputs.make<arrow::ipc::RecordBatchWriter>(Output{"AOD", "CALO"}, batch->schema());
231  } else if (meta["description"] == "MUON" && (readMask & AODTypeMask::Muon)) {
232  writer = outputs.make<arrow::ipc::RecordBatchWriter>(Output{"AOD", "MUON"}, batch->schema());
233  } else if (meta["description"] == "VZERO" && (readMask & AODTypeMask::VZero)) {
234  writer = outputs.make<arrow::ipc::RecordBatchWriter>(Output{"AOD", "VZERO"}, batch->schema());
235  } else if (meta["description"] == "COLLISION" && (readMask & AODTypeMask::Collisions)) {
236  writer = outputs.make<arrow::ipc::RecordBatchWriter>(Output{"AOD", "COLLISION"}, batch->schema());
237  } else if (meta["description"] == "TIMEFRAME" && (readMask & AODTypeMask::Timeframe)) {
238  writer = outputs.make<arrow::ipc::RecordBatchWriter>(Output{"AOD", "TIMEFRAME"}, batch->schema());
239  } else {
240  continue;
241  }
242  auto writeStatus = writer->WriteRecordBatch(*batch);
243  if (writeStatus.ok() == false) {
244  throw std::runtime_error("Error while writing record");
245  }
246  }
247  }
248  if (ferror(pipe)) {
249  LOG(ERROR) << "Error while reading from PIPE";
250  }
251  pclose(pipe);
252  });
253  })};
254 
255  return callback;
256 }

258 AlgorithmSpec AODReaderHelpers::rootFileReaderCallback()
259 {
260  auto callback = AlgorithmSpec{adaptStateful([](ConfigParamRegistry const& options,
261  DeviceSpec const& spec) {
262  std::vector<std::string> filenames;
263  auto filename = options.get<std::string>("aod-file");
264 
265  // If option starts with a @, we consider the file as text which contains a list of
266  // files.
267  if (filename.size() && filename[0] == '@') {
268  try {
269  filename.erase(0, 1);
270  std::ifstream filelist(filename);
271  while (std::getline(filelist, filename)) {
272  filenames.push_back(filename);
273  }
274  } catch (...) {
275  LOG(error) << "Unable to process file list: " << filename;
276  }
277  } else {
278  filenames.push_back(filename);
279  }
280 
281  uint64_t readMask = calculateReadMask(spec.outputs, header::DataOrigin{"AOD"});
282  auto counter = std::make_shared<int>(0);
283  return adaptStateless([readMask,
284  counter,
285  filenames](DataAllocator& outputs, ControlService& control, DeviceSpec const& device) {
286  // Each parallel reader reads the files whose index is associated to
287  // their inputTimesliceId
288  assert(device.inputTimesliceId < device.maxInputTimeslices);
289  size_t fi = (*counter * device.maxInputTimeslices) + device.inputTimesliceId;
290  if (fi >= filenames.size()) {
291  LOG(info) << "All input files processed";
292  control.endOfStream();
293  control.readyToQuit(QuitRequest::Me);
294  return;
295  }
296  auto f = filenames[fi];
297  LOG(INFO) << "Processing " << f;
298  auto infile = std::make_unique<TFile>(f.c_str());
299  *counter += 1;
300  if (infile.get() == nullptr || infile->IsOpen() == false) {
301  LOG(ERROR) << "File not found: " + f;
302  return;
303  }
304 
306  if (readMask & AODTypeMask::Collisions) {
307  std::unique_ptr<TTreeReader> reader = std::make_unique<TTreeReader>("O2events", infile.get());
308  auto& collisionBuilder = outputs.make<TableBuilder>(Output{"AOD", "COLLISION"});
309  RootTableBuilderHelpers::convertASoA<o2::aod::Collisions>(collisionBuilder, *reader);
310  }
311 
312  if (readMask & AODTypeMask::Tracks) {
313  std::unique_ptr<TTreeReader> reader = std::make_unique<TTreeReader>("O2tracks", infile.get());
314  auto& trackParBuilder = outputs.make<TableBuilder>(Output{"AOD", "TRACKPAR"});
315  RootTableBuilderHelpers::convertASoA<o2::aod::Tracks>(trackParBuilder, *reader);
316  }
317 
318  if (readMask & AODTypeMask::TracksCov) {
319  std::unique_ptr<TTreeReader> covReader = std::make_unique<TTreeReader>("O2tracks", infile.get());
320  auto& trackParCovBuilder = outputs.make<TableBuilder>(Output{"AOD", "TRACKPARCOV"});
321  RootTableBuilderHelpers::convertASoA<o2::aod::TracksCov>(trackParCovBuilder, *covReader);
322  }
323 
324  if (readMask & AODTypeMask::TracksExtra) {
325  std::unique_ptr<TTreeReader> extraReader = std::make_unique<TTreeReader>("O2tracks", infile.get());
326  auto& extraBuilder = outputs.make<TableBuilder>(Output{"AOD", "TRACKEXTRA"});
327  RootTableBuilderHelpers::convertASoA<o2::aod::TracksExtra>(extraBuilder, *extraReader);
328  }
329 
330  if (readMask & AODTypeMask::Calo) {
331  std::unique_ptr<TTreeReader> extraReader = std::make_unique<TTreeReader>("O2calo", infile.get());
332  auto& extraBuilder = outputs.make<TableBuilder>(Output{"AOD", "CALO"});
333  RootTableBuilderHelpers::convertASoA<o2::aod::Calos>(extraBuilder, *extraReader);
334  }
335 
336  if (readMask & AODTypeMask::Muon) {
337  std::unique_ptr<TTreeReader> muReader = std::make_unique<TTreeReader>("O2muon", infile.get());
338  auto& muBuilder = outputs.make<TableBuilder>(Output{"AOD", "MUON"});
339  RootTableBuilderHelpers::convertASoA<o2::aod::Muons>(muBuilder, *muReader);
340  }
341 
342  if (readMask & AODTypeMask::VZero) {
343  std::unique_ptr<TTreeReader> vzReader = std::make_unique<TTreeReader>("O2vzero", infile.get());
344  auto& vzBuilder = outputs.make<TableBuilder>(Output{"AOD", "VZERO"});
345  RootTableBuilderHelpers::convertASoA<o2::aod::Muons>(vzBuilder, *vzReader);
346  }
347 
348  // Candidates as described by Gianmichele example
349  if (readMask & AODTypeMask::DZeroFlagged) {
350  std::unique_ptr<TTreeReader> dzReader = std::make_unique<TTreeReader>("fTreeDzeroFlagged", infile.get());
351 
352  TTreeReaderValue<float> c0(*dzReader, "d_len_ML");
353  TTreeReaderValue<int> c1(*dzReader, "cand_type_ML");
354  TTreeReaderValue<float> c2(*dzReader, "cos_p_ML");
355  TTreeReaderValue<float> c3(*dzReader, "cos_p_xy_ML");
356  TTreeReaderValue<float> c4(*dzReader, "d_len_xy_ML");
357  TTreeReaderValue<float> c5(*dzReader, "eta_prong0_ML");
358  TTreeReaderValue<float> c6(*dzReader, "eta_prong1_ML");
359  TTreeReaderValue<float> c7(*dzReader, "imp_par_prong0_ML");
360  TTreeReaderValue<float> c8(*dzReader, "imp_par_prong1_ML");
361  TTreeReaderValue<float> c9(*dzReader, "imp_par_xy_ML");
362  TTreeReaderValue<float> c10(*dzReader, "inv_mass_ML");
363  TTreeReaderValue<float> c11(*dzReader, "max_norm_d0d0exp_ML");
364  TTreeReaderValue<float> c12(*dzReader, "norm_dl_xy_ML");
365  TTreeReaderValue<float> c13(*dzReader, "pt_cand_ML");
366  TTreeReaderValue<float> c14(*dzReader, "pt_prong0_ML");
367  TTreeReaderValue<float> c15(*dzReader, "pt_prong1_ML");
368  TTreeReaderValue<float> c16(*dzReader, "y_cand_ML");
369  TTreeReaderValue<float> c17(*dzReader, "phi_cand_ML");
370  TTreeReaderValue<float> c18(*dzReader, "eta_cand_ML");
371  TTreeReaderValue<int> c19(*dzReader, "cand_evtID_ML");
372  TTreeReaderValue<int> c20(*dzReader, "cand_fileID_ML");
373 
374  auto& dzBuilder = outputs.make<TableBuilder>(Output{"AOD", "DZEROFLAGGED"});
375  RootTableBuilderHelpers::convertTTree(dzBuilder, *dzReader,
376  c0, c1, c2, c3, c4, c5, c6, c7,
377  c8, c9, c10, c11, c12, c13, c14,
378  c15, c16, c17, c18, c19, c20);
379  }
380  });
381  })};
382 
383  return callback;
384 }
