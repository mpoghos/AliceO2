55 std::string MergerInfrastructureBuilder::validateConfig()
56 {
57  std::string error;
58  const std::string preamble = "mergers::MergerInfrastructureBuilder error: ";
59  if (mInfrastructureName.empty()) {
60  error += preamble + "the infrastructure name is empty\n";
61  }
62  if (mInputs.empty()) {
63  error += preamble + "no inputs specified\n";
64  }
65  if (DataSpecUtils::validate(mOutputSpec) == false) {
66  error += preamble + "invalid output\n";
67  }
68 
69  if (mConfig.topologySize.value == TopologySize::NumberOfLayers && mConfig.topologySize.param < 1) {
70  error += preamble + "number of layers less than 1 (" + std::to_string(mConfig.topologySize.param) + ")\n";
71  }
72  if (mConfig.topologySize.value == TopologySize::ReductionFactor && mConfig.topologySize.param < 2) {
73  error += preamble + "reduction factor smaller than 2 (" + std::to_string(mConfig.topologySize.param) + ")\n";
74  }
75 
76  if (mConfig.publicationDecision.value == PublicationDecision::WhenXInputsUpdated && (mConfig.publicationDecision.param <= 0.0 || mConfig.publicationDecision.param > 1.0)) {
77  error += preamble + "parameter for PublicationDecision::WhenXInputsUpdated is not inside range (0, 1], it is " + std::to_string(mConfig.publicationDecision.param) + "\n";
78  }
79 
80  if (mConfig.ownershipMode.value == OwnershipMode::Full && mConfig.mergingTime.value != MergingTime::BeforePublication) {
81  error += preamble + "with OwnershipMode::Full, only MergingTime::BeforePublication is allowed.";
82  }
83 
84  return error;
85 }

87 framework::WorkflowSpec MergerInfrastructureBuilder::generateInfrastructure()
88 {
89  if (std::string error = validateConfig(); !error.empty()) {
90  throw std::runtime_error(error);
91  }
92 
93  //todo: remember about range inputs!!! // solution: check if an input is concrete?
94  framework::WorkflowSpec workflow;
95  auto layerInputs = mInputs;
96 
97  // preparing some numbers
98  auto mergersPerLayer = computeNumberOfMergersPerLayer(layerInputs.size());
99 
100  // actual topology generation
101  MergerBuilder mergerBuilder;
102  mergerBuilder.setName(mInfrastructureName);
103  for (size_t layer = 1; layer < mergersPerLayer.size(); layer++) {
104 
105  size_t numberOfMergers = mergersPerLayer[layer];
106  size_t inputsPerMerger = layerInputs.size() / numberOfMergers;
107  size_t inputsPerMergerRemainder = layerInputs.size() % numberOfMergers;
108 
109  MergerConfig layerConfig = mConfig;
110  if (layer > 1 && mConfig.ownershipMode.value == OwnershipMode::Integral) {
111  layerConfig.ownershipMode = {OwnershipMode::Full}; // in Integral mode only the first layer should integrate
112  layerConfig.timespan = {Timespan::LastDifference}; // and objects that are merged should not be used again
113  }
114  mergerBuilder.setConfig(layerConfig);
115 
116  framework::Inputs nextLayerInputs;
117  auto inputsRangeBegin = layerInputs.begin();
118 
119  for (size_t m = 0; m < numberOfMergers; m++) {
120 
121  mergerBuilder.setTopologyPosition(layer, m);
122 
123  auto inputsRangeEnd = inputsRangeBegin + inputsPerMerger + (m < inputsPerMergerRemainder);
124  mergerBuilder.setInputSpecs(framework::Inputs(inputsRangeBegin, inputsRangeEnd));
125  inputsRangeBegin = inputsRangeEnd;
126 
127  if (numberOfMergers == 1) {
128  assert(layer == mergersPerLayer.size() - 1);
129  // the last layer => use the specified external OutputSpec
130  mergerBuilder.setOutputSpec(mOutputSpec);
131  }
132 
133  auto merger = mergerBuilder.buildSpec();
134 
135  auto input = DataSpecUtils::matchingInput(merger.outputs.at(0));
136  input.binding = "in";
137  nextLayerInputs.push_back(input);
138 
139  workflow.emplace_back(std::move(merger));
140  }
141  layerInputs = nextLayerInputs; //todo: could be optimised with pointers
142  }
143 
144  return workflow;
145 }

147 std::vector<size_t> MergerInfrastructureBuilder::computeNumberOfMergersPerLayer(const size_t inputs) const
148 {
149  std::vector<size_t> mergersPerLayer{inputs};
150  if (mConfig.topologySize.value == TopologySize::NumberOfLayers) {
151  // _ _
152  // | L - i | where:
153  // | ----- | L - number of layers
154  // |V| --- | |V| L | i - layer index (0 - input layer)
155  // | |i --- | | |0 | M_i - number of mergers in i layer
156  // | |
157  //
158 
159  size_t L = mConfig.topologySize.param;
160  for (size_t i = 1; i <= L; i++) {
161  mergersPerLayer.push_back(static_cast<size_t>(ceil(pow(inputs, (L - i) / static_cast<double>(L)))));
162  }
163 
164  } else { // mConfig.topologySize.value == TopologySize::ReductionFactor
165  // _ _
166  // | |V| | where:
167  // |V| --- | | |i-1 | R - reduction factor
168  // | |i --- | -------- | i - layer index (0 - input layer)
169  // | R | M_i - number of mergers in i layer
170  //
171 
172  double R = mConfig.topologySize.param;
173  size_t Mi, prevMi = inputs;
174  do {
175  Mi = static_cast<size_t>(ceil(prevMi / R));
176  mergersPerLayer.push_back(Mi);
177  prevMi = Mi;
178  } while (Mi > 1);
179  }
180 
181  return mergersPerLayer;
182 }
