42 void TrackInterpolation::process()
43 {
44  // main processing function
45  loadEntryForTrees(0); // TODO for now all input data is stored in trees with a single entry
46 
47  std::set<unsigned int> tracksDone; // to store indices of ITS-TPC matched tracks that have been processed
48 
49  LOG(info) << "Processing " << mTOFMatchVecInput->size() << " ITS-TPC-TOF matched tracks out of "
50  << mITSTPCTrackVecInput->size() << " ITS-TPC matched tracks";
51 
52  // TODO reserve only a fraction of the needed space for all tracks? How many tracks pass on average the quality cuts with how many TPC clusters?
53  mTrackData.reserve(mTOFMatchVecInput->size());
54  mClRes.reserve(mTOFMatchVecInput->size() * param::NPadRows);
55 
56  int nTracksTPC = mTPCTrackVecInput->size();
57 
58  for (const auto& trkTOF : *mTOFMatchVecInput) {
59  // process ITS-TPC-TOF matched tracks
60  if (!trackPassesQualityCuts(mITSTPCTrackVecInput->at(trkTOF.first.getIndex()))) {
61  continue;
62  }
63  mTrackData.emplace_back();
64  if (!interpolateTrackITSTOF(trkTOF)) {
65  mTrackData.pop_back();
66  continue;
67  }
68  mTrackData.back().nTracksInEvent = nTracksTPC;
69  tracksDone.insert(trkTOF.first.getIndex());
70  }
71 
72  LOG(info) << "Could process " << tracksDone.size() << " ITS-TPC-TOF matched tracks successfully";
73 
74  if (mDoITSOnlyTracks) {
75  size_t nTracksDoneITS = 0;
76  size_t nTracksSkipped = 0;
77  for (std::size_t iTrk = 0; iTrk < mITSTPCTrackVecInput->size(); ++iTrk) {
78  // process ITS-TPC matched tracks that were not matched to TOF
79  if (tracksDone.find(iTrk) != tracksDone.end()) {
80  // track also has a matching cluster in TOF and has already been processed
81  ++nTracksSkipped;
82  continue;
83  }
84  const auto& trk = mITSTPCTrackVecInput->at(iTrk);
85  if (!trackPassesQualityCuts(trk, false)) {
86  continue;
87  }
88  mTrackData.emplace_back();
89  const auto& trkTPC = mTPCTrackVecInput->at(trk.getRefTPC());
90  const auto& trkITS = mITSTrackVecInput->at(trk.getRefITS());
91  if (!extrapolateTrackITS(trkITS, trkTPC, trk.getTimeMUS().getTimeStamp(), trk.getRefTPC())) {
92  mTrackData.pop_back();
93  continue;
94  }
95  mTrackData.back().nTracksInEvent = nTracksTPC;
96  ++nTracksDoneITS;
97  }
98  LOG(info) << "Could process " << nTracksDoneITS << " ITS-TPC matched tracks successfully";
99  LOG(info) << "Skipped " << nTracksSkipped << " tracks, as they were successfully propagated to TOF";
100  }
101 
102  if (mTreeOutTrackData) {
103  mTreeOutTrackData->Fill();
104  }
105  if (mTreeOutClusterRes) {
106  mTreeOutClusterRes->Fill();
107  }
108 }

110 bool TrackInterpolation::trackPassesQualityCuts(const o2::dataformats::TrackTPCITS& matchITSTPC, bool hasOuterPoint) const
111 {
112  // apply track quality cuts (assume different settings for track with and without points in TRD or TOF)
113  const auto& trkTPC = mTPCTrackVecInput->at(matchITSTPC.getRefTPC());
114  const auto& trkITS = mITSTrackVecInput->at(matchITSTPC.getRefITS());
115  if (hasOuterPoint) {
116  // track has a match in TRD or TOF
117  if (trkTPC.getNClusterReferences() < param::MinTPCNCls ||
118  trkITS.getNumberOfClusters() < param::MinITSNCls) {
119  return false;
120  }
121  if (trkTPC.getChi2() / trkTPC.getNClusterReferences() > param::MaxTPCChi2 ||
122  trkITS.getChi2() / trkITS.getNumberOfClusters() > param::MaxITSChi2) {
123  return false;
124  }
125  } else {
126  // ITS-TPC only track
127  if (trkTPC.getNClusterReferences() < param::MinTPCNClsNoOuterPoint ||
128  trkITS.getNumberOfClusters() < param::MinITSNClsNoOuterPoint) {
129  return false;
130  }
131  if (trkTPC.getChi2() / trkTPC.getNClusterReferences() > param::MaxTPCChi2 ||
132  trkITS.getChi2() / trkITS.getNumberOfClusters() > param::MaxITSChi2) {
133  return false;
134  }
135  }
136  return true;
137 }

139 bool TrackInterpolation::interpolateTrackITSTOF(const std::pair<o2::dataformats::EvIndex<>, o2::dataformats::MatchInfoTOF>& matchTOF)
140 {
141  // get TPC cluster residuals to ITS-TOF only tracks
142  size_t trkIdx = mTrackData.size() - 1;
143  auto propagator = o2::base::Propagator::Instance();
144  const auto& matchITSTPC = mITSTPCTrackVecInput->at(matchTOF.first.getIndex());
145  const auto& clTOF = mTOFClusterVecInput->at(matchTOF.second.getTOFClIndex());
146  //const int clTOFSec = (TMath::ATan2(-clTOF.getY(), -clTOF.getX()) + o2::constants::math::PI) * o2::constants::math::Rad2Deg * 0.05; // taken from TOF cluster class as there is no const getter for the sector
147  const int clTOFSec = clTOF.getCount();
148  const float clTOFAlpha = o2::utils::Sector2Angle(clTOFSec);
149  const auto& trkTPC = mTPCTrackVecInput->at(matchITSTPC.getRefTPC());
150  const auto& trkITS = mITSTrackVecInput->at(matchITSTPC.getRefITS());
151  auto trkWork = trkITS.getParamOut();
152  // reset the cache array (sufficient to set )
153  for (auto& elem : mCache) {
154  elem.clAvailable = 0;
155  }
156  mTrackData[trkIdx].clIdx.setFirstEntry(mClRes.size()); // reference the first cluster residual belonging to this track
157  //printf("=== New Track with pt = %.3f, nClsTPC = %i ===\n", trkWork.getQ2Pt(), trkTPC.getNClusterReferences());
158 
159  // store the TPC cluster positions in the cache
160  float clusterTimeBinOffset = matchITSTPC.getTimeMUS().getTimeStamp() / mTPCTimeBinMUS;
161  for (int iCl = trkTPC.getNClusterReferences(); iCl--;) {
162  uint8_t sector, row;
163  uint32_t clusterIndexInRow;
164  trkTPC.getClusterReference(*mTPCTrackClIdxVecInput, iCl, sector, row, clusterIndexInRow);
165  const auto& clTPC = trkTPC.getCluster(*mTPCTrackClIdxVecInput, iCl, *mTPCClusterIdxStruct, sector, row);
166  float clTPCX;
167  std::array<float, 2> clTPCYZ;
168  mFastTransform->TransformIdeal(sector, row, clTPC.getPad(), clTPC.getTime(), clTPCX, clTPCYZ[0], clTPCYZ[1], clusterTimeBinOffset);
169  sector %= SECTORSPERSIDE;
170  mCache[row].clAvailable = 1;
171  mCache[row].clY = clTPCYZ[0];
172  mCache[row].clZ = clTPCYZ[1];
173  mCache[row].clAngle = o2::utils::Sector2Angle(sector);
174  }
175 
176  // first extrapolate through TPC and store track position at each pad row
177  for (int iRow = 0; iRow < param::NPadRows; ++iRow) {
178  if (!mCache[iRow].clAvailable) {
179  continue;
180  }
181  if (!trkWork.rotate(mCache[iRow].clAngle)) {
182  LOG(debug) << "Failed to rotate track during first extrapolation";
183  return false;
184  }
185  if (!propagator->PropagateToXBxByBz(trkWork, param::RowX[iRow], o2::constants::physics::MassPionCharged, mMaxSnp, mMaxStep, mMatCorr)) {
186  LOG(debug) << "Failed on first extrapolation";
187  return false;
188  }
189  mCache[iRow].y[ExtOut] = trkWork.getY();
190  mCache[iRow].z[ExtOut] = trkWork.getZ();
191  mCache[iRow].sy2[ExtOut] = trkWork.getSigmaY2();
192  mCache[iRow].szy[ExtOut] = trkWork.getSigmaZY();
193  mCache[iRow].sz2[ExtOut] = trkWork.getSigmaZ2();
194  mCache[iRow].phi[ExtOut] = trkWork.getSnp();
195  mCache[iRow].tgl[ExtOut] = trkWork.getTgl();
196  //printf("Track alpha at row %i: %.2f, Y(%.2f), Z(%.2f)\n", iRow, trkWork.getAlpha(), trkWork.getY(), trkWork.getZ());
197  }
198 
199  // now continue to TOF and update track with TOF cluster
200  if (!trkWork.rotate(clTOFAlpha)) {
201  LOG(debug) << "Failed to rotate into TOF cluster sector frame";
202  return false;
203  }
204  //float ca, sa;
205  //o2::utils::sincosf(clTOFAlpha, sa, ca);
206  //float clTOFX = clTOF.getX() * ca + clTOF.getY() * sa; // cluster x in sector coordinate frame
207  //std::array<float, 2> clTOFYZ{ -clTOF.getX() * sa + clTOF.getY() * ca, clTOF.getZ() }; // cluster y and z in sector coordinate frame
208  float clTOFX = clTOF.getX();
209  std::array<float, 2> clTOFYZ{clTOF.getY(), clTOF.getZ()};
210  std::array<float, 3> clTOFCov{mSigYZ2TOF, 0.f, mSigYZ2TOF}; // assume no correlation between y and z and equal cluster error sigma^2 = (3cm)^2 / 12
211  if (!propagator->PropagateToXBxByBz(trkWork, clTOFX, o2::constants::physics::MassPionCharged, mMaxSnp, mMaxStep, mMatCorr)) {
212  LOG(debug) << "Failed final propagation to TOF radius";
213  return false;
214  }
215  if (!trkWork.update(clTOFYZ, clTOFCov)) {
216  LOG(debug) << "Failed to update extrapolated ITS track with TOF cluster";
217  return false;
218  }
219 
220  // go back through the TPC and store updated track positions
221  for (int iRow = param::NPadRows; iRow--;) {
222  if (!mCache[iRow].clAvailable) {
223  continue;
224  }
225  if (!trkWork.rotate(mCache[iRow].clAngle)) {
226  LOG(debug) << "Failed to rotate track during back propagation";
227  return false;
228  }
229  if (!propagator->PropagateToXBxByBz(trkWork, param::RowX[iRow], o2::constants::physics::MassPionCharged, mMaxSnp, mMaxStep, mMatCorr)) {
230  LOG(debug) << "Failed on back propagation";
231  //printf("trkX(%.2f), clX(%.2f), clY(%.2f), clZ(%.2f), alphaTOF(%.2f)\n", trkWork.getX(), param::RowX[iRow], clTOFYZ[0], clTOFYZ[1], clTOFAlpha);
232  return false;
233  }
234  mCache[iRow].y[ExtIn] = trkWork.getY();
235  mCache[iRow].z[ExtIn] = trkWork.getZ();
236  mCache[iRow].sy2[ExtIn] = trkWork.getSigmaY2();
237  mCache[iRow].szy[ExtIn] = trkWork.getSigmaZY();
238  mCache[iRow].sz2[ExtIn] = trkWork.getSigmaZ2();
239  mCache[iRow].phi[ExtIn] = trkWork.getSnp();
240  mCache[iRow].tgl[ExtIn] = trkWork.getTgl();
241  }
242 
243  // calculate weighted mean at each pad row (assume for now y and z are uncorrelated) and store residuals to TPC clusters
244  unsigned short deltaRow = 0;
245  unsigned short nMeasurements = 0;
246  for (int iRow = 0; iRow < param::NPadRows; ++iRow) {
247  if (!mCache[iRow].clAvailable) {
248  ++deltaRow;
249  continue;
250  }
251  float wTotY = 1.f / mCache[iRow].sy2[ExtOut] + 1.f / mCache[iRow].sy2[ExtIn];
252  float wTotZ = 1.f / mCache[iRow].sz2[ExtOut] + 1.f / mCache[iRow].sz2[ExtIn];
253  mCache[iRow].y[Int] = (mCache[iRow].y[ExtOut] / mCache[iRow].sy2[ExtOut] + mCache[iRow].y[ExtIn] / mCache[iRow].sy2[ExtIn]) / wTotY;
254  mCache[iRow].z[Int] = (mCache[iRow].z[ExtOut] / mCache[iRow].sz2[ExtOut] + mCache[iRow].z[ExtIn] / mCache[iRow].sz2[ExtIn]) / wTotZ;
255 
256  // simple average w/o weighting for angles
257  mCache[iRow].phi[Int] = (mCache[iRow].phi[ExtOut] + mCache[iRow].phi[ExtIn]) / 2.f;
258  mCache[iRow].tgl[Int] = (mCache[iRow].tgl[ExtOut] + mCache[iRow].tgl[ExtIn]) / 2.f;
259 
260  TPCClusterResiduals res;
261  res.setDY(mCache[iRow].clY - mCache[iRow].y[Int]);
262  res.setDZ(mCache[iRow].clZ - mCache[iRow].z[Int]);
263  res.setY(mCache[iRow].y[Int]);
264  res.setZ(mCache[iRow].z[Int]);
265  res.setPhi(mCache[iRow].phi[Int]);
266  res.setTgl(mCache[iRow].tgl[Int]);
267  res.sec = o2::utils::Angle2Sector(mCache[iRow].clAngle);
268  res.dRow = deltaRow;
269  res.row = iRow;
270  mClRes.push_back(std::move(res));
271  ++nMeasurements;
272  deltaRow = 1;
273  }
274 
275  mTrackData[trkIdx].trkId = matchITSTPC.getRefTPC();
276  mTrackData[trkIdx].eta = trkTPC.getEta();
277  mTrackData[trkIdx].phi = trkTPC.getSnp();
278  mTrackData[trkIdx].qPt = trkTPC.getQ2Pt();
279  mTrackData[trkIdx].chi2TPC = trkTPC.getChi2();
280  mTrackData[trkIdx].chi2ITS = trkITS.getChi2();
281  mTrackData[trkIdx].nClsTPC = trkTPC.getNClusterReferences();
282  mTrackData[trkIdx].nClsITS = trkITS.getNumberOfClusters();
283  mTrackData[trkIdx].clIdx.setEntries(nMeasurements);
284 
285  return true;
286 }

288 bool TrackInterpolation::extrapolateTrackITS(const o2::its::TrackITS& trkITS, const TrackTPC& trkTPC, float trkTime, int trkIdTPC)
289 {
290  // extrapolate ITS-only track through TPC and store residuals to TPC clusters in the output vectors
291  size_t trkIdx = mTrackData.size() - 1;
292  mTrackData[trkIdx].clIdx.setFirstEntry(mClRes.size());
293  auto trk = trkITS.getParamOut();
294  float clusterTimeBinOffset = trkTime / mTPCTimeBinMUS;
295  auto propagator = o2::base::Propagator::Instance();
296  unsigned short rowPrev = 0;
297  unsigned short nMeasurements = 0;
298  for (int iCl = trkTPC.getNClusterReferences(); iCl--;) {
299  uint8_t sector, row;
300  uint32_t clusterIndexInRow;
301  trkTPC.getClusterReference(*mTPCTrackClIdxVecInput, iCl, sector, row, clusterIndexInRow);
302  const auto& cl = trkTPC.getCluster(*mTPCTrackClIdxVecInput, iCl, *mTPCClusterIdxStruct, sector, row);
303  float x = 0, y = 0, z = 0;
304  mFastTransform->TransformIdeal(sector, row, cl.getPad(), cl.getTime(), x, y, z, clusterTimeBinOffset);
305  if (!trk.rotate(o2::utils::Sector2Angle(sector))) {
306  return false;
307  }
308  if (!propagator->PropagateToXBxByBz(trk, x, o2::constants::physics::MassPionCharged, mMaxSnp, mMaxStep, mMatCorr)) {
309  return false;
310  }
311  TPCClusterResiduals res;
312  res.setDY(y - trk.getY());
313  res.setDY(z - trk.getZ());
314  res.setY(trk.getY());
315  res.setZ(trk.getZ());
316  res.setPhi(trk.getSnp());
317  res.setTgl(trk.getTgl());
318  res.sec = o2::utils::Angle2Sector(trk.getAlpha());
319  res.dRow = row - rowPrev;
320  res.row = row;
321  rowPrev = row;
322  mClRes.push_back(std::move(res));
323  ++nMeasurements;
324  }
325  mTrackData[trkIdx].trkId = trkIdTPC;
326  mTrackData[trkIdx].eta = trkTPC.getEta();
327  mTrackData[trkIdx].phi = trkTPC.getSnp();
328  mTrackData[trkIdx].qPt = trkTPC.getQ2Pt();
329  mTrackData[trkIdx].chi2TPC = trkTPC.getChi2();
330  mTrackData[trkIdx].chi2ITS = trkITS.getChi2();
331  mTrackData[trkIdx].nClsTPC = trkTPC.getNClusterReferences();
332  mTrackData[trkIdx].nClsITS = trkITS.getNumberOfClusters();
333  mTrackData[trkIdx].clIdx.setEntries(nMeasurements);
334 
335  return true;
336 }

351 void TrackInterpolation::attachInputTrees()
