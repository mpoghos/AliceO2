78 struct Stat {
79  double mean{0};
80  double rms{0};
81  double q{0};
82  int n{0};
83  void incr(int v)
84  {
85  n++;
86  auto newMean = mean + (v - mean) / n;
87  q += (v - newMean) * (v - mean);
88  rms = sqrt(q / n);
89  mean = newMean;
90  }
91 };

96  return os;

98 template <typename FORMAT, typename CHARGESUM, typename RDH>
99 std::map<std::string, Stat> rawdump(std::string input, DumpOptions opt)
100 {
101  std::ifstream in(input.c_str(), std::ios::binary);
102  if (!in.good()) {
103  std::cout << "could not open file " << input << "\n";
104  return {};
105  }
106  constexpr size_t pageSize = 8192;
107 
108  std::array<uint8_t, pageSize> buffer;
109  gsl::span<uint8_t> sbuffer(buffer);
110 
111  size_t ndigits{0};
112 
113  std::map<std::string, int> uniqueDS;
114  std::map<std::string, int> uniqueChannel;
115  std::map<std::string, Stat> statChannel;
116 
117  memset(&buffer[0], 0, buffer.size());
118  auto channelHandler = [&ndigits, &uniqueDS, &uniqueChannel, &statChannel](DsElecId dsId,
119  uint8_t channel, o2::mch::raw::SampaCluster sc) {
120  auto s = asString(dsId);
121  uniqueDS[s]++;
122  auto ch = fmt::format("{}-CH{}", s, channel);
123  uniqueChannel[ch]++;
124  auto& stat = statChannel[ch];
125  for (auto d = 0; d < sc.nofSamples(); d++) {
126  stat.incr(sc.samples[d]);
127  }
128  ++ndigits;
129  };
130 
131  auto cruLink2solar = o2::mch::raw::createCruLink2SolarMapper<ElectronicMapperGenerated>();
132 
133  size_t nrdhs{0};
134  auto rdhHandler = [&](const RDH& rdh) -> std::optional<RDH> {
135  nrdhs++;
136  if (opt.showRDHs()) {
137  std::cout << nrdhs << "--" << rdh << "\n";
138  }
139  auto r = rdh;
140  auto cruId = r.cruID;
141  if (opt.cruId().has_value()) {
142  // force cruId to externally given value
143  cruId = opt.cruId().value();
144  }
145  auto linkId = rdhLinkId(r);
146  auto solar = cruLink2solar(o2::mch::raw::CruLinkId(cruId, linkId, opt.deId()));
147  if (!solar.has_value()) {
148  std::cout << fmt::format("ERROR - Could not get solarUID from CRU,LINK=({},{},{})\n",
149  cruId, linkId, opt.deId());
150  return std::nullopt;
151  }
152  r.feeId = solar.value();
153  return r;
154  };
155 
156  o2::mch::raw::Decoder decode = o2::mch::raw::createDecoder<FORMAT, CHARGESUM, RDH>(rdhHandler, channelHandler);
157 
158  std::vector<std::chrono::microseconds> timers;
159 
160  size_t npages{0};
161  DecoderStat decStat;
162 
163  while (npages < opt.maxNofRDHs() && in.read(reinterpret_cast<char*>(&buffer[0]), pageSize)) {
164  npages++;
165  decStat = decode(sbuffer);
166  }
167 
168  if (!opt.json()) {
169  std::cout << ndigits << " digits seen - " << nrdhs << " RDHs seen - " << npages << " npages read\n";
170  std::cout << "#unique DS=" << uniqueDS.size() << " #unique Channel=" << uniqueChannel.size() << "\n";
171  std::cout << decStat << "\n";
172  }
173  return statChannel;
174 }

176 void output(const std::map<std::string, Stat>& channels)
177 {
178  rapidjson::StringBuffer buffer;
179  rapidjson::Writer<rapidjson::StringBuffer> writer(buffer);
180 
181  writer.StartObject();
182  writer.Key("channels");
183  writer.StartArray();
184  for (auto s : channels) {
185  writer.StartObject();
186  writer.Key("id");
187  writer.String(s.first.c_str());
188  writer.Key("ped");
189  writer.Double(s.second.mean);
190  writer.Key("noise");
191  writer.Double(s.second.rms);
192  writer.Key("nof_samples");
193  writer.Int(s.second.n);
194  writer.EndObject();
195  }
196  writer.EndArray();
197  writer.EndObject();
198  std::cout << buffer.GetString() << "\n";
199 }

201 int main(int argc, char* argv[])
202 {
203  std::string prefix;
204  std::vector<int> detElemIds;
205  std::string inputFile;
206  po::variables_map vm;
207  po::options_description generic("Generic options");
208  unsigned int nrdhs{0};
209  unsigned int deId{0};
210  bool showRDHs{false};
211  bool userLogic{false}; // default format is bareformat...
212  bool chargeSum{false}; //... in sample mode
213  bool jsonOutput{false};
214 
215  // clang-format off
216  generic.add_options()
217  ("help,h", "produce help message")
218  ("input-file,i", po::value<std::string>(&inputFile)->required(), "input file name")
219  ("nrdhs,n", po::value<unsigned int>(&nrdhs), "number of RDHs to go through")
220  ("showRDHs,s",po::bool_switch(&showRDHs),"show RDHs")
221  ("userLogic,u",po::bool_switch(&userLogic),"user logic format")
222  ("chargeSum,c",po::bool_switch(&chargeSum),"charge sum format")
223  ("json,j",po::bool_switch(&jsonOutput),"output means and rms in json format")
224  ("de,d",po::value<unsigned int>(&deId)->required(),"detection element id of the data to be decoded")
225  ("cru",po::value<uint16_t>(),"force cruId")
226  ;
227  // clang-format on
228 
229  po::options_description cmdline;
230  cmdline.add(generic);
231 
232  po::store(po::command_line_parser(argc, argv).options(cmdline).run(), vm);
233 
234  if (vm.count("help")) {
235  std::cout << generic << "\n";
236  return 2;
237  }
238 
239  try {
240  po::notify(vm);
241  } catch (boost::program_options::error& e) {
242  std::cout << "Error: " << e.what() << "\n";
243  exit(1);
244  }
245 
246  DumpOptions opt(deId, nrdhs, showRDHs, jsonOutput);
247  std::map<std::string, Stat> statChannel;
248 
249  if (vm.count("cru")) {
250  opt.cruId(vm["cru"].as<uint16_t>());
251  }
252  if (userLogic) {
253  if (chargeSum) {
254  statChannel = rawdump<UserLogicFormat, ChargeSumMode, RDHv4>(inputFile, opt);
255  } else {
256  statChannel = rawdump<UserLogicFormat, SampleMode, RDHv4>(inputFile, opt);
257  }
258  } else {
259  if (chargeSum) {
260  statChannel = rawdump<BareFormat, ChargeSumMode, RDHv4>(inputFile, opt);
261  } else {
262  statChannel = rawdump<BareFormat, SampleMode, RDHv4>(inputFile, opt);
263  }
264  }
265 
266  if (jsonOutput) {
267  output(statChannel);
268  }
269  return 0;
270 }
