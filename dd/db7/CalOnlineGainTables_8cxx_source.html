27 int CalOnlineGainTables::getArrayOffset(int det, int row, int col) const
28 {
29  FeeParam* mFeeParam = FeeParam::instance();

32  int detoffset = det * 128;
33  int mcmoffset = rob * (mFeeParam->mgkNmcmRob) + mcm;

42 float CalOnlineGainTables::getGainCorrectionFactor(int det, int row, int col) const
43 {
44  //notes for my sanity ...
45  //det gives us the [0-540]
46  //row and col give us the [0-128] 16*rob+mcm
47  int arrayoffset = getArrayOffset(det, row, col);
48  int channel = getChannel(col);
49  float GainCorrectionFactor = 0.0;
50  if (mGainTable[arrayoffset].mAdcdac == 0) {
51  if (mGainTable[arrayoffset].mFGFN[channel] < 0) {
52  GainCorrectionFactor = -1.0;
53  } else if (mGainTable[arrayoffset].mFGFN[channel] > 511) {
54  GainCorrectionFactor = CalOnlineGainTables::UnDef;
55  } else
56  GainCorrectionFactor = (mGainTable[arrayoffset].mFGFN[channel] / 2048.) + 0.875;
57  } else {
58  float ADCCorrection = (1. / (1. + ((float)mGainTable[arrayoffset].mAdcdac / 31.) * 0.4 / 1.05));
59  GainCorrectionFactor = ADCCorrection * (((mGainTable[arrayoffset].mFGFN[channel]) / 2048.) + 0.875);
60  }
61  return GainCorrectionFactor;
62 }
