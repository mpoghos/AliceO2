114  GPUd() int LoadTrack(const T& trk, const int label = -1, const int* nTrkltsOffline = nullptr, const int labelOffline = -1)
115  {
116  if (mNTracks >= mNMaxTracks) {
117 #ifndef GPUCA_GPUCODE
118  GPUError("Error: Track dropped (no memory available) -> must not happen");
119 #endif
120  return (1);
121  }
122  if (!trk.CheckNumericalQuality()) {
123  return (0);
124  }
125  if (CAMath::Abs(trk.getEta()) > mMaxEta) {
126  return (0);
127  }
128  if (trk.getPt() < mMinPt) {
129  return (0);
130  }
131 #ifdef GPUCA_ALIROOT_LIB
132  new (&mTracks[mNTracks]) GPUTRDTrack(trk); // We need placement new, since the class is virtual
133 #else
134  mTracks[mNTracks] = trk;
135 #endif
136  mTracks[mNTracks].SetTPCtrackId(mNTracks);
137  if (label >= 0) {
138  mTracks[mNTracks].SetLabel(label);
139  }
140  if (nTrkltsOffline) {
141  for (int i = 0; i < 4; ++i) {
142  mTracks[mNTracks].SetNtrackletsOffline(i, nTrkltsOffline[i]); // see GPUTRDTrack.h for information on the index
143  }
144  }
145  mTracks[mNTracks].SetLabelOffline(labelOffline);
146  mNTracks++;
147  return (0);
148  }
