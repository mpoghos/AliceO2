306  // type trait to determine the type of the storage
307  // the storage type is different depending on the type, because
308  // - binary branches are supported
309  // - the ROOT TTree API makes a difference between fundamental types where the pointer to variable is
310  // passed and complex objects with dictionary where the pointer to a pointer variable needs to be
311  // passed. The former case will allways include a copy of the value, the latter uses the deserialized
312  // object by pointer
313  // It appears, that also vectors of fundamental and also messageable tyapes can be treated in the same
314  // way as fundamental types, as long as there is a corresponding TClass implementation. ROOT creates those
315  // for vectors of fundamental types, for all other types, ClassDefNV macro is needed in the class
316  // declaration together with entry in LinkDef.
317  template <typename T, typename _ = void>
318  struct StructureElementTypeTrait {
319  };

387  // the default method creates branch using address to store variable
388  // Note: the type of the store variable is pointer to object for objects with a ROOT TClass
389  // interface, or plain type for all others
390  // specialized on specialization_id
391  template <typename S, typename std::enable_if_t<!std::is_same<S, BinaryBranchSpecialization>::value, int> = 0>
392  TBranch* createBranch(TTree* tree, const char* name, size_t branchIdx)
393  {
394  return tree->Branch(name, &(mStore.at(branchIdx)));

397  // specialization for binary buffers indicated by const char*
398  // a variable binary data branch is written, the size of each entry is stored in a size branch
399  // specialized on specialization_id
400  template <typename S, typename std::enable_if_t<std::is_same<S, BinaryBranchSpecialization>::value, int> = 0>
401  TBranch* createBranch(TTree* tree, const char* name, size_t branchIdx)
402  {
403  // size variable to write the size of the data branch
404  std::get<2>(mStore.at(branchIdx)) = 1;
405  // branch name of the size branch
406  std::string sizeBranchName = std::string(name) + "Size";
407  // leaf list: one leaf of type unsinged int
408  std::string leafList = sizeBranchName + "/i";
409  std::get<1>(mStore.at(branchIdx)) = tree->Branch(sizeBranchName.c_str(), &(std::get<2>(mStore.at(branchIdx))), leafList.c_str());
410  return tree->Branch(name, &(std::get<0>(mStore.at(branchIdx))));
411  }

414  void setupInstance(std::vector<BranchSpec>& specs, TTree* tree)
415  {
416  // recursing through the tree structure by simply using method of the previous type,
417  // i.e. the base class method.
418  PrevT::setupInstance(specs, tree);
419  constexpr size_t SpecIndex = STAGE - 1;
420  specs[SpecIndex].classinfo = TClass::GetClass(typeid(value_type));
421  if (std::is_same<value_type, const char*>::value == false && std::is_fundamental<value_type>::value == false &&
422  specs[SpecIndex].classinfo == nullptr) {
423  // for all non-fundamental types but the special case for binary chunks, a dictionary is required
424  // FIXME: find a reliable way to check that the type has been specified in the LinkDef
425  // Only then the required functionality for streaming the type to the branch is available.
426  // If e.g. a standard container of some ROOT serializable type has not been specified in
427  // LinkDef, the functionality is not available and the attempt to stream will simply crash.
428  // Unfortunately, a class info object can be extracted for the type, so this check does not help
429  throw std::runtime_error(std::to_string(SpecIndex) + ": no dictionary available for non-fundamental type " + typeid(value_type).name());
430  }
431  size_t branchIdx = 0;
432  mStore.resize(specs[SpecIndex].names.size());
433  for (auto const& name : specs[SpecIndex].names) {
434  specs[SpecIndex].branches.at(branchIdx) = createBranch<specialization_id>(tree, name.c_str(), branchIdx);
435  if (specs[SpecIndex].branches.at(branchIdx) == nullptr) {
436  throw std::runtime_error(std::to_string(SpecIndex) + ": can not create branch " + name + " for type " + typeid(value_type).name() + " - LinkDef entry missing?");
437  }
438  LOG(INFO) << SpecIndex << ": branch " << name << " set up";
439  branchIdx++;
440  }
441  }

453  // specialization for non-messageable types with ROOT dictionary
454  // for non-trivial structs, the address of the pointer to the objects needs to be used
455  // in order to directly use the pointer to extracted object
456  // store is a pointer to object
457  template <typename S, typename std::enable_if_t<std::is_same<S, ROOTTypeSpecialization>::value, int> = 0>
458  void fillData(InputContext& context, const char* key, TBranch* branch, size_t branchIdx)
459  {
460  auto data = context.get<typename std::add_pointer<value_type>::type>(key);
461  // this is ugly but necessary because of the TTree API does not allow a const
462  // object as input. Have to rely on that ROOT treats the object as const
463  mStore[branchIdx] = const_cast<value_type*>(data.get());
464  branch->Fill();
465  }

467  // specialization for binary buffers using const char*
468  // this writes both the data branch and a size branch
469  template <typename S, typename std::enable_if_t<std::is_same<S, BinaryBranchSpecialization>::value, int> = 0>
470  void fillData(InputContext& context, const char* key, TBranch* branch, size_t branchIdx)
471  {
472  auto data = context.get<gsl::span<char>>(key);
473  std::get<2>(mStore.at(branchIdx)) = data.size();
474  std::get<1>(mStore.at(branchIdx))->Fill();
475  std::get<0>(mStore.at(branchIdx)).resize(data.size());
476  memcpy(std::get<0>(mStore.at(branchIdx)).data(), data.data(), data.size());
477  branch->Fill();
478  }

480  // specialization for vectors of messageable types
481  template <typename S, typename std::enable_if_t<std::is_same<S, MessageableVectorSpecialization>::value, int> = 0>
482  void fillData(InputContext& context, const char* key, TBranch* branch, size_t branchIdx)
483  {
484  using ValueType = typename value_type::value_type;
485  static_assert(is_messageable<ValueType>::value, "logical error: should be correctly selected by StructureElementTypeTrait");
486  // if the value type is messagable and has a ROOT dictionary, two serialization methods are possible
487  // for the moment, the InputRecord API can not handle both with the same call
488  try {
489  // try extracting from message with serialization method NONE, throw runtime error
490  // if message is serialized
491  auto data = context.get<gsl::span<ValueType>>(key);
492  value_type clone(data.begin(), data.end());
493  mStore[branchIdx] = &clone;
494  branch->Fill();
495  } catch (const std::runtime_error& e) {
496  if constexpr (has_root_dictionary<value_type>::value == true) {
497  // try extracting from message with serialization method ROOT
498  auto data = context.get<typename std::add_pointer<value_type>::type>(key);
499  mStore[branchIdx] = const_cast<value_type*>(data.get());
500  branch->Fill();
501  } else {
502  // the type has no ROOT dictionary, re-throw exception
503  throw e;
504  }
505  }
506  }

508  // process previous stage and this stage
509  void process(InputContext& context, std::vector<BranchSpec>& specs)
510  {
511  // recursing through the tree structure by simply using method of the previous type,
512  // i.e. the base class method.
513  PrevT::process(context, specs);
514  constexpr size_t SpecIndex = STAGE - 1;
515  BranchSpec const& spec = specs[SpecIndex];
516  // loop over all defined inputs
517  for (auto const& key : spec.keys) {
518  auto dataref = context.get(key.c_str());
519  size_t branchIdx = 0;
520  if (spec.getIndex) {
521  branchIdx = spec.getIndex(dataref);
522  if (branchIdx == ~(size_t)0) {
523  // this indicates skipping
524  continue;
525  }
526  }
527  fillData<specialization_id>(context, key.c_str(), spec.branches.at(branchIdx), branchIdx);
528  }
529  }

537  template <typename BASE, typename T, typename... Args>
538  std::enable_if_t<std::is_base_of<BranchDef<typename T::type, typename T::key_type, typename T::key_extractor>, T>::value, std::unique_ptr<TreeStructureInterface>>
539  createTreeStructure(T def, Args&&... args)
540  {
541  mBranchSpecs.push_back({{}, {def.branchName}});
542  auto& spec = mBranchSpecs.back();
543 
544  // extract the internal keys for the list of provided input definitions
545  size_t idx = 0;
546  spec.keys.resize(def.keys.size());
547  std::generate(spec.keys.begin(), spec.keys.end(), [&def, &idx] { return T::key_extractor::asString(def.keys[idx++]); });
548  mBranchSpecs.back().branches.resize(def.nofBranches, nullptr);
549  // the number of branches has to match the number of inputs but can be larger depending
550  // on the exact functionality provided with the getIndex callback. In any case, the
551  // callbacks only need to be propagated if multiple branches are defined
552  assert(def.nofBranches >= spec.keys.size());
553  // a getIndex function makes only sense if there are multiple branches
554  assert(def.nofBranches == 1 || def.getIndex);
555  if (def.nofBranches > 1) {
556  assert(def.getIndex && def.getName);
557  mBranchSpecs.back().getIndex = def.getIndex;
558  mBranchSpecs.back().getName = def.getName;
559  mBranchSpecs.back().names.resize(def.nofBranches);
560 
561  // fill the branch names by calling the getName callback
562  idx = 0;
563  std::generate(mBranchSpecs.back().names.begin(), mBranchSpecs.back().names.end(),
564  [&def, &idx]() { return def.getName(def.branchName, idx++); });
565  }
566  using type = TreeStructureElement<typename T::type, BASE>;
567  return std::move(createTreeStructure<type>(std::forward<Args>(args)...));
568  }
