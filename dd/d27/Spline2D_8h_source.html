261 {
263  const Spline1D& gridU = getGridU();
264  const Spline1D& gridV = getGridV();
265  int nu = gridU.getNumberOfKnots();
266  int iv = iKnot / nu;
267  int iu = iKnot % nu;
268  u = gridU.getKnot(iu).u;
269  v = gridV.getKnot(iv).u;

272 template <typename T>
273 GPUdi() void Spline2D::interpolate(int Ndim, GPUgeneric() const T* parameters, float u, float v, GPUgeneric() T Suv[/*Ndim*/]) const
274 {
275  // Get interpolated value for f(u,v) using parameters[getNumberOfParameters()]
276 
277  const Spline1D& gridU = getGridU();
278  const Spline1D& gridV = getGridV();
279  int nu = gridU.getNumberOfKnots();
280  int iu = gridU.getKnotIndex(u);
281  int iv = gridV.getKnotIndex(v);
282 
283  const Spline1D::Knot& knotU = gridU.getKnot(iu);
284  const Spline1D::Knot& knotV = gridV.getKnot(iv);
285 
286  const int Ndim2 = Ndim * 2;
287  const int Ndim4 = Ndim * 4;
288 
289  // X:=Sx, Y:=Sy, Z:=Sz
290 
291  const T* par00 = parameters + (nu * iv + iu) * Ndim4; // values { {X,Y,Z}, {X,Y,Z}'v, {X,Y,Z}'u, {X,Y,Z}''vu } at {u0, v0}
292  const T* par10 = par00 + Ndim4; // values { ... } at {u1, v0}
293  const T* par01 = par00 + Ndim4 * nu; // values { ... } at {u0, v1}
294  const T* par11 = par01 + Ndim4; // values { ... } at {u1, v1}
295 
296  T Su0[Ndim4]; // values { {X,Y,Z,X'v,Y'v,Z'v}(v0), {X,Y,Z,X'v,Y'v,Z'v}(v1) }, at u0
297  T Du0[Ndim4]; // derivatives {}'_u at u0
298  T Su1[Ndim4]; // values { {X,Y,Z,X'v,Y'v,Z'v}(v0), {X,Y,Z,X'v,Y'v,Z'v}(v1) }, at u1
299  T Du1[Ndim4]; // derivatives {}'_u at u1
300 
301  for (int i = 0; i < Ndim2; i++) {
302  Su0[i] = par00[i];
303  Su0[Ndim2 + i] = par01[i];
304 
305  Du0[i] = par00[Ndim2 + i];
306  Du0[Ndim2 + i] = par01[Ndim2 + i];
307 
308  Su1[i] = par10[i];
309  Su1[Ndim2 + i] = par11[i];
310 
311  Du1[i] = par10[Ndim2 + i];
312  Du1[Ndim2 + i] = par11[Ndim2 + i];
313  }
314 
315  T parU[Ndim4]; // interpolated values { {X,Y,Z,X'v,Y'v,Z'v}(v0), {X,Y,Z,X'v,Y'v,Z'v}(v1) } at u
316  gridU.interpolate<T>(Ndim4, knotU, Su0, Du0, Su1, Du1, u, parU);
317 
318  const T* Sv0 = parU + 0;
319  const T* Dv0 = parU + Ndim;
320  const T* Sv1 = parU + Ndim2;
321  const T* Dv1 = parU + Ndim2 + Ndim;
322 
323  gridV.interpolate<T>(Ndim, knotV, Sv0, Dv0, Sv1, Dv1, v, Suv);
324 }
