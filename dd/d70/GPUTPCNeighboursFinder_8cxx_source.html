22 GPUd() void GPUTPCNeighboursFinder::Thread<0>(int /*nBlocks*/, int nThreads, int iBlock, int iThread, GPUsharedref() MEM_LOCAL(GPUSharedMemory) & s, processorType& tracker)
23 {
24  //* find neighbours
25 
26 #ifdef GPUCA_GPUCODE
27  for (unsigned int i = iThread; i < sizeof(MEM_PLAIN(GPUTPCRow)) / sizeof(int); i += nThreads) {
28  reinterpret_cast<GPUsharedref() int*>(&s.mRow)[i] = reinterpret_cast<GPUglobalref() int*>(&tracker.SliceDataRows()[iBlock])[i];
29  if (iBlock >= 2 && iBlock < GPUCA_ROW_COUNT - 2) {
30  reinterpret_cast<GPUsharedref() int*>(&s.mRowUp)[i] = reinterpret_cast<GPUglobalref() int*>(&tracker.SliceDataRows()[iBlock + 2])[i];
31  reinterpret_cast<GPUsharedref() int*>(&s.mRowDown)[i] = reinterpret_cast<GPUglobalref() int*>(&tracker.SliceDataRows()[iBlock - 2])[i];
32  }
33  }
34  GPUbarrier();
35 #endif
36  if (iThread == 0) {
37  s.mIRow = iBlock;
38  if (s.mIRow < GPUCA_ROW_COUNT) {
39 #ifdef GPUCA_GPUCODE
40  GPUsharedref() const MEM_LOCAL(GPUTPCRow)& row = s.mRow;
41 #else
42  GPUglobalref() const MEM_GLOBAL(GPUTPCRow)& row = tracker.Row(s.mIRow);
43 #endif
44  s.mNHits = row.NHits();
45 
46  if ((s.mIRow >= 2) && (s.mIRow <= GPUCA_ROW_COUNT - 3)) {
47  s.mIRowUp = s.mIRow + 2;
48  s.mIRowDn = s.mIRow - 2;
49 
50  // references to the rows above and below
51 
52 #ifdef GPUCA_GPUCODE
53  GPUsharedref() const MEM_LOCAL(GPUTPCRow)& rowUp = s.mRowUp;
54  GPUsharedref() const MEM_LOCAL(GPUTPCRow)& rowDn = s.mRowDown;
55 #else
56  GPUglobalref() const MEM_GLOBAL(GPUTPCRow)& rowUp = tracker.Row(s.mIRowUp);
57  GPUglobalref() const MEM_GLOBAL(GPUTPCRow)& rowDn = tracker.Row(s.mIRowDn);
58 #endif
59  // the axis perpendicular to the rows
60  const float xDn = rowDn.X();
61  const float x = row.X();
62  const float xUp = rowUp.X();
63 
64  // number of hits in rows above and below
65  s.mUpNHits = tracker.Row(s.mIRowUp).NHits();
66  s.mDnNHits = tracker.Row(s.mIRowDn).NHits();
67 
68  // distance of the rows (absolute and relative)
69  s.mUpDx = xUp - x;
70  s.mDnDx = xDn - x;
71  s.mUpTx = xUp / x;
72  s.mDnTx = xDn / x;
73  // UpTx/DnTx is used to move the HitArea such that central events are preferred (i.e. vertices
74  // coming from y = 0, z = 0).
75 
76  // s.mGridUp = tracker.Row( s.mIRowUp ).Grid();
77  // s.mGridDn = tracker.Row( s.mIRowDn ).Grid();
78  }
79  }
80  }
81  GPUbarrier();
82 
83  if ((s.mIRow <= 1) || (s.mIRow >= GPUCA_ROW_COUNT - 2)) {
84 #ifdef GPUCA_GPUCODE
85  GPUsharedref() const MEM_LOCAL(GPUTPCRow)& row = s.mRow;
86 #else
87  GPUglobalref() const MEM_GLOBAL(GPUTPCRow)& row = tracker.Row(s.mIRow);
88 #endif
89  for (int ih = iThread; ih < s.mNHits; ih += nThreads) {
90  tracker.SetHitLinkUpData(row, ih, CALINK_INVAL);
91  tracker.SetHitLinkDownData(row, ih, CALINK_INVAL);
92  }
93  return;
94  }
95 
96  float chi2Cut = 3.f * 3.f * 4 * (s.mUpDx * s.mUpDx + s.mDnDx * s.mDnDx);
97 // float chi2Cut = 3.*3.*(s.mUpDx*s.mUpDx + s.mDnDx*s.mDnDx ); //SG
98 #ifdef GPUCA_GPUCODE
99  GPUsharedref() const MEM_LOCAL(GPUTPCRow)& row = s.mRow;
100  GPUsharedref() const MEM_LOCAL(GPUTPCRow)& rowUp = s.mRowUp;
101  GPUsharedref() const MEM_LOCAL(GPUTPCRow)& rowDn = s.mRowDown;
102 #else
103  GPUglobalref() const MEM_GLOBAL(GPUTPCRow)& row = tracker.Row(s.mIRow);
104  GPUglobalref() const MEM_GLOBAL(GPUTPCRow)& rowUp = tracker.Row(s.mIRowUp);
105  GPUglobalref() const MEM_GLOBAL(GPUTPCRow)& rowDn = tracker.Row(s.mIRowDn);
106 #endif
107  const float y0 = row.Grid().YMin();
108  const float z0 = row.Grid().ZMin();
109  const float stepY = row.HstepY();
110  const float stepZ = row.HstepZ();
111 
112  for (int ih = iThread; ih < s.mNHits; ih += nThreads) {
113  int linkUp = -1;
114  int linkDn = -1;
115 
116  if (s.mDnNHits > 0 && s.mUpNHits > 0) {
117 // coordinates of the hit in the current row
118 #if defined(GPUCA_TEXTURE_FETCH_NEIGHBORS)
119  cahit2 tmpval = tex1Dfetch(gAliTexRefu2, ((char*)tracker.Data().HitData() - tracker.Data().GPUTextureBase()) / sizeof(cahit2) + row.HitNumberOffset() + ih);
120  const float y = y0 + tmpval.x * stepY;
121  const float z = z0 + tmpval.y * stepZ;
122 #else
123  const float y = y0 + tracker.HitDataY(row, ih) * stepY;
124  const float z = z0 + tracker.HitDataZ(row, ih) * stepZ;
125 #endif // GPUCA_TEXTURE_FETCH_NEIGHBORS
126 
127 #if GPUCA_NEIGHBOURS_FINDER_MAX_NNEIGHUP > 0
128  GPUsharedref() calink* neighUp = s.mB[iThread];
129  GPUsharedref() float2* yzUp = s.mA[iThread];
130  calink neighUp2[GPUCA_MAXN - GPUCA_NEIGHBOURS_FINDER_MAX_NNEIGHUP];
131  float2 yzUp2[GPUCA_MAXN - GPUCA_NEIGHBOURS_FINDER_MAX_NNEIGHUP];
132 #else
133  calink neighUp[GPUCA_MAXN];
134  float2 yzUp[GPUCA_MAXN];
135 #endif // GPUCA_NEIGHBOURS_FINDER_MAX_NNEIGHUP > 0
136 
137  int nNeighUp = 0;
138  GPUTPCHitArea areaDn, areaUp;
139 
140  const float kAngularMultiplier = tracker.Param().rec.SearchWindowDZDR;
141  const float kAreaSize = tracker.Param().rec.NeighboursSearchArea;
142  areaUp.Init(rowUp, tracker.Data(), y * s.mUpTx, kAngularMultiplier != 0.f ? z : (z * s.mUpTx), kAreaSize, kAngularMultiplier != 0.f ? (s.mUpDx * kAngularMultiplier) : kAreaSize);
143  areaDn.Init(rowDn, tracker.Data(), y * s.mDnTx, kAngularMultiplier != 0.f ? z : (z * s.mDnTx), kAreaSize, kAngularMultiplier != 0.f ? (-s.mDnDx * kAngularMultiplier) : kAreaSize);
144 
145  do {
146  GPUTPCHit h;
147  int i = areaUp.GetNext(tracker, rowUp, tracker.Data(), &h);
148  if (i < 0) {
149  break;
150  }
151 
152 #if GPUCA_NEIGHBOURS_FINDER_MAX_NNEIGHUP > 0
153  if (nNeighUp >= GPUCA_NEIGHBOURS_FINDER_MAX_NNEIGHUP) {
154  neighUp2[nNeighUp - GPUCA_NEIGHBOURS_FINDER_MAX_NNEIGHUP] = (calink)i;
155  yzUp2[nNeighUp - GPUCA_NEIGHBOURS_FINDER_MAX_NNEIGHUP] = CAMath::MakeFloat2(s.mDnDx * (h.Y() - y), s.mDnDx * (h.Z() - z));
156  } else
157 #endif
158  {
159  neighUp[nNeighUp] = (calink)i;
160  yzUp[nNeighUp] = CAMath::MakeFloat2(s.mDnDx * (h.Y() - y), s.mDnDx * (h.Z() - z));
161  }
162  if (++nNeighUp >= GPUCA_MAXN) {
163  // GPUInfo("Neighbors buffer ran full...");
164  break;
165  }
166  } while (1);
167 
168  if (nNeighUp > 0) {
169  int bestDn = -1, bestUp = -1;
170  float bestD = 1.e10f;
171 
172  int nNeighDn = 0;
173  do {
174  GPUTPCHit h;
175  int i = areaDn.GetNext(tracker, rowDn, tracker.Data(), &h);
176  if (i < 0) {
177  break;
178  }
179 
180  nNeighDn++;
181  float2 yzdn = CAMath::MakeFloat2(s.mUpDx * (h.Y() - y), s.mUpDx * (h.Z() - z));
182 
183  for (int iUp = 0; iUp < nNeighUp; iUp++) {
184 #if GPUCA_NEIGHBOURS_FINDER_MAX_NNEIGHUP > 0
185  float2 yzup = iUp >= GPUCA_NEIGHBOURS_FINDER_MAX_NNEIGHUP ? yzUp2[iUp - GPUCA_NEIGHBOURS_FINDER_MAX_NNEIGHUP] : yzUp[iUp];
186 #else
187  float2 yzup = yzUp[iUp];
188 #endif
189  float dy = yzdn.x - yzup.x;
190  float dz = yzdn.y - yzup.y;
191  float d = dy * dy + dz * dz;
192  if (d < bestD) {
193  bestD = d;
194  bestDn = i;
195  bestUp = iUp;
196  }
197  }
198  } while (1);
199 
200  if (bestD <= chi2Cut) {
201 #if GPUCA_NEIGHBOURS_FINDER_MAX_NNEIGHUP > 0
202  linkUp = bestUp >= GPUCA_NEIGHBOURS_FINDER_MAX_NNEIGHUP ? neighUp2[bestUp - GPUCA_NEIGHBOURS_FINDER_MAX_NNEIGHUP] : neighUp[bestUp];
203 #else
204  linkUp = neighUp[bestUp];
205 #endif
206  linkDn = bestDn;
207  }
208  }
209  }
210 
211  tracker.SetHitLinkUpData(row, ih, linkUp);
212  tracker.SetHitLinkDownData(row, ih, linkDn);
213  }
