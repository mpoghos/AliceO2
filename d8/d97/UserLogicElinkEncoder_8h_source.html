71 void append(uint64_t prefix, std::vector<uint64_t>& buffer, int& index, uint64_t& word, int data)
72 {
73  word |= static_cast<uint64_t>(data) << (index * 10);
74  --index;
75  if (index < 0) {
76  buffer.emplace_back(prefix | word);
77  index = 4;
78  word = 0;
79  }
80 }

82 template <typename CHARGESUM>
83 void ElinkEncoder<UserLogicFormat, CHARGESUM>::addChannelData(uint8_t chId,
84  const std::vector<SampaCluster>& data)
85 {
86  if (data.empty()) {
87  throw std::invalid_argument("cannot add empty data");
88  }
89  assertNotMixingClusters<CHARGESUM>(data);
90 
91  int error{0}; // FIXME: what to do with error ?
92 
93  uint64_t b9 = dsId64(mElinkId) | error64(error);
94 
95  const uint64_t sync = sampaSync().uint64();
96 
97  if (!mHasSync) {
98  mBuffer.emplace_back(b9 | sync);
99  mHasSync = true;
100  }
101 
102  auto header = buildHeader(mElinkId, chId, data);
103  mBuffer.emplace_back(b9 | header.uint64());
104 
105  mCurrent10BitIndex = 4;
106  CHARGESUM chargeSum;
107  uint64_t word{0};
108  for (auto& cluster : data) {
109  append(b9, mBuffer, mCurrent10BitIndex, word, cluster.nofSamples());
110  append(b9, mBuffer, mCurrent10BitIndex, word, cluster.timestamp);
111  if (chargeSum() == true) {
112  append(b9, mBuffer, mCurrent10BitIndex, word, cluster.chargeSum & 0x3FF);
113  append(b9, mBuffer, mCurrent10BitIndex, word, (cluster.chargeSum & 0xFFC00) >> 10);
114  } else {
115  for (auto& s : cluster.samples) {
116  append(b9, mBuffer, mCurrent10BitIndex, word, s);
117  }
118  }
119  }
120  while (mCurrent10BitIndex != 4) {
121  append(b9, mBuffer, mCurrent10BitIndex, word, 0);
122  }
123 }

132 template <typename CHARGESUM>
133 size_t ElinkEncoder<UserLogicFormat, CHARGESUM>::moveToBuffer(std::vector<uint64_t>& buffer, uint64_t prefix)
134 {
135  if (mBuffer.empty()) {
136  return 0;
137  }
138  auto n = buffer.size();
139  buffer.reserve(n + mBuffer.size());
140  for (auto& b : mBuffer) {
141  buffer.emplace_back(b | prefix);
142  }
143  clear();
144  return buffer.size() - n;
145 }
