55 void doTest(const o2::mid::EventType& inEventType, const std::map<uint16_t, std::vector<o2::mid::ColumnData>>& inData, const std::vector<o2::mid::ROFRecord>& rofRecords, const std::vector<o2::mid::ColumnData>& data)
56 {
57  BOOST_TEST(rofRecords.size() == inData.size());
58  auto inItMap = inData.begin();
59  for (auto rofIt = rofRecords.begin(); rofIt != rofRecords.end(); ++rofIt) {
60  BOOST_TEST(static_cast<int>(rofIt->eventType) == static_cast<int>(inEventType));
61  BOOST_TEST(rofIt->interactionRecord.bc == inItMap->first);
62  BOOST_TEST(rofIt->nEntries == inItMap->second.size());
63  auto sortedIn = sortData(inItMap->second, 0, inItMap->second.size());
64  auto sortedOut = sortData(data, rofIt->firstEntry, rofIt->firstEntry + rofIt->nEntries);
65  for (size_t icol = 0; icol < sortedOut.size(); ++icol) {
66  // size_t inCol = icol - rofIt->firstEntry;
67  BOOST_TEST(sortedOut[icol].deId == sortedIn[icol].deId);
68  BOOST_TEST(sortedOut[icol].columnId == sortedIn[icol].columnId);
69  BOOST_TEST(sortedOut[icol].getNonBendPattern() == sortedIn[icol].getNonBendPattern());
70  for (int iline = 0; iline < 4; ++iline) {
71  BOOST_TEST(sortedOut[icol].getBendPattern(iline) == sortedIn[icol].getBendPattern(iline));
72  }
73  }
74  ++inItMap;
75  }
76 }
