64 framework::DataProcessorSpec MergerBuilder::buildSpec()
65 {
66  framework::DataProcessorSpec merger;
67 
68  merger.name = mergerIdString() + "-" + mName + std::to_string(mLayer) + "l-" + std::to_string(mId);
69 
70  merger.inputs = mInputSpecs;
71 
72  merger.outputs.push_back(mOutputSpec);
73  framework::DataAllocator::SubSpecificationType subSpec = DataSpecUtils::getOptionalSubSpec(mOutputSpec).value();
74  if (DataSpecUtils::validate(mOutputSpec) == false) {
75  // inner layer => generate output spec according to scheme
76  subSpec = mergerSubSpec(mLayer, mId);
77  merger.outputs[0] = OutputSpec{{mergerOutputBinding()},
78  mergerDataOrigin(),
79  mergerDataDescription(mName),
80  subSpec}; // it servers as a unique merger output ID
81  } else {
82  // last layer
83  merger.outputs[0].binding = {mergerOutputBinding()};
84  }
85 
86  merger.algorithm = framework::adaptFromTask<Merger>(mConfig, subSpec);
87 
88  if (mConfig.publicationDecision.value == PublicationDecision::EachNSeconds) {
89  merger.inputs.push_back({"timer-publish", "MRGR", mergerDataDescription("timer-" + mName), mergerSubSpec(mLayer, mId), framework::Lifetime::Timer});
90  merger.options.push_back({"period-timer-publish", framework::VariantType::Int, static_cast<int>(mConfig.publicationDecision.param * 1000000), {"timer period"}});
91  }
92 
93  return std::move(merger);
94 }
