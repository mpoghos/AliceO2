21 void Reconstructor::Process(const Digit& digit, RecPoint& recPoint) const
22 {
23  o2::InteractionRecord intRecord = digit.GetInteractionRecord();
24  recPoint.SetInteractionRecord(intRecord);
25  Double_t eventTime = o2::InteractionRecord::bc2ns(intRecord.bc, intRecord.orbit);
26  recPoint.SetTimeFromDigit(eventTime);
27 
28  std::vector<o2::fdd::ChannelData> channelData = digit.GetChannelData();
29  recPoint.SetChannelData(channelData);
30 
31  //Compute charge weighted average time
32  Double_t timeFDA = 0, timeFDC = 0;
33  Double_t weightFDA = 0.0, weightFDC = 0.0;
34 
35  for (auto& channel : channelData) {
36  Float_t adc = channel.mChargeADC;
37  Float_t time = channel.mTime;
38  //LOG(INFO) <<adc <<" "<<time<<FairLogger::endl;
39  if (time == o2::InteractionRecord::DummyTime)
40  continue;
41  Float_t timeErr = 1;
42  if (adc > 1)
43  timeErr = 1 / adc;
44  if (channel.mPMNumber < 8) {
45  timeFDC += time / (timeErr * timeErr);
46  weightFDC += 1. / (timeErr * timeErr);
47  } else {
48  timeFDA += time / (timeErr * timeErr);
49  weightFDA += 1. / (timeErr * timeErr);
50  }
51  }
52  timeFDA = (weightFDA > 1) ? timeFDA /= weightFDA : o2::InteractionRecord::DummyTime;
53  timeFDC = (weightFDC > 1) ? timeFDC /= weightFDC : o2::InteractionRecord::DummyTime;
54 
55  recPoint.SetMeanTimeFDA(timeFDA);
56  recPoint.SetMeanTimeFDC(timeFDC);
57 }
