234 bool Compressor::decoderRead()
235 {
236  if (!mDecoderFile.is_open()) {
237  std::cout << colorRed << "--- no input file is open"
238  << colorReset
239  << std::endl;
240  return true;
241  }
242 
243  char* inputPointer = mDecoderBuffer;
244  mDecoderFile.read(inputPointer, 64);
245  mDecoderBufferSize = 64;
246  auto rdh = reinterpret_cast<o2::header::RAWDataHeader*>(inputPointer);
247  while (!rdh->stop) {
248  mDecoderFile.read(inputPointer + rdh->headerSize, rdh->offsetToNext - rdh->headerSize);
249  mDecoderBufferSize += (rdh->offsetToNext - rdh->headerSize);
250  inputPointer += rdh->offsetToNext;
251  mDecoderFile.read(inputPointer, 64);
252  mDecoderBufferSize += 64;
253  rdh = reinterpret_cast<o2::header::RAWDataHeader*>(inputPointer);
254  }
255 
257  if (mDecoderFile.eof()) {
258  std::cout << colorRed << "--- Nothing else to read"
259  << colorReset
260  << std::endl;
261  return true;
262  }
263 
264 #ifdef DECODER_VERBOSE
265  if (mDecoderVerbose) {
266  std::cout << colorBlue
267  << "--- DECODER READ HBF: " << mDecoderBufferSize << " bytes"
268  << colorReset
269  << std::endl;
270  }
271 #endif
272 
273  return false;
274 }

301 bool Compressor::processHBF()
302 {
303 
304 #ifdef DECODER_VERBOSE
305  if (mDecoderVerbose) {
306  std::cout << colorBlue
307  << "--- PROCESS HBF"
308  << colorReset
309  << std::endl;
310  }
311 #endif
312 
313  mDecoderRDH = reinterpret_cast<o2::header::RAWDataHeader*>(mDecoderPointer);
314  mEncoderRDH = reinterpret_cast<o2::header::RAWDataHeader*>(mEncoderPointer);
315  auto rdh = mDecoderRDH;
316 
318  while (!rdh->stop) {
319 
320 #ifdef DECODER_VERBOSE
321  if (mDecoderVerbose) {
322  std::cout << colorBlue
323  << "--- RDH open/continue detected"
324  << colorReset
325  << std::endl;
326  o2::raw::HBFUtils::printRDH(*rdh);
327  }
328 #endif
329 
330  auto headerSize = rdh->headerSize;
331  auto memorySize = rdh->memorySize;
332  auto offsetToNext = rdh->offsetToNext;
333  auto drmPayload = memorySize - headerSize;
334 
336  std::memcpy(mDecoderSaveBuffer + mDecoderSaveBufferDataSize, reinterpret_cast<char*>(rdh) + headerSize, drmPayload);
337  mDecoderSaveBufferDataSize += drmPayload;
338 
340  rdh = reinterpret_cast<o2::header::RAWDataHeader*>(reinterpret_cast<char*>(rdh) + offsetToNext);
341 
343  if (reinterpret_cast<char*>(rdh) < mDecoderBuffer + mDecoderBufferSize)
344  continue;
345 
347  return true;
348  }
349 
350 #ifdef DECODER_VERBOSE
351  if (mDecoderVerbose) {
352  std::cout << colorBlue
353  << "--- RDH close detected"
354  << colorReset
355  << std::endl;
356  o2::raw::HBFUtils::printRDH(*rdh);
357  }
358 #endif
359 
361  std::memcpy(mEncoderPointer, mDecoderRDH, mDecoderRDH->headerSize);
362  mEncoderPointer = reinterpret_cast<uint32_t*>(reinterpret_cast<char*>(mEncoderPointer) + rdh->headerSize);
363 
365  mDecoderPointer = reinterpret_cast<uint32_t*>(mDecoderSaveBuffer);
366  mDecoderPointerMax = reinterpret_cast<uint32_t*>(mDecoderSaveBuffer + mDecoderSaveBufferDataSize);
367  while (mDecoderPointer < mDecoderPointerMax) {
368  mEventCounter++;
369  if (processDRM())
370  break;
371  }
372  mDecoderSaveBufferDataSize = 0;
373 
375  if (mDecoderFatal) {
376  mFatalCounter++;
377  mEncoderPointer = mEncoderPointerStart;
378  }
379 
380  if (mDecoderError)
381  mErrorCounter++;
382 
384  mEncoderRDH->memorySize = reinterpret_cast<char*>(mEncoderPointer) - reinterpret_cast<char*>(mEncoderRDH);
385  mEncoderRDH->offsetToNext = mEncoderRDH->memorySize;
386 
389  mEncoderRDH = reinterpret_cast<o2::header::RAWDataHeader*>(mEncoderPointer);
390  std::memcpy(mEncoderRDH, rdh, rdh->headerSize);
391  mEncoderRDH->memorySize = rdh->headerSize;
392  mEncoderRDH->offsetToNext = mEncoderRDH->memorySize;
393  mEncoderPointer = reinterpret_cast<uint32_t*>(reinterpret_cast<char*>(mEncoderPointer) + rdh->headerSize);
394 
395 #ifdef DECODER_VERBOSE
396  if (mDecoderVerbose) {
397  std::cout << colorBlue
398  << "--- END PROCESS HBF"
399  << colorReset
400  << std::endl;
401  }
402 #endif
403 
405  mDecoderPointer = reinterpret_cast<uint32_t*>(reinterpret_cast<char*>(rdh) + rdh->offsetToNext);
406 
408  if (reinterpret_cast<char*>(mDecoderPointer) < mDecoderBuffer + mDecoderBufferSize)
409  return false;
410 
412  return true;
413 }

415 bool Compressor::processDRM()
416 {
417 
418 #ifdef DECODER_VERBOSE
419  if (mDecoderVerbose) {
420  std::cout << colorBlue << "--- PROCESS DRM"
421  << colorReset
422  << std::endl;
423  }
424 #endif
425 
427  mDecoderNextWord = 1;
428  mDecoderError = false;
429  mDecoderFatal = false;
430  mEncoderPointerStart = mEncoderPointer;
431 
433  if (!IS_DRM_COMMON_HEADER(*mDecoderPointer)) {
434 #ifdef DECODER_VERBOSE
435  printf("%s %08x [ERROR] fatal error %s \n", colorRed, *mDecoderPointer, colorReset);
436 #endif
437  mDecoderFatal = true;
438  return true;
439  }
440  mDecoderSummary.tofDataHeader = *mDecoderPointer;
441 #ifdef DECODER_VERBOSE
442  if (mDecoderVerbose) {
443  auto tofDataHeader = reinterpret_cast<raw::TOFDataHeader_t*>(mDecoderPointer);
444  auto bytePayload = tofDataHeader->bytePayload;
445  printf(" %08x TOF Data Header (bytePayload=%d) \n", *mDecoderPointer, bytePayload);
446  }
447 #endif
448 #ifdef DECODER_PARANOID
449  if (decoderParanoid())
450  return true;
451 #endif
452  decoderNext();
453 
455  mDecoderSummary.tofOrbit = *mDecoderPointer;
456 #ifdef DECODER_VERBOSE
457  if (mDecoderVerbose) {
458  auto tofOrbit = reinterpret_cast<raw::TOFOrbit_t*>(mDecoderPointer);
459  auto orbit = tofOrbit->orbit;
460  printf(" %08x TOF Orbit (orbit=%d) \n", *mDecoderPointer, orbit);
461  }
462 #endif
463 #ifdef DECODER_PARANOID
464  if (decoderParanoid())
465  return true;
466 #endif
467  decoderNext();
468 
470  if (!IS_DRM_GLOBAL_HEADER(*mDecoderPointer)) {
471 #ifdef DECODER_VERBOSE
472  printf("%s %08x [ERROR] fatal error %s \n", colorRed, *mDecoderPointer, colorReset);
473 #endif
474  mDecoderFatal = true;
475  return true;
476  }
477  mDecoderSummary.drmDataHeader = *mDecoderPointer;
478 #ifdef DECODER_VERBOSE
479  if (mDecoderVerbose) {
480  auto drmDataHeader = reinterpret_cast<raw::DRMDataHeader_t*>(mDecoderPointer);
481  auto drmId = drmDataHeader->drmId;
482  printf(" %08x DRM Data Header (drmId=%d) \n", *mDecoderPointer, drmId);
483  }
484 #endif
485 #ifdef DECODER_PARANOID
486  if (decoderParanoid())
487  return true;
488 #endif
489  decoderNext();
490 
492  mDecoderSummary.drmHeadW1 = *mDecoderPointer;
493 #ifdef DECODER_VERBOSE
494  if (mDecoderVerbose) {
495  auto drmHeadW1 = reinterpret_cast<raw::DRMHeadW1_t*>(mDecoderPointer);
496  auto partSlotMask = drmHeadW1->partSlotMask;
497  auto clockStatus = drmHeadW1->clockStatus;
498  auto drmHSize = drmHeadW1->drmHSize;
499  printf(" %08x DRM Header Word 1 (partSlotMask=0x%03x, clockStatus=%d, drmHSize=%d) \n", *mDecoderPointer, partSlotMask, clockStatus, drmHSize);
500  }
501 #endif
502 #ifdef DECODER_PARANOID
503  if (decoderParanoid())
504  return true;
505 #endif
506  decoderNext();
507 
509  mDecoderSummary.drmHeadW2 = *mDecoderPointer;
510 #ifdef DECODER_VERBOSE
511  if (mDecoderVerbose) {
512  auto drmHeadW2 = reinterpret_cast<raw::DRMHeadW2_t*>(mDecoderPointer);
513  auto enaSlotMask = drmHeadW2->enaSlotMask;
514  auto faultSlotMask = drmHeadW2->faultSlotMask;
515  auto readoutTimeOut = drmHeadW2->readoutTimeOut;
516  printf(" %08x DRM Header Word 2 (enaSlotMask=0x%03x, faultSlotMask=%d, readoutTimeOut=%d) \n", *mDecoderPointer, enaSlotMask, faultSlotMask, readoutTimeOut);
517  }
518 #endif
519 #ifdef DECODER_PARANOID
520  if (decoderParanoid())
521  return true;
522 #endif
523  decoderNext();
524 
526  mDecoderSummary.drmHeadW3 = *mDecoderPointer;
527 #ifdef DECODER_VERBOSE
528  if (mDecoderVerbose) {
529  auto drmHeadW3 = reinterpret_cast<raw::DRMHeadW3_t*>(mDecoderPointer);
530  auto gbtBunchCnt = drmHeadW3->gbtBunchCnt;
531  auto locBunchCnt = drmHeadW3->locBunchCnt;
532  printf(" %08x DRM Header Word 3 (gbtBunchCnt=%d, locBunchCnt=%d) \n", *mDecoderPointer, gbtBunchCnt, locBunchCnt);
533  }
534 #endif
535 #ifdef DECODER_PARANOID
536  if (decoderParanoid())
537  return true;
538 #endif
539  decoderNext();
540 
542  mDecoderSummary.drmHeadW4 = *mDecoderPointer;
543 #ifdef DECODER_VERBOSE
544  if (mDecoderVerbose) {
545  printf(" %08x DRM Header Word 4 \n", *mDecoderPointer);
546  }
547 #endif
548 #ifdef DECODER_PARANOID
549  if (decoderParanoid())
550  return true;
551 #endif
552  decoderNext();
553 
555  mDecoderSummary.drmHeadW5 = *mDecoderPointer;
556 #ifdef DECODER_VERBOSE
557  if (mDecoderVerbose) {
558  printf(" %08x DRM Header Word 5 \n", *mDecoderPointer);
559  }
560 #endif
561 #ifdef DECODER_PARANOID
562  if (decoderParanoid())
563  return true;
564 #endif
565  decoderNext();
566 
568  *mEncoderPointer = 0x80000000;
569  *mEncoderPointer |= GET_DRMHEADW2_ENASLOTMASK(mDecoderSummary.drmHeadW2) << 12;
570  *mEncoderPointer |= GET_DRMDATAHEADER_DRMID(mDecoderSummary.drmDataHeader) << 24;
571  *mEncoderPointer |= GET_DRMHEADW3_GBTBUNCHCNT(mDecoderSummary.drmHeadW3);
572 #ifdef ENCODER_VERBOSE
573  if (mEncoderVerbose) {
574  auto crateHeader = reinterpret_cast<compressed::CrateHeader_t*>(mEncoderPointer);
575  auto bunchID = crateHeader->bunchID;
576  auto drmID = crateHeader->drmID;
577  auto slotEnableMask = crateHeader->slotEnableMask;
578  printf("%s %08x Crate header (drmID=%d, bunchID=%d, slotEnableMask=0x%x) %s \n", colorGreen, *mEncoderPointer, drmID, bunchID, slotEnableMask, colorReset);
579  }
580 #endif
581  encoderNext();
582 
584  *mEncoderPointer = mDecoderSummary.tofOrbit;
585 #ifdef ENCODER_VERBOSE
586  if (mEncoderVerbose) {
587  auto crateOrbit = reinterpret_cast<compressed::CrateOrbit_t*>(mEncoderPointer);
588  auto orbitID = crateOrbit->orbitID;
589  printf("%s %08x Crate orbit (orbitID=%d) %s \n", colorGreen, *mEncoderPointer, orbitID, colorReset);
590  }
591 #endif
592  encoderNext();
593 
595  while (true) {
597  if (IS_LTM_GLOBAL_HEADER(*mDecoderPointer)) {
598 
599 #ifdef DECODER_VERBOSE
600  if (mDecoderVerbose) {
601  printf(" %08x LTM Global Header \n", *mDecoderPointer);
602  }
603 #endif
604 #ifdef DECODER_PARANOID
605  if (decoderParanoid())
606  return true;
607 #endif
608  decoderNext();
609 
611  while (true) {
613  if (IS_LTM_GLOBAL_TRAILER(*mDecoderPointer)) {
614 #ifdef DECODER_VERBOSE
615  if (mDecoderVerbose) {
616  printf(" %08x LTM Global Trailer \n", *mDecoderPointer);
617  }
618 #endif
619 #ifdef DECODER_PARANOID
620  if (decoderParanoid())
621  return true;
622 #endif
623  decoderNext();
624  break;
625  }
626 
627 #ifdef DECODER_VERBOSE
628  if (mDecoderVerbose) {
629  printf(" %08x LTM data \n", *mDecoderPointer);
630  }
631 #endif
632 #ifdef DECODER_PARANOID
633  if (decoderParanoid())
634  return true;
635 #endif
636  decoderNext();
637  }
638  }
639 
641  if (IS_TRM_GLOBAL_HEADER(*mDecoderPointer) && GET_TRMDATAHEADER_SLOTID(*mDecoderPointer) > 2) {
642  uint32_t slotId = GET_TRMDATAHEADER_SLOTID(*mDecoderPointer);
643  int itrm = slotId - 3;
644  mDecoderSummary.trmDataHeader[itrm] = *mDecoderPointer;
645 #ifdef DECODER_VERBOSE
646  if (mDecoderVerbose) {
647  auto trmDataHeader = reinterpret_cast<raw::TRMDataHeader_t*>(mDecoderPointer);
648  auto eventWords = trmDataHeader->eventWords;
649  auto eventCnt = trmDataHeader->eventCnt;
650  auto emptyBit = trmDataHeader->emptyBit;
651  printf(" %08x TRM Data Header (slotId=%u, eventWords=%d, eventCnt=%d, emptyBit=%01x) \n", *mDecoderPointer, slotId, eventWords, eventCnt, emptyBit);
652  }
653 #endif
654 #ifdef DECODER_PARANOID
655  if (decoderParanoid())
656  return true;
657 #endif
658  decoderNext();
659 
661  while (true) {
663  if (IS_TRM_CHAINA_HEADER(*mDecoderPointer) && GET_TRMCHAINHEADER_SLOTID(*mDecoderPointer) == slotId) {
664  mDecoderSummary.trmChainHeader[itrm][0] = *mDecoderPointer;
665  mDecoderSummary.hasHits[itrm][0] = false;
666  mDecoderSummary.hasErrors[itrm][0] = false;
667 #ifdef DECODER_VERBOSE
668  if (mDecoderVerbose) {
669  auto trmChainHeader = reinterpret_cast<raw::TRMChainHeader_t*>(mDecoderPointer);
670  auto bunchCnt = trmChainHeader->bunchCnt;
671  printf(" %08x TRM Chain-A Header (slotId=%u, bunchCnt=%d) \n", *mDecoderPointer, slotId, bunchCnt);
672  }
673 #endif
674 #ifdef DECODER_PARANOID

1004 {

1006 
1008  int firstFilledFrame = 255;

1010 
1012  for (int ichain = 0; ichain < 2; ++ichain) {
1013 
1014  if (!mDecoderSummary.hasHits[itrm][ichain])
1015  continue;
1016 
1018  for (int itdc = 0; itdc < 15; ++itdc) {
1019 
1020  auto nhits = mDecoderSummary.trmDataHits[ichain][itdc];
1021  if (nhits == 0)
1022  continue;
1023 
1025  for (int ihit = 0; ihit < nhits; ++ihit) {
1026 
1027  auto lhit = mDecoderSummary.trmDataHit[ichain][itdc][ihit];
1028  if (GET_TRMDATAHIT_PSBITS(lhit) != 0x1) // must be a leading hit
1029  continue;
1030 
1031  auto chan = GET_TRMDATAHIT_CHANID(lhit);
1032  auto hitTime = GET_TRMDATAHIT_TIME(lhit);
1033  auto eBit = GET_TRMDATAHIT_EBIT(lhit);
1034  uint32_t totWidth = 0;
1035 
1036  // check next hits for packing
1037  for (int jhit = ihit + 1; jhit < nhits; ++jhit) {
1038  auto thit = mDecoderSummary.trmDataHit[ichain][itdc][jhit];
1039  if (GET_TRMDATAHIT_PSBITS(thit) == 0x2 && GET_TRMDATAHIT_CHANID(thit) == chan) { // must be a trailing hit from same channel
1040  totWidth = (GET_TRMDATAHIT_TIME(thit) - hitTime) / Geo::RATIO_TOT_TDC_BIN; // compute TOT
1041  lhit = 0x0; // mark as used
1042  break;
1043  }
1044  }
1045 
1046  auto iframe = hitTime >> 13;
1047  auto phit = mSpiderSummary.nFramePackedHits[iframe];
1048 
1049  mSpiderSummary.FramePackedHit[iframe][phit] = 0x00000000;
1050  mSpiderSummary.FramePackedHit[iframe][phit] |= (totWidth & 0x7FF) << 0;
1051  mSpiderSummary.FramePackedHit[iframe][phit] |= (hitTime & 0x1FFF) << 11;
1052  mSpiderSummary.FramePackedHit[iframe][phit] |= chan << 24;
1053  mSpiderSummary.FramePackedHit[iframe][phit] |= itdc << 27;
1054  mSpiderSummary.FramePackedHit[iframe][phit] |= ichain << 31;
1055  mSpiderSummary.nFramePackedHits[iframe]++;
1056 
1057  if (iframe < firstFilledFrame)
1058  firstFilledFrame = iframe;
1059  if (iframe > lastFilledFrame)
1060  lastFilledFrame = iframe;
1061  }
1062 
1063  mDecoderSummary.trmDataHits[ichain][itdc] = 0;
1064  }
1065  }
1066 
1068  for (int iframe = firstFilledFrame; iframe < lastFilledFrame + 1; iframe++) {
1069 
1071  if (mSpiderSummary.nFramePackedHits[iframe] == 0)
1072  continue;
1073 
1074  // encode Frame Header
1075  *mEncoderPointer = 0x00000000;
1076  *mEncoderPointer |= slotId << 24;
1077  *mEncoderPointer |= iframe << 16;
1078  *mEncoderPointer |= mSpiderSummary.nFramePackedHits[iframe];
1079 #ifdef ENCODER_VERBOSE
1080  if (mEncoderVerbose) {
1081  auto FrameHeader = reinterpret_cast<compressed::FrameHeader_t*>(mEncoderPointer);
1082  auto NumberOfHits = FrameHeader->numberOfHits;
1083  auto FrameID = FrameHeader->frameID;
1084  auto TRMID = FrameHeader->trmID;
1085  printf("%s %08x Frame header (TRMID=%d, FrameID=%d, NumberOfHits=%d) %s \n", colorGreen, *mEncoderPointer, TRMID, FrameID, NumberOfHits, colorReset);
1086  }
1087 #endif
1088  encoderNext();
1089 
1090  // packed hits
1091  for (int ihit = 0; ihit < mSpiderSummary.nFramePackedHits[iframe]; ++ihit) {
1092  *mEncoderPointer = mSpiderSummary.FramePackedHit[iframe][ihit];
1093 #ifdef ENCODER_VERBOSE
1094  if (mEncoderVerbose) {
1095  auto PackedHit = reinterpret_cast<compressed::PackedHit_t*>(mEncoderPointer);
1096  auto Chain = PackedHit->chain;
1097  auto TDCID = PackedHit->tdcID;
1098  auto Channel = PackedHit->channel;
1099  auto Time = PackedHit->time;
1100  auto TOT = PackedHit->tot;
1101  printf("%s %08x Packed hit (Chain=%d, TDCID=%d, Channel=%d, Time=%d, TOT=%d) %s \n", colorGreen, *mEncoderPointer, Chain, TDCID, Channel, Time, TOT, colorReset);
1102  }
1103 #endif
1104  encoderNext();
1105  }
1106

1108  }
