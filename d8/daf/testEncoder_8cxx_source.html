83 int estimateUserLogicSize(int nofDS, int maxNofChPerDS)
84 {
85  size_t headerSize = 2; // equivalent to 2 64-bits words
86  // one 64-bits header and one 64-bits data per channel
87  // plus one sync per DS
88  // (assuming data = just one sample)
89  size_t ndata = (maxNofChPerDS * 2) + nofDS;
90  return 8 * (ndata + headerSize); // size in bytes
91 }

93 int estimateBareSize(int nofDS, int maxNofChPerGBT)
94 {
95  size_t headerSize = 2; // equivalent to 2 64-bits words
96  size_t nbits = nofDS * 50 + maxNofChPerGBT * 90;
97  size_t n128bitwords = nbits / 2;
98  size_t n64bitwords = n128bitwords * 2;
99  return 8 * (n64bitwords + headerSize); // size in bytes
100 }
