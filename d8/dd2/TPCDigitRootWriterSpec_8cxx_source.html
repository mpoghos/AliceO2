47 TBranch* getOrMakeBranch(TTree& tree, std::string basename, int sector, T* ptr)
48 {
49  std::stringstream stream;
50  stream << basename << "_" << sector;
51  const auto brname = stream.str();
52  if (auto br = tree.GetBranch(brname.c_str())) {
53  br->SetAddress(static_cast<void*>(&ptr));
54  return br;
55  }
56  // otherwise make it
57  return tree.Branch(brname.c_str(), ptr);
58 }

63 DataProcessorSpec getTPCDigitRootWriterSpec(int numberofsourcedevices)
64 {
65  // assign input names to each channel
66  auto digitchannelname = std::make_shared<std::vector<std::string>>();
67  auto triggerchannelname = std::make_shared<std::vector<std::string>>();
68  auto labelchannelname = std::make_shared<std::vector<std::string>>();

71  {

74  digitchannelname->push_back(ss.str());
75  }
76  {
77  std::stringstream ss;
78  ss << "triggerinput" << i;
79  triggerchannelname->push_back(ss.str());
80  }
81  {
82  std::stringstream ss;
83  ss << "labelinput" << i;
84  labelchannelname->push_back(ss.str());
85  }
86  {
87  std::stringstream ss;
88  ss << "commonmodeinput" << i;
89  commonModechannelname->push_back(ss.str());

93  auto initFunction = [numberofsourcedevices, digitchannelname, labelchannelname, triggerchannelname, commonModechannelname](InitContext& ic) {
94  // get the option from the init context
95  auto filename = ic.options().get<std::string>("tpc-digit-outfile");
96  auto treename = ic.options().get<std::string>("treename");
97 
98  auto outputfile = std::make_shared<TFile>(filename.c_str(), "RECREATE");
99  auto outputtree = std::make_shared<TTree>(treename.c_str(), treename.c_str());
100 
101  // container for cashed grouping of digits
102  auto trigP2Sect = std::make_shared<std::array<std::vector<DigiGroupRef>, 36>>();
103 
104  // the callback to be set as hook at stop of processing for the framework
105  auto finishWriting = [outputfile, outputtree]() {
106  // check/verify number of entries (it should be same in all branches)
107 
108  // will return a TObjArray
109  const auto brlist = outputtree->GetListOfBranches();
110  int entries = -1; // init to -1 (as unitialized)
111  for (TObject* entry : *brlist) {
112  auto br = static_cast<TBranch*>(entry);
113  int brentries = br->GetEntries();
114  if (entries == -1) {
115  entries = brentries;
116  } else {
117  if (brentries != entries && !TString(br->GetName()).Contains("CommonMode")) {
118  LOG(WARNING) << "INCONSISTENT NUMBER OF ENTRIES IN BRANCH " << br->GetName() << ": " << entries << " vs " << brentries;
119  }
120  }
121  }
122  if (entries > 0) {
123  outputtree->SetEntries(entries);
124  }
125  outputtree->Write();
126  outputfile->Close();
127  };
128  ic.services().get<CallbackService>().set(CallbackService::Id::Stop, finishWriting);
129 
130  // set up the processing function
131  // using by-copy capture of the worker instance shared pointer
132  // the shared pointer makes sure to clean up the instance when the processing
133  // function gets out of scope
134  auto processingFct = [outputfile, outputtree, trigP2Sect, digitchannelname, labelchannelname,
135  triggerchannelname, commonModechannelname, numberofsourcedevices](ProcessingContext& pc) {
136  static bool finished = false;
137  if (finished) {
138  // avoid being executed again when marked as finished;
139  return;
140  }
141 
142  static int finishchecksum = 0;
143  static int invocation = 0;
144  invocation++;
145  // need to record which channel has completed in order to decide when we can shutdown
146  static std::vector<bool> digitsdone;
147  static std::vector<bool> labelsdone;
148  static std::vector<bool> triggersdone;
149  static std::vector<bool> commonmodedone;
150  if (invocation == 1) {
151  digitsdone.resize(numberofsourcedevices, false);
152  labelsdone.resize(numberofsourcedevices, false);
153  triggersdone.resize(numberofsourcedevices, false);
154  commonmodedone.resize(numberofsourcedevices, false);
155  }
156 
157  // find out if all source devices (channels) are done
158  // by means of a simple checksum
159  auto isComplete = [numberofsourcedevices](int i) {
160  if (i == numberofsourcedevices * (numberofsourcedevices + 1) / 2) {
161  return true;
162  }
163  return false;
164  };
165 
166  // extracts the sector from header of an input
167  auto extractSector = [&pc](const char* inputname) {
168  auto sectorHeader = DataRefUtils::getHeader<o2::tpc::TPCSectorHeader*>(pc.inputs().get(inputname));
169  if (!sectorHeader) {
170  LOG(FATAL) << "Missing sector header in TPC data";
171  }
172  return sectorHeader->sector;
173  };
174 
175  int sector = -1; // TPC sector for which data was received
176  for (int d = 0; d < numberofsourcedevices; ++d) {
177  const auto dname = digitchannelname->operator[](d);
178  const auto lname = labelchannelname->operator[](d);
179  const auto tname = triggerchannelname->operator[](d);
180  const auto cname = commonModechannelname->operator[](d);
181  if (pc.inputs().isValid(tname.c_str())) {
182  sector = extractSector(tname.c_str());
183  LOG(INFO) << "HAVE TRIGGER DATA FOR SECTOR " << sector << " ON CHANNEL " << d;
184  if (sector <= -1) {
185  if (sector != -2) {
186  triggersdone[d] = true;
187  }
188  } else {
189  auto triggers = pc.inputs().get<std::vector<DigiGroupRef>>(tname.c_str());
190  (*trigP2Sect.get())[sector] = std::move(triggers);
191  const auto& trigS = (*trigP2Sect.get())[sector];
192  LOG(INFO) << "GOT Triggers of sector " << sector << " | SIZE " << trigS.size();
193  }
194  }
195 
196  // probe which channel has data and of what kind
197  // a) probe for digits:
198  if (pc.inputs().isValid(dname.c_str())) {
199  sector = extractSector(dname.c_str());
200  LOG(INFO) << "HAVE DIGIT DATA FOR SECTOR " << sector << " ON CHANNEL " << d;
201  if (sector <= -1) {
202  if (sector != -2) {
203  digitsdone[d] = true;
204  }
205  } else {
206  // have to do work ...
207  // the digits
208  auto digiData = pc.inputs().get<std::vector<o2::tpc::Digit>>(dname.c_str());
209  LOG(INFO) << "DIGIT SIZE " << digiData.size();
210  const auto& trigS = (*trigP2Sect.get())[sector];
211  if (!trigS.size()) {
212  LOG(FATAL) << "Digits for sector " << sector << " are received w/o info on grouping in triggers";
213  } else { // check consistency of Ndigits with that of expected from the trigger
214  int nExp = trigS.back().getFirstEntry() + trigS.back().getEntries() - trigS.front().getFirstEntry();
215  if (nExp != digiData.size()) {
216  LOG(ERROR) << "Number of digits " << digiData.size() << " is inconsistent with expectation " << nExp
217  << " from digits grouping for sector " << sector;
218  }
219  }
220 
221  {
222  if (trigS.size() == 1) { // just 1 entry (continous mode?), use digits directly
223  // connect this to a particular branch
224  auto digP = &digiData;
225  auto br = getOrMakeBranch(*outputtree.get(), "TPCDigit", sector, digP);
226  br->Fill();
227  br->ResetAddress();
228  } else { // triggered mode (>1 entrie will be written)
229  std::vector<o2::tpc::Digit> digGroup; // group of digits related to single trigger
230  auto digGroupPtr = &digGroup;
231  auto br = getOrMakeBranch(*outputtree.get(), "TPCDigit", sector, digGroupPtr);
232  for (auto grp : trigS) {
233  digGroup.clear();
234  for (int i = 0; i < grp.getEntries(); i++) {
235  digGroup.emplace_back(digiData[grp.getFirstEntry() + i]); // fetch digits of given trigger
236  }
237  br->Fill();
238  }
239  br->ResetAddress();
240  }
241  }
242  }
243  } // end digit case
244 
245  // b) probe for labels
246  if (pc.inputs().isValid(lname.c_str())) {
247  sector = extractSector(lname.c_str());
248  LOG(INFO) << "HAVE LABEL DATA FOR SECTOR " << sector << " ON CHANNEL " << d;
249  if (sector <= -1) {
250  if (sector != -2) {
251  labelsdone[d] = true;
252  }
253  } else {
254  // the labels
255  auto labeldata = pc.inputs().get<o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>(lname.c_str());
256  auto labeldataRaw = labeldata.get();
257  LOG(INFO) << "MCTRUTH ELEMENTS " << labeldataRaw->getIndexedSize()
258  << " WITH " << labeldataRaw->getNElements() << " LABELS";
259  const auto& trigS = (*trigP2Sect.get())[sector];
260  if (!trigS.size()) {
261  LOG(FATAL) << "MCTruth for sector " << sector << " are received w/o info on grouping in triggers";
262  } else {
263  int nExp = trigS.back().getFirstEntry() + trigS.back().getEntries() - trigS.front().getFirstEntry();
264  if (nExp != labeldataRaw->getIndexedSize()) {
265  LOG(ERROR) << "Number of indexed (label) slots " << labeldataRaw->getIndexedSize()
266  << " is inconsistent with expectation " << nExp
267  << " from digits grouping for sector " << sector;
268  }
269  }
270  {
271  if (trigS.size() == 1) { // just 1 entry (continous mode?), use labels directly
272  auto br = getOrMakeBranch(*outputtree.get(), "TPCDigitMCTruth", sector, &labeldataRaw);
273  br->Fill();
274  br->ResetAddress();
275  } else {
276  o2::dataformats::MCTruthContainer<o2::MCCompLabel> lblGroup; // labels for group of digits related to single trigger
277  auto lblGroupPtr = &lblGroup;
278  auto br = getOrMakeBranch(*outputtree.get(), "TPCDigitMCTruth", sector, &lblGroupPtr);
279  for (auto grp : trigS) {
280  lblGroup.clear();
281  for (int i = 0; i < grp.getEntries(); i++) {
282  auto lbls = labeldataRaw->getLabels(grp.getFirstEntry() + i);
283  lblGroup.addElements(i, lbls);
284  }
285  br->Fill();
286  }
287  }
288  }
289  }
290  } // end label case
291 
292  // c) probe for common mode:
293  if (pc.inputs().isValid(cname.c_str())) {
294  sector = extractSector(cname.c_str());
295  LOG(INFO) << "HAVE COMMON MODE DATA FOR SECTOR " << sector << " ON CHANNEL " << d;
296  const auto& trigS = (*trigP2Sect.get())[sector];
297  if (sector <= -1) {
298  if (sector != -2) {
299  commonmodedone[d] = true;
300  }
301  } else {
302  auto commonModeData = pc.inputs().get<std::vector<o2::tpc::CommonMode>>(cname.c_str());
303  LOG(INFO) << "COMMON MODE SIZE " << commonModeData.size();
304 
305  if (!trigS.size()) {
306  LOG(FATAL) << "CommonMode for sector " << sector << " are received w/o info on grouping in triggers";
307  }
308  {
309  auto digC = &commonModeData;
310  auto br = getOrMakeBranch(*outputtree.get(), "TPCCommonMode", sector, digC);
311  br->Fill();
312  br->ResetAddress();
313  }
314  }
315  } // end common mode case
316 
317  if (labelsdone[d] && digitsdone[d] && commonmodedone[d]) {
318  LOG(INFO) << "CHANNEL " << d << " DONE ";
319 
320  // we must increase the checksum only once
321  // prevent this by invalidating ...
322  labelsdone[d] = false;
323  digitsdone[d] = false;
324  commonmodedone[d] = false;
325 
326  finishchecksum += (d + 1); // + 1 since d starts at 0 ... important for the checksum test
327  if (isComplete(finishchecksum)) {
328  finished = true;
329  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
330  return;
331  }
332  }
333  }
334  };
335 
336  // return the actual processing function as a lambda function using variables
337  // of the init function
338  return processingFct;
339  };
340 
341  std::vector<InputSpec> inputs;
342  for (int d = 0; d < numberofsourcedevices; ++d) {
343  inputs.emplace_back(InputSpec{(*digitchannelname.get())[d].c_str(), "TPC", "DIGITS",
344  static_cast<SubSpecificationType>(d), Lifetime::Timeframe}); // digit input
345  inputs.emplace_back(InputSpec{(*triggerchannelname.get())[d].c_str(), "TPC", "DIGTRIGGERS",
346  static_cast<SubSpecificationType>(d), Lifetime::Timeframe}); // groupping in triggers
347  inputs.emplace_back(InputSpec{(*labelchannelname.get())[d].c_str(), "TPC", "DIGITSMCTR",
348  static_cast<SubSpecificationType>(d), Lifetime::Timeframe});
349  inputs.emplace_back(InputSpec{(*commonModechannelname.get())[d].c_str(), "TPC", "COMMONMODE",
350  static_cast<SubSpecificationType>(d), Lifetime::Timeframe}); // digit input
351  }
352 
353  return DataProcessorSpec{
354  "TPCDigitWriter",
355  inputs,
356  {}, // no output
357  AlgorithmSpec(initFunction),
358  Options{
359  {"tpc-digit-outfile", VariantType::String, "tpcdigits.root", {"Name of the input file"}},
360  {"treename", VariantType::String, "o2sim", {"Name of tree for tracks"}},
361  }};
362 }
