37 std::vector<DataProcessorSpec> defineDataProcessing(ConfigContext const&)
38 {
39  // define a template workflow with processors to be executed in a pipeline
40  std::vector<DataProcessorSpec> workflowSpecs{
41  {"processor1",
42  Inputs{
43  {"input", "TST", "TRIGGER", 0, Lifetime::Timeframe}},
44  Outputs{
45  {{"output"}, "TST", "PREPROC", 0, Lifetime::Timeframe}},
46  AlgorithmSpec{[](ProcessingContext& ctx) {
47  for (auto const& input : ctx.inputs()) {
48  auto const& parallelContext = ctx.services().get<ParallelContext>();
49  LOG(DEBUG) << "instance " << parallelContext.index1D() << " of " << parallelContext.index1DSize() << ": "
50  << *input.spec << ": " << *((int*)input.payload);
51  auto const* dataheader = DataRefUtils::getHeader<o2::header::DataHeader*>(input);
52  //auto& data = ctx.outputs().make<int>(OutputRef{"output", dataheader->subSpecification});
53  auto& data = ctx.outputs().make<int>(Output{"TST", "PREPROC", dataheader->subSpecification, Lifetime::Timeframe});
54  ASSERT_ERROR(ctx.inputs().get<int>(input.spec->binding.c_str()) == parallelContext.index1D());
55  data = parallelContext.index1D();
56  }
57  }}},
58  {"processor2",
59  Inputs{
60  {"input", "TST", "PREPROC", 0, Lifetime::Timeframe}},
61  Outputs{
62  {{"output"}, "TST", "DATA", 0, Lifetime::Timeframe},
63  {{"metadt"}, "TST", "META", 0, Lifetime::Timeframe}},
64  AlgorithmSpec{[](ProcessingContext& ctx) {
65  for (auto const& input : ctx.inputs()) {
66  auto const& parallelContext = ctx.services().get<ParallelContext>();
67  LOG(DEBUG) << "instance " << parallelContext.index1D() << " of " << parallelContext.index1DSize() << ": "
68  << *input.spec << ": " << *((int*)input.payload);
69  ASSERT_ERROR(ctx.inputs().get<int>(input.spec->binding.c_str()) == parallelContext.index1D());
70  auto const* dataheader = DataRefUtils::getHeader<o2::header::DataHeader*>(input);
71  // TODO: there is a bug in the API for using OutputRef, returns an rvalue which can not be bound to
72  // lvalue reference
73  //auto& data = ctx.outputs().make<int>(OutputRef{"output", dataheader->subSpecification});
74  auto& data = ctx.outputs().make<int>(Output{"TST", "DATA", dataheader->subSpecification, Lifetime::Timeframe});
75  data = ctx.inputs().get<int>(input.spec->binding.c_str());
76  //auto& meta = ctx.outputs().make<int>(OutputRef{"metadt", dataheader->subSpecification});
77  auto& meta = ctx.outputs().make<int>(Output{"TST", "META", dataheader->subSpecification, Lifetime::Timeframe});
78  meta = dataheader->subSpecification;
79  }
80  }}},
81  };
82 
83  // create parallel pipelines from the template workflow, the number of parallel channel is defined by
84  // nParallelChannels and is distributed among the pipelines
85  std::vector<o2::header::DataHeader::SubSpecificationType> subspecs(nParallelChannels);
86  std::generate(subspecs.begin(), subspecs.end(), [counter = std::make_shared<int>(0)]() { return 0x1 << (*counter)++; });
87  workflowSpecs = parallelPipeline(
88  workflowSpecs, nPipelines,
89  [&subspecs]() { return subspecs.size(); },
90  [&subspecs](size_t index) { return subspecs[index]; });
91 
92  // define a producer process with outputs for all subspecs
93  auto producerOutputs = [&subspecs]() {
94  Outputs outputs;
95  for (auto const& subspec : subspecs) {
96  outputs.emplace_back("TST", "TRIGGER", subspec, Lifetime::Timeframe);
97  }
98  return outputs;
99  };
100 
101  // we keep the correspondence between the subspec and the instance which serves this particular subspec
102  // this is checked in the final consumer
103  auto checkMap = std::make_shared<std::unordered_map<o2::header::DataHeader::SubSpecificationType, int>>();
104  workflowSpecs.emplace_back(DataProcessorSpec{
105  "trigger",
106  Inputs{},
107  producerOutputs(),
108  AlgorithmSpec{[subspecs, checkMap, counter = std::make_shared<int>(0)](ProcessingContext& ctx) {
109  if (*counter < nRolls) {
110  size_t pipeline = 0;
111  size_t channels = subspecs.size();
112  std::vector<size_t> multiplicities(nPipelines);
113  for (pipeline = 0; pipeline < nPipelines; pipeline++) {
114  multiplicities[pipeline] = channels / (nPipelines - pipeline) + ((channels % (nPipelines - pipeline)) > 0 ? 1 : 0);
115  channels -= multiplicities[pipeline];
116  }
117  ASSERT_ERROR(channels == 0);
118  size_t index = 0;
119  auto end = subspecs.size();
120  for (pipeline = 0; index < end; index++) {
121  if (multiplicities[pipeline] == 0) {
122  continue;
123  }
124  ctx.outputs().make<int>(Output{"TST", "TRIGGER", subspecs[index], Lifetime::Timeframe}) = pipeline;
125  (*checkMap)[subspecs[index]] = pipeline;
126  multiplicities[pipeline++]--;
127  if (pipeline >= nPipelines) {
128  pipeline = 0;
129  }
130  }
131  ASSERT_ERROR(index == subspecs.size());
132  (*counter)++;
133  }
134  if (*counter == nRolls) {
135  ctx.services().get<ControlService>().endOfStream();
136  ctx.services().get<ControlService>().readyToQuit(QuitRequest::Me);
137  }
138  }}});
139 
140  // the final consumer
141  workflowSpecs.emplace_back(DataProcessorSpec{
142  "consumer",
143  mergeInputs({{"datain", "TST", "DATA", 0, Lifetime::Timeframe},
144  {"metain", "TST", "META", 0, Lifetime::Timeframe}},
145  subspecs.size(),
146  [&subspecs](InputSpec& input, size_t index) {
147  DataSpecUtils::updateMatchingSubspec(input, subspecs[index]);
148  }),
149  Outputs(),
150  AlgorithmSpec{adaptStateless([checkMap](InputRecord& inputs, CallbackService& callbacks, ControlService&) {
151  callbacks.set(CallbackService::Id::EndOfStream, [](EndOfStreamContext& ctx) {
152  ctx.services().get<ControlService>().readyToQuit(QuitRequest::All);
153  });
154  for (auto const& input : inputs) {
155  LOG(DEBUG) << "consuming : " << *input.spec << ": " << *((int*)input.payload);
156  auto const* dataheader = DataRefUtils::getHeader<o2::header::DataHeader*>(input);
157  if (input.spec->binding.compare(0, 6, "datain") == 0) {
158  ASSERT_ERROR((*checkMap)[dataheader->subSpecification] == inputs.get<int>(input.spec->binding.c_str()));
159  }
160  }
161  })}});
162 
163  return workflowSpecs;
164 }
