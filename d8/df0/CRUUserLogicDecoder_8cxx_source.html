35 void CRUUserLogicDecoder::process(gsl::span<const raw::RawUnit> bytes)
36 {
38  reset();
39  mBuffer.setBuffer(bytes);
40 
41  // Each data block consists of:
42  // 28 bits of event type
43  // up to 8 local boards info consisting of at most 136 bits
44  // So, if we want to have a full block,
45  // the buffer size must be at least 28 + 8 * 136 = 1116 bits
46  // i.e. 35 uint32_t
47  while (mBuffer.hasNext(35)) {
48  processBlock();
49  }
50 }

52 bool CRUUserLogicDecoder::processBlock()
53 {
55  size_t firstEntry = mData.size();
56 
57  uint8_t eventWord = mBuffer.next(sNBitsEventWord);
58  uint16_t localClock = mBuffer.next(sNBitsLocalClock);
59  uint8_t crateId = mBuffer.next(sNBitsCrateId);
60  int nFiredBoards = mBuffer.next(sNBitsNFiredBoards);
61 
62  EventType eventType = EventType::Standard;
63  if (crateparams::isCalibration(eventWord)) {
64  eventType = EventType::Noise;
65  } else if (crateparams::isFET(eventWord)) {
66  eventType = EventType::Dead;
67  }
68  InteractionRecord intRec(localClock, mBuffer.getRDH()->triggerOrbit);
69 
70  for (int iboard = 0; iboard < nFiredBoards; ++iboard) {
71  uint8_t locId = mBuffer.next(sNBitsLocId);
72  uint8_t firedChambers = mBuffer.next(sNBitsFiredChambers);
73  mData.push_back({0, 0, crateparams::makeUniqueLocID(crateId, locId), firedChambers});
74  for (int ich = 0; ich < 4; ++ich) {
75  if ((firedChambers >> ich) & 0x1) {
76  mData.back().patternsBP[ich] = mBuffer.next(sNBitsFiredStrips);
77  mData.back().patternsNBP[ich] = mBuffer.next(sNBitsFiredStrips);
78  }
79  }
80  }
81 
82  mROFRecords.emplace_back(intRec, eventType, firstEntry, mData.size() - firstEntry);
83 
84  mBuffer.skipOverhead();
85 
86  return true;
87 }
