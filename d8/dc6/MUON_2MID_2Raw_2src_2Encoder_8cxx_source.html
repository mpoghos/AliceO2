33 void Encoder::newHeader(const InteractionRecord& ir)
34 {
36  std::vector<InteractionRecord> HBIRVec;
37  InteractionRecord irFrom = mLastIR.isDummy() ? mHBFUtils.getFirstIR() : mLastIR + 1;
38  mHBFUtils.fillHBIRvector(HBIRVec, irFrom, ir);
39  mLastIR = ir;
40  for (auto& hbIr : HBIRVec) {
41  auto rdh = mHBFUtils.createRDH<header::RAWDataHeader>(hbIr);
42  for (auto linkEncIt = mCRUUserLogicEncoders.begin(); linkEncIt != mCRUUserLogicEncoders.end(); ++linkEncIt) {
43  // The link ID is sequential
44  uint16_t feeId = linkEncIt - mCRUUserLogicEncoders.begin();
45  linkEncIt->newHeader(feeId, rdh);
46  }
47  }
48 }

57 bool Encoder::convertData(gsl::span<const ColumnData> data)
58 {
60  mROData.clear();
61  for (auto& col : data) {
62  for (int iline = 0; iline < 4; ++iline) {
63  if (col.getBendPattern(iline) == 0) {
64  continue;
65  }
66  auto uniqueLocId = mCrateMapper.deLocalBoardToRO(col.deId, col.columnId, iline);
67  auto& roData = mROData[uniqueLocId];
68  roData.boardId = crateparams::getLocId(uniqueLocId);
69  int ich = detparams::getChamber(col.deId);
70  roData.firedChambers |= (1 << ich);
71  roData.patternsBP[ich] = col.getBendPattern(iline);
72  roData.patternsNBP[ich] = col.getNonBendPattern();
73  }
74  }
75  return mROData.empty();
76 }

78 void Encoder::process(gsl::span<const ColumnData> data, const InteractionRecord& ir, EventType eventType)
79 {
81  newHeader(ir);
82 
83  if (convertData(data)) {
84  return;
85  }
86 
87  std::vector<LocalBoardRO> localBoardROs;
88  localBoardROs.reserve(8);
89  uint16_t lastROId = crateparams::sNGBTs + 1;
90  // The local board ID is defined as:
91  // board ID in crate [0-3]
92  // crate ID [4-7]
93  // With this definition, the local boards within the same crate have a sequential ID
94  // This is important since it means that the boards within the same crate are ordered in the mROData map.
95  // So, we can cumulate the strip pattern of one link, and process them when we switch to another link
96  for (auto& item : mROData) {
97  auto crateId = crateparams::getCrateId(item.first);
98  auto roId = crateparams::makeROId(crateId, crateparams::getGBTIdFromBoardInCrate(item.second.boardId));
99  if (roId != lastROId) {
100  // We are in a new link
101  // Let us check that this is not just the first board
102  if (lastROId < crateparams::sNGBTs) {
103  // We process the collected boards since they belong to the same link
104  mCRUUserLogicEncoders[lastROId].process(localBoardROs, ir.bc, eventType);
105  // And we clear the vector so that we can start collecting the board of a new link
106  localBoardROs.clear();
107  }
108  lastROId = roId;
109  }
110  localBoardROs.emplace_back(item.second);
111  }
112 
113  // With the current logic, the last read link is not processed
114  // Let us process it here
115  if (!localBoardROs.empty()) {
116  mCRUUserLogicEncoders[lastROId].process(localBoardROs, ir.bc, eventType);
117  }
118 }
