25 void CRUUserLogicEncoder::add(int value, unsigned int nBits)
26 {
28  for (int ibit = 0; ibit < nBits; ++ibit) {
29  if (mBitIndex == raw::sElementSizeInBits) {
30  mBitIndex = 0;
31  if (mBytes.size() == mHeaderIndex + sMaxPageSize) {
32  completePage(false);
33  // Copy previous header
34  std::vector<raw::RawUnit> oldHeader(mBytes.begin() + mHeaderIndex, mBytes.begin() + mHeaderIndex + raw::sHeaderSizeInElements);
35  mHeaderIndex = mBytes.size();
36  std::copy(oldHeader.begin(), oldHeader.end(), std::back_inserter(mBytes));
37  // And increase the page counter
38  ++getRDH()->pageCnt;
39  }
40  mBytes.push_back(0);
41  }
42  bool isOn = (value >> ibit) & 0x1;
43  if (isOn) {
44  mBytes.back() |= (1 << mBitIndex);
45  }
46  ++mBitIndex;
47  }
48 }

72 void CRUUserLogicEncoder::completePage(bool stop)
73 {
75  if (mBytes.empty()) {
76  return;
77  }
78  auto pageSizeInElements = mBytes.size() - mHeaderIndex;
79  if (mHeaderOffset && pageSizeInElements < sMaxPageSize) {
80  // Write zeros up to the end of the page
81  mBytes.resize(mHeaderIndex + sMaxPageSize, 0);
82  }
83 
84  // This has to be done after the resize
85  auto rdh = getRDH();
86  rdh->memorySize = pageSizeInElements * raw::sElementSizeInBytes;
87  rdh->offsetToNext = (mBytes.size() - mHeaderIndex) * raw::sElementSizeInBytes;
88  rdh->stop = stop;
89 }

101 void CRUUserLogicEncoder::process(gsl::span<const LocalBoardRO> sortedData, const uint16_t bc, EventType eventType)
102 {
104 
105  // FIXME: Finalize the final format of the event word together with the CRU + RO team
106  uint16_t eventWord = 0;
107  if (eventType == EventType::Noise) {
108  eventWord = 0x8;
109  } else if (eventType == EventType::Dead) {
110  eventWord = 0xc;
111  }
112  add(eventWord, sNBitsEventWord);
113  add(bc, sNBitsLocalClock);
114  add(crateparams::getCrateIdFromROId(getRDH()->feeId), sNBitsCrateId);
115  add(sortedData.size(), sNBitsNFiredBoards);
116  for (auto& loc : sortedData) {
117  add(loc.boardId, sNBitsLocId);
118  add(loc.firedChambers, sNBitsFiredChambers);
119  for (int ich = 0; ich < 4; ++ich) {
120  if ((loc.firedChambers >> ich) & 0x1) {
121  add(loc.patternsBP[ich], sNBitsFiredStrips);
122  add(loc.patternsNBP[ich], sNBitsFiredStrips);
123  }
124  }
