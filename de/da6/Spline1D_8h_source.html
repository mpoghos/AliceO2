313 template <typename T>
314 GPUdi() void Spline1D::interpolate(int Ndim, const Spline1D::Knot& knotL,
315  GPUgeneric() const T Sl[/*Ndim*/], GPUgeneric() const T Dl[/*Ndim*/],
316  GPUgeneric() const T Sr[/*Ndim*/], GPUgeneric() const T Dr[/*Ndim*/],
317  float u, GPUgeneric() T Su[/*Ndim*/])
318 {
323 
324  T uu = T(u - knotL.u);
325  T li = T(knotL.Li);
326  T x = uu * li; // scaled u
327  for (int dim = 0; dim < Ndim; ++dim) {
328  T df = (Sr[dim] - Sl[dim]) * li;
329  T a = Dl[dim] + Dr[dim] - df - df;
330  T b = df - Dl[dim] - a;
331  Su[dim] = ((a * x + b) * x + Dl[dim]) * uu + Sl[dim];
332  }
333 
334  /* another way to calculate f(u):
335  T uu = T(u - knotL.u);
336  T x = uu * T(knotL.Li); // scaled u
337  T xm1 = x-1;
338  T x2 = x * x;
339  float cSr = x2*(3-2*x);
340  float cSl = 1-cSr;
341  float cDl = x*xm1*xm1*knotL.L;
342  float cDr = x2*xm1*knotL.L;
343  return cSl*Sl + cSr*Sr + cDl*Dl + cDr*Dr;
344  */
345 }

349 {
353  int iknot = getKnotIndex(u);
