28 double Digitizer::get_time(const std::vector<double>& times)
29 {
30  mHist->Reset();
31  mHistsum->Reset();
32  mHistshift->Reset();
33 
35  for (auto time : times) {
36  for (int bin = mHist->FindBin(time); bin < mHist->GetSize(); ++bin)
37  if (mHist->GetBinCenter(bin) > time)
38  mHist->AddBinContent(bin, signalForm_i(mHist->GetBinCenter(bin) - time));
39  }
41  for (int c = 0; c < mHist->GetSize(); c += mNoisePeriod) {
42  double val = gRandom->Gaus(0, parameters.mNoiseVar);
43  for (int bin = 0; bin < mHist->GetSize(); ++bin)
44  mHist->AddBinContent(bin, val * sinc(TMath::Pi() * (bin - c) / (double)mNoisePeriod));
45  for (int bin = 0; bin < mBinshift; ++bin)
46  mHistshift->AddBinContent(bin, val * sinc(TMath::Pi() * (bin - c) / (double)mNoisePeriod));
47  }
49  for (int bin = 0; bin < mHist->GetSize() - mBinshift; ++bin)
50  mHistshift->SetBinContent(bin + mBinshift, mHist->GetBinContent(bin));
51 
53  mHist->Scale(-1);
54  mHistsum->Add(mHistshift, mHist, 5, 1);
55  for (int bin = 1; bin < mHist->GetSize(); ++bin) {
57  if (mHistsum->GetBinContent(bin - 1) < 0 && mHistsum->GetBinContent(bin) >= 0) {
59  if (std::abs(mHist->GetBinContent(bin)) > 3) {
60  //std::cout << "Amp high enough: " << mHist->GetBinContent(bin) << " mV of " << maxBin << " mV at " << mHist->GetBinCenter(bin) << " ns\n";
61  return mHistsum->GetBinCenter(bin);
62  }
63  }
64  }
65  std::cout << "CFD failed to find peak\n";
66  return 1e10;
67 }

69 void Digitizer::process(const std::vector<o2::ft0::HitType>* hits, o2::ft0::Digit* digit, std::vector<std::vector<double>>& channel_times)
70 
71 {
72 
73  auto sorted_hits{*hits};
74  std::sort(sorted_hits.begin(), sorted_hits.end(), [](o2::ft0::HitType const& a, o2::ft0::HitType const& b) {
75  return a.GetTrackID() < b.GetTrackID();
76  });
77  digit->setTime(mEventTime);
78  digit->setInteractionRecord(mIntRecord);
79 
80  //Calculating signal time, amplitude in mean_time +- time_gate --------------
81  std::vector<o2::ft0::ChannelData>& channel_data = digit->getChDgData();
82  if (channel_data.size() == 0) {
83  channel_data.reserve(parameters.mMCPs);
84  for (int i = 0; i < parameters.mMCPs; ++i)
85  channel_data.emplace_back(o2::ft0::ChannelData{i, 0, 0, 0});
86  }
87  if (channel_times.size() == 0)
88  channel_times.resize(parameters.mMCPs);
89  Int_t parent = -10;
90  assert(digit->getChDgData().size() == parameters.mMCPs);
91  for (auto& hit : sorted_hits) {
92  if (hit.GetEnergyLoss() > 0)
93  continue;
94  Int_t hit_ch = hit.GetDetectorID();
95  Bool_t is_A_side = (hit_ch < 4 * parameters.nCellsA);
96  Float_t time_compensate = is_A_side ? parameters.A_side_cable_cmps : parameters.C_side_cable_cmps;
97  Double_t hit_time = hit.GetTime() - time_compensate;
98 
99  Bool_t is_hit_in_signal_gate = (abs(hit_time) < parameters.mSignalWidth * .5);
100 
101  if (is_hit_in_signal_gate) {
102  channel_data[hit_ch].numberOfParticles++;
103  channel_times[hit_ch].push_back(hit_time);
104  }
105 
106  //charge particles in MCLabel
107  Int_t parentID = hit.GetTrackID();
108  if (parentID != parent) {
109  o2::ft0::MCLabel label(hit.GetTrackID(), mEventID, mSrcID, hit_ch);
110  int lblCurrent;
111  if (mMCLabels) {
112  lblCurrent = mMCLabels->getIndexedSize(); // this is the size of mHeaderArray;
113  mMCLabels->addElement(lblCurrent, label);
114  }
115  parent = parentID;
116  }
117  }
118 }

121 void Digitizer::smearCFDtime(o2::ft0::Digit* digit, std::vector<std::vector<double>> const& channel_times)
122 {
123  //smeared CFD time for 50ps
124  std::vector<o2::ft0::ChannelData> mChDgDataArr;
125  for (const auto& d : digit->getChDgData()) {
126  Int_t mcp = d.ChId;
127  Float_t amp = (parameters.mMip_in_V * d.numberOfParticles / parameters.mPe_in_mip);
128  int chain = (std::rand() % 2) ? 1 : 0;
129  if (amp > parameters.mCFD_trsh) {
130  double smeared_time = 1000. * (get_time(channel_times[mcp]) - parameters.mCfdShift);
131  if (smeared_time < 1e9)
132  mChDgDataArr.emplace_back(o2::ft0::ChannelData{mcp, int(smeared_time / parameters.channelWidth), int(parameters.mV_2_Nchannels * amp), chain});
133  }
134  }
135  digit->setChDgData(std::move(mChDgDataArr));
136 }

139 void Digitizer::setTriggers(o2::ft0::Digit* digit)
140 {
141 
142  // Calculating triggers -----------------------------------------------------
143  int n_hit_A = 0, n_hit_C = 0, mean_time_A = 0, mean_time_C = 0;
144  int summ_ampl_A = 0, summ_ampl_C = 0;
145  int vertex_time;
146 
147  for (const auto& d : digit->getChDgData()) {
148  int mcp = d.ChId;
149  int cfd = d.CFDTime;
150  int amp = d.QTCAmpl;
151  if (std::abs(cfd) > parameters.mTime_trg_gate / 2.)
152  continue;
153 
154  Bool_t is_A_side = (mcp <= 4 * parameters.nCellsA);
155  if (is_A_side) {
156  n_hit_A++;
157  summ_ampl_A += amp;
158  mean_time_A += cfd;
159  } else {
160  n_hit_C++;
161  summ_ampl_C += amp;
162  mean_time_C += cfd;
163  }
164  }
165 
166  Bool_t is_A = n_hit_A > 0;
167  Bool_t is_C = n_hit_C > 0;
168  Bool_t is_Central = summ_ampl_A + summ_ampl_C >= parameters.mtrg_central_trh;
169  Bool_t is_SemiCentral = summ_ampl_A + summ_ampl_C >= parameters.mtrg_semicentral_trh;
170 
171  mean_time_A = is_A ? mean_time_A / n_hit_A : 0;
172  mean_time_C = is_C ? mean_time_C / n_hit_C : 0;
173  vertex_time = (mean_time_A - mean_time_C) * 0.5;
174  Bool_t is_Vertex = is_A && is_C && (std::abs(vertex_time) < parameters.mtrg_vertex);
175 
176  //filling digit
177  digit->setTriggers(is_A, is_C, is_Central, is_SemiCentral, is_Vertex);
178 
179  // Debug output -------------------------------------------------------------
180  LOG(DEBUG) << "\n\nTest digizing data ===================";
181 
182  LOG(INFO) << "Event ID: " << mEventID << " Event Time " << mEventTime;
183  LOG(INFO) << "N hit A: " << n_hit_A << " N hit C: " << n_hit_C << " summ ampl A: " << summ_ampl_A
184  << " summ ampl C: " << summ_ampl_C << " mean time A: " << mean_time_A
185  << " mean time C: " << mean_time_C;
186 
187  LOG(INFO) << "IS A " << is_A << " IS C " << is_C << " is Central " << is_Central
188  << " is SemiCentral " << is_SemiCentral << " is Vertex " << is_Vertex;
189 
190  LOG(DEBUG) << "======================================\n\n";
191  // --------------------------------------------------------------------------
192 }

194 void Digitizer::initParameters()
195 {
196  mEventTime = 0;
197  float signal_width = 0.5 * parameters.mSignalWidth;
198  mHist = new TH1F("time_histogram", "", 1000, -signal_width, signal_width);
199  mHistsum = new TH1F("time_sum", "", 1000, -signal_width, signal_width);
200  mHistshift = new TH1F("time_shift", "", 1000, -signal_width, signal_width);
201 
202  mNoisePeriod = parameters.mNoisePeriod / mHist->GetBinWidth(0);
203  mBinshift = int(parameters.mCFDShiftPos / (parameters.mSignalWidth / 1000.));
204 }
