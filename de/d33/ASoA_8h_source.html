141 template <typename T, typename ChunkingPolicy = Chunked>
142 class ColumnIterator : ChunkingPolicy
143 {
144  public:
149  ColumnIterator(arrow::Column const* column)
150  : mColumn{column},
151  mCurrentPos{nullptr},
152  mFirstIndex{0},
153  mCurrentChunk{0}
154  {
155  auto chunks = mColumn->data();
156  auto array = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
157  mCurrent = array->raw_values();
158  mLast = mCurrent + array->length();
159  }
160 
161  ColumnIterator() = default;
162  ColumnIterator(ColumnIterator<T, ChunkingPolicy> const&) = default;
163  ColumnIterator<T, ChunkingPolicy>& operator=(ColumnIterator<T, ChunkingPolicy> const&) = default;
164 
165  ColumnIterator(ColumnIterator<T, ChunkingPolicy>&&) = default;
166  ColumnIterator<T, ChunkingPolicy>& operator=(ColumnIterator<T, ChunkingPolicy>&&) = default;
167 
169  void nextChunk() const
170  {
171  auto chunks = mColumn->data();
172  auto previousArray = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
173  mFirstIndex += previousArray->length();
174  mCurrentChunk++;
175  auto array = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
176  mCurrent = array->raw_values() - mFirstIndex;
177  mLast = mCurrent + array->length() + mFirstIndex;
178  }
179 
180  void prevChunk() const
181  {
182  auto chunks = mColumn->data();
183  auto previousArray = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
184  mFirstIndex -= previousArray->length();
185  mCurrentChunk--;
186  auto array = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
187  mCurrent = array->raw_values() - mFirstIndex;
188  mLast = mCurrent + array->length() + mFirstIndex;
189  }
190 
191  void moveToChunk(int chunk)
192  {
193  if (mCurrentChunk < chunk) {
194  while (mCurrentChunk != chunk) {
195  nextChunk();
196  }
197  } else {
198  while (mCurrentChunk != chunk) {
199  prevChunk();
200  }
201  }
202  }
203 
205  void moveToEnd()
206  {
207  mCurrentChunk = mColumn->data()->num_chunks() - 1;
208  auto chunks = mColumn->data();
209  auto array = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
210  assert(array.get());
211  mFirstIndex = mColumn->length() - array->length();
212  mCurrent = array->raw_values() - mFirstIndex;
213  mLast = mCurrent + array->length() + mFirstIndex;
214  }
215 
216  T const& operator*() const
217  {
218  if constexpr (ChunkingPolicy::chunked) {
219  if (O2_BUILTIN_UNLIKELY(((mCurrent + *mCurrentPos) >= mLast))) {
220  nextChunk();
221  }
222  }
223  return *(mCurrent + *mCurrentPos);
224  }
225 
226  // Move to the chunk which containts element pos
227  ColumnIterator<T>& moveToPos()
228  {
229  // If we get outside range of the current chunk, go to the next.
230  if constexpr (ChunkingPolicy::chunked) {
231  while (O2_BUILTIN_UNLIKELY((mCurrent + *mCurrentPos) >= mLast)) {
232  nextChunk();
233  }
234  }
235  return *this;
236  }
237 
238  // Move to the chunk which containts element pos
239  ColumnIterator<T>& checkNextChunk()
240  {
241  if constexpr (ChunkingPolicy::chunked) {
242  if (O2_BUILTIN_LIKELY((mCurrent + *mCurrentPos) <= mLast)) {
243  return *this;
244  }
245  nextChunk();
246  }
247  return *this;
248  }
249 
250  mutable T const* mCurrent;
251  int64_t const* mCurrentPos;
252  mutable T const* mLast;
253  arrow::Column const* mColumn;
254  mutable int mFirstIndex;
255  mutable int mCurrentChunk;
256 };
