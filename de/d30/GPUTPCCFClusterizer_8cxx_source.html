29 GPUd() void GPUTPCCFClusterizer::computeClustersImpl(int nBlocks, int nThreads, int iBlock, int iThread, GPUSharedMemory& smem,
30  const Array2D<PackedCharge>& chargeMap,
31  const deprecated::Digit* digits,
32  uint clusternum,
33  uint maxClusterPerRow,
34  uint* clusterInRow,
35  tpc::ClusterNative* clusterByRow)
36 {
37  uint idx = get_global_id(0);
38 
39  // For certain configurations dummy work items are added, so the total
40  // number of work items is dividable by 64.
41  // These dummy items also compute the last cluster but discard the result.
42  deprecated::Digit myDigit = digits[CAMath::Min(idx, clusternum - 1)];
43 
44  ChargePos pos(myDigit);
45 
46  ClusterAccumulator pc;
47 #if defined(BUILD_CLUSTER_SCRATCH_PAD)
48  /* #if defined(BUILD_CLUSTER_SCRATCH_PAD) && defined(GPUCA_GPUCODE) */
49  /* #if 0 */
50  buildClusterScratchPad(
51  chargeMap,
52  pos,
53  smem.posBcast,
54  smem.buf,
55  smem.innerAboveThreshold,
56  &pc);
57 #else
58  buildClusterNaive(chargeMap, &pc, pos);
59 #endif
60 
61  if (idx >= clusternum) {
62  return;
63  }
64  pc.finalize(myDigit);
65 
66  tpc::ClusterNative myCluster;
67  pc.toNative(myDigit, myCluster);
68 
69 #if defined(CUT_QTOT)
70  bool aboveQTotCutoff = (pc.Q > QTOT_CUTOFF);
71 #else
72  bool aboveQTotCutoff = true;
73 #endif
74 
75  if (aboveQTotCutoff) {
76  sortIntoBuckets(
77  myCluster,
78  myDigit.row,
79  maxClusterPerRow,
80  clusterInRow,
81  clusterByRow);
82  }
83 }

105 GPUd() void GPUTPCCFClusterizer::addCorner(
106  const Array2D<PackedCharge>& chargeMap,
107  ClusterAccumulator* myCluster,
108  const ChargePos& pos,
109  Delta2 d)
110 {
111  Charge q = addInnerCharge(chargeMap, myCluster, pos, d);
112 
113  if (q > CHARGE_THRESHOLD) {
114  addOuterCharge(chargeMap, myCluster, pos, {GlobalPad(2 * d.x), d.y});
115  addOuterCharge(chargeMap, myCluster, pos, {d.x, Timestamp(2 * d.y)});
116  addOuterCharge(chargeMap, myCluster, pos, {GlobalPad(2 * d.x), Timestamp(2 * d.y)});
117  }

133 GPUd() void GPUTPCCFClusterizer::updateClusterScratchpadInner(
134  ushort lid,
135  ushort N,
136  const PackedCharge* buf,
137  ClusterAccumulator* cluster,
138  uchar* innerAboveThreshold)
139 {
140  uchar aboveThreshold = 0;
141 
142  LOOP_UNROLL_ATTR for (ushort i = 0; i < N; i++)
143  {
144  Delta2 d = CfConsts::InnerNeighbors[i];
145 
146  PackedCharge p = buf[N * lid + i];
147 
148  Charge q = cluster->updateInner(p, d);
149 
150  aboveThreshold |= (uchar(q > CHARGE_THRESHOLD) << i);
151  }
152 
153  innerAboveThreshold[lid] = aboveThreshold;
154 
155  GPUbarrier();
156 }

159  ushort lid,

162  ushort offset,

176 GPUd() void GPUTPCCFClusterizer::buildClusterScratchPad(
177  const Array2D<PackedCharge>& chargeMap,
178  ChargePos pos,
179  ChargePos* posBcast,
180  PackedCharge* buf,
181  uchar* innerAboveThreshold,
182  ClusterAccumulator* myCluster)
183 {
184  ushort ll = get_local_id(0);
185 
186  posBcast[ll] = pos;
187  GPUbarrier();
188 
189  CfUtils::blockLoad<PackedCharge>(
190  chargeMap,
191  SCRATCH_PAD_WORK_GROUP_SIZE,
192  SCRATCH_PAD_WORK_GROUP_SIZE,
193  ll,
194  0,
195  8,
196  CfConsts::InnerNeighbors,
197  posBcast,
198  buf);
199  updateClusterScratchpadInner(
200  ll,
201  8,
202  buf,
203  myCluster,
204  innerAboveThreshold);
205 
206  ushort wgSizeHalf = (SCRATCH_PAD_WORK_GROUP_SIZE + 1) / 2;
207 
208  bool inGroup1 = ll < wgSizeHalf;
209 
210  ushort llhalf = (inGroup1) ? ll : (ll - wgSizeHalf);
211 
212  CfUtils::condBlockLoad(
213  chargeMap,
214  wgSizeHalf,
215  SCRATCH_PAD_WORK_GROUP_SIZE,
216  ll,
217  0,
218  16,
219  CfConsts::OuterNeighbors,
220  posBcast,
221  innerAboveThreshold,
222  buf);
223 
224  if (inGroup1) {
225  updateClusterScratchpadOuter(
226  llhalf,
227  16,
228  16,
229  0,
230  buf,
231  myCluster);
232  }
233 
234 #if defined(GPUCA_GPUCODE)
235  CfUtils::condBlockLoad(
236  chargeMap,
237  wgSizeHalf,
238  SCRATCH_PAD_WORK_GROUP_SIZE,
239  ll,
240  0,
241  16,
242  CfConsts::OuterNeighbors,
243  posBcast + wgSizeHalf,
244  innerAboveThreshold + wgSizeHalf,
245  buf);
246  if (!inGroup1) {
247  updateClusterScratchpadOuter(
248  llhalf,
249  16,
250  16,
251  0,
252  buf,
253  myCluster);
254  }
255 #endif
256 }

259  const Array2D<PackedCharge>& chargeMap,
260  ClusterAccumulator* myCluster,
261  const ChargePos& pos)
262 {
263  // Add charges in top left corner:
264  // O O o o o
265  // O I i i o
266  // o i c i o
267  // o i i i o
268  // o o o o o
269  addCorner(chargeMap, myCluster, pos, {-1, -1});
270 
271  // Add upper charges
272  // o o O o o
273  // o i I i o
274  // o i c i o
275  // o i i i o
276  // o o o o o
277  addLine(chargeMap, myCluster, pos, {0, -1});
278 
279  // Add charges in top right corner:
280  // o o o O O
281  // o i i I O
282  // o i c i o
283  // o i i i o
284  // o o o o o
285  addCorner(chargeMap, myCluster, pos, {1, -1});
286 
287  // Add left charges
288  // o o o o o
289  // o i i i o
290  // O I c i o
291  // o i i i o
292  // o o o o o
293  addLine(chargeMap, myCluster, pos, {-1, 0});
294 
295  // Add right charges
296  // o o o o o
297  // o i i i o
298  // o i c I O
299  // o i i i o
300  // o o o o o
301  addLine(chargeMap, myCluster, pos, {1, 0});
302 
303  // Add charges in bottom left corner:
304  // o o o o o
305  // o i i i o
306  // o i c i o
307  // O I i i o
308  // O O o o o
309  addCorner(chargeMap, myCluster, pos, {-1, 1});
310 
311  // Add bottom charges
312  // o o o o o
313  // o i i i o
314  // o i c i o
315  // o i I i o
316  // o o O o o
317  addLine(chargeMap, myCluster, pos, {0, 1});

321  // o i i i o
322  // o i c i o
323  // o i i I O
324  // o o o O O
325  addCorner(chargeMap, myCluster, pos, {1, 1});
