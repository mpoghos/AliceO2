41  template <typename T>
42  T get(const char* key) const
43  {
44  assert(mRetriever);
45  try {
46  if constexpr (std::is_same_v<T, int>) {
47  return mRetriever->getInt(key);
48  } else if constexpr (std::is_same_v<T, int64_t>) {
49  return mRetriever->getInt64(key);
50  } else if constexpr (std::is_same_v<T, float>) {
51  return mRetriever->getFloat(key);
52  } else if constexpr (std::is_same_v<T, double>) {
53  return mRetriever->getDouble(key);
54  } else if constexpr (std::is_same_v<T, std::string>) {
55  return mRetriever->getString(key);
56  } else if constexpr (std::is_same_v<T, bool>) {
57  return mRetriever->getBool(key);
58  } else if constexpr (std::is_same_v<T, boost::property_tree::ptree>) {
59  return mRetriever->getPTree(key);
60  } else if constexpr (std::is_constructible_v<T, boost::property_tree::ptree>) {
61  return T{mRetriever->getPTree(key)};
62  } else if constexpr (std::is_constructible_v<T, boost::property_tree::ptree> == false) {
63  static_assert(std::is_constructible_v<T, boost::property_tree::ptree> == false,
64  "Not a basic type and no constructor from ptree provided");
65  }
66  } catch (std::exception& e) {
67  throw std::invalid_argument(std::string("missing option: ") + key + " (" + e.what() + ")");
68  } catch (...) {
69  throw std::invalid_argument(std::string("error parsing option: ") + key);
70  }
71  }
