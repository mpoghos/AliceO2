116  static void set(S* store, T value) { *(reinterpret_cast<T*>(store)) = value; }

128 // Poor man variant class. Does not take ownership of anything passed to it.
129 // FIXME: we should really use C++17 std::variant when it
130 // comes about
131 class Variant
132 {
133  using storage_t = std::aligned_union<8, int, int64_t, const char*, float, double, bool>::type;
134 
135  public:
136  Variant(VariantType type = VariantType::Unknown) : mType{type} {}
137 
138  template <typename T>
139  Variant(T value) : mType{variant_trait<T>::type()}
140  {
141  variant_helper<storage_t, decltype(value)>::set(&mStore, value);
142  }
143 
144  template <typename T>
145  Variant(std::initializer_list<T>) : mType{VariantType::Unknown}
146  {
147  static_assert(sizeof(T) == 0,
148  "brace-enclosed initializer list forbidden for Variant"
149  "\n did you accidentally put braces around the default value?");
150  }
151 
152  Variant(const Variant& other) : mType(other.mType)
153  {
154  // In case this is a string we need to duplicate it to avoid
155  // double deletion.
156  if (mType == variant_trait<const char*>::type()) {
157  variant_helper<storage_t, const char*>::set(&mStore, other.get<const char*>());
158  } else {
159  mStore = other.mStore;
160  }
161  }
162 
163  Variant(Variant&& other) : mType(other.mType)
164  {
165  // In case this is a string we need to duplicate it to avoid
166  // double deletion.
167  if (mType == variant_trait<const char*>::type()) {
168  mStore = other.mStore;
169  *reinterpret_cast<char**>(&(other.mStore)) = nullptr;
170  } else {
171  mStore = other.mStore;
172  }
173  }
174 
175  ~Variant()
176  {
177  // In case we allocated a string out of bound, we
178  // should delete it.
179  if (mType == variant_trait<const char*>::type() || mType == variant_trait<char*>::type()) {
180  free(*reinterpret_cast<void**>(&mStore));
181  }
182  }
183 
184  void operator=(const Variant& other)
185  {
186  if (mType == variant_trait<const char*>::type()) {
187  variant_helper<storage_t, const char*>::set(&mStore, other.get<const char*>());
188  } else {
189  mStore = other.mStore;
190  }
191  }
192 
193  template <typename T>
194  T get() const
195  {
196  if (mType != variant_trait<T>::type()) {
197  throw std::runtime_error("Mismatch between types");
198  }
199  return variant_helper<storage_t, T>::get(&mStore);
200  }
201 
202  template <typename T>
203  void set(T value)
204  {
205  return variant_helper<storage_t, T>::set(mStore, value);
206  }
207 
208  VariantType type() const { return mType; }
209 
210  private:
211  friend std::ostream& operator<<(std::ostream& oss, Variant const& val);
212  storage_t mStore;
213  VariantType mType;
214 };
