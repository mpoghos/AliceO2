78  // info on the smallest block of data to be read when fetching the HBF
79  struct LinkBlock {
80  enum { StartTF = 0x1,
81  StartHB = 0x1 << 1,
82  StartSP = 0x1 << 2,
83  EndHB = 0x1 << 3 };
84  size_t offset = 0; // where data of the block starts
85  uint32_t size = 0; // block size
86  uint32_t tfID = 0; // tf counter (from 0)
87  uint32_t orbit = 0; // orbit starting the block
88  uint16_t fileID = 0; // file id where the block is located
89  uint8_t flags = 0; // different flags
90  LinkBlock() = default;
91  LinkBlock(int fid, size_t offs) : offset(offs), fileID(fid) {}
92  void setFlag(uint8_t fl, bool v = true)
93  {
94  if (v)
95  flags |= fl;
96  else
97  flags &= ~fl;
98  }
99  bool testFlag(uint8_t fl) const { return (flags & fl) == fl; }
100  void print(const std::string& pref = "") const;
101  };
