98 bool Encoder::flush()
99 {
100  bool allempty = true;
101  for (int i = 0; i < 72; i++)
102  if (mIntegratedBytes[i])
103  allempty = false;
104 
105  if (allempty)
106  return true;
107 
108  // write superpages
109  for (int i = 0; i < 72; i++) {
110  int icru = i / NLINKSPERCRU;
111  mFileCRU[icru].write(mBuffer[i], mIntegratedBytes[i]);
112  }
113 
114  for (int i = 0; i < 72; i++)
115  flush(i);
116 
117  memset(mBuffer[0], 0, mSize * 72);
118 
119  return false;
120 }

130 bool Encoder::alloc(long size)
131 {
132  if (size < 500000)
133  size = 500000;
134 
135  mSize = size;
136 
137  mBufferLocal.resize(size * 72);
138 
139  mBuffer[0] = mBufferLocal.data();
140  memset(mBuffer[0], 0, mSize * 72);
141  mStart[0] = reinterpret_cast<Union_t*>(mBuffer[0]);
142  mUnion[0] = mStart[0]; // rewind
143 
144  for (int i = 1; i < 72; i++) {
145  mBuffer[i] = mBuffer[i - 1] + size;
146  mStart[i] = reinterpret_cast<Union_t*>(mBuffer[i]);
147  mUnion[i] = mStart[i]; // rewind
148  }
149  return false;
150 }

152 void Encoder::encodeTRM(const std::vector<Digit>& summary, Int_t icrate, Int_t itrm, int& istart) // encode one TRM assuming digit vector sorted by electronic index
153 // return next TRM index (-1 if not in the same crate)
154 // start to convert digiti from istart --> then update istart to the starting position of the new TRM
155 {
156  if (mVerbose)
157  printf("Crate %d: encode TRM %d \n", icrate, itrm);
158 
159  // TRM HEADER
160  mUnion[icrate]->trmDataHeader.slotId = itrm;
161  mUnion[icrate]->trmDataHeader.eventWords = 0; // to be filled at the end
162  mUnion[icrate]->trmDataHeader.eventCnt = mEventCounter;
163  mUnion[icrate]->trmDataHeader.emptyBit = 0;
164  mUnion[icrate]->trmDataHeader.dataId = 4;
165  nextWord(icrate);
166 
167  // LOOP OVER CHAINS
168  for (int ichain = 0; ichain < 2; ichain++) {
169  // CHAIN HEADER
170  mUnion[icrate]->trmChainHeader.slotId = itrm;
171  mUnion[icrate]->trmChainHeader.bunchCnt = mIR.bc;
172  mUnion[icrate]->trmChainHeader.mbz = 0;
173  mUnion[icrate]->trmChainHeader.dataId = 2 * ichain;
174  nextWord(icrate);
175 
176  while (istart < summary.size()) { // fill hits
178  int whatChain = summary[istart].getElChainIndex();
179  if (whatChain != ichain)
180  break;
181  int whatTRM = summary[istart].getElTRMIndex();
182  if (whatTRM != itrm)
183  break;
184  int whatCrate = summary[istart].getElCrateIndex();
185  if (whatCrate != icrate)
186  break;
187 
188  int hittimeTDC = (summary[istart].getBC() - mEventCounter * Geo::BC_IN_WINDOW) * 1024 + summary[istart].getTDC(); // time in TDC bin within the TOF WINDOW
189 
190  if (hittimeTDC < 0) {
191  LOG(ERROR) << "Negative hit encoded " << hittimeTDC << ", something went wrong in filling readout window";
192  printf("%d %d %d\n", summary[istart].getBC(), mEventCounter * Geo::BC_IN_WINDOW, summary[istart].getTDC());
193  }
194  // leading time
195  mUnion[icrate]->trmDataHit.time = hittimeTDC;
196  mUnion[icrate]->trmDataHit.chanId = summary[istart].getElChIndex();
197  mUnion[icrate]->trmDataHit.tdcId = summary[istart].getElTDCIndex();
198  mUnion[icrate]->trmDataHit.dataId = 0xa;
199  nextWord(icrate);
200 
201  // trailing time
202  mUnion[icrate]->trmDataHit.time = hittimeTDC + summary[istart].getTOT() * Geo::RATIO_TOT_TDC_BIN;
203  mUnion[icrate]->trmDataHit.chanId = summary[istart].getElChIndex();
204  mUnion[icrate]->trmDataHit.tdcId = summary[istart].getElTDCIndex();
205  mUnion[icrate]->trmDataHit.dataId = 0xc;
206  nextWord(icrate);
207 
208  istart++;
209  }
210 
211  // CHAIN TRAILER
212  mUnion[icrate]->trmChainTrailer.status = 0;
213  mUnion[icrate]->trmChainTrailer.mbz = 0;
214  mUnion[icrate]->trmChainTrailer.eventCnt = mEventCounter;
215  mUnion[icrate]->trmChainTrailer.dataId = 1 + 2 * ichain;
216  nextWord(icrate);
217  }
218 
219  // TRM TRAILER
220  mUnion[icrate]->trmDataTrailer.trailerMark = 3;
221  mUnion[icrate]->trmDataTrailer.eventCRC = 0; // to be implemented
222  mUnion[icrate]->trmDataTrailer.tempValue = 0;
223  mUnion[icrate]->trmDataTrailer.tempAddress = 0;
224  mUnion[icrate]->trmDataTrailer.tempChain = 0;
225  mUnion[icrate]->trmDataTrailer.tempAck = 0;
226  mUnion[icrate]->trmDataTrailer.lutErrorBit = 0;
227  mUnion[icrate]->trmDataTrailer.dataId = 5;
228  nextWord(icrate);
229 }

231 bool Encoder::encode(std::vector<std::vector<o2::tof::Digit>> digitWindow, int tofwindow) // pass a vector of digits in a TOF readout window, tof window is the entry in the vector-of-vector of digits needed to extract bunch id and orbit
232 {
233  if (digitWindow.size() != Geo::NWINDOW_IN_ORBIT) {
234  printf("Something went wrong in encoding: digitWindow=%lu different from %d\n", digitWindow.size(), Geo::NWINDOW_IN_ORBIT);
235  return 999;
236  }
237 
238  for (int i = 0; i < 72; i++) {
239  if (mIntegratedBytes[i] + 100000 > mSize)
240  flush();
241  }
242 
243  for (int iwin = 0; iwin < Geo::NWINDOW_IN_ORBIT; iwin++) {
244  std::vector<o2::tof::Digit>& summary = digitWindow.at(iwin);
245  // caching electronic indexes in digit array
246  for (auto dig = summary.begin(); dig != summary.end(); dig++) {
247  int digitchannel = dig->getChannel();
248  dig->setElectronicIndex(Geo::getECHFromCH(digitchannel));
249  }
250 
251  // sorting by electroni indexes
252  std::sort(summary.begin(), summary.end(),
253  [](Digit a, Digit b) { return a.getElectronicIndex() < b.getElectronicIndex(); });
254  }
255 
256 #ifdef VERBOSE
257  if (mVerbose)
258  std::cout << "-------- START ENCODE EVENT ----------------------------------------" << std::endl;
259 #endif
260  auto start = std::chrono::high_resolution_clock::now();
261 
262  mEventCounter = tofwindow; // tof window index
263  mIR.orbit = mEventCounter / Geo::NWINDOW_IN_ORBIT;
264 
265  if (!(mIR.orbit % 256)) { // new TF
266  flush();
267  }
268 
269  for (int i = 0; i < 72; i++) {
270  // add RDH open
271  mRDH[i] = reinterpret_cast<o2::header::RAWDataHeader*>(mUnion[i]);
272  mNextWordStatus[i] = false;
273  openRDH(i);
274  }
275 
276  // encode all windows
277  for (int iwin = 0; iwin < Geo::NWINDOW_IN_ORBIT; iwin++) {
278  mEventCounter = tofwindow + iwin; // tof window index
279 
280  std::vector<o2::tof::Digit>& summary = digitWindow.at(iwin);
281 
282  mIR.bc = ((mEventCounter % Geo::NWINDOW_IN_ORBIT) * Geo::BC_IN_ORBIT) / Geo::NWINDOW_IN_ORBIT; // bunch crossing in the current orbit at the beginning of the window
283 
284  int icurrentdigit = 0;
285  // TOF data header
286  for (int i = 0; i < 72; i++) {
287  mTOFDataHeader[i] = reinterpret_cast<TOFDataHeader_t*>(mUnion[i]);
288  mTOFDataHeader[i]->bytePayload = 0; // event length in byte (to be filled later)
289  mTOFDataHeader[i]->mbz = 0;
290  mTOFDataHeader[i]->dataId = 4;
291  nextWord(i);
292 
293  mUnion[i]->tofOrbit.orbit = mIR.orbit;
294  nextWord(i);
295 
296  mDRMDataHeader[i] = reinterpret_cast<DRMDataHeader_t*>(mUnion[i]);
297  mDRMDataHeader[i]->slotId = 1;
298  mDRMDataHeader[i]->eventWords = 0; // event length in word (to be filled later) --> word = byte/4
299  mDRMDataHeader[i]->drmId = i;
300  mDRMDataHeader[i]->dataId = 4;
301  nextWord(i);
302 
303  mUnion[i]->drmHeadW1.slotId = 1;
304  mUnion[i]->drmHeadW1.partSlotMask = (i % 2 == 0 ? 0x7fc : 0x7fe);
305  mUnion[i]->drmHeadW1.clockStatus = 2;
306  mUnion[i]->drmHeadW1.drmhVersion = 0x11;
307  mUnion[i]->drmHeadW1.drmHSize = 5;
308  mUnion[i]->drmHeadW1.mbz = 0;
309  mUnion[i]->drmHeadW1.dataId = 4;
310  nextWord(i);
311 
312  mUnion[i]->drmHeadW2.slotId = 1;
313  mUnion[i]->drmHeadW2.enaSlotMask = (i % 2 == 0 ? 0x7fc : 0x7fe);
314  mUnion[i]->drmHeadW2.mbz = 0;
315  mUnion[i]->drmHeadW2.faultSlotMask = 0;
316  mUnion[i]->drmHeadW2.readoutTimeOut = 0;
317  mUnion[i]->drmHeadW2.dataId = 4;
318  nextWord(i);
319 
320  mUnion[i]->drmHeadW3.slotId = 1;
321  mUnion[i]->drmHeadW3.gbtBunchCnt = mIR.bc;
322  mUnion[i]->drmHeadW3.locBunchCnt = 0;
323  mUnion[i]->drmHeadW3.dataId = 4;
324  nextWord(i);
325 
326  mUnion[i]->drmHeadW4.slotId = 1;
327  mUnion[i]->drmHeadW4.tempValue = 0;
328  mUnion[i]->drmHeadW4.mbza = 0;
329  mUnion[i]->drmHeadW4.tempAddress = 0;
330  mUnion[i]->drmHeadW4.mbzb = 0;
331  mUnion[i]->drmHeadW4.dataId = 4;
332  nextWord(i);
333 
334  mUnion[i]->drmHeadW5.slotId = 1;
335  mUnion[i]->drmHeadW5.eventCRC = 0;
336  mUnion[i]->drmHeadW5.irq = 0;
337  mUnion[i]->drmHeadW5.mbz = 0;
338  mUnion[i]->drmHeadW5.dataId = 4;
339  nextWord(i);
340 
341  int trm0 = 4 - (i % 2);
342  for (int itrm = trm0; itrm < 13; itrm++) {
343  encodeTRM(summary, i, itrm, icurrentdigit);
344  }
345 
346  mUnion[i]->drmDataTrailer.slotId = 1;
347  mUnion[i]->drmDataTrailer.locEvCnt = mEventCounter;
348  mUnion[i]->drmDataTrailer.mbz = 0;
349  mUnion[i]->drmDataTrailer.dataId = 5;
350  nextWord(i);
351  mUnion[i]->data = 0x70000000;
352  nextWord(i);
353 
354  mTOFDataHeader[i]->bytePayload = getSize(mTOFDataHeader[i], mUnion[i]);
355  mDRMDataHeader[i]->eventWords = mTOFDataHeader[i]->bytePayload / 4;
356  }
357 
358  // check that all digits were used
359  if (icurrentdigit < summary.size()) {
360  LOG(ERROR) << "Not all digits are been used : only " << icurrentdigit << " of " << summary.size();
361  }
362  }
363 
364  for (int i = 0; i < 72; i++) {
365  // adjust RDH open with the size
366  mRDH[i]->memorySize = getSize(mRDH[i], mUnion[i]);
367  mRDH[i]->offsetToNext = mRDH[i]->memorySize;
368  mIntegratedBytes[i] += mRDH[i]->offsetToNext;
369 
370  // add RDH close
371  closeRDH(i);
372  mIntegratedBytes[i] += mRDH[i]->offsetToNext;
373  mUnion[i] = reinterpret_cast<Union_t*>(nextPage(mRDH[i], mRDH[i]->offsetToNext));
374  }
375 
376  mStartRun = false;
377 
378  auto finish = std::chrono::high_resolution_clock::now();
379  std::chrono::duration<double> elapsed = finish - start;
380  mIntegratedTime = elapsed.count();
381 
382 #ifdef VERBOSE
383  int allBytes = mIntegratedAllBytes;
384  for (int i = 0; i < 72; i++)
385  allBytes += mIntegratedBytes[i];
386  if (mVerbose && mIntegratedTime)
387  std::cout << "-------- END ENCODE EVENT ------------------------------------------"
388  << " " << allBytes << " words"
389  << " | " << 1.e3 * mIntegratedTime << " ms"
390  << " | " << 1.e-6 * allBytes / mIntegratedTime << " MB/s (average)"
391  << std::endl;
392 #endif
393 
394  return false;
395 }

397 void Encoder::openRDH(int icrate)
398 {
399  *mRDH[icrate] = mHBFSampler.createRDH<o2::header::RAWDataHeader>(mIR);
400 
401  // word1
402  mRDH[icrate]->memorySize = mRDH[icrate]->headerSize;
403  mRDH[icrate]->offsetToNext = mRDH[icrate]->memorySize;
404  mRDH[icrate]->linkID = icrate % NLINKSPERCRU;
405  mRDH[icrate]->packetCounter = mNRDH[icrate];
406  mRDH[icrate]->cruID = icrate / NLINKSPERCRU;
407  mRDH[icrate]->feeId = icrate;
408 
409  // word2
410  mRDH[icrate]->triggerOrbit = mIR.orbit; // to be checked
411  mRDH[icrate]->heartbeatOrbit = mIR.orbit;
412 
413  // word4
414  mRDH[icrate]->triggerBC = 0; // to be checked (it should be constant)
415  mRDH[icrate]->heartbeatBC = 0;
416  mRDH[icrate]->triggerType = o2::trigger::HB | o2::trigger::ORBIT;
417  if (mStartRun) {
418  // mRDH[icrate]->triggerType |= mIsContinuous ? o2::trigger::SOC : o2::trigger::SOT;
419  mRDH[icrate]->triggerType |= o2::trigger::SOT;
420  }
421  if (!(mIR.orbit % 256))
422  mRDH[icrate]->triggerType |= o2::trigger::TF;
423 
424  // word6
425  mRDH[icrate]->pageCnt = 0;
426 
427  char* shift = reinterpret_cast<char*>(mRDH[icrate]);
428 
429  mUnion[icrate] = reinterpret_cast<Union_t*>(shift + mRDH[icrate]->headerSize);
430  mNRDH[icrate]++;
431 }
