53 void RawReader::run(ProcessingContext& pc)
54 {
55  if (mState != 1) {
56  return;
57  }
58 
59  printf("Run TOF compressed decoding\n");
60 
61  mState = 2;
62 
63  o2::tof::compressed::Decoder decoder;
64 
65  decoder.open(mFilename.c_str());
66  decoder.setVerbose(0);
67 
68  // decode raw to digit here
69  std::vector<o2::tof::Digit> digitsTemp;
70 
71  printf("start decoding raw\n");
72  decoder.decode();
73  printf("end decoding raw\n");
74 
75  std::vector<o2::tof::Digit>* alldigits = decoder.getDigitPerTimeFrame();
76  std::vector<o2::tof::ReadoutWindowData>* row = decoder.getReadoutWindowData();
77 
78  int n_tof_window = row->size();
79  int n_orbits = n_tof_window / 3;
80  int digit_size = alldigits->size();
81 
82  LOG(INFO) << "TOF: N tof window decoded = " << n_tof_window << "(orbits = " << n_orbits << ") with " << digit_size << " digits";
83 
84  // add digits in the output snapshot
85  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "DIGITS", 0, Lifetime::Timeframe}, *alldigits);
86  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "READOUTWINDOW", 0, Lifetime::Timeframe}, *row);
87 
88  static o2::parameters::GRPObject::ROMode roMode = o2::parameters::GRPObject::CONTINUOUS;
89 
90  LOG(INFO) << "TOF: Sending ROMode= " << roMode << " to GRPUpdater";
91  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "ROMode", 0, Lifetime::Timeframe}, roMode);
92 
93  //pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
94  pc.services().get<ControlService>().endOfStream();
95 }
