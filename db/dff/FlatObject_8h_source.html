119 template <typename T>
120 T* resizeArray(T*& ptr, int oldSize, int newSize, T* newPtr = nullptr)
121 {
122  // Resize array pointed by ptr. T must be a POD class.
123  // If the non-null newPtr is provided, use it instead of allocating a new one.
124  // In this case it is up to the user to ensure that it has at least newSize slots allocated.
125  // Return original array pointer, so that the user can manage previously allocate memory
126  if (oldSize < 0) {
127  oldSize = 0;
128  }
129  if (newSize > 0) {
130  if (!newPtr) {
131  newPtr = new T[newSize];
132  }
133  int mcp = std::min(newSize, oldSize);
134  std::memmove(newPtr, ptr, mcp * sizeof(T));
135  if (newSize > oldSize) {
136  std::memset(newPtr + mcp, 0, (newSize - oldSize) * sizeof(T));
137  }
138  }
139  T* oldPtr = ptr;
140  ptr = newPtr;
141  return oldPtr;
142 }

354 inline void FlatObject::cloneFromObject(const FlatObject& obj, char* newFlatBufferPtr)
355 {
359  // that we want to relocate the obj to external buffer)
360 
361  assert(obj.isConstructed());
362 
363  // providing *this with internal buffer as obj makes sens only if we want to conver it to object with PROVIDED external buffer
364  assert(!(!newFlatBufferPtr && obj.mFlatBufferPtr == mFlatBufferPtr && obj.isBufferInternal()));
365 
366  char* oldPtr = resizeArray(mFlatBufferPtr, mFlatBufferSize, obj.mFlatBufferSize, newFlatBufferPtr);
367 
368  if (isBufferInternal()) {
369  delete[] oldPtr; // delete old buffer if owned
370  }
371  mFlatBufferSize = obj.mFlatBufferSize;
372  mFlatBufferContainer = newFlatBufferPtr ? nullptr : mFlatBufferPtr; // external buffer is not provided, make object to own the buffer
373  std::memcpy(mFlatBufferPtr, obj.mFlatBufferPtr, obj.mFlatBufferSize);
374  mConstructionMask = (unsigned int)ConstructionState::Constructed;
375 }

442 template <class T>
443 inline std::string FlatObject::stressTest(T& obj)
444 {
448  std::string err;
449 
450  if (!obj.isConstructed())
451  return "tested object is not constructed!";
452 
453  T tst;
454  tst.cloneFromObject(obj, nullptr);
455  if (!tst.isConstructed() || !tst.isBufferInternal())
456  return "error at cloneFromObject()!";
457 
458  obj.destroy();
459 
460  char* buf0 = tst.releaseInternalBuffer();
461  char* buf1 = new char[tst.getFlatBufferSize()];
462  char* buf2 = new char[tst.getFlatBufferSize()];
463  std::memcpy(buf1, tst.getFlatBufferPtr(), tst.getFlatBufferSize());
464  tst.setActualBufferAddress(buf1);
465  delete[] buf0;
466 
467  tst.setFutureBufferAddress(buf2);
468  std::memcpy(buf2, buf1, tst.getFlatBufferSize());
469  delete[] buf1;
470 
471  if (tst.isBufferInternal())
472  return err = "error, buffer should be external!";
473 
474  tst.adoptInternalBuffer(buf2);
475  if (!tst.isBufferInternal())
476  return err = "error, buffer should be internal!";
477 
478  obj.cloneFromObject(tst, nullptr);
479  if (!obj.isBufferInternal())
480  return err = "error, buffer should be internal!";
481 
482  return err;
483 }
