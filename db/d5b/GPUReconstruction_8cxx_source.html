83 int GPUReconstruction::Init()
84 {
85  if (mDeviceProcessingSettings.memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_AUTO) {
86  mDeviceProcessingSettings.memoryAllocationStrategy = IsGPU() ? GPUMemoryResource::ALLOCATION_GLOBAL : GPUMemoryResource::ALLOCATION_INDIVIDUAL;
87  }
88  if (mDeviceProcessingSettings.eventDisplay) {
89  mDeviceProcessingSettings.keepAllMemory = true;
90  }
91  if (mDeviceProcessingSettings.debugLevel >= 4) {
92  mDeviceProcessingSettings.keepAllMemory = true;
93  }
94  if (mDeviceProcessingSettings.debugLevel < 6) {
95  mDeviceProcessingSettings.debugMask = 0;
96  }
97  if (mDeviceProcessingSettings.debugLevel < 1) {
98  mDeviceProcessingSettings.deviceTimers = false;
99  }
100  if (GetDeviceProcessingSettings().debugLevel >= 6 && GetDeviceProcessingSettings().comparableDebutOutput) {
101  mDeviceProcessingSettings.nTPCClustererLanes = 1;
102  }
103 
104 #ifndef HAVE_O2HEADERS
105  mRecoSteps.setBits(RecoStep::ITSTracking, false);
106  mRecoSteps.setBits(RecoStep::TRDTracking, false);
107  mRecoSteps.setBits(RecoStep::TPCConversion, false);
108  mRecoSteps.setBits(RecoStep::TPCCompression, false);
109  mRecoSteps.setBits(RecoStep::TPCdEdx, false);
110 #endif
111  mRecoStepsGPU &= mRecoSteps;
112  mRecoStepsGPU &= AvailableRecoSteps();
113  if (!IsGPU()) {
114  mRecoStepsGPU.set((unsigned char)0);
115  }
116  if (!IsGPU()) {
117  mDeviceProcessingSettings.trackletConstructorInPipeline = mDeviceProcessingSettings.trackletSelectorInPipeline = false;
118  }
119  if (param().rec.NonConsecutiveIDs) {
120  param().rec.DisableRefitAttachment = 0xFF;
121  }
122  if (!mDeviceProcessingSettings.trackletConstructorInPipeline) {
123  mDeviceProcessingSettings.trackletSelectorInPipeline = false;
124  }
125 
126 #ifdef WITH_OPENMP
127  if (mDeviceProcessingSettings.nThreads <= 0) {
128  mDeviceProcessingSettings.nThreads = omp_get_max_threads();
129  } else {
130  omp_set_num_threads(mDeviceProcessingSettings.nThreads);
131  }
132 #else
133  mDeviceProcessingSettings.nThreads = 1;
134 #endif
135 
136  mDeviceMemorySize = mHostMemorySize = 0;
137  for (unsigned int i = 0; i < mChains.size(); i++) {
138  mChains[i]->RegisterPermanentMemoryAndProcessors();
139  size_t memGpu, memHost;
140  mChains[i]->MemorySize(memGpu, memHost);
141  mDeviceMemorySize += memGpu;
142  mHostMemorySize += memHost;
143  }
144  if (mDeviceProcessingSettings.forceMemoryPoolSize && mDeviceProcessingSettings.forceMemoryPoolSize <= 2 && CanQueryMaxMemory()) {
145  mDeviceMemorySize = mDeviceProcessingSettings.forceMemoryPoolSize;
146  } else if (mDeviceProcessingSettings.forceMemoryPoolSize > 2) {
147  mDeviceMemorySize = mHostMemorySize = mDeviceProcessingSettings.forceMemoryPoolSize;
148  }
149 
150  for (unsigned int i = 0; i < mProcessors.size(); i++) {
151  (mProcessors[i].proc->*(mProcessors[i].RegisterMemoryAllocation))();
152  }
153 
154  if (InitDevice()) {
155  return 1;
156  }
157  if (IsGPU()) {
158  for (unsigned int i = 0; i < mChains.size(); i++) {
159  mChains[i]->RegisterGPUProcessors();
160  }
161  }
162  AllocateRegisteredPermanentMemory();
163 
164  for (unsigned int i = 0; i < mChains.size(); i++) {
165  if (mChains[i]->Init()) {
166  return 1;
167  }
168  }
169  for (unsigned int i = 0; i < mProcessors.size(); i++) {
170  (mProcessors[i].proc->*(mProcessors[i].InitializeProcessor))();
171  }
172 
173  if (IsGPU()) {
174  const auto threadContext = GetThreadContext();
175  WriteToConstantMemory((char*)&processors()->param - (char*)processors(), &param(), sizeof(param()), -1);
176  }
177 
178  mInitialized = true;
179  return 0;
180 }

190 int GPUReconstruction::Exit()
191 {
192  mChains.clear(); // Make sure we destroy a possible ITS GPU tracker before we call the destructors
193  mHostConstantMem.reset(); // Reset these explicitly before the destruction of other members unloads the library
194  if (mDeviceProcessingSettings.memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_INDIVIDUAL) {
195  for (unsigned int i = 0; i < mMemoryResources.size(); i++) {
196  if (mMemoryResources[i].mReuse != -1) {
197  continue;
198  }
199  operator delete(mMemoryResources[i].mPtrDevice);
200  mMemoryResources[i].mPtr = mMemoryResources[i].mPtrDevice = nullptr;
201  }
202  }
203  mMemoryResources.clear();
204  if (mInitialized) {
205  ExitDevice();
206  }
207  mInitialized = false;
208  return 0;
209 }

214 size_t GPUReconstruction::AllocateRegisteredMemory(GPUProcessor* proc)
215 {
216  if (mDeviceProcessingSettings.debugLevel >= 5) {
217  GPUInfo("Allocating memory %p", (void*)proc);
218  }
219  size_t total = 0;
220  for (unsigned int i = 0; i < mMemoryResources.size(); i++) {
221  if ((proc == nullptr ? !mMemoryResources[i].mProcessor->mAllocateAndInitializeLate : mMemoryResources[i].mProcessor == proc) && !(mMemoryResources[i].mType & GPUMemoryResource::MEMORY_CUSTOM)) {
222  total += AllocateRegisteredMemory(i);
223  }
224  }
225  if (mDeviceProcessingSettings.debugLevel >= 5) {
226  GPUInfo("Allocating memory done");
227  }
228  return total;

231 size_t GPUReconstruction::AllocateRegisteredPermanentMemory()
232 {
233  if (mDeviceProcessingSettings.debugLevel >= 5) {
234  GPUInfo("Allocating Permanent Memory");
235  }
236  int total = 0;
237  for (unsigned int i = 0; i < mMemoryResources.size(); i++) {
238  if ((mMemoryResources[i].mType & GPUMemoryResource::MEMORY_PERMANENT) && mMemoryResources[i].mPtr == nullptr) {
239  total += AllocateRegisteredMemory(i);
240  }
241  }
242  mHostMemoryPermanent = mHostMemoryPool;
243  mDeviceMemoryPermanent = mDeviceMemoryPool;
244  if (mDeviceProcessingSettings.debugLevel >= 5) {
245  GPUInfo("Permanent Memory Done");
246  }
247  return total;
248 }

250 size_t GPUReconstruction::AllocateRegisteredMemoryHelper(GPUMemoryResource* res, void*& ptr, void*& memorypool, void* memorybase, size_t memorysize, void* (GPUMemoryResource::*setPtr)(void*))
251 {
252  if (res->mReuse != -1) {
253  ptr = (&ptr == &res->mPtrDevice) ? mMemoryResources[res->mReuse].mPtrDevice : mMemoryResources[res->mReuse].mPtr;
254  size_t retVal = (char*)((res->*setPtr)(ptr)) - (char*)(ptr);
255  if (retVal > mMemoryResources[res->mReuse].mSize) {
256  throw std::bad_alloc();
257  }
258  if (mDeviceProcessingSettings.debugLevel >= 5) {
259  std::cout << "Reused " << res->mName << ": " << retVal << "\n";
260  }
261  return retVal;
262  }
263  if (memorypool == nullptr) {
264  GPUInfo("Memory pool uninitialized");
265  throw std::bad_alloc();
266  }
267  ptr = memorypool;
268  memorypool = (char*)((res->*setPtr)(memorypool));
269  size_t retVal = (char*)memorypool - (char*)ptr;
270  if (IsGPU() && retVal == 0) { // Transferring 0 bytes might break some GPU backends, but we cannot simply skip the transfer, or we will break event dependencies
271  GPUProcessor::getPointerWithAlignment<GPUProcessor::MIN_ALIGNMENT, char>(memorypool, retVal = GPUProcessor::MIN_ALIGNMENT);
272  }
273  if ((size_t)((char*)memorypool - (char*)memorybase) > memorysize) {
274  std::cout << "Memory pool size exceeded (" << res->mName << ": " << (char*)memorypool - (char*)memorybase << " < " << memorysize << "\n";
275  throw std::bad_alloc();
276  }
277  memorypool = (void*)((char*)memorypool + GPUProcessor::getAlignment<GPUCA_MEMALIGN>(memorypool));
278  if (mDeviceProcessingSettings.debugLevel >= 5) {
279  std::cout << "Allocated " << res->mName << ": " << retVal << " - available: " << memorysize - ((char*)memorypool - (char*)memorybase) << "\n";
280  }
281  return retVal;
282 }

284 size_t GPUReconstruction::AllocateRegisteredMemory(short ires, GPUOutputControl* control)
285 {
286  GPUMemoryResource* res = &mMemoryResources[ires];
287  if ((res->mType & GPUMemoryResource::MEMORY_PERMANENT) && res->mPtr != nullptr) {
288  ResetRegisteredMemoryPointers(ires);
289  } else if (mDeviceProcessingSettings.memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_INDIVIDUAL && (control == nullptr || control->OutputType == GPUOutputControl::AllocateInternal)) {
290  if (!(res->mType & GPUMemoryResource::MEMORY_EXTERNAL)) {
291  if (res->mPtrDevice && res->mReuse == -1) {
292  operator delete(res->mPtrDevice);
293  }
294  res->mSize = (size_t)res->SetPointers((void*)1) - 1;
295  if (res->mReuse != -1) {
296  if (res->mSize > mMemoryResources[res->mReuse].mSize) {
297  GPUError("Invalid reuse, insufficient size: %lld < %lld", (long long int)mMemoryResources[res->mReuse].mSize, (long long int)res->mSize);
298  throw std::bad_alloc();
299  }
300  res->mPtrDevice = mMemoryResources[res->mReuse].mPtrDevice;
301  } else {
302  res->mPtrDevice = operator new(res->mSize + GPUProcessor::MIN_ALIGNMENT);
303  }
304  res->mPtr = GPUProcessor::alignPointer<GPUProcessor::MIN_ALIGNMENT>(res->mPtrDevice);
305  res->SetPointers(res->mPtr);
306  if (mDeviceProcessingSettings.debugLevel >= 5) {
307  std::cout << (res->mReuse != -1 ? "Reused " : "Allocated ") << res->mName << ": " << res->mSize << "\n";
308  }
309  }
310  } else {
311  if (res->mPtr != nullptr) {
312  GPUError("Double allocation! (%s)", res->mName);
313  throw std::bad_alloc();
314  }
315  if ((!IsGPU() || (res->mType & GPUMemoryResource::MEMORY_HOST) || mDeviceProcessingSettings.keepAllMemory) && !(res->mType & GPUMemoryResource::MEMORY_EXTERNAL)) {
316  if (control && control->OutputType == GPUOutputControl::UseExternalBuffer) {
317  res->mSize = AllocateRegisteredMemoryHelper(res, res->mPtr, control->OutputPtr, control->OutputBase, control->OutputMaxSize, &GPUMemoryResource::SetPointers);
318  } else {
319  res->mSize = AllocateRegisteredMemoryHelper(res, res->mPtr, mHostMemoryPool, mHostMemoryBase, mHostMemorySize, &GPUMemoryResource::SetPointers);
320  }
321  }
322  if (IsGPU() && (res->mType & GPUMemoryResource::MEMORY_GPU)) {
323  if (res->mProcessor->mDeviceProcessor == nullptr) {
324  GPUError("Device Processor not set (%s)", res->mName);
325  throw std::bad_alloc();
326  }
327  size_t size = AllocateRegisteredMemoryHelper(res, res->mPtrDevice, mDeviceMemoryPool, mDeviceMemoryBase, mDeviceMemorySize, &GPUMemoryResource::SetDevicePointers);
328 
329  if (!(res->mType & GPUMemoryResource::MEMORY_HOST) || (res->mType & GPUMemoryResource::MEMORY_EXTERNAL)) {
330  res->mSize = size;
331  } else if (size != res->mSize) {
332  GPUError("Inconsistent device memory allocation (%s)", res->mName);
333  throw std::bad_alloc();
334  }
335  }
336  }
337  if (res->mReuse != -1) {
338  return 0;
339  } else {
340  return res->mSize;
341  }
342 }

344 void* GPUReconstruction::AllocateUnmanagedMemory(size_t size, int type)
345 {
346  if (type != GPUMemoryResource::MEMORY_HOST && (!IsGPU() || type != GPUMemoryResource::MEMORY_GPU)) {
347  throw std::bad_alloc();
348  }
349  if (mDeviceProcessingSettings.memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_INDIVIDUAL) {
350  mUnmanagedChunks.emplace_back(new char[size + GPUProcessor::MIN_ALIGNMENT]);
351  return GPUProcessor::alignPointer<GPUProcessor::MIN_ALIGNMENT>(mUnmanagedChunks.back().get());
352  } else {
353  void* pool = type == GPUMemoryResource::MEMORY_GPU ? mDeviceMemoryPool : mHostMemoryPool;
354  void* base = type == GPUMemoryResource::MEMORY_GPU ? mDeviceMemoryBase : mHostMemoryBase;
355  size_t poolsize = type == GPUMemoryResource::MEMORY_GPU ? mDeviceMemorySize : mHostMemorySize;
356  char* retVal;
357  GPUProcessor::computePointerWithAlignment(pool, retVal, size);
358  if ((size_t)((char*)pool - (char*)base) > poolsize) {
359  throw std::bad_alloc();
360  }
361  return retVal;
362  }
363 }

418 void GPUReconstruction::PrintMemoryStatistics()
419 {
420  std::map<std::string, std::array<size_t, 3>> sizes;
421  for (unsigned int i = 0; i < mMemoryResources.size(); i++) {
422  auto& res = mMemoryResources[i];
423  auto& x = sizes[res.mName];
424  if (res.mPtr) {
425  x[0] += res.mSize;
426  }
427  if (res.mPtrDevice) {
428  x[1] += res.mSize;
429  }
430  if (res.mType & GPUMemoryResource::MemoryType::MEMORY_PERMANENT) {
431  x[2] = 1;
432  }
433  }
434  for (auto it = sizes.begin(); it != sizes.end(); it++) {
435  printf("Allocation %30s %s: Size %'13lld / %'13lld\n", it->first.c_str(), it->second[2] ? "P" : " ", (long long int)it->second[0], (long long int)it->second[1]);
436  }
437  if (GetDeviceProcessingSettings().memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_GLOBAL) {
438  printf("Memory Allocation: Host %'lld / %'lld (Permanent %'lld), Device %'lld / %'lld, (Permanent %'lld) %d chunks\n",
439  ptrDiff(mHostMemoryPool, mHostMemoryBase), (long long int)mHostMemorySize, ptrDiff(mHostMemoryPermanent, mHostMemoryBase),
440  ptrDiff(mDeviceMemoryPool, mDeviceMemoryBase), (long long int)mDeviceMemorySize, ptrDiff(mDeviceMemoryPermanent, mDeviceMemoryBase), (int)mMemoryResources.size());
441  }
442  for (unsigned int i = 0; i < mChains.size(); i++) {
443  mChains[i]->PrintMemoryStatistics();
444  }
445 }
