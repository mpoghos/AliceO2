78 //____________________________________________
79 size_t RawFileReader::LinkData::getNextHBFSize() const
80 {
81  // estimate the memory size of the next HBF to read
82  // The blocks are guaranteed to not cover more than 1 HB
83  size_t sz = 0;
84  int ibl = nextBlock2Read, nbl = blocks.size();
85  while (ibl < nbl && (blocks[ibl].orbit == blocks[nextBlock2Read].orbit)) {
86  sz += blocks[ibl].size;
87  ibl++;
88  }
89  return sz;
90 }

92 //____________________________________________
93 size_t RawFileReader::LinkData::readNextHBF(char* buff)
94 {
95  // read data of the next complete HB, buffer of getNextHBFSize() must be allocated in advance
96  size_t sz = 0;
97  int ibl = nextBlock2Read, nbl = blocks.size();
98  bool error = false;
99  while (ibl < nbl) {
100  const auto& blc = blocks[ibl];
101  if (blc.orbit != blocks[nextBlock2Read].orbit) {
102  break;
103  }
104  ibl++;
105  auto fl = reader->mFiles[blc.fileID];
106  if (fseek(fl, blc.offset, SEEK_SET) || fread(buff + sz, 1, blc.size, fl) != blc.size) {
107  LOGF(ERROR, "Failed to read for the %s a bloc:", describe());
108  blc.print();
109  error = true;
110  }
111  sz += blc.size;
112  }
113  nextBlock2Read = ibl;
114  return error ? 0 : sz; // in case of the error we ignore the data
115 }

119 {
120  // estimate the memory size of the next TF to read
121  // (assuming nextBlock2Read is at the start of the TF)

131 //_____________________________________________________________________
132 size_t RawFileReader::LinkData::readNextTF(char* buff)
133 {
134  // read next complete TF, buffer of getNextTFSize() must be allocated in advance
135  size_t sz = 0;
136  int ibl0 = nextBlock2Read, nbl = blocks.size();
137  bool error = false;
138  while (nextBlock2Read < nbl && (blocks[nextBlock2Read].tfID == blocks[ibl0].tfID)) { // nextBlock2Read is incremented by the readNextHBF!
139  auto szb = readNextHBF(buff + sz);
140  if (!szb) {
141  error = true;
142  }
143  sz += szb;
144  }
145  return error ? 0 : sz; // in case of the error we ignore the data
146 }

162 //____________________________________________
163 size_t RawFileReader::LinkData::getLargestSuperPage() const
164 {
165  // estimate largest super page size
166  size_t szMax = 0, szLast = 0;
167  for (const auto& bl : blocks) {
168  if (bl.testFlag(LinkBlock::StartSP)) { // account previous SPage and start accumulation of the next one
169  if (szLast > szMax) {
170  szMax = szLast;
171  }
172  szLast = 0;
173  }
174  szLast += bl.size;
175  }
176  return szLast > szMax ? szLast : szMax;
177 }

196 //_____________________________________________________________________
197 bool RawFileReader::LinkData::preprocessCRUPage(const RDH& rdh, bool newSPage)
198 {
199  // account RDH in statistics
200  bool ok = true;
201  bool newTF = false, newHB = false;
202 
203  if (rdh.feeId != rdhl.feeId) { // make sure links with different FEEID were not assigned same subspec
204  LOGF(ERROR, "Same SubSpec is found for %s with different RDH.feeId", describe());
205  LOGF(ERROR, "old RDH assigned SubSpec=0x%-8d:", subspec);
206  HBFUtils::dumpRDH(rdhl);
207  LOGF(ERROR, "new RDH assigned SubSpec=0x%-8d:", subspec);
208  HBFUtils::dumpRDH(rdh);
209  throw std::runtime_error("Confliting SubSpecs are provided");
210  ok = false;
211  nErrors++;
212  }
213 
214  if (rdh.pageCnt == 0) {
215  newTF = (rdh.triggerType & o2::trigger::TF);
216  newHB = (rdh.triggerType & (o2::trigger::ORBIT | o2::trigger::HB)) == (o2::trigger::ORBIT | o2::trigger::HB);
217  } else if (reader->mCheckErrors & (0x1 << ErrWrongPageCounterIncrement)) {
218  // check increasing pageCnt
219  if (nCRUPages && (rdh.pageCnt != ((rdhl.pageCnt + 1) & 0xffff))) { // skip for very 1st page
220  LOG(ERROR) << ErrNames[ErrWrongPageCounterIncrement]
221  << " old=" << int(rdh.pageCnt) << " new=" << int(rdhl.pageCnt);
222  ok = false;
223  nErrors++;
224  }
225  }
226 
227  if (reader->mCheckErrors) {
228  if (nCRUPages) {
229  // check increasing (or wrapping) packetCounter
230  if ((rdh.packetCounter != ((rdhl.packetCounter + 1) & 0xff)) &&
231  (reader->mCheckErrors & (0x1 << ErrWrongPacketCounterIncrement))) { // skip for very 1st page
232  LOG(ERROR) << ErrNames[ErrWrongPacketCounterIncrement]
233  << " new=" << int(rdh.packetCounter) << " old=" << int(rdhl.packetCounter);
234  ok = false;
235  nErrors++;
236  }
237  // check if number of HBFs in the TF is as expected
238  if (newTF) {
239  if (nHBFinTF != reader->mNominalHBFperTF &&
240  (reader->mCheckErrors & (0x1 << ErrWrongHBFsPerTF))) {
241  LOG(ERROR) << ErrNames[ErrWrongHBFsPerTF] << ": "
242  << nHBFinTF << " instead of " << reader->mNominalHBFperTF;
243  ok = false;
244  nErrors++;
245  }
246  nHBFinTF = 0; // reset
247  }
248 
249  } else { // make sure data starts with TF and HBF
250  if ((!newTF || !newHB || rdh.pageCnt != 0) &&
251  (reader->mCheckErrors & (0x1 << ErrWrongFirstPage))) {
252  LOG(ERROR) << ErrNames[ErrWrongFirstPage];
253  ok = false;
254  nErrors++;
255  }
256  }
257  }
258 
259  if (newHB) {
260  if (reader->mCheckErrors) {
261  nHBFinTF++;
262  if (rdh.stop && (reader->mCheckErrors & (0x1 << ErrHBFStopOnFirstPage))) {
263  LOG(ERROR) << ErrNames[ErrHBFStopOnFirstPage] << " @ HBF#" << nHBFrames;
264  ok = false;
265  nErrors++;
266  }
267  if (openHB && (reader->mCheckErrors & (0x1 << ErrHBFNoStop))) {
268  LOG(ERROR) << ErrNames[ErrHBFNoStop] << " @ HBF#" << nHBFrames;
269  ok = false;
270  nErrors++;
271  }
272  if ((reader->mCheckErrors & (0x1 << ErrHBFJump)) &&
273  (nCRUPages && // skip this check for the very 1st RDH
274  !(HBFUtils::getHBBC(rdh) == HBFUtils::getHBBC(rdhl) &&
275  HBFUtils::getHBOrbit(rdh) == HBFUtils::getHBOrbit(rdhl) + 1))) {
276  LOG(ERROR) << ErrNames[ErrHBFJump] << " @ HBF#" << nHBFrames << " New HB orbit/bc="
277  << HBFUtils::getHBOrbit(rdh) << '/' << int(HBFUtils::getHBBC(rdh))
278  << " is not incremented by 1 orbit wrt Old HB orbit/bc="
279  << HBFUtils::getHBOrbit(rdhl) << '/' << int(HBFUtils::getHBBC(rdhl));
280  ok = false;
281  nErrors++;
282  }
283  } // end of check errors
284  openHB = true;
285  nHBFrames++;
286  }
287  if (rdh.stop) {
288  openHB = false;
289  }
290 
291  if (newTF || newSPage || newHB) {
292  auto& bl = blocks.emplace_back(reader->mCurrentFileID, reader->mPosInFile);
293  if (newTF) {
294  nTimeFrames++;
295  bl.setFlag(LinkBlock::StartTF);
296  if (reader->mCheckErrors & (0x1 << ErrNoSuperPageForTF)) {
297  if (reader->mMultiLinkFile && !newSPage) {
298  LOG(ERROR) << ErrNames[ErrNoSuperPageForTF] << " @ TF#" << nTimeFrames;
299  ok = false;
300  nErrors++;
301  }
302  } // end of check errors
303  }
304  bl.orbit = HBFUtils::getHBOrbit(rdh);
305  bl.tfID = nTimeFrames - 1;
306 
307  if (newSPage) {
308  nSPages++;
309  bl.setFlag(LinkBlock::StartSP);
310  }
311  if (newHB) {
312  bl.setFlag(LinkBlock::StartHB);
313  }
314  }
315  blocks.back().setFlag(LinkBlock::EndHB, rdh.stop); // last processed RDH defines this flag
316  blocks.back().size += rdh.offsetToNext;
317  rdhl = rdh;
318  nCRUPages++;
319  if (!ok) {
320  LOG(ERROR) << " ^^^Problem(s) was encountered at offset " << reader->mPosInFile << " of file " << reader->mCurrentFileID;
321  HBFUtils::printRDH(rdh);
322  } else if (reader->mVerbosity > 1) {
323  if (reader->mVerbosity > 2) {
324  HBFUtils::dumpRDH(rdh);
325  } else {
326  HBFUtils::printRDH(rdh);
327  }
328  }
329  return true;
330 }

344 int RawFileReader::getLinkLocalID(const RDH& rdh, const o2::header::DataOrigin orig)
345 {
346  // get id of the link subspec. in the parser (create entry if new)
347  LinkSubSpec_t subspec = HBFUtils::getSubSpec(rdh);
348  LinkSpec_t spec = createSpec(orig, subspec);
349  auto entryMap = mLinkEntries.find(spec);
350  if (entryMap == mLinkEntries.end()) { // need to register a new link
351  int n = mLinkEntries.size();
352  mLinkEntries[spec] = n;
353  auto& lnk = mLinksData.emplace_back(rdh, this);
354  lnk.subspec = subspec;
355  lnk.origin = mDataSpecs[mCurrentFileID].first;
356  lnk.description = mDataSpecs[mCurrentFileID].second;
357  lnk.spec = spec;
358  return n;
359  }
360  return entryMap->second;
361 }

363 //_____________________________________________________________________
364 bool RawFileReader::preprocessFile(int ifl)
365 {
366  // preprocess file, check RDH data, build statistics
367  FILE* fl = mFiles[ifl];
368  mCurrentFileID = ifl;
369  RDH rdh;
370 
371  LinkSpec_t specPrev = 0xffffffffffffffff;
372  int lIDPrev = -1;
373  mMultiLinkFile = false;
374  rewind(fl);
375  long int nr = 0;
376  mPosInFile = 0;
377  int nRDHread = 0;
378  bool ok = true;
379  int readBytes = sizeof(RDH);
380  while ((nr = fread(&rdh, 1, readBytes, fl))) {
381  if (nr < readBytes) {
382  LOG(ERROR) << "EOF was unexpected, only " << nr << " bytes were read for RDH";
383  ok = false;
384  break;
385  }
386  if (!(ok = HBFUtils::checkRDH(rdh))) {
387  break;
388  }
389  nRDHread++;
390  LinkSpec_t spec = createSpec(mDataSpecs[mCurrentFileID].first, HBFUtils::getSubSpec(rdh));
391  int lID = lIDPrev;
392  if (spec != specPrev) { // link has changed
393  specPrev = spec;
394  if (lIDPrev != -1) {
395  mMultiLinkFile = true;
396  }
397  lID = getLinkLocalID(rdh, mDataSpecs[mCurrentFileID].first);
398  }
399  bool newSPage = lID != lIDPrev;
400  mLinksData[lID].preprocessCRUPage(rdh, newSPage);
401  //
402  mPosInFile += rdh.offsetToNext;
403  if (fseek(fl, mPosInFile, SEEK_SET)) {
404  break;
405  }
406  lIDPrev = lID;
407  }
408  mPosInFile = ftell(fl);
409  if (lIDPrev != -1) { // close last block
410  auto& lastBlock = mLinksData[lIDPrev].blocks.back();
411  lastBlock.size = mPosInFile - lastBlock.offset;
412  }
413 
414  LOGF(INFO, "File %3d : %9li bytes scanned, %6d RDH read for %4d links from %s",
415  mCurrentFileID, mPosInFile, nRDHread, int(mLinkEntries.size()), mFileNames[mCurrentFileID]);
416  return ok;
417 }

449 bool RawFileReader::addFile(const std::string& sname, o2::header::DataOrigin origin, o2::header::DataDescription desc)
450 {
451  if (mInitDone) {
452  LOG(ERROR) << "Cannot add new files after initialization";
453  return false;
454  }
455  auto inFile = fopen(sname.c_str(), "rb");
456  bool ok = true;
457  if (!inFile) {
458  LOG(ERROR) << "Failed to open input file " << sname;
459  ok = false;
460  }
461  if (origin == o2h::gDataOriginInvalid) {
462  LOG(ERROR) << "Invalid data origin " << origin.as<std::string>() << " for file " << sname;
463  ok = false;
464  }
465  if (desc == o2h::gDataDescriptionInvalid) {
466  LOG(ERROR) << "Invalid data description " << desc.as<std::string>() << " for file " << sname;
467  ok = false;
468  }
469  if (!ok) {
470  fclose(inFile);
471  }
472 
473  mFileNames.push_back(sname);
474  mFiles.push_back(inFile);
475  mDataSpecs.emplace_back(origin, desc);
476  return true;
477 }

479 //_____________________________________________________________________
480 bool RawFileReader::init()
481 {
482  // make initialization, preprocess files and chack for errors if asked
483 
484  for (int i = 0; i < NErrorsDefined; i++) {
485  if (mCheckErrors & (0x1 << i))
486  LOGF(INFO, "perform check for /%s/", ErrNames[i].data());
487  }
488 
489  int nf = mFiles.size();
490  bool ok = true;
491  for (int i = 0; i < nf; i++) {
492  ok &= preprocessFile(i);
493  }
494  mOrderedIDs.resize(mLinksData.size());
495  for (int i = mLinksData.size(); i--;) {
496  mOrderedIDs[i] = i;
497  if (mNTimeFrames < mLinksData[i].nTimeFrames) {
498  mNTimeFrames = mLinksData[i].nTimeFrames;
499  }
500  }
501  std::sort(mOrderedIDs.begin(), mOrderedIDs.end(),
502  [& links = mLinksData](int a, int b) { return links[a].spec < links[b].spec; });
503 
504  size_t maxSP = 0, maxTF = 0;
505 
506  LOGF(INFO, "Summary of preprocessing:");
507  for (int i = 0; i < int(mLinksData.size()); i++) {
508  const auto& link = getLink(i);
509  auto msp = link.getLargestSuperPage();
510  auto mtf = link.getLargestTF();
511  if (maxSP < msp) {
512  maxSP = msp;
513  }
514  if (maxTF < mtf) {
515  maxTF = mtf;
516  }
517  std::stringstream counts;
518  counts << "Lnk" << std::setw(4) << std::left << i << "| ";
519  link.print(mVerbosity, counts.str());
520  if (msp > mNominalSPageSize) {
521  LOGF(WARNING, " Attention: largest superpage %zu B exceeds expected %d B",
522  msp, mNominalSPageSize);
523  }
524  // min max orbits
525  if (link.blocks.front().orbit < mOrbitMin) {
526  mOrbitMin = link.blocks.front().orbit;
527  }
528  if (link.blocks.back().orbit > mOrbitMax) {
529  mOrbitMax = link.blocks.back().orbit;
530  }
531  if ((mCheckErrors & (0x1 << ErrWrongNumberOfTF)) && (mNTimeFrames != link.nTimeFrames)) {
532  LOGF(ERROR, "%s for %s: %u TFs while %u were seen for other links", ErrNames[ErrWrongNumberOfTF],
533  link.describe(), link.nTimeFrames, mNTimeFrames);
534  }
535  }
536  LOGF(INFO, "First orbit: %d, Last orbit: %d", mOrbitMin, mOrbitMax);
537  LOGF(INFO, "Largest super-page: %zu B, largest TF: %zu B", maxSP, maxTF);
538  if (!mCheckErrors) {
539  LOGF(INFO, "Detailed data format check was disabled");
540  }
541  mInitDone = true;
542 
543  return ok;
544 }

597 //_____________________________________________________________________
598 RawFileReader::InputsMap RawFileReader::parseInput(const std::string& confUri)
599 {
600  // read input files from configuration
601  std::map<std::pair<o2h::DataOrigin, o2h::DataDescription>, std::vector<std::string>> entries;
602 
603  if (confUri.empty()) {
604  throw std::runtime_error("Input configuration file is not provided");
605  }
606  std::string confFile = confUri;
607  ConfigFile cfg;
608  // ConfigFile file requires input as file:///absolute_path or file:local
609  if (confFile.rfind("file:", 0) != 0) {
