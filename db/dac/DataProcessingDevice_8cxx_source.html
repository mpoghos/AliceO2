102 void DataProcessingDevice::Init()
103 {
104  // For some reason passing rateLogging does not work anymore.
105  // This makes sure we never have more than one notification per minute.
106  for (auto& x : fChannels) {
107  for (auto& c : x.second) {
108  if (c.GetRateLogging() < MIN_RATE_LOGGING) {
109  c.UpdateRateLogging(MIN_RATE_LOGGING);
110  }
111  }
112  }
113  auto optionsRetriever(std::make_unique<FairOptionsRetriever>(mSpec.options, GetConfig()));
114  mConfigRegistry = std::move(std::make_unique<ConfigParamRegistry>(std::move(optionsRetriever)));
115 
116  mExpirationHandlers.clear();
117 
118  auto distinct = DataRelayerHelpers::createDistinctRouteIndex(mSpec.inputs);
119  int i = 0;
120  for (auto& di : distinct) {
121  auto& route = mSpec.inputs[di];
122  if (route.configurator.has_value() == false) {
123  i++;
124  continue;
125  }
126  ExpirationHandler handler{
127  RouteIndex{i++},
128  route.matcher.lifetime,
129  route.configurator->creatorConfigurator(*mConfigRegistry),
130  route.configurator->danglingConfigurator(*mConfigRegistry),
131  route.configurator->expirationConfigurator(*mConfigRegistry)};
132  mExpirationHandlers.emplace_back(std::move(handler));
133  }

135  auto& monitoring = mServiceRegistry.get<Monitoring>();
136  monitoring.enableBuffering(MONITORING_QUEUE_SIZE);
137  static const std::string dataProcessorIdMetric = "dataprocessor_id";
138  static const std::string dataProcessorIdValue = mSpec.name;
139  monitoring.addGlobalTag("dataprocessor_id", dataProcessorIdValue);
140 
141  if (mInit) {
142  InitContext initContext{*mConfigRegistry, mServiceRegistry};
143  mStatefulProcess = mInit(initContext);
144  }
145  mState.inputChannelInfos.resize(mSpec.inputChannels.size());
149  for (size_t ci = 0; ci < mSpec.inputChannels.size(); ++ci) {
150  auto& name = mSpec.inputChannels[ci].name;
151  if (name.find("from_internal-dpl-clock") == 0) {
152  mState.inputChannelInfos[ci].state = InputChannelState::Pull;
153  } else if (name.find("from_internal-dpl-ccdb-backend") == 0) {
154  mState.inputChannelInfos[ci].state = InputChannelState::Pull;
155  }
156  }

191  monitoring.send(Metric{stats.lastElapsedTimeMs, "elapsed_time_ms"}.addTag(Key::Subsystem, Value::DPL));

200  .addTag(Key::Subsystem, Value::DPL));
201  monitoring.send(Metric{(stats.lastTotalProcessedSize / (stats.lastLatency.maxLatency ? stats.lastLatency.maxLatency : 1) / 1000), "input_rate_mb_s"}
202  .addTag(Key::Subsystem, Value::DPL));
203 
204  lastSent = currentTime;
205  O2_SIGNPOST_END(MonitoringStatus::ID, MonitoringStatus::SEND, 0, 0, O2_SIGNPOST_BLUE);
206  };

209  auto flushMetrics = [& stats = mStats,
210  &relayer = mRelayer,
211  &lastFlushed = mLastMetricFlushedTimestamp,
212  &currentTime = mBeginIterationTimestamp,
213  &monitoring = mServiceRegistry.get<Monitoring>()]()
214  -> void {
215  if (currentTime - lastFlushed < 1000) {
216  return;
217  }
218 
219  O2_SIGNPOST_START(MonitoringStatus::ID, MonitoringStatus::FLUSH, 0, 0, O2_SIGNPOST_RED);
220  // Send all the relevant metrics for the relayer to update the GUI
221  // FIXME: do a delta with the previous version if too many metrics are still
222  // sent...
223  for (size_t si = 0; si < stats.relayerState.size(); ++si) {
224  auto state = stats.relayerState[si];
225  monitoring.send({state, "data_relayer/" + std::to_string(si)});
226  }
227  relayer.sendContextState();
228  monitoring.flushBuffer();
229  lastFlushed = currentTime;
230  O2_SIGNPOST_END(MonitoringStatus::ID, MonitoringStatus::FLUSH, 0, 0, O2_SIGNPOST_RED);

233  auto switchState = [& control = mServiceRegistry.get<ControlService>(),
234  &state = mState.streaming](StreamingState newState) {
235  state = newState;
236  control.notifyStreamingState(state);
237  };
238 
239  auto now = std::chrono::high_resolution_clock::now();
240  mBeginIterationTimestamp = (uint64_t)std::chrono::duration<double, std::milli>(now.time_since_epoch()).count();
241 
242  mServiceRegistry.get<CallbackService>()(CallbackService::Id::ClockTick);
243  // Wether or not we had something to do.
244  bool active = false;
245  // Notice that in case there are no input channels we should the allDone
246  // state will have to be set explicitly in the code, like we do, for example
247  // for implicit data sources or timers. This also implies that whatever is
248  // time driven will have to signal EndOfStream itself.
249  bool allDone = mSpec.inputChannels.empty() == false;
250  // Whether or not all the channels are completed
251  for (size_t ci = 0; ci < mSpec.inputChannels.size(); ++ci) {
252  auto& channel = mSpec.inputChannels[ci];
253  auto& info = mState.inputChannelInfos[ci];
254  if (info.state != InputChannelState::Completed) {
255  allDone = false;
256  }
257  if (info.state != InputChannelState::Running) {
258  continue;
259  }
260  FairMQParts parts;
261  auto result = this->Receive(parts, channel.name, 0, 0);
262  if (result > 0) {
263  this->handleData(parts, info);
264  active |= this->tryDispatchComputation();
265  }
266  }
267  if (active == false) {
268  mServiceRegistry.get<CallbackService>()(CallbackService::Id::Idle);
269  }
270  mRelayer.processDanglingInputs(mExpirationHandlers, mServiceRegistry);
271  this->tryDispatchComputation();
272 
273  sendRelayerMetrics();
274  flushMetrics();
275 
276  // If we got notified that all the sources are done, we call the EndOfStream
277  // callback and return false. Notice that what happens next is actually
278  // dependent on the callback, not something which is controlled by the
279  // framework itself.
280  if (allDone == true && mState.streaming == StreamingState::Streaming) {
281  switchState(StreamingState::EndOfStreaming);
282  }
283 
284  if (mState.streaming == StreamingState::EndOfStreaming) {
285  // We keep processing data until we are Idle.
286  // FIXME: not sure this is the correct way to drain the queues, but
287  // I guess we will see.
288  while (this->tryDispatchComputation()) {
289  mRelayer.processDanglingInputs(mExpirationHandlers, mServiceRegistry);
290  }
291  sendRelayerMetrics();
292  flushMetrics();
293  mContextRegistry.get<MessageContext>()->clear();
294  mContextRegistry.get<StringContext>()->clear();
295  mContextRegistry.get<ArrowContext>()->clear();
296  mContextRegistry.get<RawBufferContext>()->clear();
297  EndOfStreamContext eosContext{mServiceRegistry, mAllocator};
298  mServiceRegistry.get<CallbackService>()(CallbackService::Id::EndOfStream, eosContext);
299  DataProcessor::doSend(*this, *mContextRegistry.get<MessageContext>());
300  DataProcessor::doSend(*this, *mContextRegistry.get<StringContext>());
301  DataProcessor::doSend(*this, *mContextRegistry.get<ArrowContext>());
302  DataProcessor::doSend(*this, *mContextRegistry.get<RawBufferContext>());
303  for (auto& channel : mSpec.outputChannels) {
304  DataProcessingHelpers::sendEndOfStream(*this, channel);
305  }
306  // This is needed because the transport is deleted before the device.
307  mRelayer.clear();
308  switchState(StreamingState::Idle);
309  return true;
310  }
311  return true;

323 bool DataProcessingDevice::handleData(FairMQParts& parts, InputChannelInfo& info)
324 {
325  assert(mSpec.inputChannels.empty() == false);
326  assert(parts.Size() > 0);
327 
328  // Initial part. Let's hide all the unnecessary and have
329  // simple lambdas for each of the steps I am planning to have.
330  assert(!mSpec.inputs.empty());
331 
332  // These duplicate references are created so that each function
333  // does not need to know about the whole class state, but I can
334  // fine grain control what is exposed at each state.
335  auto& monitoringService = mServiceRegistry.get<Monitoring>();
336  StateMonitoring<DataProcessingStatus>::moveTo(DataProcessingStatus::IN_DPL_OVERHEAD);
337  ScopedExit metricFlusher([&monitoringService] {
338  StateMonitoring<DataProcessingStatus>::moveTo(DataProcessingStatus::IN_DPL_OVERHEAD);
339  });
340 
341  enum struct InputType {
342  Invalid,
343  Data,
344  SourceInfo
345  };
346 
347  // This is how we validate inputs. I.e. we try to enforce the O2 Data model
348  // and we do a few stats. We bind parts as a lambda captured variable, rather
349  // than an input, because we do not want the outer loop actually be exposed
350  // to the implementation details of the messaging layer.
351  auto getInputTypes = [& stats = mStats, &parts, &info]() -> std::optional<std::vector<InputType>> {
352  stats.inputParts = parts.Size();
353 
354  if (parts.Size() % 2) {
355  return std::nullopt;
356  }
357  std::vector<InputType> results(parts.Size() / 2, InputType::Invalid);
358 
359  for (size_t hi = 0; hi < parts.Size() / 2; ++hi) {
360  auto pi = hi * 2;
361  auto sih = o2::header::get<SourceInfoHeader*>(parts.At(pi)->GetData());
362  if (sih) {
363  info.state = sih->state;
364  results[hi] = InputType::SourceInfo;
365  continue;
366  }
367  auto dh = o2::header::get<DataHeader*>(parts.At(pi)->GetData());
368  if (!dh) {
369  results[hi] = InputType::Invalid;
370  LOGP(error, "Header is not a DataHeader?");
371  continue;
372  }
373  if (dh->payloadSize != parts.At(pi + 1)->GetSize()) {
374  results[hi] = InputType::Invalid;
375  LOGP(error, "DataHeader payloadSize mismatch");
376  continue;
377  }
378  auto dph = o2::header::get<DataProcessingHeader*>(parts.At(pi)->GetData());
379  if (!dph) {
380  results[hi] = InputType::Invalid;
381  LOGP(error, "Header stack does not contain DataProcessingHeader");
382  continue;
383  }
384  results[hi] = InputType::Data;
385  }
386  return results;
387  };

389  auto reportError = [& device = *this](const char* message) {
390  device.error(message);
391  };
392 
393  auto handleValidMessages = [&parts, &relayer = mRelayer, &reportError](std::vector<InputType> const& types) {
394  // We relay execution to make sure we have a complete set of parts
395  // available.
396  for (size_t pi = 0; pi < (parts.Size() / 2); ++pi) {
397  switch (types[pi]) {
398  case InputType::Data: {
399  auto headerIndex = 2 * pi;
400  auto payloadIndex = 2 * pi + 1;
401  assert(payloadIndex < parts.Size());
402  auto relayed = relayer.relay(std::move(parts.At(headerIndex)),
403  std::move(parts.At(payloadIndex)));
404  if (relayed == DataRelayer::WillNotRelay) {
405  reportError("Unable to relay part.");
406  }
407  } break;
408  case InputType::SourceInfo: {
409 
410  } break;
411  case InputType::Invalid: {
412  reportError("Invalid part found.");
413  } break;
414  }
415  }
416  };
417 
418  // Second part. This is the actual outer loop we want to obtain, with
419  // implementation details which can be read. Notice how most of the state
420  // is actually hidden. For example we do not expose what "input" is. This
421  // will allow us to keep the same toplevel logic even if the actual meaning
422  // of input is changed (for example we might move away from multipart
423  // messages). Notice also that we need to act diffently depending on the
424  // actual CompletionOp we want to perform. In particular forwarding inputs
425  // also gets rid of them from the cache.
426  auto inputTypes = getInputTypes();
427  if (bool(inputTypes) == false) {
428  reportError("Parts should come in couples. Dropping it.");
429  return true;
430  }
431  handleValidMessages(*inputTypes);
432  return true;
433 }

435 bool DataProcessingDevice::tryDispatchComputation()
436 {
437  // This is the actual hidden state for the outer loop. In case we decide we
438  // want to support multithreaded dispatching of operations, I can simply
439  // move these to some thread local store and the rest of the lambdas
440  // should work just fine.
441  std::vector<DataRelayer::RecordAction> completed;
442  std::vector<std::unique_ptr<FairMQMessage>> currentSetOfInputs;
443 
444  auto& allocator = mAllocator;
445  auto& context = *mContextRegistry.get<MessageContext>();
446  auto& device = *this;
447  auto& errorCallback = mError;
448  auto& errorCount = mErrorCount;
449  auto& forwards = mSpec.forwards;
450  auto& inputsSchema = mSpec.inputs;
451  auto& processingCount = mProcessingCount;
452  auto& rdfContext = *mContextRegistry.get<ArrowContext>();
453  auto& relayer = mRelayer;
454  auto& serviceRegistry = mServiceRegistry;
455  auto& statefulProcess = mStatefulProcess;
456  auto& statelessProcess = mStatelessProcess;
457  auto& stringContext = *mContextRegistry.get<StringContext>();
458  auto& timingInfo = mTimingInfo;
459  auto& timesliceIndex = mServiceRegistry.get<TimesliceIndex>();
460  auto& rawContext = *mContextRegistry.get<RawBufferContext>();
461 
462  // These duplicate references are created so that each function
463  // does not need to know about the whole class state, but I can
464  // fine grain control what is exposed at each state.
465  // FIXME: I should use a different id for this state.
466  auto& monitoringService = mServiceRegistry.get<Monitoring>();
467  StateMonitoring<DataProcessingStatus>::moveTo(DataProcessingStatus::IN_DPL_OVERHEAD);
468  ScopedExit metricFlusher([&monitoringService] {
469  StateMonitoring<DataProcessingStatus>::moveTo(DataProcessingStatus::IN_DPL_OVERHEAD);
470  });
471 
472  auto reportError = [&device](const char* message) {
473  device.error(message);
474  };
475 
476  // For the moment we have a simple "immediately dispatch" policy for stuff
477  // in the cache. This could be controlled from the outside e.g. by waiting
478  // for a few sets of inputs to arrive before we actually dispatch the
479  // computation, however this can be defined at a later stage.
480  auto canDispatchSomeComputation = [&completed, &relayer]() -> bool {
481  completed = relayer.getReadyToProcess();
482  return completed.empty() == false;
483  };
484 
485  // We use this to get a list with the actual indexes in the cache which
486  // indicate a complete set of inputs. Notice how I fill the completed
487  // vector and return it, so that I can have a nice for loop iteration later
488  // on.
489  auto getReadyActions = [&relayer, &completed, &stats = mStats]() -> std::vector<DataRelayer::RecordAction> {
490  stats.pendingInputs = (int)relayer.getParallelTimeslices() - completed.size();
491  stats.incomplete = completed.empty() ? 1 : 0;
492  return completed;
493  };
494 
495  // This is needed to convert from a pair of pointers to an actual DataRef
496  // and to make sure the ownership is moved from the cache in the relayer to
497  // the execution.
498  auto fillInputs = [&relayer, &inputsSchema, &currentSetOfInputs](TimesliceSlot slot) -> InputRecord {
499  currentSetOfInputs = std::move(relayer.getInputsForTimeslice(slot));
500  InputSpan span{[&currentSetOfInputs](size_t i) -> char const* {
501  return currentSetOfInputs.at(i) ? static_cast<char const*>(currentSetOfInputs.at(i)->GetData()) : nullptr;
502  },
503  currentSetOfInputs.size()};
504  return InputRecord{inputsSchema, std::move(span)};
505  };
506 
507  // This is the thing which does the actual computation. No particular reason
508  // why we do the stateful processing before the stateless one.
509  // PROCESSING:{START,END} is done so that we can trigger on begin / end of processing
510  // in the GUI.
511  auto dispatchProcessing = [&processingCount, &allocator, &statefulProcess, &statelessProcess, &monitoringService,
512  &context, &stringContext, &rdfContext, &rawContext, &serviceRegistry, &device](TimesliceSlot slot, InputRecord& record) {
513  if (statefulProcess) {
514  ProcessingContext processContext{record, serviceRegistry, allocator};
515  StateMonitoring<DataProcessingStatus>::moveTo(DataProcessingStatus::IN_DPL_USER_CALLBACK);
516  statefulProcess(processContext);
517  StateMonitoring<DataProcessingStatus>::moveTo(DataProcessingStatus::IN_DPL_OVERHEAD);
518  processingCount++;
519  }
520  if (statelessProcess) {
521  ProcessingContext processContext{record, serviceRegistry, allocator};
522  StateMonitoring<DataProcessingStatus>::moveTo(DataProcessingStatus::IN_DPL_USER_CALLBACK);
523  statelessProcess(processContext);
524  StateMonitoring<DataProcessingStatus>::moveTo(DataProcessingStatus::IN_DPL_OVERHEAD);
525  processingCount++;
526  }
527 
528  DataProcessor::doSend(device, context);
529  DataProcessor::doSend(device, stringContext);
530  DataProcessor::doSend(device, rdfContext);
531  DataProcessor::doSend(device, rawContext);
532  };
533 
534  // Error handling means printing the error and updating the metric
535  auto errorHandling = [&errorCallback, &monitoringService, &serviceRegistry](std::exception& e, InputRecord& record) {
536  StateMonitoring<DataProcessingStatus>::moveTo(DataProcessingStatus::IN_DPL_ERROR_CALLBACK);
537  LOG(ERROR) << "Exception caught: " << e.what() << std::endl;
538  if (errorCallback) {
539  monitoringService.send({1, "error"});
540  ErrorContext errorContext{record, serviceRegistry, e};
541  errorCallback(errorContext);
542  }
543  StateMonitoring<DataProcessingStatus>::moveTo(DataProcessingStatus::IN_DPL_OVERHEAD);
544  };
545 
546  // I need a preparation step which gets the current timeslice id and
547  // propagates it to the various contextes (i.e. the actual entities which
548  // create messages) because the messages need to have the timeslice id into
549  // it.
550  auto prepareAllocatorForCurrentTimeSlice = [&timingInfo, &stringContext, &rdfContext, &rawContext, &context, &relayer, &timesliceIndex](TimesliceSlot i) {
551  auto timeslice = timesliceIndex.getTimesliceForSlot(i);
552  timingInfo.timeslice = timeslice.value;
553  context.clear();
554  stringContext.clear();
555  rdfContext.clear();
556  rawContext.clear();
557  };
558 
559  // When processing them, timers will have to be cleaned up
560  // to avoid double counting them.
561  // This was actually the easiest solution we could find for
562  // O2-646.
563  auto cleanTimers = [&currentSetOfInputs](TimesliceSlot slot, InputRecord& record) {
564  assert(record.size() * 2 == currentSetOfInputs.size());
565  for (size_t ii = 0, ie = record.size(); ii < ie; ++ii) {
566  DataRef input = record.getByPos(ii);
567  if (input.spec->lifetime != Lifetime::Timer) {
568  continue;
569  }
570  if (input.header == nullptr || input.payload == nullptr) {
571  continue;
572  }
573  // This will hopefully delete the message.
574  std::unique_ptr<FairMQMessage> header = std::move(currentSetOfInputs[ii * 2]);
575  std::unique_ptr<FairMQMessage> payload = std::move(currentSetOfInputs[ii * 2 + 1]);
576  }
577  };

579  // This is how we do the forwarding, i.e. we push
580  // the inputs which are shared between this device and others
581  // to the next one in the daisy chain.
582  // FIXME: do it in a smarter way than O(N^2)
583  auto forwardInputs = [&reportError, &forwards, &device, &currentSetOfInputs](TimesliceSlot slot, InputRecord& record) {
584  assert(record.size() * 2 == currentSetOfInputs.size());
585  for (size_t ii = 0, ie = record.size(); ii < ie; ++ii) {
586  DataRef input = record.getByPos(ii);
587 
588  // If is now possible that the record is not complete when
589  // we forward it, because of a custom completion policy.
590  // this means that we need to skip the empty entries in the
591  // record for being forwarded.
592  if (input.header == nullptr || input.payload == nullptr) {
593  continue;
594  }
595  auto sih = o2::header::get<SourceInfoHeader*>(input.header);
596  if (sih) {
597  continue;
598  }
599 
600  auto dh = o2::header::get<DataHeader*>(input.header);
601  if (!dh) {
602  reportError("Header is not a DataHeader?");
603  continue;
604  }
605  auto dph = o2::header::get<DataProcessingHeader*>(input.header);
606  if (!dph) {
607  reportError("Header stack does not contain DataProcessingHeader");
608  continue;
609  }
610 
611  auto& header = currentSetOfInputs[ii * 2];
612  auto& payload = currentSetOfInputs[ii * 2 + 1];
613 
614  for (auto forward : forwards) {
615  if (DataSpecUtils::match(forward.matcher, dh->dataOrigin, dh->dataDescription, dh->subSpecification) && (dph->startTime % forward.maxTimeslices) == forward.timeslice) {
616 
617  if (header.get() == nullptr) {
618  // FIXME: this should not happen, however it's actually harmless and
619  // we can simply discard it for the moment.
620  // LOG(ERROR) << "Missing header! " << dh->dataDescription.as<std::string>();
621  continue;
622  }
623  auto fdph = o2::header::get<DataProcessingHeader*>(header.get()->GetData());
624  if (fdph == nullptr) {
625  LOG(ERROR) << "Forwarded data does not have a DataProcessingHeader";
626  continue;
627  }
628  auto fdh = o2::header::get<DataHeader*>(header.get()->GetData());
629  if (fdh == nullptr) {
630  LOG(ERROR) << "Forwarded data does not have a DataHeader";
631  continue;
632  }
633  FairMQParts forwardedParts;
634  forwardedParts.AddPart(std::move(header));
635  forwardedParts.AddPart(std::move(payload));
636  assert(forwardedParts.Size() == 2);
637  assert(o2::header::get<DataProcessingHeader*>(forwardedParts.At(0)->GetData()));
638  // FIXME: this should use a correct subchannel
639  device.Send(forwardedParts, forward.channel, 0);
640  }
641  }
642  }
643  };
644 
645  auto calculateInputRecordLatency = [](InputRecord const& record, auto now) -> DataProcessingStats::InputLatency {
646  DataProcessingStats::InputLatency result{static_cast<int>(-1), 0};
647 
648  auto currentTime = (uint64_t)std::chrono::duration<double, std::milli>(now.time_since_epoch()).count();
649  for (auto& item : record) {
650  auto* header = o2::header::get<DataProcessingHeader*>(item.header);
651  if (header == nullptr) {
652  continue;
653  }
654  int partLatency = currentTime - header->creation;
655  result.minLatency = std::min(result.minLatency, partLatency);
656  result.maxLatency = std::max(result.maxLatency, partLatency);
657  }
658  return result;
659  };
660 
661  auto calculateTotalInputRecordSize = [](InputRecord const& record) -> int {
662  size_t totalInputSize = 0;
663  for (auto& item : record) {
664  auto* header = o2::header::get<DataHeader*>(item.header);
665  if (header == nullptr) {
666  continue;
667  }
668  totalInputSize += header->payloadSize;
669  }
670  return totalInputSize;
671  };
672 
673  auto switchState = [& control = mServiceRegistry.get<ControlService>(),
674  &state = mState.streaming](StreamingState newState) {
675  state = newState;
676  control.notifyStreamingState(state);
677  };
678 
679  if (canDispatchSomeComputation() == false) {
680  return false;
681  }
682 
683  for (auto action : getReadyActions()) {
684  if (action.op == CompletionPolicy::CompletionOp::Wait) {
685  continue;
686  }
687 
688  prepareAllocatorForCurrentTimeSlice(TimesliceSlot{action.slot});
689  InputRecord record = fillInputs(action.slot);
690  if (action.op == CompletionPolicy::CompletionOp::Discard) {
691  if (forwards.empty() == false) {
692  forwardInputs(action.slot, record);
693  continue;
694  }
695  }
696  auto tStart = std::chrono::high_resolution_clock::now();
697  for (size_t ai = 0; ai != record.size(); ai++) {
698  auto cacheId = action.slot.index * record.size() + ai;
699  auto state = record.isValid(ai) ? 2 : 0;
700  mStats.relayerState.resize(std::max(cacheId + 1, mStats.relayerState.size()), 0);
701  mStats.relayerState[cacheId] = state;
702  }
703  try {
704  if (mState.quitRequested == false) {
705  dispatchProcessing(action.slot, record);
706  }
707  } catch (std::exception& e) {
708  errorHandling(e, record);
709  }
710  for (size_t ai = 0; ai != record.size(); ai++) {
711  auto cacheId = action.slot.index * record.size() + ai;
712  auto state = record.isValid(ai) ? 3 : 0;
713  mStats.relayerState.resize(std::max(cacheId + 1, mStats.relayerState.size()), 0);
714  mStats.relayerState[cacheId] = state;
715  }
716  auto tEnd = std::chrono::high_resolution_clock::now();
717  mStats.lastElapsedTimeMs = std::chrono::duration<double, std::milli>(tEnd - tStart).count();
718  mStats.lastTotalProcessedSize = calculateTotalInputRecordSize(record);
719  mStats.lastLatency = calculateInputRecordLatency(record, tStart);
720  // We forward inputs only when we consume them. If we simply Process them,
721  // we keep them for next message arriving.
722  if (action.op == CompletionPolicy::CompletionOp::Consume) {
723  if (forwards.empty() == false) {
724  forwardInputs(action.slot, record);
725  }
726  } else if (action.op == CompletionPolicy::CompletionOp::Process) {
727  cleanTimers(action.slot, record);
728  }
729  }
730  // We now broadcast the end of stream if it was requested
731  if (mState.streaming == StreamingState::EndOfStreaming) {
732  for (auto& channel : mSpec.outputChannels) {
733  DataProcessingHelpers::sendEndOfStream(*this, channel);
734  }
735  switchState(StreamingState::Idle);
736  }
737 
738  return true;
739 }
