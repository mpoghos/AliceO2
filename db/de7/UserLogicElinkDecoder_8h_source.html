235 template <typename CHARGESUM>
236 struct StateMachine_ : public msm::front::state_machine_def<StateMachine_<CHARGESUM>> {
237 
238  typedef WaitingSync initial_state;
239 
240  struct transition_table : mpl::vector<
241  // clang-format off
242  // Start Event Next Action Guard
243  Row< WaitingSync , NewData , WaitingHeader , foundSync , isSync >,
244 
245  Row< WaitingHeader , NewData , WaitingSize , ActionSequence_<
246  mpl::vector<setData,
247  readHeader>> , Not_<isSync> >,
248  Row< WaitingSize , NewData , WaitingSize , setData , And_<moreWordsToRead,
249  Not_<moreDataAvailable>> >,
250  Row< WaitingSize , none , WaitingTime , readSize<CHARGESUM> , And_<moreDataAvailable,
251  moreWordsToRead> >,
252  Row< WaitingTime , NewData , WaitingTime , setData , And_<moreWordsToRead,
253  Not_<moreDataAvailable>> >,
254  Row< WaitingTime , none , WaitingSample , readTime , And_<moreSampleToRead,
255  moreDataAvailable> >,
256 
257  Row< WaitingSample , NewData , WaitingSample , setData , moreSampleToRead >,
258  Row< WaitingSample , none , WaitingHeader , none , Not_<moreSampleToRead> >,
259 
260  Row< WaitingSample , none , WaitingSample , readSample<CHARGESUM> , And_<moreDataAvailable,
261  moreSampleToRead> >,
262  Row< WaitingSample , none , WaitingSize , none , And_<moreWordsToRead,
263  Not_<moreSampleToRead>>>
264  // clang-format on
265  > {
266  };
267  uint16_t data10(uint64_t value, size_t index) const
268  {
269  if (index < 0 || index >= masks.size()) {
270  std::cout << fmt::format("index {} is out of range\n", index);
271  return 0;
272  }
273  uint64_t m = masks[index];
274  return static_cast<uint16_t>(((value & m) >> (masks.size() - 1 - index) * 10) & 0x3FF);
275  }
276  uint8_t channelNumber(const SampaHeader& sh)
277  {
278  return sh.channelAddress() + (sh.chipAddress() % 2) * 32;
279  }
280  uint16_t pop10()
281  {
282  auto rv = data10(data, maskIndex);
283  nof10BitWords = std::max(0, nof10BitWords - 1);
284  maskIndex = std::min(masks.size(), maskIndex + 1);
285  return rv;
286  }
287  void addSample(uint16_t sample)
288  {
289 #ifdef ULDEBUG
290  std::cout << "sample = " << sample << "\n";
291 #endif
292  samples.emplace_back(sample);
293  if (clusterSize == 0) {
294  // a cluster is ready, send it
295  channelHandler(dsId,
296  channelNumber(sampaHeader),
297  SampaCluster(clusterTime, samples));
298  samples.clear();
299  }
300  }
301  void addChargeSum(uint16_t b, uint16_t a)
302  {
303  // a cluster is ready, send it
304  uint32_t q = (((static_cast<uint32_t>(a) & 0x3FF) << 10) | (static_cast<uint32_t>(b) & 0x3FF));
305 #ifdef ULDEBUG
306  std::cout << "chargeSum = " << q << "\n";
307 #endif
308  channelHandler(dsId,
309  channelNumber(sampaHeader),
310  SampaCluster(clusterTime, q));
311  }
312 
313  // Replaces the default no-transition response.
314  template <class FSM, class Event>
315  void no_transition(Event const& e, FSM&, int state)
316  {
317  std::cout << "no transition from state " << state
318  << " on event " << typeid(e).name() << std::endl;
319  }
320 
321  // masks used to access groups of 10 bits in a 50 bits range
322  std::array<uint64_t, 5>
323  masks = {0x3FF0000000000, 0xFFC0000000, 0x3FF00000, 0xFFC00, 0x3FF};
324  DsElecId dsId{0, 0, 0};
325  uint16_t nof10BitWords{0};
326  uint16_t clusterSize{0};
327  uint16_t clusterTime{0};
328  uint64_t data{0};
329  size_t maskIndex{0};
330  size_t nofSync{0};
331  std::vector<uint16_t> samples;
332  SampaHeader sampaHeader;
333  SampaChannelHandler channelHandler;
334 };
