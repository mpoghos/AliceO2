43 GPUd() void GPUTPCCFDecodeZS::decode(GPUTPCClusterFinder& clusterer, GPUSharedMemory& s, int nBlocks, int nThreads, int iBlock, int iThread)
44 {
45  const unsigned int slice = clusterer.mISlice;
46  const unsigned int endpoint = iBlock;
47  const GPUTrackingInOutZS::GPUTrackingInOutZSSlice& zs = clusterer.GetConstantMem()->ioPtrs.tpcZS->slice[slice];
48  if (zs.count[endpoint] == 0) {
49  return;
50  }
51  deprecated::PackedDigit* digits = clusterer.mPdigits;
52  const size_t nDigits = clusterer.mPmemory->nDigitsOffset[endpoint];
53  unsigned int rowOffsetCounter = 0;
54  if (iThread == 0) {
55  const int region = endpoint / 2;
56  s.nRowsRegion = clusterer.Param().tpcGeometry.GetRegionRows(region);
57  s.regionStartRow = clusterer.Param().tpcGeometry.GetRegionStart(region);
58  s.nThreadsPerRow = CAMath::Max(1u, nThreads / ((s.nRowsRegion + (endpoint & 1)) / 2));
59  s.rowStride = nThreads / s.nThreadsPerRow;
60  const unsigned char* page = (const unsigned char*)zs.zsPtr[endpoint][0];
61  const TPCZSHDR* hdr = reinterpret_cast<const TPCZSHDR*>(page + sizeof(o2::header::RAWDataHeader));
62  const bool decode12bit = hdr->version == 2;
63  s.decodeBits = decode12bit ? TPCZSHDR::TPC_ZS_NBITS_V2 : TPCZSHDR::TPC_ZS_NBITS_V1;
64  s.decodeBitsFactor = 1.f / (1 << (s.decodeBits - 10));
65  }
66  GPUbarrier();
67  const unsigned int myRow = iThread / s.nThreadsPerRow;
68  const unsigned int mySequence = iThread % s.nThreadsPerRow;
69  for (unsigned int i = 0; i < zs.count[endpoint]; i++) {
70  for (unsigned int j = 0; j < zs.nZSPtr[endpoint][i]; j++) {
71  const unsigned int* pageSrc = (const unsigned int*)(((const unsigned char*)zs.zsPtr[endpoint][i]) + j * TPCZSHDR::TPC_ZS_PAGE_SIZE);
72  GPUbarrier();
73  CA_SHARED_CACHE_REF(&s.ZSPage[0], pageSrc, TPCZSHDR::TPC_ZS_PAGE_SIZE, unsigned int, pageCache);
74  GPUbarrier();
75  const unsigned char* page = (const unsigned char*)pageCache;
76  const unsigned char* pagePtr = page + sizeof(o2::header::RAWDataHeader);
77  const TPCZSHDR* hdr = reinterpret_cast<const TPCZSHDR*>(pagePtr);
78  pagePtr += sizeof(*hdr);
79  unsigned int mask = (1 << s.decodeBits) - 1;
80  int timeBin = hdr->timeOffset;
81  for (int l = 0; l < hdr->nTimeBins; l++) {
82  pagePtr += (pagePtr - page) & 1; //Ensure 16 bit alignment
83  const TPCZSTBHDR* tbHdr = reinterpret_cast<const TPCZSTBHDR*>(pagePtr);
84  if ((tbHdr->rowMask & 0x7FFF) == 0) {
85  pagePtr += 2;
86  continue;
87  }
88  const int rowOffset = s.regionStartRow + ((endpoint & 1) ? (s.nRowsRegion / 2) : 0);
89  const int nRows = (endpoint & 1) ? (s.nRowsRegion - s.nRowsRegion / 2) : (s.nRowsRegion / 2);
90  const int nRowsUsed = CAMath::Popcount((unsigned int)(tbHdr->rowMask & 0x7FFF));
91  pagePtr += 2 * nRowsUsed;
92  GPUbarrier();
93  if (iThread == 0) {
94  for (int n = 0; n < nRowsUsed; n++) {
95  s.RowClusterOffset[n] = rowOffsetCounter;
96  const unsigned char* rowData = n == 0 ? pagePtr : (page + tbHdr->rowAddr1()[n - 1]);
97  rowOffsetCounter += rowData[2 * *rowData]; // Sum up number of ADC samples per row to compute offset in target buffer
98  }
99  }
100  GPUbarrier();
101  if (myRow < s.rowStride) {
102  for (int m = myRow; m < nRows; m += s.rowStride) {
103  if ((tbHdr->rowMask & (1 << m)) == 0) {
104  continue;
105  }
106  const int rowPos = CAMath::Popcount((unsigned int)(tbHdr->rowMask & ((1 << m) - 1)));
107  size_t nDigitsTmp = nDigits + s.RowClusterOffset[rowPos];
108  const unsigned char* rowData = rowPos == 0 ? pagePtr : (page + tbHdr->rowAddr1()[rowPos - 1]);
109  const int nSeqRead = *rowData;
110  const int nSeqPerThread = (nSeqRead + s.nThreadsPerRow - 1) / s.nThreadsPerRow;
111  const int mySequenceStart = mySequence * nSeqPerThread;
112  const int mySequenceEnd = CAMath::Min(mySequenceStart + nSeqPerThread, nSeqRead);
113  if (mySequenceEnd > mySequenceStart) {
114  const unsigned char* adcData = rowData + 2 * nSeqRead + 1;
115  const unsigned int nSamplesStart = mySequenceStart ? rowData[2 * mySequenceStart] : 0;
116  nDigitsTmp += nSamplesStart;
117  unsigned int nADCStartBits = nSamplesStart * s.decodeBits;
118  const unsigned int nADCStart = (nADCStartBits + 7) / 8;
119  const int nADC = (rowData[2 * mySequenceEnd] * s.decodeBits + 7) / 8;
120  adcData += nADCStart;
121  nADCStartBits &= 0x7;
122  unsigned int byte = 0, bits = 0;
123  if (nADCStartBits) { // % 8 != 0
124  bits = 8 - nADCStartBits;
125  byte = ((*(adcData - 1) & (0xFF ^ ((1 << nADCStartBits) - 1)))) >> nADCStartBits;
126  }
127  int nSeq = mySequenceStart;
128  int seqLen = nSeq ? (rowData[(nSeq + 1) * 2] - rowData[nSeq * 2]) : rowData[2];
129  Pad pad = rowData[nSeq++ * 2 + 1];
130  for (int n = nADCStart; n < nADC; n++) {
131  byte |= *(adcData++) << bits;
132  bits += 8;
133  while (bits >= s.decodeBits) {
134  if (seqLen == 0) {
135  seqLen = rowData[(nSeq + 1) * 2] - rowData[nSeq * 2];
136  pad = rowData[nSeq++ * 2 + 1];
137  }
138  digits[nDigitsTmp++] = deprecated::PackedDigit{(float)(byte & mask) * s.decodeBitsFactor, (Timestamp)(timeBin + l), pad++, (Row)(rowOffset + m)};
139  byte = byte >> s.decodeBits;
140  bits -= s.decodeBits;
141  seqLen--;
142  }
143  }
144  }
145  }
146  }
147  if (nRowsUsed > 1) {
148  pagePtr = page + tbHdr->rowAddr1()[nRowsUsed - 2];
149  }
150  pagePtr += 2 * *pagePtr; // Go to entry for last sequence length
151  pagePtr += 1 + (*pagePtr * s.decodeBits + 7) / 8; // Go to beginning of next time bin
152  }
153  }
154  }
155 }
