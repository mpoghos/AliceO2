199 GPUh() void GPUTPCTracker::WriteOutput()
200 {
201  mOutput->SetNTracks(0);
202  mOutput->SetNLocalTracks(0);
203  mOutput->SetNTrackClusters(0);
204 
205  if (mCommonMem->nTracks == 0) {
206  return;
207  }
208  if (mCommonMem->nTracks > GPUCA_MAX_SLICE_NTRACK) {
209  GPUError("Maximum number of tracks exceeded, cannot store");
210  return;
211  }
212 
213  int nStoredHits = 0;
214  int nStoredTracks = 0;
215  int nStoredLocalTracks = 0;
216 
217  GPUTPCSliceOutTrack* out = mOutput->FirstTrack();
218 
219  trackSortData* trackOrder = new trackSortData[mCommonMem->nTracks];
220  for (unsigned int i = 0; i < mCommonMem->nTracks; i++) {
221  trackOrder[i].fTtrack = i;
222  trackOrder[i].fSortVal = mTracks[trackOrder[i].fTtrack].NHits() / 1000.f + mTracks[trackOrder[i].fTtrack].Param().GetZ() * 100.f + mTracks[trackOrder[i].fTtrack].Param().GetY();
223  }
224  std::sort(trackOrder, trackOrder + mCommonMem->nLocalTracks, SortComparison<trackSortData>);
225  std::sort(trackOrder + mCommonMem->nLocalTracks, trackOrder + mCommonMem->nTracks, SortComparison<trackSortData>);
226 
227  for (unsigned int iTrTmp = 0; iTrTmp < mCommonMem->nTracks; iTrTmp++) {
228  const int iTr = trackOrder[iTrTmp].fTtrack;
229  GPUTPCTrack& iTrack = mTracks[iTr];
230 
231  out->SetParam(iTrack.Param());
232  out->SetLocalTrackId(iTrack.LocalTrackId());
233  int nClu = 0;
234  int iID = iTrack.FirstHitID();
235 
236  for (int ith = 0; ith < iTrack.NHits(); ith++) {
237  const GPUTPCHitId& ic = mTrackHits[iID + ith];
238  int iRow = ic.RowIndex();
239  int ih = ic.HitIndex();
240 
241  const GPUTPCRow& row = mData.Row(iRow);
242 #ifdef GPUCA_ARRAY_BOUNDS_CHECKS
243  if (ih >= row.NHits() || ih < 0) {
244  GPUError("Array out of bounds access (Sector Row) (Hit %d / %d - NumC %d): Sector %d Row %d Index %d", ith, iTrack.NHits(), NHitsTotal(), mISlice, iRow, ih);
245  fflush(stdout);
246  continue;
247  }
248 #endif
249  int clusterIndex = mData.ClusterDataIndex(row, ih);
250 
251 #ifdef GPUCA_ARRAY_BOUNDS_CHECKS
252  if (clusterIndex >= NHitsTotal() || clusterIndex < 0) {
253  GPUError("Array out of bounds access (Cluster Data) (Hit %d / %d - NumC %d): Sector %d Row %d Hit %d, Clusterdata Index %d", ith, iTrack.NHits(), NHitsTotal(), mISlice, iRow, ih, clusterIndex);
254  fflush(stdout);
255  continue;
256  }
257 #endif
258 
259  float origX, origY, origZ;
260  unsigned char flags;
261  unsigned short amp;
262  int id;
263  if (Param().earlyTpcTransform) {
264  origX = mData.ClusterData()[clusterIndex].x;
265  origY = mData.ClusterData()[clusterIndex].y;
266  origZ = mData.ClusterData()[clusterIndex].z;
267  flags = mData.ClusterData()[clusterIndex].flags;
268  amp = mData.ClusterData()[clusterIndex].amp;
269  id = mData.ClusterData()[clusterIndex].id;
270  } else {
271  const ClusterNativeAccess& cls = *mConstantMem->ioPtrs.clustersNative;
272  id = clusterIndex + cls.clusterOffset[mISlice][0];
273  GPUTPCConvertImpl::convert(*mConstantMem, mISlice, iRow, cls.clustersLinear[id].getPad(), cls.clustersLinear[id].getTime(), origX, origY, origZ);
274  flags = cls.clustersLinear[id].getFlags();
275  amp = cls.clustersLinear[id].qTot;
276  }
277  GPUTPCSliceOutCluster c;
278  c.Set(id, iRow, flags, amp, origX, origY, origZ);
279 #ifdef GPUCA_TPC_RAW_PROPAGATE_PAD_ROW_TIME
280  c.mPad = mData.ClusterData()[clusterIndex].pad;
281  c.mTime = mData.ClusterData()[clusterIndex].time;
282 #endif
283  out->SetCluster(nClu, c);
284  nClu++;
285  }
286 
287  nStoredTracks++;
288  if (iTr < mCommonMem->nLocalTracks) {
289  nStoredLocalTracks++;
290  }
291  nStoredHits += nClu;
292  out->SetNClusters(nClu);
293  out = out->NextTrack();
294  }
295  delete[] trackOrder;
296 
297  mOutput->SetNTracks(nStoredTracks);
298  mOutput->SetNLocalTracks(nStoredLocalTracks);
299  mOutput->SetNTrackClusters(nStoredHits);
300  if (Param().debugLevel >= 3) {
301  GPUInfo("Slice %d, Output: Tracks %d, local tracks %d, hits %d", mISlice, nStoredTracks, nStoredLocalTracks, nStoredHits);
302  }
303 }

305 GPUh() int GPUTPCTracker::PerformGlobalTrackingRun(GPUTPCTracker& sliceSource, int iTrack, int rowIndex, float angle, int direction)
306 {
307  /*for (int j = 0;j < mTracks[j].NHits();j++)
308  {
309  GPUInfo("Hit %3d: Row %3d: X %3.7lf Y %3.7lf", j, mTrackHits[mTracks[iTrack].FirstHitID() + j].RowIndex(), Row(mTrackHits[mTracks[iTrack].FirstHitID() + j].RowIndex()).X(),
310  (float) Data().HitDataY(Row(mTrackHits[mTracks[iTrack].FirstHitID() + j].RowIndex()), mTrackHits[mTracks[iTrack].FirstHitID() + j].HitIndex()) * Row(mTrackHits[mTracks[iTrack].FirstHitID() + j].RowIndex()).HstepY() + Row(mTrackHits[mTracks[iTrack].FirstHitID() + j].RowIndex()).Grid().YMin());
311  }*/
312 
313  GPUTPCTrackParam tParam;
314  tParam.InitParam();
315  tParam.SetCov(0, 0.05);
316  tParam.SetCov(2, 0.05);
317  tParam.SetCov(5, 0.001);
318  tParam.SetCov(9, 0.001);
319  tParam.SetCov(14, 0.05);
320  tParam.SetParam(sliceSource.mTracks[iTrack].Param());
321 
322  // GPUInfo("Parameters X %f Y %f Z %f SinPhi %f DzDs %f QPt %f SignCosPhi %f", tParam.X(), tParam.Y(), tParam.Z(), tParam.SinPhi(), tParam.DzDs(), tParam.QPt(), tParam.SignCosPhi());
323  if (!tParam.Rotate(angle, GPUCA_MAX_SIN_PHI)) {
324  return (0);
325  }
326  // GPUInfo("Rotated X %f Y %f Z %f SinPhi %f DzDs %f QPt %f SignCosPhi %f", tParam.X(), tParam.Y(), tParam.Z(), tParam.SinPhi(), tParam.DzDs(), tParam.QPt(), tParam.SignCosPhi());
327 
328  int maxRowGap = 10;
329  GPUTPCTrackLinearisation t0(tParam);
330  do {
331  rowIndex += direction;
332  if (!tParam.TransportToX(Row(rowIndex).X(), t0, Param().ConstBz, GPUCA_MAX_SIN_PHI)) {
333  return (0); // Reuse t0 linearization until we are in the next sector
334  }
335  // GPUInfo("Transported X %f Y %f Z %f SinPhi %f DzDs %f QPt %f SignCosPhi %f (MaxY %f)", tParam.X(), tParam.Y(), tParam.Z(), tParam.SinPhi(), tParam.DzDs(), tParam.QPt(), tParam.SignCosPhi(), Row(rowIndex).MaxY());
336  if (--maxRowGap == 0) {
337  return (0);
338  }
339  } while (fabsf(tParam.Y()) > Row(rowIndex).MaxY());
340 
341  float err2Y, err2Z;
342  GetErrors2Seeding(rowIndex, tParam.Z(), tParam.SinPhi(), tParam.DzDs(), err2Y, err2Z);
343  if (tParam.GetCov(0) < err2Y) {
344  tParam.SetCov(0, err2Y);
345  }
346  if (tParam.GetCov(2) < err2Z) {
347  tParam.SetCov(2, err2Z);
348  }
349 
350  int nHits = GPUTPCTrackletConstructor::GPUTPCTrackletConstructorGlobalTracking(*this, tParam, rowIndex, direction, 0);
351  if (nHits >= GPUCA_GLOBAL_TRACKING_MIN_HITS) {
352  // GPUInfo("%d hits found", nHits);
353  unsigned int hitId = CAMath::AtomicAdd(&mCommonMem->nTrackHits, nHits);
354  if ((hitId + nHits) >= mNMaxTrackHits) {
355  mCommonMem->kernelError = GPUCA_ERROR_GLOBAL_TRACKING_TRACK_HIT_OVERFLOW;
356  CAMath::AtomicExch(&mCommonMem->nTrackHits, mNMaxTrackHits);
357  return (0);
358  }
359  unsigned int trackId = CAMath::AtomicAdd(&mCommonMem->nTracks, 1);
360  if (direction == 1) {
361  int i = 0;
362  while (i < nHits) {
363 #ifdef GPUCA_EXTERN_ROW_HITS
364  const calink rowHit = TrackletRowHits()[rowIndex * *NTracklets()];
365 #else
366  const calink rowHit = Tracklet(0).RowHit(rowIndex);
367 #endif
368  if (rowHit != CALINK_INVAL) {
369  // GPUInfo("New track: entry %d, row %d, hitindex %d", i, rowIndex, mTrackletRowHits[rowIndex * mCommonMem->nTracklets]);
370  mTrackHits[hitId + i].Set(rowIndex, rowHit);
371  // if (i == 0) tParam.TransportToX(Row(rowIndex).X(), Param().ConstBz(), GPUCA_MAX_SIN_PHI); //Use transport with new linearisation, we have changed the track in between - NOT needed, fitting will always start at outer end of global track!
372  i++;
373  }
374  rowIndex++;
375  }
376  } else {
377  int i = nHits - 1;
378  while (i >= 0) {
379 #ifdef GPUCA_EXTERN_ROW_HITS
380  const calink rowHit = TrackletRowHits()[rowIndex * *NTracklets()];
381 #else
382  const calink rowHit = Tracklet(0).RowHit(rowIndex);
383 #endif
384  if (rowHit != CALINK_INVAL) {
385  // GPUInfo("New track: entry %d, row %d, hitindex %d", i, rowIndex, mTrackletRowHits[rowIndex * mCommonMem->nTracklets]);
386  mTrackHits[hitId + i].Set(rowIndex, rowHit);
387  i--;
388  }
389  rowIndex--;
390  }
391  }
392  GPUTPCTrack& track = mTracks[trackId];
393  track.SetAlive(1);
394  track.SetParam(tParam.GetParam());
395  track.SetNHits(nHits);
396  track.SetFirstHitID(hitId);
397  track.SetLocalTrackId((sliceSource.mISlice << 24) | sliceSource.mTracks[iTrack].LocalTrackId());
398  }
399 
400  return (nHits >= GPUCA_GLOBAL_TRACKING_MIN_HITS);
401 }

403 GPUh() void GPUTPCTracker::PerformGlobalTracking(GPUTPCTracker& sliceTarget, bool right)
404 {
405  for (int i = 0; i < mCommonMem->nLocalTracks; i++) {
406  {
407  const int tmpHit = mTracks[i].FirstHitID();
408  if (mTrackHits[tmpHit].RowIndex() >= GPUCA_GLOBAL_TRACKING_MIN_ROWS && mTrackHits[tmpHit].RowIndex() < GPUCA_GLOBAL_TRACKING_RANGE) {
409  int rowIndex = mTrackHits[tmpHit].RowIndex();
410  const GPUTPCRow& row = Row(rowIndex);
411  float Y = (float)Data().HitDataY(row, mTrackHits[tmpHit].HitIndex()) * row.HstepY() + row.Grid().YMin();
412  if (sliceTarget.mCommonMem->nTracks >= sliceTarget.mNMaxTracks) {
413  sliceTarget.mCommonMem->kernelError = GPUCA_ERROR_GLOBAL_TRACKING_TRACK_OVERFLOW;
414  return;
415  }
416  if (!right && Y < -row.MaxY() * GPUCA_GLOBAL_TRACKING_Y_RANGE_LOWER) {
417  // GPUInfo("Track %d, lower row %d, left border (%f of %f)", i, mTrackHits[tmpHit].RowIndex(), Y, -row.MaxY());
418  sliceTarget.PerformGlobalTrackingRun(*this, i, rowIndex, -Param().DAlpha, -1);
419  }
420  if (right && Y > row.MaxY() * GPUCA_GLOBAL_TRACKING_Y_RANGE_LOWER) {
421  // GPUInfo("Track %d, lower row %d, right border (%f of %f)", i, mTrackHits[tmpHit].RowIndex(), Y, row.MaxY());
422  sliceTarget.PerformGlobalTrackingRun(*this, i, rowIndex, Param().DAlpha, -1);
423  }
424  }
425  }
426 
427  {
428  const int tmpHit = mTracks[i].FirstHitID() + mTracks[i].NHits() - 1;
429  if (mTrackHits[tmpHit].RowIndex() < GPUCA_ROW_COUNT - GPUCA_GLOBAL_TRACKING_MIN_ROWS && mTrackHits[tmpHit].RowIndex() >= GPUCA_ROW_COUNT - GPUCA_GLOBAL_TRACKING_RANGE) {
430  int rowIndex = mTrackHits[tmpHit].RowIndex();
431  const GPUTPCRow& row = Row(rowIndex);
432  float Y = (float)Data().HitDataY(row, mTrackHits[tmpHit].HitIndex()) * row.HstepY() + row.Grid().YMin();
433  if (sliceTarget.mCommonMem->nTracks >= sliceTarget.mNMaxTracks) {
434  sliceTarget.mCommonMem->kernelError = GPUCA_ERROR_GLOBAL_TRACKING_TRACK_OVERFLOW;
435  return;
436  }
437  if (!right && Y < -row.MaxY() * GPUCA_GLOBAL_TRACKING_Y_RANGE_UPPER) {
438  // GPUInfo("Track %d, upper row %d, left border (%f of %f)", i, mTrackHits[tmpHit].RowIndex(), Y, -row.MaxY());
439  sliceTarget.PerformGlobalTrackingRun(*this, i, rowIndex, -Param().DAlpha, 1);
440  }
441  if (right && Y > row.MaxY() * GPUCA_GLOBAL_TRACKING_Y_RANGE_UPPER) {
442  // GPUInfo("Track %d, upper row %d, right border (%f of %f)", i, mTrackHits[tmpHit].RowIndex(), Y, row.MaxY());
443  sliceTarget.PerformGlobalTrackingRun(*this, i, rowIndex, Param().DAlpha, 1);
444  }
445  }
446  }
447  }
