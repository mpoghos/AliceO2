38  int status = 0;

40  // store buffer

61  mCurrentSize = SSizeOfUShort;

66  } else {
67  // read buffer otherwise
68  status = readBuffer();

70 
71  } else {
72

74  mNPreClusters = nullptr;
75

78  status = -ENOBUFS;

89 int PreClusterBlock::startPreCluster(const DigitStruct& digit)
90 {
92 
93  assert(mWriteMode);
94 
95  // could move back to constructor (or reset)
96  if (mCurrentSize > mSize4PreCluster) {
97  LOG(ERROR) << "The buffer is too small to store a new precluster.";
98  mLastNDigits = nullptr;
99  return -ENOBUFS;
100  }
101 
102  // use current buffer position to store number of digits
103  // and increment buffer
104  mLastNDigits = reinterpret_cast<uint16_t*>(mBuffer);
105  *mLastNDigits = 1;
106  mBuffer = (reinterpret_cast<uint16_t*>(mBuffer) + 1);
107  mCurrentSize += SSizeOfUShort;
108 
109  // store digit and increment buffer
110  auto lastDigit = reinterpret_cast<DigitStruct*>(mBuffer);
111  *lastDigit = digit;
112  mBuffer = (reinterpret_cast<DigitStruct*>(mBuffer) + 1);
113  mCurrentSize += SSizeOfDigit;
114 
115  // increment number of pre-clusters
116  (*mNPreClusters) += 1;
117 
118  // insert in list
119  PreClusterStruct preCluster = {*mLastNDigits, lastDigit};
120  mPreClusters.push_back(preCluster);
121 
122  return 0;
123 }

126 int PreClusterBlock::addDigit(const DigitStruct& digit)
127 {
129 
130  assert(mWriteMode);
131  if (mCurrentSize > mSize4Digit) {
132  LOG(ERROR) << "The buffer is too small to store a new digit.";
133  return -ENOBUFS;
134  }
135 
136  if (!mLastNDigits) {
137  LOG(ERROR) << "No precluster to attach the new digit to.";
138  return -EPERM;
139  }
140 
141  // increment number of digits
142  *mLastNDigits += 1;
143 
144  // assign digit to the buffer and increment buffer
145  *(reinterpret_cast<DigitStruct*>(mBuffer)) = digit;
146  mBuffer = (reinterpret_cast<DigitStruct*>(mBuffer) + 1);
147  mCurrentSize += SSizeOfDigit;
148 
149  // increment number of digits in the stored cluster
150  if (!mPreClusters.empty()) {
151  ++mPreClusters.back().nDigits;
152  }
153 
154  return 0;
155 }

158 int PreClusterBlock::readBuffer()
159 {
162 
163  assert(!mWriteMode);
164 
165  // make sure mNPreClusters was assigned
166  if (mNPreClusters == nullptr) {
167  return -EILSEQ;
168  }
169 
170  // initialize status
171  int status = 0;
172 
173  // rolling pre-cluster
174  PreClusterStruct preCluster = {};
175 
176  // loop over
177  for (int i = 0; i < *mNPreClusters; ++i) {
178 
179  // store number of digits from buffer and increment
180  uint16_t* nDigits = nullptr;
181  if (mCurrentSize < mSize) {
182  nDigits = reinterpret_cast<uint16_t*>(mBuffer);
183  mBuffer = (reinterpret_cast<uint16_t*>(mBuffer) + 1);
184  mCurrentSize += SSizeOfUShort;
185  } else {
186  LOG(ERROR) << "Cannot read the expected number of preclusters.";
187  status = -EILSEQ;
188  break;
189  }
190 
191  // read the digits
192  if (nDigits && (*nDigits) > 0 && mCurrentSize + (*nDigits) * SSizeOfDigit <= mSize) {
193 
194  auto digit = reinterpret_cast<DigitStruct*>(mBuffer);
195  mBuffer = (reinterpret_cast<DigitStruct*>(mBuffer) + (*nDigits));
196  mCurrentSize += (*nDigits) * SSizeOfDigit;
197 
198  // store
199  preCluster.nDigits = *nDigits;
200  preCluster.digits = digit;
201  mPreClusters.push_back(preCluster);
202 
203  } else {
204 
205  if (*nDigits == 0) {
206  LOG(ERROR) << "The precluster cannot contain 0 digit.";
207  } else {
208  LOG(ERROR) << "Cannot read the expected number of digits.";
209  }
210 
211  status = -EILSEQ;
212  break;
213  }
214  }
215 
216  // sanity check on read size
217  if (mCurrentSize != mSize && status >= 0) {
218  LOG(ERROR) << "The number of bytes read differs from the buffer size";
219  status = -EILSEQ;
220  }
221 
222  // reset in case of negative status
223  if (status < 0) {
224  mCurrentSize = mSize + 1;
225  mNPreClusters = nullptr;
226  mPreClusters.clear();
227  }
228 
229  return status;
230 }
