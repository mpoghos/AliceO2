152 int Spline2D::test(const bool draw, const bool drawDataPoints)
153 {
154  using namespace std;
155 
156  const int Ndim = 3;
157 
158  const int Fdegree = 4;
159 
160  double Fcoeff[Ndim][4 * (Fdegree + 1) * (Fdegree + 1)];
161 
162  int nKnots = 4;
163  const int nAxiliaryPoints = 1;
164  int uMax = nKnots * 3;
165 
166  auto F = [&](float u, float v, float Fuv[]) -> void {
167  double uu = u * TMath::Pi() / uMax;
168  double vv = v * TMath::Pi() / uMax;
169  for (int dim = 0; dim < Ndim; dim++) {
170  double f = 0; // Fcoeff[dim][0]/2;
171  for (int i = 1; i <= Fdegree; i++) {
172  double cosu = TMath::Cos(i * uu);
173  double sinu = TMath::Sin(i * uu);
174  for (int j = 1; j <= Fdegree; j++) {
175  double* c = &(Fcoeff[dim][4 * (i * Fdegree + j)]);
176  double cosv = TMath::Cos(j * vv);
177  double sinv = TMath::Sin(j * vv);
178  f += c[0] * cosu * cosv;
179  f += c[1] * cosu * sinv;
180  f += c[2] * sinu * cosv;
181  f += c[3] * sinu * sinv;
182  }
183  }
184  Fuv[dim] = f;
185  }
186  };
187 
188  TCanvas* canv = nullptr;
189  TNtuple* nt = nullptr;
190  TNtuple* knots = nullptr;
191 
192  auto ask = [&]() -> bool {
193  if (!canv) {
194  return 0;
195  }
196  canv->Update();
197  cout << "type 'q ' to exit" << endl;
198  std::string str;
199  std::getline(std::cin, str);
200  return (str != "q" && str != ".q");
201  };
202 
203  std::cout << "Test 2D interpolation with the compact spline" << std::endl;
204 
205  int nTries = 10;
206 
207  if (draw) {
208  canv = new TCanvas("cQA", "Spline1D QA", 2000, 1000);
209  nTries = 10000;
210  }
211 
212  long double statDf = 0;
213  long double statDf1D = 0;
214  long double statN = 0;
215 
216  for (int seed = 1; seed < nTries + 1; seed++) {
217  //cout << "next try.." << endl;
218 
219  gRandom->SetSeed(seed);
220 
221  for (int dim = 0; dim < Ndim; dim++) {
222  for (int i = 0; i < 4 * (Fdegree + 1) * (Fdegree + 1); i++) {
223  Fcoeff[dim][i] = gRandom->Uniform(-1, 1);
224  }
225  }
226 
227  GPUCA_NAMESPACE::gpu::SplineHelper2D helper;
228  GPUCA_NAMESPACE::gpu::Spline2D spline;
229  // spline.constructKnotsRegular(nKnots, nKnots);
230 
231  do {
232  int knotsU[nKnots], knotsV[nKnots];
233  knotsU[0] = 0;
234  knotsV[0] = 0;
235  double du = 1. * uMax / (nKnots - 1);
236  for (int i = 1; i < nKnots; i++) {
237  knotsU[i] = (int)(i * du); // + gRandom->Uniform(-du / 3, du / 3);
238  knotsV[i] = (int)(i * du); // + gRandom->Uniform(-du / 3, du / 3);
239  }
240  knotsU[nKnots - 1] = uMax;
241  knotsV[nKnots - 1] = uMax;
242  spline.constructKnots(nKnots, knotsU, nKnots, knotsV);
243 
244  if (nKnots != spline.getGridU().getNumberOfKnots() ||
245  nKnots != spline.getGridV().getNumberOfKnots()) {
246  cout << "warning: n knots changed during the initialisation " << nKnots
247  << " -> " << spline.getNumberOfKnots() << std::endl;
248  continue;
249  }
250  } while (0);
251 
252  std::string err = FlatObject::stressTest(spline);
253  if (!err.empty()) {
254  cout << "error at FlatObject functionality: " << err << endl;
255  return -1;
256  } else {
257  // cout << "flat object functionality is ok" << endl;
258  }
259 
260  int err2 = helper.setSpline(spline, nAxiliaryPoints, nAxiliaryPoints);
261  if (err2 != 0) {
262  cout << "Error by spline construction: " << helper.getLastError()
263  << std::endl;
264  return -1;
265  }
266 
267  // Ndim-D spline
268 
269  std::unique_ptr<float[]> parameters = helper.constructParameters(Ndim, F, 0., uMax, 0., uMax);
270 
271  // 1-D splines for each of Ndim dimensions
272 
273  std::unique_ptr<float[]> parameters1D[Ndim];
274  {
275  std::unique_ptr<float[]> dataPoints1D[Ndim];
276  for (int dim = 0; dim < Ndim; dim++) {
277  parameters1D[dim].reset(new float[spline.getNumberOfParameters(1)]);
278  dataPoints1D[dim].reset(new float[helper.getNumberOfDataPoints()]);
279  }
280 
281  int nPointsU = helper.getNumberOfDataPointsU();
282  int nPointsV = helper.getNumberOfDataPointsV();
283  for (int ipv = 0; ipv < nPointsV; ipv++) {
284  float v = helper.getHelperV().getDataPoint(ipv).u;
285  for (int ipu = 0; ipu < nPointsU; ipu++) {
286  float u = helper.getHelperU().getDataPoint(ipu).u;
287  float Fuv[Ndim];
288  F(u, v, Fuv);
289  int ip = ipv * nPointsU + ipu;
290  for (int dim = 0; dim < Ndim; dim++) {
291  dataPoints1D[dim][ip] = Fuv[dim];
292  }
293  }
294  }
295 
296  for (int dim = 0; dim < Ndim; dim++) {
297  helper.constructParameters(1, dataPoints1D[dim].get(), parameters1D[dim].get());
298  }
299  }
300 
301  double stepU = .1;
302  for (double u = 0; u < uMax; u += stepU) {
303  for (double v = 0; v < uMax; v += stepU) {
304  float f[Ndim];
305  F(u, v, f);
306  float s[Ndim];
307  float s1;
308  spline.interpolate(Ndim, parameters.get(), u, v, s);
309  for (int dim = 0; dim < Ndim; dim++) {
310  statDf += (s[dim] - f[dim]) * (s[dim] - f[dim]);
311  spline.interpolate(1, parameters1D[dim].get(), u, v, &s1);
312  statDf1D += (s[dim] - s1) * (s[dim] - s1);
313  }
314  statN += Ndim;
315  // cout << u << " " << v << ": f " << f << " s " << s << " df "
316  // << s - f << " " << sqrt(statDf / statN) << std::endl;
317  }
318  }
319  // cout << "Spline2D standard deviation : " << sqrt(statDf / statN)
320  // << std::endl;
321 
322  if (draw) {
323  delete nt;
324  delete knots;
325  nt = new TNtuple("nt", "nt", "u:v:f:s");
326  knots = new TNtuple("knots", "knots", "type:u:v:s");
327 
328  double stepU = .1;
329  for (double u = 0; u < uMax; u += stepU) {
330  for (double v = 0; v < uMax; v += stepU) {
331  float f[Ndim];
332  F(u, v, f);
333  float s[Ndim];
334  spline.interpolate(Ndim, parameters.get(), u, v, s);
335  nt->Fill(u, v, f[0], s[0]);
336  }
337  }
338  nt->SetMarkerStyle(8);
339 
340  nt->SetMarkerSize(.5);
341  nt->SetMarkerColor(kBlue);
342  nt->Draw("s:u:v", "", "");
343 
344  nt->SetMarkerColor(kGray);
345  nt->SetMarkerSize(2.);
346  nt->Draw("f:u:v", "", "same");
347 
348  nt->SetMarkerSize(.5);
349  nt->SetMarkerColor(kBlue);
350  nt->Draw("s:u:v", "", "same");
351 
352  for (int i = 0; i < nKnots; i++) {
353  for (int j = 0; j < nKnots; j++) {
354  double u = spline.getGridU().getKnot(i).u;
355  double v = spline.getGridV().getKnot(j).u;
356  float s[Ndim];
357  spline.interpolate(Ndim, parameters.get(), u, v, s);
358  knots->Fill(1, u, v, s[0]);
359  }
360  }
361 
362  knots->SetMarkerStyle(8);
363  knots->SetMarkerSize(1.5);
364  knots->SetMarkerColor(kRed);
365  knots->SetMarkerSize(1.5);
366  knots->Draw("s:u:v", "type==1", "same"); // knots
367 
368  if (drawDataPoints) {
369  for (int ipu = 0; ipu < helper.getHelperU().getNumberOfDataPoints(); ipu++) {
370  const SplineHelper1D::DataPoint& pu = helper.getHelperU().getDataPoint(ipu);
371  for (int ipv = 0; ipv < helper.getHelperV().getNumberOfDataPoints(); ipv++) {
372  const SplineHelper1D::DataPoint& pv = helper.getHelperV().getDataPoint(ipv);
373  if (pu.isKnot && pv.isKnot) {
374  continue;
375  }
376  float s[Ndim];
377  spline.interpolate(Ndim, parameters.get(), pu.u, pv.u, s);
378  knots->Fill(2, pu.u, pv.u, s[0]);
379  }
380  }
381  knots->SetMarkerColor(kBlack);
382  knots->SetMarkerSize(1.);
383  knots->Draw("s:u:v", "type==2", "same"); // data points
384  }
385 
386  if (!ask()) {
387  break;
388  }
389  }
390  }
391  // delete canv;
392  // delete nt;
393  // delete knots;
394 
395  statDf = sqrt(statDf / statN);
396  statDf1D = sqrt(statDf1D / statN);
397 
398  cout << "\n std dev for Compact Spline : " << statDf << std::endl;
399  cout << " mean difference between 1-D and " << Ndim
400  << "-D splines : " << statDf1D << std::endl;
401 
402  if (statDf < 0.15 && statDf1D < 1.e-20) {
403  cout << "Everything is fine" << endl;
404  } else {
405  cout << "Something is wrong!!" << endl;
406  return -2;
407  }
408  return 0;
409 }
