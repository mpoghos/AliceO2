50 template <typename RDH, typename PAYLOADDECODER>
51 DecoderStat PageParser<RDH, PAYLOADDECODER>::parse(gsl::span<uint8_t> buffer)
52 {
53  RDH originalRDH;
54  const size_t nofRDHWords = sizeof(originalRDH);
55  size_t index{0};
56  uint64_t nbytes{0};
57 
58  while ((index + nofRDHWords) < buffer.size()) {
59  originalRDH = createRDH<RDH>(buffer.subspan(index, nofRDHWords));
60  if (!isValid(originalRDH)) {
61  std::cout << "Got an invalid RDH\n";
62  impl::dumpBuffer(buffer.subspan(index, nofRDHWords));
63  return mStats;
64  }
65  if (hasOrbitJump(rdhOrbit(originalRDH), mOrbit)) {
66  ++mStats.nofOrbitJumps;
67  mDecoder.reset();
68  } else if (rdhOrbit(originalRDH) != mOrbit) {
69  ++mStats.nofOrbitSeen;
70  }
71  mOrbit = rdhOrbit(originalRDH);
72  auto rdhOpt = mRdhHandler(originalRDH);
73  if (!rdhOpt.has_value()) {
74  break;
75  }
76  auto rdh = rdhOpt.value();
77  int payloadSize = rdhPayloadSize(rdh);
78  size_t n = static_cast<size_t>(payloadSize);
79  if (n) {
80  size_t pos = static_cast<size_t>(index + nofRDHWords);
81  mDecoder.process(rdh, buffer.subspan(pos, n));
82  nbytes += n + nofRDHWords;
83  }
84  index += rdh.offsetToNext;
85  }
86  mStats.nofBytesUsed += nbytes;
87  return mStats;
88 }
