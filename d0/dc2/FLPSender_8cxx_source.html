56 void FLPSender::Run()
57 {
58  // base buffer, to be copied from for every timeframe body (zero-copy)
59  FairMQMessagePtr baseMsg(NewMessage(mEventSize));
60 
61  // store the channel reference to avoid traversing the map on every loop iteration
62  FairMQChannel& dataInChannel = fChannels.at(mInChannelName).at(0);
63 
64  while (compatibility::FairMQ13<FairMQDevice>::IsRunning(this)) {
65  // initialize f2e header
66  auto* header = new f2eHeader;
67  if (mTestMode > 0) {
68  // test-mode: receive and store id part in the buffer.
69  FairMQMessagePtr id(NewMessage());
70  if (dataInChannel.Receive(id) > 0) {
71  header->timeFrameId = *(static_cast<uint16_t*>(id->GetData()));
72  header->flpIndex = mIndex;
73  } else {
74  // if nothing was received, try again
75  delete header;
76  continue;
77  }
78  } else {
79  // regular mode: use the id generated locally
80  header->timeFrameId = mTimeFrameId;
81  header->flpIndex = mIndex;
82 
83  if (++mTimeFrameId == UINT16_MAX - 1) {
84  mTimeFrameId = 0;
85  }
86  }
87 
88  FairMQParts parts;
89 
90  parts.AddPart(NewMessage(
91  header, sizeof(f2eHeader), [](void* data, void* hint) { delete static_cast<f2eHeader*>(hint); }, header));
92  parts.AddPart(NewMessage());
93 
94  // save the arrival time of the message.
95  mArrivalTime.push(steady_clock::now());
96 
97  if (mTestMode > 0) {
98  // test-mode: initialize and store data part in the buffer.
99  parts.At(1)->Copy(*baseMsg);
100  mSTFBuffer.push(move(parts));
101  } else {
102  // regular mode: receive data part from input
103  if (dataInChannel.Receive(parts.At(1)) >= 0) {
104  mSTFBuffer.push(move(parts));
105  } else {
106  // if nothing was received, try again
107  continue;
108  }
109  }
110 
111  // if offset is 0 - send data out without staggering.
112  if (mSendOffset == 0 && mSTFBuffer.size() > 0) {
113  sendFrontData();
114  } else if (mSTFBuffer.size() > 0) {
115  if (duration_cast<milliseconds>(steady_clock::now() - mArrivalTime.front()).count() >= (mSendDelay * mSendOffset)) {
116  sendFrontData();
117  } else {
118  // LOG(INFO) << "buffering...";
119  }
120  }
121  }
122 }

124 inline void FLPSender::sendFrontData()
125 {
126  f2eHeader header = *(static_cast<f2eHeader*>(mSTFBuffer.front().At(0)->GetData()));
127  uint16_t currentTimeframeId = header.timeFrameId;
128 
129  // for which EPN is the message?
130  int direction = currentTimeframeId % mNumEPNs;
131  // LOG(INFO) << "Sending event " << currentTimeframeId << " to EPN#" << direction << "...";
132 
133  if (Send(mSTFBuffer.front(), mOutChannelName, direction, 0) < 0) {
134  LOG(ERROR) << "Failed to queue sub-timeframe #" << currentTimeframeId << " to EPN[" << direction << "]";
135  }
136  mSTFBuffer.pop();
137  mArrivalTime.pop();
138 }
