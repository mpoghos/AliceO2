123 template <>
124 RAWDataHeaderV4 createRDH(gsl::span<uint8_t> buffer)
125 {
126  if (buffer.size() < 64) {
127  throw std::invalid_argument("buffer should be at least 64 bytes");
128  }
129  RAWDataHeaderV4 rdh;
130  rdh.word0 = eightBytes(buffer.subspan(0));
131  rdh.word1 = eightBytes(buffer.subspan(8));
132  rdh.word2 = eightBytes(buffer.subspan(16));
133  rdh.word3 = eightBytes(buffer.subspan(24));
134  rdh.word4 = eightBytes(buffer.subspan(32));
135  rdh.word5 = eightBytes(buffer.subspan(40));
136  rdh.word6 = eightBytes(buffer.subspan(48));
137  rdh.word7 = eightBytes(buffer.subspan(56));
138  return rdh;
139 }

258 template <typename RDH>
259 int forEachRDH(gsl::span<uint32_t> buffer, std::function<void(RDH&)> f)
260 {
261  RDH rdh;
262  int index{0};
263  int nrdh{0};
264  while (index < buffer.size()) {
265  memcpy(&rdh, &buffer[0] + index, sizeof(rdh));
266  if (!isValid(rdh)) {
267  break;
268  }
269  nrdh++;
270  if (f) {
271  f(rdh);
272  }
273  if (rdh.offsetToNext == 0) {
274  return -1;
275  }
276  index += rdh.offsetToNext / 4;
277  }
278  return nrdh;
279 }
