201 //_____________________________________________________________________________
202 int FeeParam::getPadColFromADC(int irob, int imcm, int iadc) const
203 {
204  //
205  // Return which pad is connected to this adc channel.
206  //
207  // Return virtual pad number even if ADC is outside chamber
208  // to keep compatibility of data processing at the edge MCM.
209  // User has to check that this is in the chamber if it is essential.
210  // Return -100 if iadc is invalid.
211  //
212  // Caution: ADC ordering in the online data is opposite to the pad column ordering.
213  // And it is not one-by-one correspondence. Precise drawing can be found in:
214  // http://wiki.kip.uni-heidelberg.de/ti/TRD/index.php/Image:ROB_MCM_numbering.pdf
215  //
216 
217  if (iadc < 0 || iadc > mgkNadcMcm)
218  return -100;
219  int mcmcol = imcm % mgkNmcmRobInCol + getRobSide(irob) * mgkNmcmRobInCol; // MCM column number on ROC [0..7]
220  int padcol = mcmcol * mgkNcolMcm + mgkNcolMcm + 1 - iadc;
221  if (padcol < 0 || padcol >= mgkNcol)
222  return -1; // this is commented because of reason above OK
223 
224  return padcol;
225 }

345 int FeeParam::extAliToAli(unsigned int dest, unsigned short linkpair, unsigned short rocType, int* mcmList, int listSize)
346 {
347  // Converts an extended ALICE ID which identifies a single MCM or a group of MCMs to
348  // the corresponding list of MCMs. Only broadcasts (127) are encoded as 127
349  // The return value is the number of MCMs in the list
350 
351  mcmList[0] = -1;
352 
353  short nmcm = 0;
354  unsigned int mcm, rob, robAB;
355  unsigned int cmA = 0, cmB = 0; // Chipmask for each A and B side
356 
357  // Default chipmask for 4 linkpairs (each bit correponds each alice-mcm)
358  static const unsigned int gkChipmaskDefLp[4] = {0x1FFFF, 0x1FFFF, 0x3FFFF, 0x1FFFF};
359 
360  rob = dest >> 7; // Extract ROB pattern from dest.
361  mcm = dest & 0x07F; // Extract MCM pattern from dest.
362  robAB = getRobAB(rob, linkpair); // Get which ROB sides are selected.
363 
364  // Abort if no ROB is selected
365  if (robAB == 0) {
366  return 0;
367  }
368 
369  // Special case
370  if (mcm == 127) {
371  if (robAB == 3) { // This is very special 127 can stay only if two ROBs are selected
372  mcmList[0] = 127; // broadcase to ALL
373  mcmList[1] = -1;
374  return 1;
375  }
376  cmA = cmB = 0x3FFFF;
377  } else if ((mcm & 0x40) != 0) { // If top bit is 1 but not 127, this is chip group.
378  if ((mcm & 0x01) != 0) {
379  cmA |= 0x04444;
380  cmB |= 0x04444;
381  } // chip_cmrg
382  if ((mcm & 0x02) != 0) {
383  cmA |= 0x10000;
384  cmB |= 0x10000;
385  } // chip_bmrg
386  if ((mcm & 0x04) != 0 && rocType == 0) {
387  cmA |= 0x20000;
388  cmB |= 0x20000;
389  } // chip_hm3
390  if ((mcm & 0x08) != 0 && rocType == 1) {
391  cmA |= 0x20000;
392  cmB |= 0x20000;
393  } // chip_hm4
394  if ((mcm & 0x10) != 0) {
395  cmA |= 0x01111;
396  cmB |= 0x08888;
397  } // chip_edge
398  if ((mcm & 0x20) != 0) {
399  cmA |= 0x0aaaa;
400  cmB |= 0x03333;
401  } // chip_norm
402  } else { // Otherwise, this is normal chip ID, turn on only one chip.
403  cmA = 1 << mcm;
404  cmB = 1 << mcm;
405  }
406 
407  // Mask non-existing MCMs
408  cmA &= gkChipmaskDefLp[linkpair];
409  cmB &= gkChipmaskDefLp[linkpair];
410  // Remove if only one side is selected
411  if (robAB == 1)
412  cmB = 0;
413  if (robAB == 2)
414  cmA = 0;
415  if (robAB == 4 && linkpair != 2)
416  cmA = cmB = 0; // Restrict to only T3A and T3B
417 
418  // Finally convert chipmask to list of slaves
419  nmcm = chipmaskToMCMlist(cmA, cmB, linkpair, mcmList, listSize);
420 
421  return nmcm;
422 }

424 short FeeParam::getRobAB(unsigned short robsel, unsigned short linkpair)
425 {
426  // Converts the ROB part of the extended ALICE ID to robs
427 
428  if ((robsel & 0x8) != 0) { // 1000 .. direct ROB selection. Only one of the 8 ROBs are used.
429  robsel = robsel & 7;
430  if ((robsel % 2) == 0 && (robsel / 2) == linkpair)
431  return 1; // Even means A side (position 0,2,4,6)
432  if ((robsel % 2) == 1 && (robsel / 2) == linkpair)
433  return 2; // Odd means B side (position 1,3,5,7)
434  return 0;
435  }
436 
437  // ROB group
438  if (robsel == 0) {
439  return 3;
440  } // Both ROB
441  if (robsel == 1) {
442  return 1;
443  } // A-side ROB
444  if (robsel == 2) {
445  return 2;
446  } // B-side ROB
447  if (robsel == 3) {
448  return 3;
449  } // Both ROB
450  if (robsel == 4) {
451  return 4;
452  } // Only T3A and T3B
453  // Other number 5 to 7 are ignored (not defined)
454 
455  return 0;
456 }

458 short FeeParam::chipmaskToMCMlist(unsigned int cmA, unsigned int cmB, unsigned short linkpair, int* mcmList, int listSize)
459 {
460  // Converts the chipmask to a list of MCMs
461 
462  short nmcm = 0;
463  short i;
464  for (i = 0; i < 18; i++) { // 18: number of MCMs on a ROB
465  if ((cmA & (1 << i)) != 0 && nmcm < listSize) {
466  mcmList[nmcm] = ((linkpair * 2) << 7) | i;
467  ++nmcm;
468  }
469  if ((cmB & (1 << i)) != 0 && nmcm < listSize) {
470  mcmList[nmcm] = ((linkpair * 2 + 1) << 7) | i;
471  ++nmcm;
472  }
473  }
474 
475  mcmList[nmcm] = -1;
476  return nmcm;
477 }

519 int FeeParam::getDyCorrection(int det, int rob, int mcm) const
520 {
521  // calculate the correction of the deflection
522  // i.e. Lorentz angle and tilt correction (if active)
523 
524  int layer = det % 6;
525 
526  float dyTilt = (mgDriftLength * std::tan(mgTiltingAngle[layer] * M_PI / 180.) *
527  getLocalZ(det, rob, mcm) * mgInvX[layer]);
528 
529  // calculate Lorentz correction
530  float dyCorr = -mOmegaTau * mgDriftLength;
531 
532  if (mTiltCorr)
533  dyCorr += dyTilt; // add tilt correction
534 
535  return (int)TMath::Nint(dyCorr * mgScalePad * mgInvWidthPad[layer]);
536 }

538 void FeeParam::getDyRange(int det, int rob, int mcm, int ch,
539  int& dyMinInt, int& dyMaxInt) const
540 {
541  // calculate the deflection range in which tracklets are accepted
542 
543  dyMinInt = mgDyMin;
544  dyMaxInt = mgDyMax;
545 
546  // deflection cut is considered for |B| > 0.1 T only
547  if (std::abs(mMagField) < 0.1)
548  return;
549 
550  float e = 0.30;
551 
552  float maxDeflTemp = getPerp(det, rob, mcm, ch) / 2. * // Sekante/2 (cm)
553  (e * 1e-2 * std::abs(mMagField) * mInvPtMin); // 1/R (1/cm)
554 
555  float phi = getPhi(det, rob, mcm, ch);
556  if (maxDeflTemp < std::cos(phi)) {
557  float maxDeflAngle = std::asin(maxDeflTemp);
558 
559  float dyMin = (mgDriftLength *
560  std::tan(phi - maxDeflAngle));
561 
562  dyMinInt = int(dyMin / mgBinDy);
563  // clipping to allowed range
564  if (dyMinInt < mgDyMin)
565  dyMinInt = mgDyMin;
566  else if (dyMinInt > mgDyMax)
567  dyMinInt = mgDyMax;
568 
569  float dyMax = (mgDriftLength *
570  std::tan(phi + maxDeflAngle));
571 
572  dyMaxInt = int(dyMax / mgBinDy);
573  // clipping to allowed range
574  if (dyMaxInt > mgDyMax)
575  dyMaxInt = mgDyMax;
576  else if (dyMaxInt < mgDyMin)
577  dyMaxInt = mgDyMin;
578  } else if (maxDeflTemp < 0.) {
579  // this must not happen
580  printf("Inconsistent calculation of sin(alpha): %f\n", maxDeflTemp);
581  } else {
582  // TRD is not reached at the given pt threshold
583  // max range
584  }
585 
586  if ((dyMaxInt - dyMinInt) <= 0) {
587  LOG(info) << "strange dy range: [" << dyMinInt << "," << dyMaxInt << "], using max range now";
588  dyMaxInt = mgDyMax;
589  dyMinInt = mgDyMin;
590  }
591 }

652 float FeeParam::getLocalZ(int det, int rob, int mcm) const
653 {
654  // get local z-position w.r.t. to the chamber boundary
655 
656  int stack = (det % 30) / 6;
657  int layer = det % 6;
658  int row = (rob / 2) * 4 + mcm / 4;
659 
660  if (stack == 2) {
661  if (row == 0)
662  return (mgZrow[layer * 6 + stack] - 0.5 * mgLengthOuterPadC0);
663  else if (row == 11)
664  return (mgZrow[layer * 6 + stack] - 1.5 * mgLengthOuterPadC0 - (row - 1) * mgLengthInnerPadC0);
665  else
