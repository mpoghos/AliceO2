165  bestLabelCount = curCount;
166  bestLabel = curLabel;

255 void GPUTPCGMMerger::UnpackSlices()
256 {
257  //* unpack the cluster information from the slice tracks and initialize track info array
258 
259  int nTracksCurrent = 0;
260 
261  const GPUTPCSliceOutTrack* firstGlobalTracks[NSLICES];
262 
263  unsigned int maxSliceTracks = 0;
264  for (int i = 0; i < NSLICES; i++) {
265  firstGlobalTracks[i] = nullptr;
266  if (mkSlices[i]->NLocalTracks() > maxSliceTracks) {
267  maxSliceTracks = mkSlices[i]->NLocalTracks();
268  }
269  }
270 
271  if (maxSliceTracks > mNMaxSingleSliceTracks) {
272  throw std::runtime_error("mNMaxSingleSliceTracks too small");
273  }
274 
275  int* TrackIds = (int*)mTmpMem;
276  for (unsigned int i = 0; i < maxSliceTracks * NSLICES; i++) {
277  TrackIds[i] = -1;
278  }
279 
280  for (int iSlice = 0; iSlice < NSLICES; iSlice++) {
281  mSliceTrackInfoIndex[iSlice] = nTracksCurrent;
282 
283  float alpha = Param().Alpha(iSlice);
284  const GPUTPCSliceOutput& slice = *(mkSlices[iSlice]);
285  const GPUTPCSliceOutTrack* sliceTr = slice.GetFirstTrack();
286 
287  for (unsigned int itr = 0; itr < slice.NLocalTracks(); itr++, sliceTr = sliceTr->GetNextTrack()) {
288  GPUTPCGMSliceTrack& track = mSliceTrackInfos[nTracksCurrent];
289  track.Set(sliceTr, alpha, iSlice);
290  if (!track.FilterErrors(this, iSlice, GPUCA_MAX_SIN_PHI, 0.1f)) {
291  continue;
292  }
293  CADEBUG(GPUInfo("INPUT Slice %d, Track %u, QPt %f DzDs %f", iSlice, itr, track.QPt(), track.DzDs()));
294  track.SetPrevNeighbour(-1);
295  track.SetNextNeighbour(-1);
296  track.SetNextSegmentNeighbour(-1);
297  track.SetPrevSegmentNeighbour(-1);
298  track.SetGlobalTrackId(0, -1);
299  track.SetGlobalTrackId(1, -1);
300  TrackIds[iSlice * maxSliceTracks + sliceTr->LocalTrackId()] = nTracksCurrent;
301  nTracksCurrent++;
302  }
303  firstGlobalTracks[iSlice] = sliceTr;
304  }
305  for (int iSlice = 0; iSlice < NSLICES; iSlice++) {
306  mSliceTrackInfoIndex[NSLICES + iSlice] = nTracksCurrent;
307 
308  float alpha = Param().Alpha(iSlice);
309  const GPUTPCSliceOutput& slice = *(mkSlices[iSlice]);
310  const GPUTPCSliceOutTrack* sliceTr = firstGlobalTracks[iSlice];
311  for (unsigned int itr = slice.NLocalTracks(); itr < slice.NTracks(); itr++, sliceTr = sliceTr->GetNextTrack()) {
312  int localId = TrackIds[(sliceTr->LocalTrackId() >> 24) * maxSliceTracks + (sliceTr->LocalTrackId() & 0xFFFFFF)];
313  if (localId == -1) {
314  continue;
315  }
316  GPUTPCGMSliceTrack& track = mSliceTrackInfos[nTracksCurrent];
317  track.Set(sliceTr, alpha, iSlice);
318  track.SetGlobalSectorTrackCov();
319  track.SetPrevNeighbour(-1);
320  track.SetNextNeighbour(-1);
321  track.SetNextSegmentNeighbour(-1);
322  track.SetPrevSegmentNeighbour(-1);
323  track.SetLocalTrackId(localId);
324  nTracksCurrent++;
325  }
326  }
327  mSliceTrackInfoIndex[2 * NSLICES] = nTracksCurrent;
328 }

330 void GPUTPCGMMerger::MakeBorderTracks(int iSlice, int iBorder, GPUTPCGMBorderTrack B[], int& nB, bool useOrigTrackParam)
331 {
332  //* prepare slice tracks for merging with next/previous/same sector
333  //* each track transported to the border line
334 
335  float fieldBz = Param().ConstBz;
336 
337  nB = 0;
338 
339  float dAlpha = Param().DAlpha / 2;
340  float x0 = 0;
341 
342  if (iBorder == 0) { // transport to the left edge of the sector and rotate horizontally
343  dAlpha = dAlpha - CAMath::Pi() / 2;
344  } else if (iBorder == 1) { // transport to the right edge of the sector and rotate horizontally
345  dAlpha = -dAlpha - CAMath::Pi() / 2;
346  } else if (iBorder == 2) { // transport to the middle of the sector and rotate vertically to the border on the left
347  x0 = Param().tpcGeometry.Row2X(63);
348  } else if (iBorder == 3) { // transport to the middle of the sector and rotate vertically to the border on the right
349  dAlpha = -dAlpha;
350  x0 = Param().tpcGeometry.Row2X(63);
351  } else if (iBorder == 4) { // transport to the middle of the sector, w/o rotation
352  dAlpha = 0;
353  x0 = Param().tpcGeometry.Row2X(63);
354  }
355 
356  const float maxSin = CAMath::Sin(60. / 180. * CAMath::Pi());
357  float cosAlpha = CAMath::Cos(dAlpha);
358  float sinAlpha = CAMath::Sin(dAlpha);
359 
360  GPUTPCGMSliceTrack trackTmp;
361  for (int itr = SliceTrackInfoFirst(iSlice); itr < SliceTrackInfoLast(iSlice); itr++) {
362  const GPUTPCGMSliceTrack* track = &mSliceTrackInfos[itr];
363 
364  if (track->PrevSegmentNeighbour() >= 0 && track->Slice() == mSliceTrackInfos[track->PrevSegmentNeighbour()].Slice()) {
365  continue;
366  }
367  if (useOrigTrackParam) {
368  if (fabsf(track->QPt()) < GPUCA_MERGER_LOOPER_QPT_LIMIT) {
369  continue;
370  }
371  const GPUTPCGMSliceTrack* trackMin = track;
372  while (track->NextSegmentNeighbour() >= 0 && track->Slice() == mSliceTrackInfos[track->NextSegmentNeighbour()].Slice()) {
373  track = &mSliceTrackInfos[track->NextSegmentNeighbour()];
374  if (track->OrigTrack()->Param().X() < trackMin->OrigTrack()->Param().X()) {
375  trackMin = track;
376  }
377  }
378  trackTmp = *trackMin;
379  track = &trackTmp;
380  trackTmp.Set(trackMin->OrigTrack(), trackMin->Alpha(), trackMin->Slice());
381  } else {
382  if (fabsf(track->QPt()) < GPUCA_MERGER_HORIZONTAL_DOUBLE_QPT_LIMIT) {
383  if (iBorder == 0 && track->NextNeighbour() >= 0) {
384  continue;
385  }
386  if (iBorder == 1 && track->PrevNeighbour() >= 0) {
387  continue;
388  }
389  }
390  }
391  GPUTPCGMBorderTrack& b = B[nB];
392 
393  if (track->TransportToXAlpha(x0, sinAlpha, cosAlpha, fieldBz, b, maxSin)) {
394  b.SetTrackID(itr);
395  b.SetNClusters(track->NClusters());
396  for (int i = 0; i < 4; i++) {
397  if (fabsf(b.Cov()[i]) >= 5.0) {
398  b.SetCov(i, 5.0);
399  }
400  }
401  if (fabsf(b.Cov()[4]) >= 0.5) {
402  b.SetCov(4, 0.5);
403  }
404  nB++;
405  }
406  }
407 }

409 void GPUTPCGMMerger::MergeBorderTracks(int iSlice1, GPUTPCGMBorderTrack B1[], int N1, int iSlice2, GPUTPCGMBorderTrack B2[], int N2, int mergeMode)
410 {
411  //* merge two sets of tracks
412  if (N1 == 0 || N2 == 0) {
413  return;
414  }
415 
416  CADEBUG(GPUInfo("\nMERGING Slices %d %d NTracks %d %d CROSS %d", iSlice1, iSlice2, N1, N2, mergeMode));
417  int statAll = 0, statMerged = 0;
418  float factor2ys = 1.5; // 1.5;//SG!!!
419  float factor2zt = 1.5; // 1.5;//SG!!!
420  float factor2k = 2.0; // 2.2;
421 
422  factor2k = 3.5 * 3.5 * factor2k * factor2k;
423  factor2ys = 3.5 * 3.5 * factor2ys * factor2ys;
424  factor2zt = 3.5 * 3.5 * factor2zt * factor2zt;
425 
426  int minNPartHits = 10; // SG!!!
427  int minNTotalHits = 20;
428 
429  GPUTPCGMBorderTrack::Range* range1 = mBorderRange[iSlice1];
430  GPUTPCGMBorderTrack::Range* range2 = mBorderRange[iSlice2] + N2;
431 
432  bool sameSlice = (iSlice1 == iSlice2);
433  {
434  for (int itr = 0; itr < N1; itr++) {
435  GPUTPCGMBorderTrack& b = B1[itr];
436  float d = CAMath::Max(0.5f, 3.5f * sqrtf(b.Cov()[1]));
437  if (fabsf(b.Par()[4]) >= 20) {
438  d *= 2;
439  } else if (d > 3) {
440  d = 3;
441  }
442  CADEBUG(
443  printf(" Input Slice 1 %d Track %d: ", iSlice1, itr); for (int i = 0; i < 5; i++) { printf("%8.3f ", b.Par()[i]); } printf(" - "); for (int i = 0; i < 5; i++) { printf("%8.3f ", b.Cov()[i]); } printf(" - D %8.3f\n", d));
444  range1[itr].fId = itr;
445  range1[itr].fMin = b.Par()[1] + b.ZOffset() - d;
446  range1[itr].fMax = b.Par()[1] + b.ZOffset() + d;
447  }
448  std::sort(range1, range1 + N1, GPUTPCGMBorderTrack::Range::CompMin);
449  if (sameSlice) {
450  for (int i = 0; i < N1; i++) {
451  range2[i] = range1[i];
452  }
453  std::sort(range2, range2 + N1, GPUTPCGMBorderTrack::Range::CompMax);
454  N2 = N1;
455  B2 = B1;
456  } else {
457  for (int itr = 0; itr < N2; itr++) {
458  GPUTPCGMBorderTrack& b = B2[itr];
459  float d = CAMath::Max(0.5f, 3.5f * sqrtf(b.Cov()[1]));
460  if (fabsf(b.Par()[4]) >= 20) {
461  d *= 2;
462  } else if (d > 3) {
463  d = 3;
464  }
465  CADEBUG(
466  printf(" Input Slice 2 %d Track %d: ", iSlice2, itr); for (int i = 0; i < 5; i++) { printf("%8.3f ", b.Par()[i]); } printf(" - "); for (int i = 0; i < 5; i++) { printf("%8.3f ", b.Cov()[i]); } printf(" - D %8.3f\n", d));
467  range2[itr].fId = itr;
468  range2[itr].fMin = b.Par()[1] + b.ZOffset() - d;
469  range2[itr].fMax = b.Par()[1] + b.ZOffset() + d;
470  }
471  std::sort(range2, range2 + N2, GPUTPCGMBorderTrack::Range::CompMax);
472  }
473  }
474 
475  int i2 = 0;
476  for (int i1 = 0; i1 < N1; i1++) {
477  GPUTPCGMBorderTrack::Range r1 = range1[i1];
478  while (i2 < N2 && range2[i2].fMax < r1.fMin) {
479  i2++;
480  }
481 
482  GPUTPCGMBorderTrack& b1 = B1[r1.fId];
483  if (b1.NClusters() < minNPartHits) {
484  continue;
485  }
486  int iBest2 = -1;
487  int lBest2 = 0;
488  statAll++;
489  for (int k2 = i2; k2 < N2; k2++) {
490  GPUTPCGMBorderTrack::Range r2 = range2[k2];
491  if (r2.fMin > r1.fMax) {
492  break;
493  }
494  if (sameSlice && (r1.fId >= r2.fId)) {
495  continue;
496  }
497  // do check
498 
499  GPUTPCGMBorderTrack& b2 = B2[r2.fId];
500 #ifdef GPUCA_MERGER_BY_MC_LABEL
501  if (GetTrackLabel(b1) != GetTrackLabel(b2)) // DEBUG CODE, match by MC label
502 #endif
503  {
504  CADEBUG(
505  printf("Comparing track %3d to %3d: ", r1.fId, r2.fId); for (int i = 0; i < 5; i++) { printf("%8.3f ", b1.Par()[i]); } printf(" - "); for (int i = 0; i < 5; i++) { printf("%8.3f ", b1.Cov()[i]); } printf("\n%28s", ""));
506  CADEBUG(
507  for (int i = 0; i < 5; i++) { printf("%8.3f ", b2.Par()[i]); } printf(" - "); for (int i = 0; i < 5; i++) { printf("%8.3f ", b2.Cov()[i]); } printf(" - %5s - ", GetTrackLabel(b1) == GetTrackLabel(b2) ? "CLONE" : "FAKE"));
508  if (b2.NClusters() < lBest2) {
509  CADEBUG2(continue, printf("!NCl1\n"));
510  }
511  if (mergeMode > 0) {
512  int maxRowDiff = mergeMode == 2 ? 1 : 3; // TODO: check cut
513  if (CAMath::Abs(b1.Row() - b2.Row()) > maxRowDiff) {
514  CADEBUG2(continue, printf("!ROW\n"));
515  }
516  if (CAMath::Abs(b1.Par()[2] - b2.Par()[2]) > 0.5 || CAMath::Abs(b1.Par()[3] - b2.Par()[3]) > 0.5) {
517  CADEBUG2(continue, printf("!CE SinPhi/Tgl\n")); // Crude cut to avoid totally wrong matches, TODO: check cut
518  }
519  }
520  if (!b1.CheckChi2Y(b2, factor2ys)) {
521  CADEBUG2(continue, printf("!Y\n"));
522  }
523  // if( !b1.CheckChi2Z(b2, factor2zt ) ) CADEBUG2(continue, printf("!NCl1\n"));
524  if (!b1.CheckChi2QPt(b2, factor2k)) {
525  CADEBUG2(continue, printf("!QPt\n"));
526  }
527  float fys = fabsf(b1.Par()[4]) < 20 ? factor2ys : (2. * factor2ys);
528  float fzt = fabsf(b1.Par()[4]) < 20 ? factor2zt : (2. * factor2zt);
529  if (!b1.CheckChi2YS(b2, fys)) {
530  CADEBUG2(continue, printf("!YS\n"));
531  }
