60 template <typename RDH, typename GBTDECODER>
61 size_t PayloadDecoder<RDH, GBTDECODER>::process(const RDH& rdh, gsl::span<uint8_t> buffer)
62 {
63  auto solarId = rdh.feeId;
64  auto c = mDecoders.find(solarId);
65  if (c == mDecoders.end()) {
66  mDecoders.emplace(solarId, GBTDECODER(solarId, mChannelHandler));
67  c = mDecoders.find(solarId);
68  }
69  return c->second.append(buffer);
70 }
