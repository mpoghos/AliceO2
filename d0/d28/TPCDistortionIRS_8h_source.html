164 GPUdi() int TPCDistortionIRS::getDistortion(int slice, int row, float u, float v, float& dx, float& du, float& dv) const
165 {
166  const IrregularSpline2D3D& spline = getSpline(slice, row);
167  const float* splineData = getSplineData(slice, row);
168  float su = 0, sv = 0;
169  mGeo.convUVtoScaledUV(slice, row, u, v, su, sv);
170  spline.getSplineVec(splineData, su, sv, dx, du, dv);
171  return 0;
172 }
