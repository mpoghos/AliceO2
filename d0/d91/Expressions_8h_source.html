42 template <typename T>
43 constexpr auto selectArrowType()
44 {
45  if constexpr (std::is_same_v<T, int>) {
46  return atype::INT32;
47  } else if constexpr (std::is_same_v<T, bool>) {
48  return atype::BOOL;
49  } else if constexpr (std::is_same_v<T, float>) {
50  return atype::FLOAT;
51  } else if constexpr (std::is_same_v<T, double>) {
52  return atype::DOUBLE;
53  } else {
54  return atype::NA;
55  }
56  O2_BUILTIN_UNREACHABLE();
57 }
