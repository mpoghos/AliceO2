43 void GPUTPCCompression::SetPointersCompressedClusters(void*& mem, T& c, unsigned int nClA, unsigned int nTr, unsigned int nClU, bool reducedClA)
44 {
45  computePointerWithAlignment(mem, c.qTotU, nClU);
46  computePointerWithAlignment(mem, c.qMaxU, nClU);
47  computePointerWithAlignment(mem, c.flagsU, nClU);
48  computePointerWithAlignment(mem, c.padDiffU, nClU);
49  computePointerWithAlignment(mem, c.timeDiffU, nClU);
50  computePointerWithAlignment(mem, c.sigmaPadU, nClU);
51  computePointerWithAlignment(mem, c.sigmaTimeU, nClU);
52  computePointerWithAlignment(mem, c.nSliceRowClusters, GPUCA_ROW_COUNT * NSLICES);
53 
54  unsigned int nClAreduced = reducedClA ? nClA - nTr : nClA;
55 
56  if (!(mMerger->Param().rec.tpcCompressionModes & GPUSettings::CompressionTrackModel)) {
57  return; // Track model disabled, do not allocate memory
58  }
59  computePointerWithAlignment(mem, c.qTotA, nClA);
60  computePointerWithAlignment(mem, c.qMaxA, nClA);
61  computePointerWithAlignment(mem, c.flagsA, nClA);
62  computePointerWithAlignment(mem, c.rowDiffA, nClAreduced);
63  computePointerWithAlignment(mem, c.sliceLegDiffA, nClAreduced);
64  computePointerWithAlignment(mem, c.padResA, nClAreduced);
65  computePointerWithAlignment(mem, c.timeResA, nClAreduced);
66  computePointerWithAlignment(mem, c.sigmaPadA, nClA);
67  computePointerWithAlignment(mem, c.sigmaTimeA, nClA);
68 
69  computePointerWithAlignment(mem, c.qPtA, nTr);
70  computePointerWithAlignment(mem, c.rowA, nTr);
71  computePointerWithAlignment(mem, c.sliceA, nTr);
72  computePointerWithAlignment(mem, c.timeA, nTr);
73  computePointerWithAlignment(mem, c.padA, nTr);
74 
75  computePointerWithAlignment(mem, c.nTrackClusters, nTr);
76 }
