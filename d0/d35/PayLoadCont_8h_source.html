189  void moveUnusedToHead()
190  {
191  auto left = getUnusedSize();
192  if (left < getOffset()) {
193  std::memcpy(mBuffer.data(), mPtr, left); // there is no overlap
194  } else {
195  std::memmove(mBuffer.data(), mPtr, left); // there is an overlap
196  }
197  mPtr = mBuffer.data();
198  mEnd = mPtr + left;
199  }

202  // (attemtint to use all free capacity) using the method provided via getNext
203  size_t append(std::function<size_t(uint8_t*, size_t)> getNext)
204  {

206  auto nRead = getNext(mEnd, getFreeCapacity());

209  }
