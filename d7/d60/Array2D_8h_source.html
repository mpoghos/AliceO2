42  GPUdi() size_t idx(const ChargePos& p) const
43  {
44  const size_t widthInTiles = (TPC_NUM_OF_PADS + Width - 1) / Width;
45 
46  const size_t tilePad = p.gpad / Width;
47  const size_t tileTime = p.time / Height;
48 
49  const size_t inTilePad = p.gpad % Width;
50  const size_t inTileTime = p.time % Height;
51 
52  return (tileTime * widthInTiles + tilePad) * (Width * Height) + inTileTime * Width + inTilePad;
53  }
54 };
