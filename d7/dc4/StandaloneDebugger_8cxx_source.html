40 void StandaloneDebugger::fillCombinatoricsTree(std::array<std::vector<Cluster>, constants::its::LayersNumberVertexer>& clusters,
41  std::vector<Tracklet> comb01,
42  std::vector<Tracklet> comb12,
43  const ROframe* event)
44 {
45  assert(event != nullptr);
46  assert(mTreeStream != nullptr);
47 
48  for (auto& combination : comb01) {
49  o2::MCCompLabel lblClus0 = event->getClusterLabels(0, clusters[0][combination.firstClusterIndex].clusterId);
50  o2::MCCompLabel lblClus1 = event->getClusterLabels(1, clusters[1][combination.secondClusterIndex].clusterId);
51  float c0z{clusters[0][combination.firstClusterIndex].zCoordinate};
52  float c1z{clusters[1][combination.secondClusterIndex].zCoordinate};
53  unsigned char isValidated{lblClus0.compare(lblClus1) == 1};
54  (*mTreeStream)
55  << "combinatorics01"
56  << "tanLambda=" << combination.tanLambda
57  << "phi=" << combination.phiCoordinate
58  << "c0z=" << c0z
59  << "c1z=" << c1z
60  << "isValidated=" << isValidated
61  << "lblClus0=" << lblClus0
62  << "lblClus1=" << lblClus1
63  << "\n";
64  }
65 
66  for (auto& combination : comb12) {
67  o2::MCCompLabel lblClus1 = event->getClusterLabels(1, clusters[1][combination.secondClusterIndex].clusterId);
68  o2::MCCompLabel lblClus2 = event->getClusterLabels(2, clusters[2][combination.secondClusterIndex].clusterId);
69  float c1z{clusters[1][combination.firstClusterIndex].zCoordinate};
70  float c2z{clusters[2][combination.secondClusterIndex].zCoordinate};
71  unsigned char isValidated{lblClus1.compare(lblClus2) == 1};
72  (*mTreeStream)
73  << "combinatorics12"
74  << "tanLambda=" << combination.tanLambda
75  << "phi=" << combination.phiCoordinate
76  << "c1z=" << c1z
77  << "c2z=" << c2z
78  << "isValidated=" << isValidated
79  << "lblClus1=" << lblClus1
80  << "lblClus2=" << lblClus2
81  << "\n";
82  }

85 void StandaloneDebugger::fillTrackletSelectionTree(std::array<std::vector<Cluster>, constants::its::LayersNumberVertexer>& clusters,
86  std::vector<Tracklet> comb01,
87  std::vector<Tracklet> comb12,
88  std::vector<std::array<int, 2>> allowedTracklets,
89  const ROframe* event)
90 {
91  assert(event != nullptr);
92  assert(mTreeStream != nullptr);
93  int id = event->getROFrameId();
94  for (auto& trackletPair : allowedTracklets) {
95  o2::MCCompLabel lblClus0 = event->getClusterLabels(0, clusters[0][comb01[trackletPair[0]].firstClusterIndex].clusterId);
96  o2::MCCompLabel lblClus1 = event->getClusterLabels(1, clusters[1][comb01[trackletPair[0]].secondClusterIndex].clusterId);
97  o2::MCCompLabel lblClus2 = event->getClusterLabels(2, clusters[2][comb12[trackletPair[1]].secondClusterIndex].clusterId);
98  unsigned char isValidated{lblClus0.compare(lblClus1) == 1 && lblClus0.compare(lblClus2) == 1};
99  float deltaPhi{comb01[trackletPair[0]].phiCoordinate - comb12[trackletPair[1]].phiCoordinate};
100  float deltaTanLambda{comb01[trackletPair[0]].tanLambda - comb12[trackletPair[1]].tanLambda};
101  mTreeStream->GetDirectory()->cd(); // in case of existing other open files
102  (*mTreeStream)
103  << "selectedTracklets"
104  << "ROframeId=" << id
105  << "deltaTanlambda=" << deltaTanLambda
106  << "deltaPhi=" << deltaPhi
107  << "isValidated=" << isValidated
108  << "cluster0z=" << clusters[0][comb01[trackletPair[0]].firstClusterIndex].zCoordinate
109  << "cluster0r=" << clusters[0][comb01[trackletPair[0]].firstClusterIndex].rCoordinate
110  << "cluster0phi=" << clusters[0][comb01[trackletPair[0]].firstClusterIndex].phiCoordinate
111  << "cluster1z=" << clusters[1][comb01[trackletPair[0]].secondClusterIndex].zCoordinate
112  << "cluster1r=" << clusters[1][comb01[trackletPair[0]].secondClusterIndex].rCoordinate
113  << "cluster1phi=" << clusters[1][comb01[trackletPair[0]].secondClusterIndex].phiCoordinate
114  << "cluster2z=" << clusters[2][comb12[trackletPair[1]].secondClusterIndex].zCoordinate
115  << "cluster2r=" << clusters[2][comb12[trackletPair[1]].secondClusterIndex].rCoordinate
116  << "lblClus0=" << lblClus0
117  << "lblClus1=" << lblClus1
118  << "lblClus2=" << lblClus2
119  << "\n";
120  }
121 }

145 void StandaloneDebugger::fillPairsInfoTree(std::vector<Line> lines, const ROframe* event)
146 {
147  assert(event != nullptr);
148  int id = event->getROFrameId();
149  for (unsigned int iLine1{0}; iLine1 < lines.size(); ++iLine1) { // compute centroids for every line pair
150  auto line1 = lines[iLine1];
151  for (unsigned int iLine2{iLine1 + 1}; iLine2 < lines.size(); ++iLine2) {
152  auto line2 = lines[iLine2];
153  ClusterLines cluster{-1, line1, -1, line2};
154  auto vtx = cluster.getVertex();
155  if (std::hypot(vtx[0], vtx[1]) < 1.98 * 1.98) {
156  float dcaPair = Line::getDCA(line1, line2);
157  (*mTreeStream)
158  << "pairInfo"
159  << "centroids"
160  << "ROframeId=" << id
161  << "xCoord=" << vtx[0]
162  << "yCoord=" << vtx[1]
163  << "zCoord=" << vtx[2]
164  << "DCApair=" << dcaPair
165  << "\n";
166  }
167  }
168  // TODO: get primary vertex montecarlo position
169  // mLinesData.push_back(Line::getDCAComponents(line1, std::array<float, 3>{0., 0., 0.}));
170  }
