47 void RawReaderTask::run(ProcessingContext& pc)
48 {
49  LOG(DEBUG) << "RawReader run";
50 
52  if (mStatus) {
53  pc.services().get<ControlService>().endOfStream();
54  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
55  return;
56  }
57 
59  int headerSize = sizeof(o2::header::RAWDataHeader);
60  char* inputPointer = mBuffer.data();
61  mFile.read(inputPointer, headerSize);
62  inputPointer += headerSize;
63  auto rdh = reinterpret_cast<o2::header::RAWDataHeader*>(inputPointer);
64  while (!rdh->stop) {
65  auto dataSize = rdh->offsetToNext - headerSize;
66  mFile.read(inputPointer, dataSize);
67  inputPointer += dataSize;
68  mFile.read(inputPointer, headerSize);
69  rdh = reinterpret_cast<o2::header::RAWDataHeader*>(inputPointer);
70  inputPointer += headerSize;
71  }
72  mBuffer.resize(inputPointer - mBuffer.data());
73 
74  auto freefct = [](void* data, void* hint) {}; // simply ignore the cleanup for the test
75  pc.outputs().adoptChunk(Output{o2::header::gDataOriginTOF, "RAWDATAFRAME", 0, Lifetime::Timeframe}, mBuffer.data(), mBuffer.size(), freefct, nullptr);
76 
78  if (mFile.eof()) {
79  LOG(WARNING) << "nothig else to read";
80  mFile.close();
81  mStatus = true;
82  }
83 }
