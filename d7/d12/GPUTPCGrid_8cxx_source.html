38 GPUd() void GPUTPCGrid::Create(float yMin, float yMax, float zMin, float zMax, float sy, float sz)
39 {
40  //* Create the grid
41  mYMin = yMin;
42  mZMin = zMin;
43 
44  mStepYInv = 1.f / sy;
45  mStepZInv = 1.f / sz;
46 
47  mNy = static_cast<unsigned int>((yMax - mYMin) * mStepYInv + 1.f);
48  mNz = static_cast<unsigned int>((zMax - mZMin) * mStepZInv + 1.f);
49 
50  mN = mNy * mNz;
51 
52  mYMax = mYMin + mNy * sy;
53  mZMax = mZMin + mNz * sz;
54 }

56 GPUd() int GPUTPCGrid::GetBin(float Y, float Z) const
57 {
58  //* get the bin pointer
59  const int yBin = static_cast<int>((Y - mYMin) * mStepYInv);
60  const int zBin = static_cast<int>((Z - mZMin) * mStepZInv);
61  const int bin = zBin * mNy + yBin;
62 #ifndef GPUCA_GPUCODE
63  assert(bin >= 0);
64  assert(bin < static_cast<int>(mN));
65 #endif
66  return bin;

69 GPUd() int GPUTPCGrid::GetBinBounded(float Y, float Z) const

105 GPUd() void GPUTPCGrid::GetBinArea(float Y, float Z, float dy, float dz, int& bin, int& ny, int& nz) const
106 {
107  Y -= mYMin;
108  int by = (int)((Y - dy) * mStepYInv);
109  ny = (int)((Y + dy) * mStepYInv) - by;
110  Z -= mZMin;
111  int bz = (int)((Z - dz) * mStepZInv);
112  nz = (int)((Z + dz) * mStepZInv) - bz;
113  if (by < 0) {
114  by = 0;
115  } else if (by >= (int)mNy) {
116  by = mNy - 1;
117  }
118  if (bz < 0) {
119  bz = 0;
120  } else if (bz >= (int)mNz) {
121  bz = mNz - 1;
122  }
123  if (by + ny >= (int)mNy) {
124  ny = mNy - 1 - by;
125  }
126  if (bz + nz >= (int)mNz) {
127  nz = mNz - 1 - bz;
128  }
129  bin = bz * mNy + by;
130 }
