40 void ClusterizerSpec::run(framework::ProcessingContext& ctx)
41 {
42  LOG(DEBUG) << "[EMCALClusterizer - run] called";
43  auto dataref = ctx.inputs().get("digits");
44  auto const* emcheader = o2::framework::DataRefUtils::getHeader<o2::emcal::EMCALBlockHeader*>(dataref);
45  if (!emcheader->mHasPayload) {
46  LOG(DEBUG) << "[EMCALClusterizer - run] No more digits" << std::endl;
47  ctx.services().get<o2::framework::ControlService>().readyToQuit(framework::QuitRequest::Me);
48  return;
49  }
50 
51  auto digits = ctx.inputs().get<std::vector<o2::emcal::Digit>>("digits");
52  LOG(DEBUG) << "[EMCALClusterizer - run] Received " << digits.size() << " digits, running clusterizer ...";
53 
54  mClusterizer.findClusters(digits); // Find clusters on digits (pass by ref)
55 
56  // Get found clusters + digit indices for output
57  // * A cluster contains a range that correspond to the vector of digit indices
58  // * The digit index vector contains the indices of the clusterized digits wrt to the original digit array
59  mOutputClusters = mClusterizer.getFoundClusters();
60  mOutputDigitIndices = mClusterizer.getFoundClustersDigitIndices();
61 
62  LOG(DEBUG) << "[EMCALClusterizer - run] Writing " << mOutputClusters->size() << " clusters ...";
63  ctx.outputs().snapshot(o2::framework::Output{o2::header::gDataOriginEMC, "CLUSTERS", 0, o2::framework::Lifetime::Timeframe}, *mOutputClusters);
64  ctx.outputs().snapshot(o2::framework::Output{o2::header::gDataOriginEMC, "INDICES", 0, o2::framework::Lifetime::Timeframe}, *mOutputDigitIndices);
65 }
