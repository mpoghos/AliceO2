51 template <typename T, typename ARRAY>
52 Status doGrouping(std::shared_ptr<ChunkedArray> chunkedArray, Datum* outputRanges)
53 {
54  TableBuilder builder;
55  auto writer = builder.persist<uint64_t, uint64_t, uint64_t>({"start", "count", "index"});
56  auto chunk = std::static_pointer_cast<ARRAY>(chunkedArray->chunk(0));
57  if (chunk->length() == 0) {
58  *outputRanges = std::move(builder.finalize());
59  return arrow::Status::OK();
60  }
61  T currentIndex = chunk->raw_values()[0];
62  T currentCount = 0;
63  T currentOffset = 0;
64  for (size_t ci = 0; ci < chunkedArray->num_chunks(); ++ci) {
65  auto chunk = chunkedArray->chunk(ci);
66  T const* data = std::static_pointer_cast<ARRAY>(chunk)->raw_values();
67  for (size_t ai = 0; ai < chunk->length(); ++ai) {
68  if (currentIndex == data[ai]) {
69  currentCount++;
70  } else if (currentIndex != -1) {
71  writer(0, currentOffset, currentCount, currentIndex);
72  currentIndex = data[ai];
73  currentOffset += currentCount;
74  currentCount = 1;
75  }
76  }
77  }
78  writer(0, currentOffset, currentCount, currentIndex);
79  *outputRanges = std::move(builder.finalize());
80  return arrow::Status::OK();
81 }

83 Status SortedGroupByKernel::Call(FunctionContext* ctx, Datum const& inputTable, Datum* outputRanges)
84 {
85  using namespace arrow;
86  if (inputTable.kind() == Datum::TABLE) {
87  auto table = util::get<std::shared_ptr<arrow::Table>>(inputTable.value);
88  auto columnIndex = table->schema()->GetFieldIndex(mOptions.columnName);
89  auto dataType = table->column(columnIndex)->type();
90  auto chunkedArray = table->column(columnIndex)->data();
91  if (dataType->id() == Type::UINT64) {
92  return doGrouping<uint64_t, arrow::UInt64Array>(chunkedArray, outputRanges);
93  } else if (dataType->id() == Type::INT64) {
94  return doGrouping<int64_t, arrow::Int64Array>(chunkedArray, outputRanges);
95  } else if (dataType->id() == Type::UINT32) {
96  return doGrouping<uint32_t, arrow::UInt32Array>(chunkedArray, outputRanges);
97  } else if (dataType->id() == Type::INT32) {
98  return doGrouping<int32_t, arrow::Int32Array>(chunkedArray, outputRanges);
99  }
100  }
101  return arrow::Status::OK();
102 }

105 arrow::Status sliceByColumn(FunctionContext* context, std::string const& key,
106  Datum const& inputTable, std::vector<Datum>* outputSlices,
107  std::vector<uint64_t>* offsets)
108 {
109  if (inputTable.kind() != Datum::TABLE) {
110  return Status::Invalid("Input Datum was not a table");
111  }
112 
113  // build all the ranges on the fly.
114  Datum outRanges;
115  SortedGroupByKernel groupBy({GroupByOptions{key}});
116  ARROW_RETURN_NOT_OK(groupBy.Call(context, inputTable, &outRanges));
117  auto ranges = util::get<std::shared_ptr<arrow::Table>>(outRanges.value);
118  outputSlices->reserve(ranges->num_rows());
119  if (offsets) {
120  offsets->reserve(ranges->num_rows());
121  }
122 
123  auto startChunks = ranges->column(0)->data();
124  assert(startChunks->num_chunks() == 1);
125  auto countChunks = ranges->column(1)->data();
126  assert(countChunks->num_chunks() == 1);
127  auto startData = std::static_pointer_cast<UInt64Array>(startChunks->chunk(0))->raw_values();
128  auto countData = std::static_pointer_cast<UInt64Array>(countChunks->chunk(0))->raw_values();
129 
130  auto table = arrow::util::get<std::shared_ptr<arrow::Table>>(inputTable.value);
131  for (size_t ri = 0; ri < ranges->num_rows(); ++ri) {
132  auto start = startData[ri];
133  auto count = countData[ri];
134  auto schema = table->schema();
135  std::vector<std::shared_ptr<Column>> slicedColumns;
136  slicedColumns.reserve(schema->num_fields());
137  for (size_t ci = 0; ci < schema->num_fields(); ++ci) {
138  slicedColumns.emplace_back(table->column(ci)->Slice(start, count));
139  }
140  outputSlices->emplace_back(Datum(arrow::Table::Make(table->schema(), slicedColumns)));
141  if (offsets) {
142  offsets->emplace_back(start);
143  }
144  }
145  return arrow::Status::OK();
146 }
