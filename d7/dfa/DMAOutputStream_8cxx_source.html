33 void DMAOutputStream::writeData(RawHeader header, gsl::span<char> buffer)
34 {
35  if (!mInitialized)
36  open();
37 
38  constexpr int PAGESIZE = 8192;
39  // Handling of the termination word (0x001d3082): The termination word is added to the payload
40  // but not included in the payload size (as done in the hardware). Therefore it has to be subtracted
41  // from the maximum possible payload size
42  constexpr int MAXNWORDS = PAGESIZE - sizeof(header) - sizeof(uint32_t);
43  bool writeNext = true;
44  int pagecounter = 0, currentindex = 0;
45  while (writeNext) {
46  int sizeRemain = buffer.size() - currentindex;
47  int nwordspage = MAXNWORDS;
48  if (sizeRemain < MAXNWORDS) {
49  // Last page
50  nwordspage = sizeRemain;
51  writeNext = false;
52  header.stop = true;
53  }
54  header.packetCounter = pagecounter;
55  header.memorySize = nwordspage + sizeof(RawHeader);
56  header.offsetToNext = 8192;
57 
58  writeDMAPage(header, gsl::span(buffer.data() + currentindex, nwordspage), PAGESIZE);
59 
60  if (writeNext) {
61  currentindex += nwordspage;
62  pagecounter++;
63  }
64  }
65 }

67 void DMAOutputStream::writeDMAPage(const RawHeader& header, gsl::span<char> payload, int pagesize)
68 {
69  std::vector<char> dmapage(pagesize);
70  o2::header::RAWDataHeaderV4* outheader = reinterpret_cast<o2::header::RAWDataHeaderV4*>(dmapage.data());
71  *outheader = header;
72  char* outpayload = dmapage.data() + sizeof(header);
73  memcpy(outpayload, payload.data(), payload.size());
