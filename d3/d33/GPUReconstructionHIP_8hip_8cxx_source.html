86 template <class T, int I, typename... Args>
87 int GPUReconstructionHIPBackend::runKernelBackend(krnlSetup& _xyz, Args... args)
88 {
89  auto& x = _xyz.x;
90  auto& z = _xyz.z;
91  if (z.evList) {
92  for (int k = 0; k < z.nEvents; k++) {
93  GPUFailedMsg(hipStreamWaitEvent(mInternals->HIPStreams[x.stream], ((hipEvent_t*)z.evList)[k], 0));
94  }
95  }
96  hipEvent_t start, stop;
97  if (mDeviceProcessingSettings.deviceTimers) {
98  GPUFailedMsg(hipEventCreate(&start));
99  GPUFailedMsg(hipEventCreate(&stop));
100  GPUFailedMsg(hipEventRecord(start, mInternals->HIPStreams[x.stream]));
101  }
102  backendInternal<T, I>::runKernelBackendInternal(_xyz, this, args...);
103  if (mDeviceProcessingSettings.deviceTimers) {
104  GPUFailedMsg(hipEventRecord(stop, mInternals->HIPStreams[x.stream]));
105  GPUFailedMsg(hipEventSynchronize(stop));
106  float v;
107  GPUFailedMsg(hipEventElapsedTime(&v, start, stop));
108  _xyz.t = v * 1.e-3;
109  GPUFailedMsg(hipEventDestroy(start));
110  GPUFailedMsg(hipEventDestroy(stop));
111  }
112  GPUFailedMsg(hipGetLastError());
113  if (z.ev) {
114  GPUFailedMsg(hipEventRecord(*(hipEvent_t*)z.ev, mInternals->HIPStreams[x.stream]));
115  }
116  return 0;
117 }

143 int GPUReconstructionHIPBackend::InitDevice_Runtime()
144 {
145  // Find best HIP device, initialize and allocate memory
146 
147  hipDeviceProp_t hipDeviceProp_t;
148 
149  int count, bestDevice = -1;
150  double bestDeviceSpeed = -1, deviceSpeed;
151  if (GPUFailedMsgI(hipGetDeviceCount(&count))) {
152  GPUError("Error getting HIP Device Count");
153  return (1);
154  }
155  if (mDeviceProcessingSettings.debugLevel >= 2) {
156  GPUInfo("Available HIP devices:");
157  }
158  const int reqVerMaj = 2;
159  const int reqVerMin = 0;
160  std::vector<bool> devicesOK(count, false);
161  for (int i = 0; i < count; i++) {
162  if (mDeviceProcessingSettings.debugLevel >= 4) {
163  GPUInfo("Examining device %d", i);
164  }
165  if (mDeviceProcessingSettings.debugLevel >= 4) {
166  GPUInfo("Obtained current memory usage for device %d", i);
167  }
168  if (GPUFailedMsgI(hipGetDeviceProperties(&hipDeviceProp_t, i))) {
169  continue;
170  }
171  if (mDeviceProcessingSettings.debugLevel >= 4) {
172  GPUInfo("Obtained device properties for device %d", i);
173  }
174  int deviceOK = true;
175  const char* deviceFailure = "";
176  if (hipDeviceProp_t.major >= 9) {
177  deviceOK = false;
178  deviceFailure = "Invalid Revision";
179  } else if (hipDeviceProp_t.major < reqVerMaj || (hipDeviceProp_t.major == reqVerMaj && hipDeviceProp_t.minor < reqVerMin)) {
180  deviceOK = false;
181  deviceFailure = "Too low device revision";
182  }
183 
184  deviceSpeed = (double)hipDeviceProp_t.multiProcessorCount * (double)hipDeviceProp_t.clockRate * (double)hipDeviceProp_t.warpSize * (double)hipDeviceProp_t.major * (double)hipDeviceProp_t.major;
185  if (mDeviceProcessingSettings.debugLevel >= 2) {
186  GPUImportant("Device %s%2d: %s (Rev: %d.%d - Mem %lld)%s %s", deviceOK ? " " : "[", i, hipDeviceProp_t.name, hipDeviceProp_t.major, hipDeviceProp_t.minor, (long long int)hipDeviceProp_t.totalGlobalMem, deviceOK ? " " : " ]", deviceOK ? "" : deviceFailure);
187  }
188  if (!deviceOK) {
189  continue;
190  }
191  devicesOK[i] = true;
192  if (deviceSpeed > bestDeviceSpeed) {
193  bestDevice = i;
194  bestDeviceSpeed = deviceSpeed;
195  } else {
196  if (mDeviceProcessingSettings.debugLevel >= 2) {
197  GPUInfo("Skipping: Speed %f < %f\n", deviceSpeed, bestDeviceSpeed);
198  }
199  }
200  }
201  if (bestDevice == -1) {
202  GPUWarning("No %sHIP Device available, aborting HIP Initialisation", count ? "appropriate " : "");
203  GPUImportant("Requiring Revision %d.%d, Mem: %lld", reqVerMaj, reqVerMin, (long long int)mDeviceMemorySize);
204  return (1);
205  }
206 
207  if (mDeviceProcessingSettings.deviceNum > -1) {
208  if (mDeviceProcessingSettings.deviceNum >= (signed)count) {
209  GPUWarning("Requested device ID %d does not exist", mDeviceProcessingSettings.deviceNum);
210  return (1);
211  } else if (!devicesOK[mDeviceProcessingSettings.deviceNum]) {
212  GPUWarning("Unsupported device requested (%d)", mDeviceProcessingSettings.deviceNum);
213  return (1);
214  } else {
215  bestDevice = mDeviceProcessingSettings.deviceNum;
216  }
217  }
218  mDeviceId = bestDevice;
219 
220  GPUFailedMsgI(hipGetDeviceProperties(&hipDeviceProp_t, mDeviceId));
221 
222  if (mDeviceProcessingSettings.debugLevel >= 2) {
223  GPUInfo("Using HIP Device %s with Properties:", hipDeviceProp_t.name);
224  GPUInfo("\ttotalGlobalMem = %lld", (unsigned long long int)hipDeviceProp_t.totalGlobalMem);
225  GPUInfo("\tsharedMemPerBlock = %lld", (unsigned long long int)hipDeviceProp_t.sharedMemPerBlock);
226  GPUInfo("\tregsPerBlock = %d", hipDeviceProp_t.regsPerBlock);
227  GPUInfo("\twarpSize = %d", hipDeviceProp_t.warpSize);
228  GPUInfo("\tmaxThreadsPerBlock = %d", hipDeviceProp_t.maxThreadsPerBlock);
229  GPUInfo("\tmaxThreadsDim = %d %d %d", hipDeviceProp_t.maxThreadsDim[0], hipDeviceProp_t.maxThreadsDim[1], hipDeviceProp_t.maxThreadsDim[2]);
230  GPUInfo("\tmaxGridSize = %d %d %d", hipDeviceProp_t.maxGridSize[0], hipDeviceProp_t.maxGridSize[1], hipDeviceProp_t.maxGridSize[2]);
231  GPUInfo("\ttotalConstMem = %lld", (unsigned long long int)hipDeviceProp_t.totalConstMem);
232  GPUInfo("\tmajor = %d", hipDeviceProp_t.major);
233  GPUInfo("\tminor = %d", hipDeviceProp_t.minor);
234  GPUInfo("\tclockRate = %d", hipDeviceProp_t.clockRate);
235  GPUInfo("\tmemoryClockRate = %d", hipDeviceProp_t.memoryClockRate);
236  GPUInfo("\tmultiProcessorCount = %d", hipDeviceProp_t.multiProcessorCount);
237  GPUInfo(" ");
238  }
239  mCoreCount = hipDeviceProp_t.multiProcessorCount;
240  mDeviceName = hipDeviceProp_t.name;
241  mDeviceName += " (HIP GPU)";
242 
243  if (hipDeviceProp_t.major < 3) {
244  GPUError("Unsupported HIP Device");
245  return (1);
246  }
247 #ifndef GPUCA_HIP_NO_CONSTANT_MEMORY
248  if (gGPUConstantMemBufferSize > hipDeviceProp_t.totalConstMem) {
249  GPUError("Insufficient constant memory available on GPU %d < %d!", (int)hipDeviceProp_t.totalConstMem, (int)gGPUConstantMemBufferSize);
250  return (1);
251  }
252 #endif
253 
254  mNStreams = std::max(mDeviceProcessingSettings.nStreams, 3);
255 
256  /*if (GPUFailedMsgI(hipDeviceSetLimit(hipLimitStackSize, GPUCA_GPU_STACK_SIZE)))
257  {
258  GPUError("Error setting HIP stack size");
259  GPUFailedMsgI(hipDeviceReset());
260  return(1);
261  }*/
262 
263  if (mDeviceMemorySize > hipDeviceProp_t.totalGlobalMem) {
264  GPUError("Insufficient GPU memory (%lld < %lld)", (long long int)hipDeviceProp_t.totalGlobalMem, (long long int)mDeviceMemorySize);
265  GPUFailedMsgI(hipDeviceReset());
266  return (1);
267  }
268  if (GPUFailedMsgI(hipMalloc(&mDeviceMemoryBase, mDeviceMemorySize))) {
269  GPUError("HIP Memory Allocation Error");
270  GPUFailedMsgI(hipDeviceReset());
271  return (1);
272  }
273  if (mDeviceProcessingSettings.debugLevel >= 1) {
274  GPUInfo("GPU Memory used: %lld (Ptr 0x%p)", (long long int)mDeviceMemorySize, mDeviceMemoryBase);
275  }
276  if (GPUFailedMsgI(hipHostMalloc(&mHostMemoryBase, mHostMemorySize))) {
277  GPUError("Error allocating Page Locked Host Memory");
278  GPUFailedMsgI(hipDeviceReset());
279  return (1);
280  }
281  if (mDeviceProcessingSettings.debugLevel >= 1) {
282  GPUInfo("Host Memory used: %lld (Ptr 0x%p)", (long long int)mHostMemorySize, mHostMemoryBase);
283  }
284 
285  if (mDeviceProcessingSettings.debugLevel >= 1) {
286  memset(mHostMemoryBase, 0, mHostMemorySize);
287  if (GPUFailedMsgI(hipMemset(mDeviceMemoryBase, 0xDD, mDeviceMemorySize))) {
288  GPUError("Error during HIP memset, trying workaround with kernel");
289  hipLaunchKernelGGL(HIP_KERNEL_NAME(gHIPMemSetWorkaround), dim3(mCoreCount), dim3(256), 0, 0, (char*)mDeviceMemoryBase, 0xDD, mDeviceMemorySize);
290  if (GPUFailedMsgI(hipGetLastError()) || GPUFailedMsgI(hipDeviceSynchronize())) {
291  GPUError("Error during HIP memset");
292  GPUFailedMsgI(hipDeviceReset());
293  return (1);
294  }
295  }
296  }
297 
298  for (int i = 0; i < mNStreams; i++) {
299  if (GPUFailedMsgI(hipStreamCreate(&mInternals->HIPStreams[i]))) {
300  GPUError("Error creating HIP Stream");
301  GPUFailedMsgI(hipDeviceReset());
302  return (1);
303  }
304  }
305 
306  void* devPtrConstantMem;
307 #ifndef GPUCA_HIP_NO_CONSTANT_MEMORY
308  if (GPUFailedMsgI(hipGetSymbolAddress(&devPtrConstantMem, HIP_SYMBOL(gGPUConstantMemBuffer)))) {
309  GPUError("Error getting ptr to constant memory");
310  GPUFailedMsgI(hipDeviceReset());
311  return 1;
312  }
313 #else
314  if (GPUFailedMsgI(hipMalloc(&devPtrConstantMem, gGPUConstantMemBufferSize))) {
315  GPUError("HIP Memory Allocation Error");
316  GPUFailedMsgI(hipDeviceReset());
317  return (1);
318  }
319 #endif
320  mDeviceConstantMem = (GPUConstantMem*)devPtrConstantMem;
321 
322  for (unsigned int i = 0; i < mEvents.size(); i++) {
323  hipEvent_t* events = (hipEvent_t*)mEvents[i].data();
324  for (unsigned int j = 0; j < mEvents[i].size(); j++) {
325  if (GPUFailedMsgI(hipEventCreate(&events[j]))) {
326  GPUError("Error creating event");
327  GPUFailedMsgI(hipDeviceReset());
328  return 1;
329  }
330  }
331  }
332 
333  GPUInfo("HIP Initialisation successfull (Device %d: %s (Frequency %d, Cores %d), %lld / %lld bytes host / global memory, Stack frame %d, Constant memory %lld)", mDeviceId, hipDeviceProp_t.name, hipDeviceProp_t.clockRate, hipDeviceProp_t.multiProcessorCount, (long long int)mHostMemorySize,
334  (long long int)mDeviceMemorySize, (int)GPUCA_GPU_STACK_SIZE, (long long int)gGPUConstantMemBufferSize);
335 
336  return (0);
337 }

368

373 size_t GPUReconstructionHIPBackend::GPUMemCpy(void* dst, const void* src, size_t size, int stream, bool toGPU, deviceEvent* ev, deviceEvent* evList, int nEvents)
374 {
375  if (mDeviceProcessingSettings.debugLevel >= 3) {
376  stream = -1;
377  }
378  if (stream == -1) {
379  SynchronizeGPU();
380  GPUFailedMsg(hipMemcpy(dst, src, size, toGPU ? hipMemcpyHostToDevice : hipMemcpyDeviceToHost));
381  } else {
382  if (evList == nullptr) {
383  nEvents = 0;
384  }
385  for (int k = 0; k < nEvents; k++) {
386  GPUFailedMsg(hipStreamWaitEvent(mInternals->HIPStreams[stream], ((hipEvent_t*)evList)[k], 0));
387  }
388  GPUFailedMsg(hipMemcpyAsync(dst, src, size, toGPU ? hipMemcpyHostToDevice : hipMemcpyDeviceToHost, mInternals->HIPStreams[stream]));
389  }
390  if (ev) {
391  GPUFailedMsg(hipEventRecord(*(hipEvent_t*)ev, mInternals->HIPStreams[stream == -1 ? 0 : stream]));
392  }
393  return size;
394 }
