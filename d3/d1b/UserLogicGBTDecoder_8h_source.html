84 template <typename CHARGESUM>
85 size_t UserLogicGBTDecoder<CHARGESUM>::append(gsl::span<uint8_t> buffer)
86 {
87  if (buffer.size() % 8) {
88  throw std::invalid_argument("buffer size should be a multiple of 8");
89  }
90  size_t n{0};
91 
92  for (size_t i = 0; i < buffer.size(); i += 8) {
93 
94  uint64_t word = (static_cast<uint64_t>(buffer[i + 0])) |
95  (static_cast<uint64_t>(buffer[i + 1]) << 8) |
96  (static_cast<uint64_t>(buffer[i + 2]) << 16) |
97  (static_cast<uint64_t>(buffer[i + 3]) << 24) |
98  (static_cast<uint64_t>(buffer[i + 4]) << 32) |
99  (static_cast<uint64_t>(buffer[i + 5]) << 40) |
100  (static_cast<uint64_t>(buffer[i + 6]) << 48) |
101  (static_cast<uint64_t>(buffer[i + 7]) << 56);
102 
103  if (word == 0) {
104  continue;
105  }
106  if (word == 0xFEEDDEEDFEEDDEED) {
107  continue;
108  }
109  int gbt = (word >> 59) & 0x1F;
110  if (gbt != mSolarId) {
111  std::cout << fmt::format("warning : solarId {} != expected {} word={:08X}\n", gbt, mSolarId, word);
112  // throw std::invalid_argument(fmt::format("gbt {} != expected {} word={:X}\n", gbt, mGbtId, word));
113  }
114 
115  uint16_t dsid = (word >> 53) & 0x3F;
116  impl::assertIsInRange("dsid", dsid, 0, 39);
117 
118  // the remaining 50 bits are passed to the ElinkDecoder
119  uint64_t data = word & UINT64_C(0x003FFFFFFFFFFFFF);
120  mElinkDecoders.at(dsid).append(data);
121  n += 8;
122  }
123  return n;
124 }
