26 enum class OwnershipMode { // todo better name
27  Full, // Mergers expect full objects each time.
28  Integral // Mergers expect objects' differences (what has changed since the previous was sent).
29 };
30 
31 enum class MergingMode {
32  Binwise, // Bins of histograms are added, TTree branches are attached, objects inside TCollections are merged correspondingly.
33  Timewise, // NOT SUPPORTED YET. Arriving objects are merged into one T*****, ordered in time.
34  Concatenate, // Arriving objects are merged into one TObjArray, no particular order.
35 };
36 
37 enum class MergingTime {
38  AfterArrival, // Merging just after new object arrives.
39  WhenXInputsCached, // Merging when 0 < X < 1 of the inputs are cached.
40  // EachNSeconds, // NOT SUPPORTED. Merging the most up-to-date inputs each N seconds //todo: maybe later (dpl makes it hard to have two timers)
41  BeforePublication // Merging just before publication.
42 };
43 
44 enum class Timespan {
45  FullHistory, // Merged object should consist of all the partial data that Mergers received..
46  LastDifference, // Merged object should consist of only data received after last publication. Merged object is reset after published.
47  // MovingWindow // NOT SUPPORTED YET. Merged object should consist of object from last N seconds. Inherit MergeInterface and implement time().
48 };
49 
50 enum class PublicationDecision {
51  EachNSeconds, // Merged object is published each N seconds.
52  WhenXInputsUpdated, // Merged object is published when 0 < X < 1 of the inputs are updated since previous publication.
53  // AfterEachMerge // NOT SUPPORTED. todo
54 };
55 
56 enum class TopologySize {
57  NumberOfLayers, // User specifies the number of layers in topology.
58  ReductionFactor // User specifies how many sources should be handled by one merger (by maximum).
59 };
60 
61 enum class UnpackingMethod {
62  NoUnpackingNeeded, // Merger treats object as it is.
63  TCollection, // NOT SUPPORTED YET. Merger treats each object as TCollection and merges each member accordingly. todo
64 };
