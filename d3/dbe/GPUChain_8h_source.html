204 template <bool Always, class T, class S, typename... Args>
205 inline void GPUChain::timeCpy(RecoStep step, bool toGPU, S T::*func, Args... args)
206 {
207  if (!Always && step != RecoStep::NoRecoStep && !(GetRecoStepsGPU() & step)) {
208  return;
209  }
210  HighResTimer* timer = nullptr;
211  size_t* bytes = nullptr;
212  if (mRec->mDeviceProcessingSettings.debugLevel >= 1) {
213  int id = mRec->getRecoStepNum(step, false);
214  if (id != -1) {
215  auto& tmp = mRec->mTimersRecoSteps[id];
216  timer = toGPU ? &tmp.timerToGPU : &tmp.timerToHost;
217  bytes = toGPU ? &tmp.bytesToGPU : &tmp.bytesToHost;
218  (toGPU ? tmp.countToGPU : tmp.countToHost)++;
219  timer->Start();
220  }
221  }
222  size_t n = (mRec->*func)(args...);
223  if (timer) {
224  SynchronizeGPU();
225  timer->Stop();
226  *bytes += n;
227  }
228 }

230 template <class T, class S, typename... Args>
231 bool GPUChain::DoDebugAndDump(GPUChain::RecoStep step, int mask, T& processor, S T::*func, Args&... args)
232 {
233  if (GetDeviceProcessingSettings().keepAllMemory) {
234  TransferMemoryResourcesToHost(step, &processor, -1, true);
235  if (GetDeviceProcessingSettings().debugLevel >= 6 && (mask == 0 || (GetDeviceProcessingSettings().debugMask & mask))) {
236  (processor.*func)(args...);
237  return true;
238  }
239  }
240  return false;
241 }
