46 struct SimpleRawWriter { // simple class to create detector payload for multiple links
47 
48  // suppose detector puts in front and end of every trigger payload some header and trailer
49 
50  RawFileWriter writer;
51 
52  //_________________________________________________________________
53  void init()
54  {
55  // init writer
56 
57  // register links
58  for (int icru = 0; icru < NCRU; icru++) {
59  std::string outFileName = "testdata_cru" + std::to_string(icru) + ".raw";
60  for (int il = 0; il < NLinkPerCRU; il++) {
61  writer.registerLink((icru << 8) + il, icru, il, 0, outFileName);
62  }
63  }
64 
65  writer.setContinuousReadout(); // in case we want to issue StartOfContinuous trigger in the beginning
66  writer.setCarryOverCallBack(this); // we want that writer to ask the detector code how to split large payloads
67  writer.setEmptyPageCallBack(this); // we want the writer to ask the detector code what to put in empty HBFs
68  }
69 
70  //_________________________________________________________________
71  void run()
72  {
73  // write payload and close outputs
74 
75  // generate interaction records for triggers to write
76  std::vector<o2::InteractionTimeRecord> irs(1000);
77  o2::steer::InteractionSampler irSampler;
78  irSampler.setInteractionRate(12000); // ~1.5 interactions per orbit
79  irSampler.init();
80  irSampler.generateCollisionTimes(irs);
81 
82  std::vector<char> buffer;
83 
84  // create payload for every interaction and push it to writer
85  for (const auto& ir : irs) {
86  for (int icru = 0; icru < NCRU; icru++) {
87  // we will create non-0 payload for all but 1st link of every CRU, the writer should take care
88  // of creating empty HBFs for the links w/o data
89  for (int il = 0; il < NLinkPerCRU; il++) {
90  int nGBT = gRandom->Poisson(HBFUtils::MAXCRUPage / HBFUtils::GBTWord * (il));
91  if (nGBT) {
92  buffer.resize((nGBT + 2) * HBFUtils::GBTWord); // reserve 16B words accounting for the Header and Trailer
93  std::memcpy(buffer.data(), PLHeader.c_str(), HBFUtils::GBTWord);
94  std::memcpy(buffer.data() + buffer.size() - HBFUtils::GBTWord, PLTrailer.c_str(), HBFUtils::GBTWord);
95  // we don't care here about the content of the payload, except the presence of header and trailer
96  } else {
97  buffer.clear();
98  }
99  writer.addData(icru, il, 0, ir, buffer);
100  }
101  }
102  }
103 
104  // for further use we write the configuration file for the output
105  {
106  std::ofstream cfgfile;
107  cfgfile.open(CFGName);
108  cfgfile << "[defaults]" << std::endl;
109  cfgfile << "dataOrigin = FLP" << std::endl;
110  cfgfile << "dataDescription = RAWDATA" << std::endl;
111  for (int i = 0; i < writer.getNOutputFiles(); i++) {
112  cfgfile << "[input-cru" << i << "]" << std::endl;
113  cfgfile << "filePath = " << writer.getOutputFileName(i) << std::endl;
114  }
115  cfgfile.close();
116  }
117 
118  writer.close(); // flush buffers and close outputs
119  }
120 
121  // optional callback functions to register in the RawFileWriter
122  //_________________________________________________________________
123  void emptyHBFMethod(const RDH& rdh, std::vector<char>& toAdd) const
124  {
125  // what we want to add for every empty page
126  toAdd.resize(HBFUtils::GBTWord);
127  std::memcpy(toAdd.data(), HBFEmpty.c_str(), HBFUtils::GBTWord);
128  }
129 
130  //_________________________________________________________________
131  int carryOverMethod(const RDH& rdh, const gsl::span<char> data, const char* ptr, int maxSize, int splitID,
132  std::vector<char>& trailer, std::vector<char>& header) const
133  {
134  // how we want to split the large payloads. The data is the full payload which was sent for writing and
135  // it is already equiped with header and trailer
136  if (maxSize <= HBFUtils::GBTWord) { // do not carry over trailer or header only
137  return 0;
138  }
139 
140  // here we simply copy the header/trailer of the payload to every CRU page of this payload
141  header.resize(HBFUtils::GBTWord);
142  std::memcpy(header.data(), &data[0], HBFUtils::GBTWord);
143  trailer.resize(HBFUtils::GBTWord);
144  std::memcpy(trailer.data(), &data[data.size() - HBFUtils::GBTWord], HBFUtils::GBTWord);
145  // since we write an extra GBT word (trailer) in the end of the CRU page, we ask to write
146  // not the block ptr : ptr+maxSize, but ptr : ptr+maxSize - GBTWord;
147  int sz = maxSize - HBFUtils::GBTWord;
148  return sz;
149  }

152 struct SimpleRawReader { // simple class to read detector raw data for multiple links
153 
154  std::unique_ptr<RawFileReader> reader;
155 
156  //_________________________________________________________________
157  void init()
158  {
159  reader = std::make_unique<RawFileReader>(CFGName); // init from configuration file
160  uint32_t errCheck = 0xffffffff;
161  errCheck ^= 0x1 << RawFileReader::ErrNoSuperPageForTF; // makes no sense for superpages not interleaved by others
162  reader->setCheckErrors(errCheck);
163  reader->init();
164  }
165 
166  //_________________________________________________________________
167  void run()
168  {
169  // read data of all links
170  if (!reader) {
171  init();
172  }
173  int nLinks = reader->getNLinks();
174  BOOST_CHECK(nLinks == NCRU * NLinkPerCRU);
175 
176  // make sure no errors detected after initialization
177  for (int il = 0; il < nLinks; il++) {
178  auto& lnk = reader->getLink(il);
179  BOOST_CHECK(lnk.nErrors == 0);
180  }
181 
182  std::vector<std::vector<char>> buffers;
183  std::vector<bool> firstHBF;
184  std::string testStr;
185  testStr.resize(HBFUtils::GBTWord);
186  buffers.resize(nLinks); // 1 buffer per link
187  firstHBF.resize(nLinks, true);
188 
189  int nLinksRead = 0;
190  do {
191  nLinksRead = 0;
192  for (int il = 0; il < nLinks; il++) {
193  auto& buff = buffers[il];
194  buff.clear();
195  auto& lnk = reader->getLink(il);
196  auto sz = lnk.getNextHBFSize();
197  if (!sz) {
198  continue;
199  }
200  buff.resize(sz);
201  BOOST_CHECK(lnk.readNextHBF(buff.data()) == sz);
202  nLinksRead++;
203  }
204  if (nLinksRead) {
205  BOOST_CHECK(nLinksRead == nLinks); // all links should have the same number of HBFs
206 
207  const auto rdhRef = *reinterpret_cast<RDH*>(buffers[0].data());
208 
209  for (int il = 0; il < nLinks; il++) {
210  auto& buff = buffers[il];
211  int hbsize = buff.size();
212  char* ptr = buff.data();
213  while (ptr < &buff.back()) { // check all RDH open/close and optional headers and trailers
214  const auto rdhi = *reinterpret_cast<RDH*>(ptr);
215  if (firstHBF[il]) { // make sure SOT or SOC is there
216  BOOST_CHECK(rdhi.triggerType & (o2::trigger::SOC | o2::trigger::SOT));
217  }
218 
219  BOOST_CHECK(HBFUtils::checkRDH(rdhi)); // check RDH validity
220  BOOST_CHECK(HBFUtils::getHBIR(rdhRef) == HBFUtils::getHBIR(rdhi)); // make sure the RDH of each link corresponds to the same BC
221  if (rdhi.stop) { // closing page must be empty
222  BOOST_CHECK(rdhi.memorySize == rdhi.headerSize);
223  } else {
224  BOOST_CHECK(rdhi.memorySize > rdhi.headerSize); // in this model all non-closing pages must contain something
225  if (rdhi.memorySize - rdhi.headerSize == HBFUtils::GBTWord) { // empty HBF will contain just a status word
226  testStr.assign(ptr + rdhi.headerSize, HBFUtils::GBTWord);
227  BOOST_CHECK(testStr == HBFEmpty);
228  } else {
229  // pages with real payload should have at least header + trailer + some payload
230  BOOST_CHECK(rdhi.memorySize - rdhi.headerSize > 2 * HBFUtils::GBTWord);
231  testStr.assign(ptr + rdhi.headerSize, HBFUtils::GBTWord);
232  BOOST_CHECK(testStr == PLHeader);
233  testStr.assign(ptr + rdhi.memorySize - HBFUtils::GBTWord, HBFUtils::GBTWord);
234  BOOST_CHECK(testStr == PLTrailer);
235  }
236  }
237  ptr += rdhi.offsetToNext;
238  }
239  firstHBF[il] = false;
240  }
241  }
242  } while (nLinksRead); // read until there is something to read
243  } // run
244 };
