56 bool FLPSyncSampler::ConditionalRun()
57 {
58  FairMQMessagePtr msg(NewSimpleMessage(mTimeFrameId));
59 
60  if (fChannels.at(mOutChannelName).at(0).Send(msg) >= 0) {
61  mTimeframeRTT[mTimeFrameId].start = steady_clock::now();
62 
63  if (++mTimeFrameId == UINT16_MAX - 1) {
64  mTimeFrameId = 0;
65  }
66  }
67 
68  // rate limiting
69  --mEventCounter;
70  while (mEventCounter == 0) {
71  this_thread::sleep_for(milliseconds(1));
72  }
73 
74  if (mMaxEvents > 0 && mTimeFrameId >= mMaxEvents) {
75  LOG(INFO) << "Reached configured maximum number of events (" << mMaxEvents << "). Exiting Run().";
76  return false;
77  }
78 
79  return true;
80 }

89 void FLPSyncSampler::ListenForAcks()
90 {
91  uint16_t id = 0;
92 
93  ofstream ofsFrames;
94  ofstream ofsTimes;
95 
96  // store round trip time measurements in a file
97  if (mStoreRTTinFile > 0) {
98  std::time_t t = system_clock::to_time_t(system_clock::now());
99  tm utc = *gmtime(&t);
100  std::stringstream s;
101  s << utc.tm_year + 1900 << "-" << utc.tm_mon + 1 << "-" << utc.tm_mday << "-" << utc.tm_hour << "-" << utc.tm_min << "-" << utc.tm_sec;
102  string name = s.str();
103  ofsFrames.open(name + "-frames.log");
104  ofsTimes.open(name + "-times.log");
105  }
106 
107  while (!mLeaving) {
108  FairMQMessagePtr idMsg(NewMessage());
109 
110  if (Receive(idMsg, mAckChannelName, 0, 1000) >= 0) {
111  id = *(static_cast<uint16_t*>(idMsg->GetData()));
112  mTimeframeRTT.at(id).end = steady_clock::now();
113  // store values in a file
114  auto elapsed = duration_cast<microseconds>(mTimeframeRTT.at(id).end - mTimeframeRTT.at(id).start);
115 
116  if (mStoreRTTinFile > 0) {
117  ofsFrames << id << "\n";
118  ofsTimes << elapsed.count() << "\n";
119  }
120 
121  LOG(INFO) << "Timeframe #" << id << " acknowledged after " << elapsed.count() << " Î¼s.";
122  }
123  }
124 
125  // store round trip time measurements in a file
126  if (mStoreRTTinFile > 0) {
127  ofsFrames.close();
128  ofsTimes.close();
129  }
130  LOG(INFO) << "Exiting Ack listener";
131 }
