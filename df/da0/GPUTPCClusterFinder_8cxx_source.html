83 size_t GPUTPCClusterFinder::getNSteps(size_t items) const
84 {
85  if (items == 0) {
86  return 0;
87  }
88  size_t c = 1;
89  size_t capacity = mScanWorkGroupSize;
90  while (items > capacity) {
91  capacity *= mScanWorkGroupSize;
92  c++;
93  }
94  return c;
95 }
