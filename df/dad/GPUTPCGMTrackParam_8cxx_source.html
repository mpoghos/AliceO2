192  }

233  noFollowCircle = false;
234 
235  lastUpdateX = mX;
236  lastLeg = clusters[ihit].leg;
237  lastRow = 255;
238  N++;
239  resetT0 = initResetT0();
240  // clang-format off
241  CADEBUG(printf("\n"));
242  CADEBUG(printf("\t%21sMirror Alpha %8.3f , X %8.3f - Y %8.3f, Z %8.3f - QPt %7.2f (%7.2f), SP %5.2f (%5.2f) %28s --- Cov sY %8.3f sZ %8.3f sSP %8.3f sPt %8.3f - YPt %8.3f\n", "", prop.GetAlpha(), mX, mP[0], mP[1], mP[4], prop.GetQPt0(), mP[2], prop.GetSinPhi0(), "", sqrtf(mC[0]), sqrtf(mC[2]), sqrtf(mC[5]), sqrtf(mC[14]), mC[10]));
243  // clang-format on
244  continue;
245  }
246  }

272  }
273  // clang-format off
274  CADEBUG(if (!CheckCov()) GPUError("INVALID COV AFTER UPDATE!!!"));

276  // clang-format on
277 
278  if (retVal == 0) // track is updated
279  {
280  noFollowCircle2 = false;
281  lastUpdateX = mX;
282  covYYUpd = mC[0];
283  nMissed = nMissed2 = 0;
284  UnmarkClusters(clusters, ihitMergeFirst, ihit, wayDirection, GPUTPCGMMergedTrackHit::flagNotFit);
285  N++;
286  ihitStart = ihit;
287  float dy = mP[0] - prop.Model().Y();
288  float dz = mP[1] - prop.Model().Z();
289  if (CAMath::Abs(mP[4]) > 10 && --resetT0 <= 0 && CAMath::Abs(mP[2]) < 0.15f && dy * dy + dz * dz > 1) {
290  CADEBUG(printf("Reinit linearization\n"));
291  prop.SetTrack(this, prop.GetAlpha());
292  }
293  if (dEdxOut && iWay == nWays - 1 && clusters[ihit].leg == clusters[maxN - 1].leg) {
294  if (merger->GetConstantMem()->ioPtrs.clustersNative == nullptr) {
295  dEdx.fillCluster(clusters[ihit].amp, 0, clusters[ihit].row, mP[2], mP[3], param);
296  } else {
297  const ClusterNative& cl = merger->GetConstantMem()->ioPtrs.clustersNative->clustersLinear[clusters[ihit].num];
298  dEdx.fillCluster(cl.qTot, cl.qMax, clusters[ihit].row, mP[2], mP[3], param);
299  }
300  }
301  } else if (retVal == 2) { // cluster far away form the track
302  if (allowModification) {
303  MarkClusters(clusters, ihitMergeFirst, ihit, wayDirection, GPUTPCGMMergedTrackHit::flagRejectDistance);
304  }
305  nMissed++;
306  nMissed2++;
307  } else {
308  break; // bad chi2 for the whole track, stop the fit
309  }
310  }
311  if (((nWays - iWay) & 1) && (clusters[0].slice < 18) == (clusters[maxN - 1].slice < 18)) {
312  ShiftZ(clusters, merger, N);
313  }
314  }
315  ConstrainSinPhi();
316 
317  bool ok = N + NTolerated >= GPUCA_TRACKLET_SELECTOR_MIN_HITS(mP[4]) && CheckNumericalQuality(covYYUpd);
318  if (!ok) {
319  return (false);
320  }
321 
322  if (dEdxOut) {
323  dEdx.computedEdx(*dEdxOut, param);
324  }
325  Alpha = prop.GetAlpha();
326  if (param.rec.TrackReferenceX <= 500) {
327  for (int k = 0; k < 3; k++) // max 3 attempts
328  {
329  int err = prop.PropagateToXAlpha(param.rec.TrackReferenceX, Alpha, 0);
330  ConstrainSinPhi();
331  if (CAMath::Abs(mP[0]) <= mX * CAMath::Tan(kSectAngle / 2.f)) {
332  break;
333  }
334  float dAngle = floor(CAMath::ATan2(mP[0], mX) / kDeg2Rad / 20.f + 0.5f) * kSectAngle;
335  Alpha += dAngle;
336  if (err || k == 2) {
337  Rotate(dAngle);
338  ConstrainSinPhi();
339  break;
340  }
341  }
342  } else if (CAMath::Abs(mP[0]) > mX * CAMath::Tan(kSectAngle / 2.f)) {
343  float dAngle = floor(CAMath::ATan2(mP[0], mX) / kDeg2Rad / 20.f + 0.5f) * kSectAngle;
344  Rotate(dAngle);
345  ConstrainSinPhi();
346  Alpha += dAngle;
347  }
348  if (Alpha > M_PI) {
349  Alpha -= 2 * M_PI;
350  } else if (Alpha <= -M_PI) {
351  Alpha += 2 * M_PI;
352  }
353 
354  return (ok);
355 }

357 GPUd() void GPUTPCGMTrackParam::MirrorTo(GPUTPCGMPropagator& prop, float toY, float toZ, bool inFlyDirection, const GPUParam& param, unsigned char row, unsigned char clusterState, bool mirrorParameters)
358 {
359  if (mirrorParameters) {
360  prop.Mirror(inFlyDirection);
361  }
362  float err2Y, err2Z;
363  prop.GetErr2(err2Y, err2Z, param, toZ, row, clusterState);
364  prop.Model().Y() = mP[0] = toY;
365  prop.Model().Z() = mP[1] = toZ;
366  if (mC[0] < err2Y) {
367  mC[0] = err2Y;
368  }
369  if (mC[2] < err2Z) {
370  mC[2] = err2Z;
371  }
372  if (CAMath::Abs(mC[5]) < 0.1f) {
373  mC[5] = mC[5] > 0 ? 0.1f : -0.1f;
374  }
375  if (mC[9] < 1.f) {
376  mC[9] = 1.f;
377  }
378  mC[1] = mC[4] = mC[6] = mC[8] = mC[11] = mC[13] = 0;
379  prop.SetTrack(this, prop.GetAlpha());
380  mNDF = -3;
381  mChi2 = 0;
382 }

384 GPUd() int GPUTPCGMTrackParam::MergeDoubleRowClusters(int ihit, int wayDirection, GPUTPCGMMergedTrackHit* clusters, const GPUTPCGMMerger* merger, GPUTPCGMPropagator& prop, float& xx, float& yy, float& zz, int maxN, float clAlpha, unsigned char& clusterState, bool rejectChi2)
385 {
386  if (ihit + wayDirection >= 0 && ihit + wayDirection < maxN && clusters[ihit].row == clusters[ihit + wayDirection].row && clusters[ihit].slice == clusters[ihit + wayDirection].slice && clusters[ihit].leg == clusters[ihit + wayDirection].leg) {
387  float maxDistY, maxDistZ;
388  prop.GetErr2(maxDistY, maxDistZ, merger->Param(), zz, clusters[ihit].row, 0);
389  maxDistY = (maxDistY + mC[0]) * 20.f;
390  maxDistZ = (maxDistZ + mC[2]) * 20.f;
391  int noReject = 0; // Cannot reject if simple estimation of y/z fails (extremely unlike case)
392  if (CAMath::Abs(clAlpha - prop.GetAlpha()) > 1.e-4f) {
393  noReject = prop.RotateToAlpha(clAlpha);
394  }
395  float projY = 0, projZ = 0;
396  if (noReject == 0) {
397  noReject |= prop.GetPropagatedYZ(xx, projY, projZ);
398  }
399  float count = 0.f;
400  xx = yy = zz = 0.f;
401  clusterState = 0;
402  while (true) {
403  const float zOffset = (clusters[ihit].slice < 18) == (clusters[0].slice < 18) ? mZOffset : -mZOffset;
404  float clx, cly, clz, clamp;
405  if (merger->Param().earlyTpcTransform) {
406  clx = clusters[ihit].x;
407  cly = clusters[ihit].y;
408  clz = clusters[ihit].z;
409  clamp = clusters[ihit].amp;
410  } else {
411  const ClusterNative& cl = merger->GetConstantMem()->ioPtrs.clustersNative->clustersLinear[clusters[ihit].num];
412  clamp = cl.qTot;
413  //merger->GetConstantMem()->calibObjects.fastTransform->TransformInTimeFrame(clusters[ihit].slice, clusters[ihit].row, cl.getPad(), cl.getTime(), clx, cly, clz, merger->Param().continuousMaxTimeBin);
414  GPUTPCConvertImpl::convert(*merger->GetConstantMem(), clusters[ihit].slice, clusters[ihit].row, cl.getPad(), cl.getTime(), clx, cly, clz);
415  }
416  float dy = cly - projY;
417  float dz = clz - zOffset - projZ;
418  if (noReject == 0 && (dy * dy > maxDistY || dz * dz > maxDistZ)) {
