72 // class branchIterator is a data iterator
73 // it basically allows to iterate over the elements of an arrow::table::Column and
74 // successively fill the branches with branchIterator::fillnext()
75 // it is used in CommonDataProcessors::table2tree
76 class branchIterator
77 {
78 
79  private:
80  const char* brn; // branch name
81  arrow::ArrayVector chs; // chunks
82  Int_t nchs; // number of chunks
83  Int_t ich; // chunk counter
84  Int_t nr; // number of rows
85  Int_t ir; // row counter
86 
87  // data buffers for each data type
88  arrow::Type::type dt;
89  std::string leaflist;
90 
91  TBranch* br = nullptr;
92 
93  char* dbuf = nullptr;
94  void* v = nullptr;
95 
96  Float_t* var_f = nullptr;
97  Double_t* var_d = nullptr;
98  UShort_t* vs = nullptr;
99  UInt_t* vi = nullptr;
100  ULong64_t* vl = nullptr;
101  Short_t* var_s = nullptr;
102  Int_t* var_i = nullptr;
103  Long64_t* var_l = nullptr;
104 
105  // initialize a branch
106  void branchini(TTree* tree, std::shared_ptr<arrow::Column> col, bool tupdate)
107  {
108 
109  if (tupdate) {
110  br = tree->GetBranch(brn);
111 
112  } else {
113 
114  // create new branch of given data type
115  switch (dt) {
116  case arrow::Type::type::FLOAT:
117  leaflist = col->name() + "/F";
118  break;
119  case arrow::Type::type::DOUBLE:
120  leaflist = col->name() + "/D";
121  break;
122  case arrow::Type::type::UINT16:
123  leaflist = col->name() + "/s";
124  break;
125  case arrow::Type::type::UINT32:
126  leaflist = col->name() + "/i";
127  break;
128  case arrow::Type::type::UINT64:
129  leaflist = col->name() + "/l";
130  break;
131  case arrow::Type::type::INT16:
132  leaflist = col->name() + "/S";
133  break;
134  case arrow::Type::type::INT32:
135  leaflist = col->name() + "/I";
136  break;
137  case arrow::Type::type::INT64:
138  leaflist = col->name() + "/L";
139  break;
140  default:
141  LOG(FATAL) << "Type not handled: " << dt << std::endl;
142  break;
143  }
144 
145  br = tree->Branch(brn, dbuf, leaflist.c_str());
146  }
147  if (!br)
148  throw std::runtime_error("Branch does not exist!");
149  };
150 
151  // initialize chunk ib
152  void dbufini(Int_t ib)
153  {
154  // get next chunk of given data type dt
155  switch (dt) {
156  case arrow::Type::type::FLOAT:
157  var_f = (Float_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::FloatType>>(chs.at(ib))->raw_values();
158  v = (void*)var_f;
159  break;
160  case arrow::Type::type::DOUBLE:
161  var_d = (Double_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::DoubleType>>(chs.at(ib))->raw_values();
162  v = (void*)var_d;
163  break;
164  case arrow::Type::type::UINT16:
165  vs = (UShort_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::UInt16Type>>(chs.at(ib))->raw_values();
166  v = (void*)vs;
167  break;
168  case arrow::Type::type::UINT32:
169  vi = (UInt_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::UInt32Type>>(chs.at(ib))->raw_values();
170  v = (void*)vi;
171  break;
172  case arrow::Type::type::UINT64:
173  vl = (ULong64_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::UInt64Type>>(chs.at(ib))->raw_values();
174  v = (void*)vl;
175  break;
176  case arrow::Type::type::INT16:
177  var_s = (Short_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::Int16Type>>(chs.at(ib))->raw_values();
178  v = (void*)var_s;
179  break;
180  case arrow::Type::type::INT32:
181  var_i = (Int_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::Int32Type>>(chs.at(ib))->raw_values();
182  v = (void*)var_i;
183  break;
184  case arrow::Type::type::INT64:
185  var_l = (Long64_t*)std::dynamic_pointer_cast<arrow::NumericArray<arrow::Int64Type>>(chs.at(ib))->raw_values();
186  v = (void*)var_l;
187  break;
188  default:
189  LOG(FATAL) << "Type not handled: " << dt << std::endl;
190  break;
191  }
192  br->SetAddress(v);
193 
194  // reset number of rows nr and row counter ir
195  nr = chs.at(ib)->length();
196  ir = 0;
197  };
198 
199  public:
200  branchIterator() : brn(nullptr),
201  nchs(0),
202  ich(0),
203  nr(0),
204  ir(0),
205  dt(arrow::Type::type::NA){};
206  branchIterator(TTree* tree, std::shared_ptr<arrow::Column> col, bool tupdate) : brn(nullptr),
207  nchs(0),
208  ich(0),
209  nr(0),
210  ir(0),
211  dt(arrow::Type::type::NA)
212  {
213  brn = col->name().c_str();
214  dt = col->type()->id();
215  chs = col->data()->chunks();
216  nchs = chs.size();
217 
218  // initialize the branch
219  branchini(tree, col, tupdate);
220 
221  ich = 0;
222  dbufini(ich);
223  LOG(DEBUG) << "datait: " << col->type()->ToString() << " with " << nchs << " chunks" << std::endl;
224  };
225  ~branchIterator();
226 
227  // return branch
228  TBranch* branch() { return br; };
229 
230  // return the data type
231  arrow::Type::type type() { return dt; };
232 
233  // fills buffer with next value
234  // returns fals if end of buffer reached
235  bool fillnext()
236  {
237 
238  // ich and ir contain the current chunk and row
239  // return the next element if available
240  if (ir >= nr) {
241  ich++;
242  if (ich < nchs) {
243  dbufini(ich);
244  } else {
245  // end of data buffer reached
246  return false;
247  }
248  } else {
249  switch (dt) {
250  case arrow::Type::type::FLOAT:
251  v = (void*)var_f++;
252  break;
253  case arrow::Type::type::DOUBLE:
254  v = (void*)var_d++;
255  break;
256  case arrow::Type::type::UINT16:
257  v = (void*)vs++;
258  break;
259  case arrow::Type::type::UINT32:
260  v = (void*)vi++;
261  break;
262  case arrow::Type::type::UINT64:
263  v = (void*)vl++;
264  break;
265  case arrow::Type::type::INT16:
266  v = (void*)var_s++;
267  break;
268  case arrow::Type::type::INT32:
269  v = (void*)var_i++;
270  break;
271  case arrow::Type::type::INT64:
272  v = (void*)var_l++;
273  break;
274  default:
275  LOG(FATAL) << "Type not handled: " << dt << std::endl;
276  v = nullptr;
277  break;
278  }
279  }
280  br->SetAddress(v);
281  ir++;
282 
283  return true;
284  };
285 };

288 void CommonDataProcessors::table2tree(TTree* tout,
289  std::shared_ptr<arrow::Table> table,
290  bool tupdate)
291 {
292 
293  LOG(DEBUG) << "table2tree: " << table->num_columns();
294 
295  // first create a vector of branchIterators
296  std::vector<branchIterator*> itbuf;
297 
298  for (int ii = 0; ii < table->num_columns(); ii++) {
299 
300  auto col = table->column(ii);
301  const char* cname = col->name().c_str();
302 
303  // for each column get a branchIterator
304  // fill the branchIterator into a std::vector
305  branchIterator* brit = new branchIterator(tout, col, tupdate);
306  itbuf.push_back(brit);
307  }
308  LOG(DEBUG) << "table2tree: size of itbuf " << itbuf.size();
309 
310  // with this loop over the columns again as long as togo is true.
311  // togo is set false when any of the data iterators returns false
312  bool togo = true;
313  while (togo) {
314  // fill the tree
315  tout->Fill();
316 
317  // update the branches
318  for (int ii = 0; ii < table->num_columns(); ii++)
319  togo &= itbuf.at(ii)->fillnext();
320  }
321 
322  // write the tree
323  tout->Write("", TObject::kOverwrite);
324 }

327 DataProcessorSpec CommonDataProcessors::getOutputObjSink(outputObjMap const& outMap)
328 {
329  auto writerFunction = [outMap](InitContext& ic) -> std::function<void(ProcessingContext&)> {
330  auto& callbacks = ic.services().get<CallbackService>();
331  auto outputObjects = std::make_shared<std::map<InputObjectRoute, InputObject>>();
332 
333  auto endofdatacb = [outputObjects](EndOfStreamContext& context) {
334  LOG(DEBUG) << "Writing merged objects to file";
335  std::string currentDirectory = "";
336  std::string currentFile = "";
337  TFile* f[OutputObjHandlingPolicy::numPolicies];
338  for (auto i = 0u; i < OutputObjHandlingPolicy::numPolicies; ++i) {
339  f[i] = nullptr;
340  }
341  for (auto& [route, entry] : *outputObjects) {
342  auto file = ROOTfileNames.find(route.policy);
343  if (file != ROOTfileNames.end()) {
344  auto filename = file->second;
345  if (f[route.policy] == nullptr) {
346  f[route.policy] = TFile::Open(filename.c_str(), "RECREATE");
347  }
348  auto nextDirectory = route.directory;
349  if ((nextDirectory != currentDirectory) || (filename != currentFile)) {
350  if (!f[route.policy]->FindKey(nextDirectory.c_str())) {
351  f[route.policy]->mkdir(nextDirectory.c_str());
352  }
353  currentDirectory = nextDirectory;
354  currentFile = filename;
355  }
356  (f[route.policy]->GetDirectory(currentDirectory.c_str()))->WriteObjectAny(entry.obj, entry.kind, entry.name.c_str());
357  }
358  }
359  for (auto i = 0u; i < OutputObjHandlingPolicy::numPolicies; ++i) {
360  if (f[i] != nullptr) {
361  f[i]->Close();
362  }
363  }
364  LOG(DEBUG) << "All outputs merged in their respective target files";
365  context.services().get<ControlService>().readyToQuit(QuitRequest::All);
366  };
367 
368  callbacks.set(CallbackService::Id::EndOfStream, endofdatacb);
369  return [outputObjects, outMap](ProcessingContext& pc) mutable -> void {
370  auto const& ref = pc.inputs().get("x");
371  if (!ref.header) {
372  LOG(ERROR) << "Header not found";
373  return;
374  }
375  if (!ref.payload) {
376  LOG(ERROR) << "Payload not found";
377  return;
378  }
379  auto datah = o2::header::get<o2::header::DataHeader*>(ref.header);
380  if (!datah) {
381  LOG(ERROR) << "No data header in stack";
382  return;
383  }
384 
385  auto objh = o2::header::get<o2::framework::OutputObjHeader*>(ref.header);
386  if (!objh) {
387  LOG(ERROR) << "No output object header in stack";
388  return;
389  }
390 
391  FairTMessage tm(const_cast<char*>(ref.payload), datah->payloadSize);
392  InputObject obj;
393  obj.kind = tm.GetClass();
394  if (obj.kind == nullptr) {
395  LOGP(error, "Cannot read class info from buffer.");
396  return;
397  }
398 
399  OutputObjHandlingPolicy policy = OutputObjHandlingPolicy::AnalysisObject;
400  if (objh)
401  policy = objh->mPolicy;
402 
403  obj.obj = tm.ReadObjectAny(obj.kind);
404  TNamed* named = static_cast<TNamed*>(obj.obj);
405  obj.name = named->GetName();
406  auto lookup = outMap.find(obj.name);
407  std::string directory{"VariousObjects"};
408  if (lookup != outMap.end()) {
409  directory = lookup->second;
410  }
411  InputObjectRoute key{obj.name, directory, policy};
412  auto existing = outputObjects->find(key);
413  if (existing == outputObjects->end()) {
414  outputObjects->insert(std::make_pair(key, obj));
415  return;
416  }
417  auto merger = existing->second.kind->GetMerge();
418  if (!merger) {
419  LOGP(error, "Already one object found for {}.", obj.name);
420  return;
421  }
422 
423  TList coll;
424  coll.Add(static_cast<TObject*>(obj.obj));
425  merger(existing->second.obj, &coll, nullptr);
426  };
427  };
428 
429  DataProcessorSpec spec{
430  "internal-dpl-global-analysis-file-sink",
431  {InputSpec("x", DataSpecUtils::dataDescriptorMatcherFrom(header::DataOrigin{"ATSK"}))},
432  Outputs{},
433  AlgorithmSpec(writerFunction),
434  {}};
435 
436  return spec;
437 }

439 // add sink for dangling AODs
440 DataProcessorSpec
441  CommonDataProcessors::getGlobalAODSink(std::vector<InputSpec> const& danglingOutputInputs)
442 {
443 
444  auto writerFunction = [danglingOutputInputs](InitContext& ic) -> std::function<void(ProcessingContext&)> {
445  LOG(DEBUG) << "======== getGlobalAODSink::Inint ==========";
446 
447  // analyze ic and take actions accordingly
448  auto fnbase = ic.options().get<std::string>("res-file");
449  auto filemode = ic.options().get<std::string>("res-mode");
450  auto keepString = ic.options().get<std::string>("keep");
451  auto ntfmerge = ic.options().get<Int_t>("ntfmerge");
452 
453  // find out if any tables need to be saved
454  bool hasOutputsToWrite = true;
455  bool usematch = false;
456 
457  // use the parameter keep to create a matcher
458  std::shared_ptr<data_matcher::DataDescriptorMatcher> matcher;
459  if (!keepString.empty()) {
460  usematch = true;
461  auto [variables, outputMatcher] = DataDescriptorQueryBuilder::buildFromKeepConfig(keepString);
462  matcher = outputMatcher;
463 
464  VariableContext context;
465  for (auto& spec : danglingOutputInputs) {
466  auto concrete = DataSpecUtils::asConcreteDataMatcher(spec);
467  if (outputMatcher->match(concrete, context)) {
468  hasOutputsToWrite = true;
469  }
470  }
471  }
472 
473  // if nothing needs to be saved then return a trivial functor
474  if (!hasOutputsToWrite) {
475  return std::move([](ProcessingContext& pc) mutable -> void {
476  static bool once = false;
477  if (!once) {
478  LOG(DEBUG) << "No dangling output to be saved.";
479  once = true;
480  }
481  std::this_thread::sleep_for(std::chrono::milliseconds(1000));
482  });
483  }
484 
485  // end of data functor is called at the end of the data stream
486  auto fout = std::make_shared<TFile>();
487  auto endofdatacb = [fout](EndOfStreamContext& context) {
488  if (fout)
489  fout->Close();
490 
491  context.services().get<ControlService>().readyToQuit(QuitRequest::All);
492  };
493 
494  auto& callbacks = ic.services().get<CallbackService>();
495  callbacks.set(CallbackService::Id::EndOfStream, endofdatacb);
496 
497  // this functor is called once per time frame
498  Int_t ntf = 0;
499  return std::move([fout, fnbase, filemode, ntf, ntfmerge, usematch, matcher](ProcessingContext& pc) mutable -> void {
500  LOG(DEBUG) << "======== getGlobalAODSink::processing ==========";
501  LOG(DEBUG) << " processing data set with " << pc.inputs().size() << " entries";
502  LOG(DEBUG) << " result are saved to " << fnbase << "_*.root";
503 
504  // return immediately if pc.inputs() is empty
505  auto ninputs = pc.inputs().size();
506  if (ninputs == 0) {
507  LOG(DEBUG) << "no inputs available!";
508  return;
509  }
510 
511  // new strategy since RDataFrame::Snapshot does not work
512  //
513  // loop over all inputs and extract the arrow::tables
514  // create a tree for each arrow::table
515  // loop over the columns of a table
516  // for each column create a branch
517  // loop over the rows of the column
518  // fill the values into the corresponding branches
519  // write the table
520  // see table2tree
521 
522  // open new file if ntfmerge time frames is reached
523  LOG(DEBUG) << "This is time frame number " << ntf;
524  bool tupdate = true;
525  std::string fname;
526  if ((ntf % ntfmerge) == 0) {
527  if (fout)
528  fout->Close();
529 
530  fname = fnbase + "_" + std::to_string((Int_t)(ntf / ntfmerge)) + ".root";
531  fout =
532  std::make_shared<TFile>(fname.c_str(), filemode.c_str());
533  tupdate = false;
534  }
535  ntf++;
536 
537  // loop over the DataRefs which are contained in pc.inputs()
538  TTree* tout = nullptr;
539  VariableContext matchingContext;
540  for (const auto& ref : pc.inputs()) {
541 
542  // is this table to be saved?
543  auto dh = DataRefUtils::getHeader<header::DataHeader*>(ref);
544  // only arrow tables are processed here
545  if (dh->payloadSerializationMethod != o2::header::gSerializationMethodArrow)
546  continue;
547 
548  // does it match the keep parameter
549  if (usematch && !matcher->match(*dh, matchingContext))
550  continue;
551 
552  // get the table name
553  auto treename = dh->dataDescription.as<std::string>();
554 
555  // get the TableConsumer and convert it into an arrow table
556  auto s = pc.inputs().get<TableConsumer>(ref.spec->binding);
557  auto table = s->asArrowTable();
558  LOG(DEBUG) << "The tree name is " << treename;
559  LOG(DEBUG) << "Number of columns " << table->num_columns();
560  LOG(DEBUG) << "Number of rows " << table->num_rows();
561 
562  // we need finite number of rows and columns
563  if (table->num_columns() == 0 || table->num_rows() == 0)
564  continue;
565 
566  // this table needs to be saved to file
567  // create the corresponding tree
568  if (tupdate) {
569  tout = (TTree*)fout->Get(treename.c_str());
570  } else {
571  tout = new TTree(treename.c_str(), treename.c_str());
572  }
573 
574  // write the table to the TTree
575  table2tree(tout, table, tupdate);
576  }
577  });
578  }; // end of writerFunction
579 
580  DataProcessorSpec spec{
581  "internal-dpl-AOD-writter",
582  danglingOutputInputs,
583  Outputs{},
584  AlgorithmSpec(writerFunction),
585  {{"res-file", VariantType::String, "AnalysisResults", {"Name of the output file"}},
586  {"res-mode", VariantType::String, "RECREATE", {"Creation mode of the result file: NEW, CREATE, RECREATE, UPDATE"}},
587  {"ntfmerge", VariantType::Int, 10, {"number of time frames to merge into one file"}},
588  {"keep", VariantType::String, "", {"Comma separated list of ORIGIN/DESCRIPTION/SUBSPECIFICATION to save in outfile"}}}};
589 
590  return spec;
591 }

593 DataProcessorSpec
594  CommonDataProcessors::getGlobalFileSink(std::vector<InputSpec> const& danglingOutputInputs,
595  std::vector<InputSpec>& unmatched)
596 {
597  auto writerFunction = [danglingOutputInputs](InitContext& ic) -> std::function<void(ProcessingContext&)> {
598  auto filename = ic.options().get<std::string>("outfile");
599  auto keepString = ic.options().get<std::string>("keep");
600 
601  if (filename.empty()) {
602  throw std::runtime_error("output file missing");
603  }
604 
605  bool hasOutputsToWrite = false;
606  auto [variables, outputMatcher] = DataDescriptorQueryBuilder::buildFromKeepConfig(keepString);
607  VariableContext context;
608  for (auto& spec : danglingOutputInputs) {
609  auto concrete = DataSpecUtils::asConcreteDataTypeMatcher(spec);
610  if (outputMatcher->match(concrete, context)) {
611  hasOutputsToWrite = true;
612  }
613  }
614  if (hasOutputsToWrite == false) {
615  return std::move([](ProcessingContext& pc) mutable -> void {
616  static bool once = false;
619  if (!once) {
620  LOG(DEBUG) << "No dangling output to be dumped.";
621  once = true;
622  }
623  std::this_thread::sleep_for(std::chrono::milliseconds(1000));
624  });
625  }
626  auto output = std::make_shared<std::ofstream>(filename.c_str(), std::ios_base::binary);
627  return std::move([output, matcher = outputMatcher](ProcessingContext& pc) mutable -> void {
628  VariableContext matchingContext;
629  LOG(DEBUG) << "processing data set with " << pc.inputs().size() << " entries";
630  for (const auto& entry : pc.inputs()) {
631  LOG(DEBUG) << " " << *(entry.spec);
632  auto header = DataRefUtils::getHeader<header::DataHeader*>(entry);
633  auto dataProcessingHeader = DataRefUtils::getHeader<DataProcessingHeader*>(entry);
634  if (matcher->match(*header, matchingContext) == false) {
635  continue;
636  }
637  output->write(reinterpret_cast<char const*>(header), sizeof(header::DataHeader));
638  output->write(reinterpret_cast<char const*>(dataProcessingHeader), sizeof(DataProcessingHeader));
639  output->write(entry.payload, o2::framework::DataRefUtils::getPayloadSize(entry));
640  LOG(DEBUG) << "wrote data, size " << o2::framework::DataRefUtils::getPayloadSize(entry);
641  }
642  });
643  };
644 
645  std::vector<InputSpec> validBinaryInputs;
646  auto onlyTimeframe = [](InputSpec const& input) {
647  return input.lifetime == Lifetime::Timeframe;
648  };
649 
650  auto noTimeframe = [](InputSpec const& input) {
651  return input.lifetime != Lifetime::Timeframe;
652  };
653 
654  std::copy_if(danglingOutputInputs.begin(), danglingOutputInputs.end(),
655  std::back_inserter(validBinaryInputs), onlyTimeframe);
656  std::copy_if(danglingOutputInputs.begin(), danglingOutputInputs.end(),
657  std::back_inserter(unmatched), noTimeframe);
658 
659  DataProcessorSpec spec{
660  "internal-dpl-global-binary-file-sink",
661  validBinaryInputs,
