175  bool next()
176  {
177  if ((mReadEntry + 1) >= mNEntries || mNEntries == 0) {
178  if (mPublishingMode == PublishingMode::Single) {
179  // stop here
180  if (mReadEntry < mNEntries) {
181  mReadEntry = mNEntries;
182  }
183  return false;
184  }
185  // start over in loop mode
186  mReadEntry = -1;
187  }
188  if (mMaxEntries > 0 && (mNofPublished + 1) >= mMaxEntries) {
189  if (mReadEntry < mNEntries) {
190  mReadEntry = mNEntries;
191  }
192  return false;
193  }
194  ++mReadEntry;
195  ++mNofPublished;
196  return true;
197  }

219  template <typename ContextType, typename... HeaderTypes>
220  bool operator()(ContextType& context,
221  HeaderTypes&&... headers) const
222  {
223  auto snapshot = [&context, &headers...](const KeyType& key, const auto& object) {
224  o2::header::Stack stack{std::forward<HeaderTypes>(headers)...};
225  context.outputs().snapshot(Output{key.origin, key.description, key.subSpec, key.lifetime, std::move(stack)}, object);
226  };
227 
228  return process(snapshot);
229  }

231  template <typename F>
232  bool process(F&& snapshot) const
233  {
234  if (mReadEntry >= mNEntries || mNEntries == 0 || (mMaxEntries > 0 && mNofPublished >= mMaxEntries)) {
235  return false;
236  }
237 
238  for (auto& spec : mBranchSpecs) {
239  char* data = nullptr;
240  spec.second->branch->SetAddress(&data);
241  spec.second->branch->GetEntry(mReadEntry);
242  if (spec.second->sizebranch == nullptr) {
243  snapshot(spec.first, std::move(ROOTSerializedByClass(*data, spec.second->classinfo)));
244  } else {
245  size_t datasize = 0;
246  spec.second->sizebranch->SetAddress(&datasize);
247  spec.second->sizebranch->GetEntry(mReadEntry);
248  auto* buffer = reinterpret_cast<BinaryDataStoreType*>(data);
249  if (buffer->size() == datasize) {
250  LOG(INFO) << "branch " << spec.second->name << ": publishing binary chunk of " << datasize << " bytes(s)";
251  snapshot(spec.first, std::move(*buffer));
252  } else {
253  LOG(ERROR) << "branch " << spec.second->name << ": inconsitent size of binary chunk "
254  << buffer->size() << " vs " << datasize;
255  BinaryDataStoreType empty;
256  snapshot(spec.first, empty);
257  }
258  }
259  auto* delfunc = spec.second->classinfo->GetDelete();
260  if (delfunc) {
261  (*delfunc)(data);
262  }
263  spec.second->branch->DropBaskets("all");
264  }
265  return true;
266  }

289  void addBranchSpec(KeyType key, const char* branchName)
290  {
291  // right now we allow the same key to appear for multiple branches
292  auto branch = mInput.GetBranch(branchName);
293  if (branch) {
294  mBranchSpecs.emplace_back(key, std::make_unique<BranchSpec>(BranchSpec{branchName}));
295  mBranchSpecs.back().second->branch = branch;
296  std::string sizebranchName = std::string(branchName) + "Size";
297  auto sizebranch = mInput.GetBranch(sizebranchName.c_str());
298  auto* classinfo = TClass::GetClass(branch->GetClassName());
299  if (!sizebranch) {
300  if (classinfo == nullptr) {
301  throw std::runtime_error(std::string("can not find class description for branch ") + branchName);
302  }
303  LOG(INFO) << "branch set up: " << branchName;
304  } else {
305  if (classinfo == nullptr || classinfo != TClass::GetClass(typeid(BinaryDataStoreType))) {
306  throw std::runtime_error("mismatching class type, expecting std::vector<char> for binary branch");
307  }
308  mBranchSpecs.back().second->sizebranch = sizebranch;
309  LOG(INFO) << "binary branch set up: " << branchName;
310  }
311  mBranchSpecs.back().second->classinfo = classinfo;
312  } else {
313  throw std::runtime_error(std::string("can not find branch ") + branchName);
314  }
315  }

326  template <size_t skip, typename U, typename... Args>
327  void parseConstructorArgs(U key, Args&&... args)
328  {
329  if constexpr (skip > 0) {
330  return parseConstructorArgs<skip - 1>(std::forward<Args>(args)...);
331  }
332  if constexpr (std::is_same<U, const char*>::value) {
333  addFile(key);
334  } else if constexpr (std::is_same<U, int>::value) {
335  mMaxEntries = key;
336  } else if constexpr (std::is_same<U, PublishingMode>::value) {
337  mPublishingMode = key;
338  } else if constexpr (sizeof...(Args) > 0) {
339  const char* arg = getCharArg(std::forward<Args>(args)...);
340  if (arg != nullptr && *arg != 0) {
341  // add branch spec if the name is not empty
342  addBranchSpec(KeyType{key}, arg);
343  }
344  return parseConstructorArgs<1>(std::forward<Args>(args)...);
345  } else {
346  static_assert(type_dependent<U>::value, "argument mismatch, allowed are: file names, int to specify number of events, publishing mode, and key-branchname pairs");
347  }
348  parseConstructorArgs<0>(std::forward<Args>(args)...);
349  }
