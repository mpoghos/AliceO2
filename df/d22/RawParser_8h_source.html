78 class ConcreteRawParser
79 {
80  public:

90  template <typename T>
91  ConcreteRawParser(T const* buffer, size_t size)
92  : mRawBuffer(reinterpret_cast<buffer_type const*>(buffer)), mSize(size)
93  {
94  static_assert(sizeof(T) == sizeof(buffer_type), "buffer required to be byte-type");
95  if (size < sizeof(header_type)) {
96  std::runtime_error("buffer too small to fit at least the page header");
97  }
98  next();
99  }
100  // FIXME: add constructor for container type buffer
101 
105  ConcreteRawParser(ConcreteRawParser const& other) = default;
106 
108  header_type const& header() const
109  {
110  return *reinterpret_cast<header_type const*>(mPosition);
111  }
112 
114  size_t size() const
115  {
116  if (mPosition == mRawBuffer + mSize) {
117  return 0;
118  }
119  header_type const& h = header();
120  // FIXME: block length disappeared in V5, check what can be used
121  return max_size - h.headerSize;
122  }
123 
125  buffer_type const* data() const
126  {
127  size_t size = this->size();
128  if (size == 0) {
129  return nullptr;
130  }
131  header_type const& h = header();
132  if (mPosition + size + h.headerSize > mRawBuffer + mSize) {
133  throw std::runtime_error("not enough data at position " + std::to_string(mPosition - mRawBuffer));
134  }
135  return mPosition + h.headerSize;
136  }
137 
139  buffer_type const* raw() const
140  {
141  if (mPosition < mRawBuffer + mSize) {
142  return mPosition;
143  }
144  return nullptr;
145  }
146 
148  size_t offset() const
149  {
150  if (mPosition < mRawBuffer + mSize) {
151  header_type const& h = header();
152  return h.headerSize;
153  }
154  return 0;
155  }

161  template <typename Processor>
162  void parse(Processor&& processor)
163  {
164  reset();
165  //auto deleter = [](buffer_type*) {};
166  do {
167  processor(data(), size());
168  //processor(std::unique_ptr<buffer_type, decltype(deleter)>(data(), deleter), size());
169  } while (next());
170  }
171 
173  bool next()
174  {
175  if (mPosition == nullptr) {
176  mPosition = mRawBuffer;
177  } else {
178  auto offset = header().offsetToNext;
179  if ((mPosition + offset + sizeof(header_type) > mRawBuffer + mSize) || (offset < sizeof(header_type))) {
180  // FIXME: should check if there is unread data left of if we just reached
181  // the end of the buffer by parsing all pages
182  mPosition = mRawBuffer + mSize;
183  return false;
184  }
185  mPosition += offset;
186  }
187  // FIXME: check page header validity: version, size
188  return true;
189  }
190 
192  void reset()
193  {
194  mPosition = mRawBuffer;
195  }
196 
200  void advance(int step)
201  {
202  if (step < 0) {
203  // set beyond the buffer
204  mPosition = mRawBuffer + mSize;
205  } else {
206  while ((step-- > 0) && next()) {
207  };
208  }
209  }
210 
213  template <typename T = self_type>
214  bool operator==(T const& other) const
215  {
216  if constexpr (std::is_same<T, self_type>::value == true) {
217  return mRawBuffer == other.mRawBuffer && mPosition == other.mPosition;
218  } else {
219  throw std::runtime_error(std::string("incompatible types for comparison ") + typeid(T).name() + "/" + typeid(self_type).name());
220  }
221  }

223  void format(std::ostream& os, FormatSpec choice = FormatSpec::Entry, const char* delimiter = "\n") const
224  {
225  RDHFormatter<header_type>::apply(os, header(), choice, delimiter);
226  }
227 
228  private:
229  buffer_type const* mRawBuffer;
230  buffer_type const* mPosition = nullptr;
231  size_t mSize;

274 template <size_t N, typename T, typename P>
275 void walk_parse(T& instances, P&& processor, size_t index)
276 {
277  if constexpr (N > 0) {
278  if (index == N - 1) {
279  std::get<N - 1>(instances).parse(processor);
280  }
281 
282  walk_parse<N - 1>(instances, processor, index);
283  }
284 }

286 template <typename U, typename T, size_t N = std::variant_size_v<T>>
287 U const* get_if(T& instances)
288 {
289  if constexpr (N > 0) {
290  auto* parser = std::get_if<N - 1>(&instances);
291  if (parser) {
292  // we are in the active instance, return header if type matches
293  using parser_type = typename std::variant_alternative<N - 1, T>::type;
294  using header_type = typename parser_type::header_type;
295  if constexpr (std::is_same<U, header_type>::value == true) {
296  return &(parser->header());
297  }
298  } else {
299  // continue walking through instances until active one is found
300  return get_if<U, T, N - 1>(instances);
301  }
302  }
303  return nullptr;
304 }
