58 Double_t fitGaus(const size_t nBins, const T* arr, const T xMin, const T xMax, std::vector<T>& param)
59 {
60  static TLinearFitter fitter(3, "pol2");
61  static TMatrixD mat(3, 3);
62  static Double_t kTol = mat.GetTol();
63  fitter.StoreData(kFALSE);
64  fitter.ClearPoints();
65  TVectorD par(3);
66  TVectorD sigma(3);
67  TMatrixD A(3, 3);
68  TMatrixD b(3, 1);
69  T rms = TMath::RMS(nBins, arr);
70  T max = TMath::MaxElement(nBins, arr);
71  T binWidth = (xMax - xMin) / T(nBins);
72 
73  Float_t meanCOG = 0;
74  Float_t rms2COG = 0;
75  Float_t sumCOG = 0;
76 
77  Float_t entries = 0;
78  Int_t nfilled = 0;
79 
80  param.resize(4);
81  param[0] = 0.;
82  param[1] = 0.;
83  param[2] = 0.;
84  param[3] = 0.;
85 
86  for (Int_t i = 0; i < nBins; i++) {
87  entries += arr[i];
88  if (arr[i] > 0)
89  nfilled++;
90  }
91 
92  // TODO: Check why this is needed
93  if (max < 4)
94  return -4;
95  if (entries < 12)
96  return -4;
97 
98  if (rms < kTol)
99  return -4;
100 
101  param[3] = entries;
102 
103  Int_t npoints = 0;
104  for (Int_t ibin = 0; ibin < nBins; ibin++) {
105  Float_t entriesI = arr[ibin];
106  if (entriesI > 1) {
107  Double_t xcenter = xMin + (ibin + 0.5) * binWidth;
108  Double_t error = 1. / TMath::Sqrt(entriesI);
109  Double_t val = TMath::Log(Float_t(entriesI));
110  fitter.AddPoint(&xcenter, val, error);
111  if (npoints < 3) {
112  A(npoints, 0) = 1;
113  A(npoints, 1) = xcenter;
114  A(npoints, 2) = xcenter * xcenter;
115  b(npoints, 0) = val;
116  meanCOG += xcenter * entriesI;
117  rms2COG += xcenter * entriesI * xcenter;
118  sumCOG += entriesI;
119  }
120  npoints++;
121  }
122  }
123 
124  Double_t chi2 = 0;
125  if (npoints >= 3) {
126  if (npoints == 3) {
127  //analytic calculation of the parameters for three points
128  A.Invert();
129  TMatrixD res(1, 3);
130  res.Mult(A, b);
131  par[0] = res(0, 0);
132  par[1] = res(0, 1);
133  par[2] = res(0, 2);
134  chi2 = -3.;
135  } else {
136  // use fitter for more than three points
137  fitter.Eval();
138  fitter.GetParameters(par);
139  fitter.GetCovarianceMatrix(mat);
140  chi2 = fitter.GetChisquare() / Double_t(npoints);
141  }
142  if (TMath::Abs(par[1]) < kTol)
143  return -4;
144  if (TMath::Abs(par[2]) < kTol)
145  return -4;
146 
147  param[1] = T(par[1] / (-2. * par[2]));
148  param[2] = T(1. / TMath::Sqrt(TMath::Abs(-2. * par[2])));
149  Double_t lnparam0 = par[0] + par[1] * param[1] + par[2] * param[1] * param[1];
150  if (lnparam0 > 307)
151  return -4;
152  param[0] = TMath::Exp(lnparam0);
153 
154  return chi2;
155  }
156 
157  if (npoints == 2) {
158  //use center of gravity for 2 points
159  meanCOG /= sumCOG;
160  rms2COG /= sumCOG;
161  param[0] = max;
162  param[1] = meanCOG;
163  param[2] = TMath::Sqrt(TMath::Abs(meanCOG * meanCOG - rms2COG));
164  chi2 = -2.;
165  }
166  if (npoints == 1) {
167  meanCOG /= sumCOG;
168  param[0] = max;
169  param[1] = meanCOG;
170  param[2] = binWidth / TMath::Sqrt(12);
171  chi2 = -1.;
172  }
173  return chi2;
174 }

193 template <typename T>
194 StatisticsData getStatisticsData(const T* arr, const size_t nBins, const double xMin, const double xMax)
195 {
196  double mean = 0;
197  double rms2 = 0;
198  double sum = 0;
199  size_t npoints = 0;
200 
201  double binWidth = (xMax - xMin) / (double)nBins;
202 
203  StatisticsData data;
204  // in case something went wrong the COG is the histogram lower limit
205  data.mCOG = xMin;
206 
207  for (size_t ibin = 0; ibin < nBins; ++ibin) {
208  double entriesI = (double)arr[ibin];
209  double xcenter = xMin + (ibin + 0.5) * binWidth; // +0.5 to shift to bin centre
210  if (entriesI > 0) {
211  mean += xcenter * entriesI;
212  rms2 += xcenter * entriesI * xcenter;
213  sum += entriesI;
214  ++npoints;
215  }
216  }
217  if (sum == 0)
218  return data;
219  mean /= sum;
220 
221  data.mCOG = mean;
222  // exception in case of only one bin is filled
223  // set the standard deviation to bin width over sqrt(12)
224  rms2 /= sum;
225  if (npoints == 1) {
226  data.mStdDev = binWidth / std::sqrt(12.);
227  } else {
228  data.mStdDev = std::sqrt(std::abs(rms2 - mean * mean));
229  }
230 
231  data.mSum = sum;
232 
233  return data;
234 }

241 template <typename T, typename R = double>
242 R median(std::vector<T> v)
243 {
244  if (v.empty()) {
245  return R{};
246  }
247  auto n = v.size() / 2;
248  nth_element(v.begin(), v.begin() + n, v.end());
249  auto med = R{v[n]};
250  if (!(v.size() & 1)) { //If the set size is even
251  auto max_it = max_element(v.begin(), v.begin() + n);
252  med = R{(*max_it + med) / 2.0};
253  }
254  return med;
255 }

291 template <typename T>
292 bool LTMUnbinned(const std::vector<T>& data, std::vector<size_t>& index, std::array<float, 7>& params, float fracKeep)
293 {
294  int nPoints = data.size();
295  std::vector<float> w(2 * nPoints);
296  int nKeep = nPoints * fracKeep;
297  if (nKeep > nPoints) {
298  nKeep = nPoints;
299  }
300  if (nKeep < 2) {
301  return false;
302  }
303  // sort in increasing order
304  SortData(data, index);
305  // build cumulants
306  double sum1 = 0.0;
307  double sum2 = 0.0;
308  for (int i = 0; i < nPoints; i++) {
309  double x = data[index[i]];
310  sum1 += x;
311  sum2 += x * x;
312  w[i] = sum1;
313  w[i + nPoints] = sum2;
314  }
315  double maxRMS = sum2 + 1e6;
316  params[0] = nKeep;
317  int limI = nPoints - nKeep + 1; // lowest possible bin to accept
318  for (int i = 0; i < limI; i++) {
319  const int limJ = i + nKeep - 1; // highest accepted bin
320  sum1 = static_cast<double>(w[limJ]) - static_cast<double>(i ? w[i - 1] : 0.);
321  sum2 = static_cast<double>(w[nPoints + limJ]) - static_cast<double>(i ? w[nPoints + i - 1] : 0.);
322  const double mean = sum1 / nKeep;
323  const double rms2 = sum2 / nKeep - mean * mean;
324  if (rms2 > maxRMS) {
325  continue;
326  }
327  maxRMS = rms2;
328  params[1] = mean;
329  params[2] = rms2;
330  params[5] = i;
331  params[6] = limJ;
332  }
333  //
334  if (params[2] < 0) {
335  LOG(error) << "Rounding error: RMS = " << params[2] << " < 0";
336  return false;
337  }
338  params[2] = std::sqrt(params[2]);
339  params[3] = params[2] / std::sqrt(params[0]); // error on mean
340  params[4] = params[3] / std::sqrt(2.0); // error on RMS
341  return true;
342 }

370 template <typename T>
371 bool LTMUnbinnedSig(const std::vector<T>& data, std::vector<size_t>& index, std::array<float, 7>& params, float fracKeepMin, float sigTgt, bool sorted = false)
372 {
373  int nPoints = data.size();
374  std::vector<double> wx(nPoints);
375  std::vector<double> wx2(nPoints);
376 
377  if (!sorted) {
378  // sort in increasing order
379  SortData(data, index);
380  } else {
381  // array is already sorted
382  std::iota(index.begin(), index.end(), 0);
383  }
384  // build cumulants
385  double sum1 = 0.0;
386  double sum2 = 0.0;
387  for (int i = 0; i < nPoints; i++) {
388  double x = data[index[i]];
389  sum1 += x;
390  sum2 += x * x;
391  wx[i] = sum1;
392  wx2[i] = sum2;
393  }
394  int keepMax = nPoints;
395  int keepMin = fracKeepMin * nPoints;
396  if (keepMin > keepMax) {
397  keepMin = keepMax;
398  }
399  float sigTgt2 = sigTgt * sigTgt;
400  //
401  while (true) {
402  double maxRMS = wx2.back() + 1e6;
403  int keepN = (keepMax + keepMin) / 2;
404  if (keepN < 2) {
405  return false;
406  }
407  params[0] = keepN;
408  int limI = nPoints - keepN + 1;
409  for (int i = 0; i < limI; ++i) {
410  const int limJ = i + keepN - 1;
411  sum1 = wx[limJ] - (i ? wx[i - 1] : 0.);
412  sum2 = wx2[limJ] - (i ? wx2[i - 1] : 0.);
413  const double mean = sum1 / keepN;
414  const double rms2 = sum2 / keepN - mean * mean;
415  if (rms2 > maxRMS) {
416  continue;
417  }
418  maxRMS = rms2;
419  params[1] = mean;
420  params[2] = rms2;
421  params[5] = i;
422  params[6] = limJ;
423  }
424  if (maxRMS < sigTgt2) {
425  keepMin = keepN;
426  } else {
427  keepMax = keepN;
428  }
429  if (keepMin >= keepMax - 1) {
430  break;
431  }
432  }
433  params[2] = std::sqrt(params[2]);
434  params[3] = params[2] / std::sqrt(params[0]); // error on mean
435  params[4] = params[3] / std::sqrt(2.0); // error on RMS
436  return true;
437 }
