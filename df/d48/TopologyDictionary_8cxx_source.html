52 void TopologyDictionary::WriteBinaryFile(string outputfile)
53 {

61  file_output.write(reinterpret_cast<char*>(&p.mNpixels), sizeof(int));
62  file_output.write(reinterpret_cast<char*>(&p.mFrequency), sizeof(double));
63  file_output.write(reinterpret_cast<char*>(&p.mIsGroup), sizeof(bool));
64  file_output.write(reinterpret_cast<char*>(&p.mPattern.mBitmap),

70 int TopologyDictionary::ReadBinaryFile(string fname)
71 {
72  mVectorOfGroupIDs.clear();
73  mFinalMap.clear();
74  for (auto& p : mSmallTopologiesLUT)
75  p = -1;
76  std::ifstream in(fname.data(), std::ios::in | std::ios::binary);
77  GroupStruct gr;
78  int groupID = 0;
79  if (!in.is_open()) {
80  LOG(ERROR) << "The file coud not be opened";
81  throw std::runtime_error("The file coud not be opened");
82  } else {
83  while (in.read(reinterpret_cast<char*>(&gr.mHash), sizeof(unsigned long))) {
84  in.read(reinterpret_cast<char*>(&gr.mErrX), sizeof(float));
85  in.read(reinterpret_cast<char*>(&gr.mErrZ), sizeof(float));
86  in.read(reinterpret_cast<char*>(&gr.mXCOG), sizeof(float));
87  in.read(reinterpret_cast<char*>(&gr.mZCOG), sizeof(float));
88  in.read(reinterpret_cast<char*>(&gr.mNpixels), sizeof(int));
89  in.read(reinterpret_cast<char*>(&gr.mFrequency), sizeof(double));
90  in.read(reinterpret_cast<char*>(&gr.mIsGroup), sizeof(bool));
91  in.read(reinterpret_cast<char*>(&gr.mPattern.mBitmap), sizeof(unsigned char) * (ClusterPattern::kExtendedPatternBytes));
92  mVectorOfGroupIDs.push_back(gr);
93  if (gr.mPattern.getUsedBytes() == 1)
94  mSmallTopologiesLUT[(gr.mPattern.getColumnSpan() - 1) * 255 + (int)gr.mPattern.mBitmap[2]] = groupID;
95  if (((gr.mHash) & 0xffffffff) != 0)
96  mFinalMap.insert(std::make_pair(gr.mHash, groupID));
97  groupID++;
98  }
99  }
100  in.close();
101  return 0;
102 }

166 double TopologyDictionary::GetFrequency(int n) const
167 {
168  if (n < 0 || n >= (int)mVectorOfGroupIDs.size()) {
169  LOG(ERROR) << "Index out of bounds";
170  throw std::range_error("Index out of bounds");
171  } else if (n == 0) {
172  return mVectorOfGroupIDs[n].mFrequency;
173  } else {
174  return mVectorOfGroupIDs[n].mFrequency - mVectorOfGroupIDs[n - 1].mFrequency;
175  }
176 }

178 Point3D<float> TopologyDictionary::getClusterCoordinates(const CompCluster& cl) const
179 {
180  Point3D<float> locCl;
181  o2::itsmft::SegmentationAlpide::detectorToLocalUnchecked(cl.getRow(), cl.getCol(), locCl);
182  locCl.SetX(locCl.X() + this->GetXcog(cl.getPatternID()));
183  locCl.SetZ(locCl.Z() + this->GetZcog(cl.getPatternID()));
184  return locCl;
