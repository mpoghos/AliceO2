33 void RawFileWriter::close()
34 {
35  // finalize all links
36  if (mFName2File.empty()) {
37  return;
38  }
39  mIRMax--; // mIRMax is pointing to the beginning of latest (among all links) HBF to open, we want just to close the last one
40  for (auto& lnk : mSSpec2Link) {
41  lnk.second.updateIR = mIRMax;
42  lnk.second.close(mIRMax);
43  lnk.second.print();
44  }
45  //
46  // close all files
47  for (auto& flh : mFName2File) {
48  LOG(INFO) << "Closing output file " << flh.first;
49  fclose(flh.second);
50  }
51  mFName2File.clear();
52 }

55 void RawFileWriter::registerLink(uint16_t fee, uint16_t cru, uint8_t link, uint8_t endpoint, const std::string& outFileName)
56 {
57  // register the GBT link and its output file
58 
59  auto sspec = HBFUtils::getSubSpec(cru, link, endpoint);
60  auto& linkData = getLinkWithSubSpec(sspec);
61  auto* file = mFName2File[outFileName];
62  if (!file) {
63  if ((file = fopen(outFileName.c_str(), "wb"))) { // if file does not exist, create it
64  mFName2File[outFileName] = file;
65  } else {
66  LOG(ERROR) << "Failed to open output file " << outFileName;
67  throw std::runtime_error(std::string("cannot open link output file ") + outFileName);
68  }
69  }
70  if (linkData.file) { // this link was already declared and associated with a file
71  if (linkData.file == file) {
72  LOGF(INFO, "Link 0x%ux was already declared with same output, do nothing", sspec);
73  return;
74  } else {
75  LOGF(ERROR, "Link 0x%ux was already connected to different output file %s", sspec, linkData.fileName);
76  throw std::runtime_error("redifinition of the link output file");
77  }
78  }
79  linkData.fileName = outFileName;
80  linkData.file = file;
81  linkData.subspec = sspec;
82  linkData.rdhCopy.feeId = fee;
83  linkData.rdhCopy.cruID = cru;
84  linkData.rdhCopy.linkID = link;
85  linkData.rdhCopy.endPointID = endpoint;
86  linkData.writer = this;
87  linkData.updateIR = mHBFUtils.getFirstIR();
88  linkData.buffer.reserve(mSuperPageSize);
89  LOGF(INFO, "Registered %s with output to %s", linkData.describe(), outFileName);
90 }

102 void RawFileWriter::addData(uint16_t cru, uint8_t lnk, uint8_t endpoint, const IR& ir, const gsl::span<char> data)
103 {
104  // add payload to relevant links
105  if (data.size() % HBFUtils::GBTWord) {
106  LOG(ERROR) << "provided payload size " << data.size() << " is not multiple of GBT word size";
107  throw std::runtime_error("payload size is not mutiple of GBT word size");
108  }
109  auto sspec = HBFUtils::getSubSpec(cru, lnk, endpoint);
110  if (!isLinkRegistered(sspec)) {
111  LOGF(ERROR, "The link for SubSpec=0x%ux(%u:%u:%u) was not registered", sspec, cru, lnk, endpoint);
112  throw std::runtime_error("data for non-registered GBT link supplied");
113  }
114  auto& link = getLinkWithSubSpec(sspec);
115  link.addData(ir, data);
116 
117  if (mIRMax < link.updateIR) { // remember highest IR seen
118  mIRMax = link.updateIR;
119  }
120 }

135 void RawFileWriter::LinkData::addData(const IR& ir, const gsl::span<char> data)
136 {
137  // add payload corresponding to IR
138  LOG(DEBUG) << "Adding " << data.size() << " bytes in IR " << ir << " to " << describe();
139 
140  if (ir >= updateIR) { // new IR exceeds or equal IR of next HBF to open, insert missed HBFs if needed
141  fillEmptyHBHs(ir);
142  }
143  // we are guaranteed to be under the valid RDH + possibly some data
144  int dataSize = data.size();
145  if (!dataSize) {
146  return;
147  }
148  const char* ptr = &data[0];
149  // in case particular detector CRU pages need to be self-consistent, when carrying-over
150  // large payload to new CRU page we may need to write optional trailer and header before
151  // and after the new RDH.
152  bool carryOver = false;
153  int splitID = 0;
154  while (dataSize > 0) {
155  if (carryOver) { // check if there is carry-over header to write in the buffer
156  addHBFPage(); // start new CRU page, if needed, the completed superpage is flushed
157  // for sure after the carryOver we have space on the CRU page, no need to check
158  LOG(DEBUG) << "Adding carryOverHeader " << sCarryOverHeader.size()
159  << " bytes in IR " << ir << " to " << describe();
160  pushBack(sCarryOverHeader.data(), sCarryOverHeader.size());
161  sCarryOverHeader.clear();
162  carryOver = false;
163  }
164  int sizeLeftSupPage = writer->mSuperPageSize - buffer.size();
165  int sizeLeftCRUPage = HBFUtils::MAXCRUPage - (int(buffer.size()) - lastRDHoffset);
166  int sizeLeft = sizeLeftCRUPage < sizeLeftSupPage ? sizeLeftCRUPage : sizeLeftSupPage;
167  if (!sizeLeft) { // this page is just over, open a new one
168  addHBFPage(); // start new CRU page, if needed, the completed superpage is flushed
169  continue;
170  }
171  if (dataSize <= sizeLeft) { // add all remaining data
172  LOG(DEBUG) << "Adding payload " << dataSize << " bytes in IR " << ir << " (carryover=" << carryOver << " ) to " << describe();
173  pushBack(ptr, dataSize);
174  dataSize = 0;
175  } else { // need to carryOver payload, determine 1st wsize bytes to write starting from ptr
176  carryOver = true;
177  int sizeActual = sizeLeft;
178  if (writer->carryOverFunc) {
179  sizeActual = writer->carryOverFunc(rdhCopy, data, ptr, sizeLeft, splitID++, sCarryOverTrailer, sCarryOverHeader);
180  }
181  LOG(DEBUG) << "Adding carry-over " << splitID - 1 << " fitted payload " << sizeActual << " bytes in IR " << ir << " to " << describe();
182  if (sizeActual < 0 || sizeActual + sCarryOverTrailer.size() > sizeLeft) {
183  throw std::runtime_error(std::string("wrong carry-over data size provided by carryOverMethod") + std::to_string(sizeActual));
184  }
185  pushBack(ptr, sizeActual); // write payload fitting to this page
186  dataSize -= sizeActual;
187  ptr += sizeActual;
188  LOG(DEBUG) << "Adding carryOverTrailer " << sCarryOverTrailer.size() << " bytes in IR "
189  << ir << " to " << describe();
190  pushBack(sCarryOverTrailer.data(), sCarryOverTrailer.size());
191  sCarryOverTrailer.clear();
192  }
193  }
194 }

197 void RawFileWriter::LinkData::addHBFPage(bool stop)
198 {
200 
201  // check if the superpage reached the size where it hase to be flushed
202  if (lastRDHoffset < 0) {
203  return; // no page was open
204  }
205  // finalize last RDH
206  auto* lastRDH = getLastRDH();
207  int psize = buffer.size() - lastRDHoffset; // set the size for the previous header RDH
208  if (stop && psize == sizeof(RDH) && writer->emptyHBFFunc) { // we are closing an empty page, does detector want to add something?
209  writer->emptyHBFFunc(*lastRDH, sEmptyHBFFiller);
210  if (sEmptyHBFFiller.size()) {
211  LOG(DEBUG) << "Adding empty HBF filler of size " << sEmptyHBFFiller.size() << " for " << describe();
212  pushBack(sEmptyHBFFiller.data(), sEmptyHBFFiller.size());
213  psize += sEmptyHBFFiller.size();
214  sEmptyHBFFiller.clear();
215  }
216  }
217  lastRDH->offsetToNext = lastRDH->memorySize = psize;
218 
219  if (writer->mVerbosity > 2) {
220  HBFUtils::printRDH(*lastRDH);
221  }
222  rdhCopy = *lastRDH;
223  int left = writer->mSuperPageSize - buffer.size();
224  if (left <= MarginToFlush) {
225  flushSuperPage();
226  }
227  rdhCopy.packetCounter = packetCounter++;
228  rdhCopy.pageCnt = pageCnt++;
229  rdhCopy.stop = stop;
230  rdhCopy.offsetToNext = rdhCopy.memorySize = sizeof(RDH);
231  lastRDHoffset = pushBack(rdhCopy); // entry of the new RDH
232  if (stop) {
233  if (rdhCopy.triggerType & o2::trigger::TF) {
234  nTFWritten++;
235  }
236  if (writer->mVerbosity > 2) {
237  HBFUtils::printRDH(rdhCopy);
238  }
239  lastRDHoffset = -1; // after closing, the previous RDH is not valid anymore
240  startOfRun = false; // signal that we are definitely not in the beginning of the run
241  }
242  //
243 }

246 void RawFileWriter::LinkData::openHBFPage(const RDH& rdhn)
247 {
249  bool forceNewPage = false;
250  if (rdhn.triggerType & o2::trigger::TF) {
251  if (writer->mVerbosity > 0) {
252  LOGF(INFO, "Starting new TF for link FEEId 0x%04x", rdhn.feeId);
253  }
254  if (writer->mStartTFOnNewSPage && nTFWritten) { // don't flush if 1st TF
255  forceNewPage = true;
256  }
257  }
258  int left = writer->mSuperPageSize - buffer.size();
259  if (forceNewPage || left <= MarginToFlush) {
260  flushSuperPage();
261  }
262  pageCnt = 0;
263  lastRDHoffset = pushBack(rdhn);
264  RDH* newRDH = getLastRDH(); // fetch new RDH
265  newRDH->packetCounter = packetCounter++;
266  newRDH->pageCnt = pageCnt++;
267  newRDH->stop = 0;
268  newRDH->memorySize = newRDH->offsetToNext = sizeof(RDH);
269  if (startOfRun && writer->isReadOutModeSet()) {
270  newRDH->triggerType |= writer->isContinuousReadout() ? o2::trigger::SOC : o2::trigger::SOT;
271  }
272  rdhCopy = *newRDH;
273 }

276 void RawFileWriter::LinkData::flushSuperPage(bool keepLastPage)
277 {
278  // write link superpage data to file
279  size_t pgSize = (lastRDHoffset < 0 || !keepLastPage) ? buffer.size() : lastRDHoffset;
280  if (writer->mVerbosity) {
281  LOGF(INFO, "Flushing super page of %u bytes for %s", pgSize, describe());
282  }
283  fwrite(buffer.data(), 1, pgSize, file); // write to file
284  auto toMove = buffer.size() - pgSize;
285  if (toMove) { // is there something left in the buffer, move it to the beginning of the buffer
286  if (toMove > pgSize) {
287  memcpy(buffer.data(), &buffer[pgSize], toMove);
288  } else {
289  memmove(buffer.data(), &buffer[pgSize], toMove);
290  }
291  buffer.resize(toMove);
292  lastRDHoffset -= pgSize;
293  } else {
294  buffer.clear();
295  lastRDHoffset = -1;
296  }
297 }

300 void RawFileWriter::LinkData::close(const IR& irf)
301 {
302  // finalize link data
303  // close open HBF, write empty HBFs until the end of the TF corresponding to irfin and detach from the stream
304  if (!file) {
305  return; // already closed
306  }
307  auto irfin = irf;
308  if (irfin < updateIR) {
309  irfin = updateIR;
310  }
311  int tf = writer->mHBFUtils.getTF(irfin);
312  auto finalIR = writer->mHBFUtils.getIRTF(tf + 1) - 1; // last IR of the current TF
313  fillEmptyHBHs(finalIR);
314  closeHBFPage(); // close last HBF
315  flushSuperPage();
316  file = nullptr;
317 }

319 //___________________________________________________________________________________
320 void RawFileWriter::LinkData::fillEmptyHBHs(const IR& ir)
321 {
322  // fill HBFs from last processed one to requested ir
323  if (!writer->mHBFUtils.fillHBIRvector(sIRWork, updateIR, ir)) {
324  return;
325  }
326  for (const auto& irdummy : sIRWork) {
327  if (writer->mVerbosity > 2) {
328  LOG(INFO) << "Adding HBF " << irdummy << " for " << describe();
329  }
330  closeHBFPage(); // close current HBF: add RDH with stop and update counters
331  rdhCopy.triggerType = 0; // reset to avoid any detector specific flags in the dummy HBFs
332  writer->mHBFUtils.updateRDH<RDH>(rdhCopy, irdummy); // update HBF orbit/bc and trigger flags
333  openHBFPage(rdhCopy); // open new HBF
334  }
335  updateIR = sIRWork.back() + o2::constants::lhc::LHCMaxBunches; // new HBF will be generated at >= this IR
336 }
