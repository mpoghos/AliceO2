46 float Response::etocharge(float edepos)
47 {
48  int nel = int(edepos * 1.e9 / 27.4);
49  float charge = 0;
50  if (nel == 0)
51  nel = 1;
52  for (int i = 1; i <= nel; i++) {
53  float arg = 0.;
54  while (!arg)
55  arg = gRandom->Rndm();
56  charge -= mChargeSlope * TMath::Log(arg);
57  }
58  //translate to fC roughly,
59  //equivalent to AliMUONConstants::DefaultADC2MV()*AliMUONConstants::DefaultA0()*AliMUONConstants::DefaultCapa() multiplication in aliroot
60  charge *= 0.61 * 1.25 * 0.2;
61  return charge;
62 }

64 double Response::chargePadfraction(float xmin, float xmax, float ymin, float ymax)
65 {
66  //see AliMUONResponseV0.cxx (inside DisIntegrate)
67  // and AliMUONMathieson.cxx (IntXY)
68  //see: https://edms.cern.ch/ui/file/1054937/1/ALICE-INT-2009-044.pdf
69  // normalise w.r.t. Pitch
70  xmin *= mInversePitch;
71  xmax *= mInversePitch;
72  ymin *= mInversePitch;
73  ymax *= mInversePitch;
74 
75  return chargefrac1d(xmin, xmax, mK2x, mSqrtK3x, mK4x) * chargefrac1d(ymin, ymax, mK2y, mSqrtK3y, mK4y);
76 }

78 double Response::chargefrac1d(float min, float max, double k2, double sqrtk3, double k4)
79 {
80  // The Mathieson function integral (1D)
81  double u1 = sqrtk3 * TMath::TanH(k2 * min);
82  double u2 = sqrtk3 * TMath::TanH(k2 * max);
83  return 2. * k4 * (TMath::ATan(u2) - TMath::ATan(u1));
84 }
