32 GPUd() bool GPUTPCCFPeakFinder::isPeakScratchPad(
33  GPUSharedMemory& smem,
34  Charge q,
35  const ChargePos& pos,
36  ushort N,
37  const Array2D<PackedCharge>& chargeMap,
38  ChargePos* posBcast,
39  PackedCharge* buf)
40 {
41  ushort ll = get_local_id(0);
42 
43  bool belowThreshold = (q <= QMAX_CUTOFF);
44 
45  ushort lookForPeaks;
46  ushort partId = CfUtils::partition(
47  smem,
48  ll,
49  belowThreshold,
50  SCRATCH_PAD_WORK_GROUP_SIZE,
51  &lookForPeaks);
52 
53  if (partId < lookForPeaks) {
54  posBcast[partId] = pos;
55  }
56  GPUbarrier();
57 
58  CfUtils::blockLoad<PackedCharge>(
59  chargeMap,
60  lookForPeaks,
61  SCRATCH_PAD_WORK_GROUP_SIZE,
62  ll,
63  0,
64  N,
65  CfConsts::InnerNeighbors,
66  posBcast,
67  buf);
68 
69  if (belowThreshold) {
70  return false;
71  }
72 
73  // Ensure q has the same float->int->float conversion error
74  // as values in chargeMap, so identical charges are actually identical
75  q = PackedCharge(q).unpack();
76 
77  int idx = N * partId;
78  bool peak = true;
79  peak = peak && buf[idx + 0].unpack() <= q;
80  peak = peak && buf[idx + 1].unpack() <= q;
81  peak = peak && buf[idx + 2].unpack() <= q;
82  peak = peak && buf[idx + 3].unpack() <= q;
83  peak = peak && buf[idx + 4].unpack() < q;
84  peak = peak && buf[idx + 5].unpack() < q;
85  peak = peak && buf[idx + 6].unpack() < q;
86  peak = peak && buf[idx + 7].unpack() < q;
87 
88  return peak;
89 }

91 GPUd() bool GPUTPCCFPeakFinder::isPeak(
92  Charge myCharge,
93  const ChargePos& pos,
94  const Array2D<PackedCharge>& chargeMap)
95 {
96  if (myCharge <= QMAX_CUTOFF) {
97  return false;
98  }
99 
100  bool peak = true;
101 
102 #define CMP_NEIGHBOR(dp, dt, cmpOp) \
103  do { \
104  PackedCharge p = chargeMap[pos.delta({dp, dt})]; \
105  const Charge otherCharge = p.unpack(); \
106  peak &= (otherCharge cmpOp myCharge); \
107  } while (false)
108 
109 #define CMP_LT CMP_NEIGHBOR(-1, -1, <=)
110 #define CMP_T CMP_NEIGHBOR(-1, 0, <=)
111 #define CMP_RT CMP_NEIGHBOR(-1, 1, <=)
112 
113 #define CMP_L CMP_NEIGHBOR(0, -1, <=)
114 #define CMP_R CMP_NEIGHBOR(0, 1, <)
115 
116 #define CMP_LB CMP_NEIGHBOR(1, -1, <)
117 #define CMP_B CMP_NEIGHBOR(1, 0, <)
118 #define CMP_RB CMP_NEIGHBOR(1, 1, <)
119 
120 #if defined(CHARGEMAP_TILING_LAYOUT)
121  CMP_LT;
122  CMP_T;
123  CMP_RT;
124  CMP_R;
125  CMP_RB;
126  CMP_B;
127  CMP_LB;
128  CMP_L;
129 #else
130  CMP_LT;
131  CMP_T;
132  CMP_RT;
133  CMP_L;
134  CMP_R;
135  CMP_LB;
136  CMP_B;
137  CMP_RB;
138 #endif
139 
140 #undef CMP_LT
141 #undef CMP_T
142 #undef CMP_RT
143 #undef CMP_L
144 #undef CMP_R
145 #undef CMP_LB
146 #undef CMP_B
147 #undef CMP_RB
148 #undef CMP_NEIGHBOR
149 
150  return peak;
151 }

153 GPUd() void GPUTPCCFPeakFinder::findPeaksImpl(int nBlocks, int nThreads, int iBlock, int iThread, GPUSharedMemory& smem,
154  const Array2D<PackedCharge>& chargeMap,
155  const Digit* digits,
156  uint digitnum,
157  uchar* isPeakPredicate,
158  Array2D<uchar>& peakMap)
159 {
160  size_t idx = get_global_id(0);
161 
162  // For certain configurations dummy work items are added, so the total
163  // number of work items is dividable by 64.
164  // These dummy items also compute the last digit but discard the result.
165  Digit myDigit = digits[CAMath::Min(idx, (size_t)(digitnum - 1))];
166 
167  ChargePos pos(myDigit);
168 
169  uchar peak;
170 #if defined(BUILD_CLUSTER_SCRATCH_PAD)
171  peak = isPeakScratchPad(smem, myDigit.charge, pos, SCRATCH_PAD_SEARCH_N, chargeMap, smem.posBcast, smem.buf);
172 #else
173  peak = isPeak(myDigit.charge, pos, chargeMap);
174 #endif
175 
176  // Exit early if dummy. See comment above.
177  bool iamDummy = (idx >= digitnum);
178  if (iamDummy) {
179  return;
180  }
181 
182  isPeakPredicate[idx] = peak;
183 
184  peakMap[pos] = (uchar(myDigit.charge > CHARGE_THRESHOLD) << 1) | peak;
185 }
