77 void EPNReceiver::DiscardIncompleteTimeframes()
78 {
79  auto it = mTimeframeBuffer.begin();
80 
81  while (it != mTimeframeBuffer.end()) {
82  if (duration_cast<milliseconds>(steady_clock::now() - (it->second).start).count() > mBufferTimeoutInMs) {
83  LOG(WARN) << "Timeframe #" << it->first << " incomplete after " << mBufferTimeoutInMs << " milliseconds, discarding";
84  mDiscardedSet.insert(it->first);
85  mTimeframeBuffer.erase(it++);
86  LOG(WARN) << "Number of discarded timeframes: " << mDiscardedSet.size();
87  } else {
88  // LOG(INFO) << "Timeframe #" << it->first << " within timeout, buffering...";
89  ++it;
90  }
