40 void DigitsWriterSpec::init(framework::InitContext& ctx)
41 {
42  // get the option from the init context
43  auto filename = ctx.options().get<std::string>("emcal-digit-outfile");
44  auto treename = ctx.options().get<std::string>("treename");
45  mOutputFile = std::make_shared<TFile>(filename.c_str(), "RECREATE");
46  mOutputTree = std::make_shared<TTree>(treename.c_str(), treename.c_str());
47  mDigits = std::make_shared<std::vector<o2::emcal::Digit>>();
48  mTriggerRecords = std::make_shared<std::vector<o2::emcal::TriggerRecord>>();
49  mFinished = false;
50 
51  // the callback to be set as hook at stop of processing for the framework
52  auto outputfile = mOutputFile;
53  auto outputtree = mOutputTree;
54  auto finishWriting = [outputfile, outputtree]() {
55  outputtree->SetEntries(1);
56  outputtree->Write();
57  outputfile->Close();
58  };
59  ctx.services().get<CallbackService>().set(CallbackService::Id::Stop, finishWriting);
60 }

62 void DigitsWriterSpec::run(framework::ProcessingContext& ctx)
63 {
64  if (mFinished)
65  return;
66 
67  // retrieve the digits from the input
68  auto indata = ctx.inputs().get<std::vector<o2::emcal::Digit>>("emcaldigits");
69  LOG(INFO) << "RECEIVED DIGITS SIZE " << indata.size();
70  *mDigits.get() = std::move(indata);
71  auto trgrecords = ctx.inputs().get<std::vector<o2::emcal::TriggerRecord>>("trgrecorddigits");
72  LOG(INFO) << "GOT " << trgrecords.size() << " TRIGGER RECORDS" << std::endl;
73  *mTriggerRecords.get() = std::move(trgrecords);
74  std::cout << "Before tree writing" << std::endl;
75 
76  // connect this to a particular branch
77  auto br = getOrMakeBranch(*mOutputTree.get(), "EMCALDigit", mDigits.get());
78  br->Fill();
79 
80  // connect trigger records branch
81  std::cout << "Before trigger record writing" << std::endl;
82  auto trgbranch = getOrMakeBranch(*mOutputTree.get(), "EMCALDigitTRGR", mTriggerRecords.get());
83  trgbranch->Fill();
84  std::cout << "After trigger digit tree writing" << std::endl;
85 
86  // retrieve labels from the input
87  auto labeldata = ctx.inputs().get<o2::dataformats::MCTruthContainer<MCLabel>*>("emcaldigitlabels");
88  LOG(INFO) << "EMCAL GOT " << labeldata->getNElements() << " LABELS ";
89  auto labeldataraw = labeldata.get();
90  // connect this to a particular branch
91  auto labelbr = getOrMakeBranch(*mOutputTree.get(), "EMCALDigitMCTruth", &labeldataraw);
92  labelbr->Fill();
93 
94  mFinished = true;
95  ctx.services().get<ControlService>().readyToQuit(QuitRequest::Me);
96 }
