152 int TPCFastTransform::writeToFile(std::string outFName, std::string name)
153 {
155  assert(isConstructed());
156 
157  if (outFName.empty())
158  outFName = "tpcFastTransform.root";
159 
160  if (name.empty())
161  name = "TPCFastTransform";
162 
163  TFile outf(outFName.data(), "recreate");
164  if (outf.IsZombie()) {
165  LOG(ERROR) << "Failed to open output file " << outFName;
166  return -1;
167  }
168 
169  bool isBufferExternal = !isBufferInternal();
170  if (isBufferExternal)
171  adoptInternalBuffer(mFlatBufferPtr);
172  outf.WriteObjectAny(this, Class(), name.data());
173  outf.Close();
174  if (isBufferExternal)
175  clearInternalBufferPtr();
176  return 0;
177 }

179 TPCFastTransform* TPCFastTransform::loadFromFile(std::string inpFName, std::string name)
180 {
182 
183  if (inpFName.empty())
184  inpFName = "tpcFastTransform.root";
185 
186  if (name.empty())
187  name = "TPCFastTransform";
188 
189  TFile inpf(inpFName.data());
190  if (inpf.IsZombie()) {
191  LOG(ERROR) << "Failed to open input file " << inpFName;
192  return nullptr;
193  }
194  TPCFastTransform* transform = reinterpret_cast<TPCFastTransform*>(inpf.GetObjectChecked(name.data(), TPCFastTransform::Class()));
195  if (!transform) {
196  LOG(ERROR) << "Failed to load " << name << " from " << inpFName;
197  return nullptr;
198  }
199  if (transform->mFlatBufferSize > 0 && transform->mFlatBufferContainer == nullptr) {
200  LOG(ERROR) << "Failed to load " << name << " from " << inpFName << ": empty flat buffer container";
201  return nullptr;
202  }
203  transform->setActualBufferAddress(transform->mFlatBufferContainer);
204  return transform;
205 }
