31 int SplineHelper2D::setSpline(const Spline2D& spline, int nAxiliaryPointsU, int nAxiliaryPointsV)
32 {
33  // Prepare creation of 2D irregular spline
34  // The should be at least one (better, two) axiliary measurements on each segnment between two knots and at least 2*nKnots measurements in total
35  // Returns 0 when the spline can not be constructed with the given nAxiliaryPoints
36 
37  int ret = 0;
38 
39  if (!spline.isConstructed()) {
40  ret = storeError(-1, "SplineHelper2D::setSpline2D: input spline is not constructed");
41  mSpline.constructKnotsRegular(2, 2);
42  } else {
43  mSpline.cloneFromObject(spline, nullptr);
44  }
45  if (mHelperU.setSpline(mSpline.getGridU(), nAxiliaryPointsU) != 0) {
46  ret = storeError(-2, "SplineHelper2D::setSpline2D: error by setting U axis");
47  }
48 
49  if (mHelperV.setSpline(mSpline.getGridV(), nAxiliaryPointsV) != 0) {
50  ret = storeError(-3, "SplineHelper2D::setSpline2D: error by setting V axis");
51  }
52 
53  return ret;
54 }

56 void SplineHelper2D::constructParameters(int Ndim, const float DataPointF[/*getNumberOfDataPoints() x Ndim*/], float parameters[/*mSpline.getNumberOfParameters(Ndim) */]) const
57 {
58  // Create 2D irregular spline in a compact way
59 
60  const int Ndim2 = 2 * Ndim;
61  const int Ndim3 = 3 * Ndim;
62  const int Ndim4 = 4 * Ndim;
63 
64  int nDataPointsU = getNumberOfDataPointsU();
65  int nDataPointsV = getNumberOfDataPointsV();
66 
67  int nKnotsU = mSpline.getGridU().getNumberOfKnots();
68  int nKnotsV = mSpline.getGridV().getNumberOfKnots();
69 
70  std::unique_ptr<float[]> rotDataPointF(new float[nDataPointsU * nDataPointsV * Ndim]); // U DataPoints x V DataPoints : rotated DataPointF for one output dimension
71  std::unique_ptr<float[]> Dv(new float[nKnotsV * nDataPointsU * Ndim]); // V knots x U DataPoints
72 
73  std::unique_ptr<float[]> parU(new float[mHelperU.getSpline().getNumberOfParameters(Ndim)]);
74  std::unique_ptr<float[]> parV(new float[mHelperV.getSpline().getNumberOfParameters(Ndim)]);
75 
76  // rotated data points (u,v)->(v,u)
77 
78  for (int ipu = 0; ipu < nDataPointsU; ipu++) {
79  for (int ipv = 0; ipv < nDataPointsV; ipv++) {
80  for (int dim = 0; dim < Ndim; dim++) {
81  rotDataPointF[Ndim * (ipu * nDataPointsV + ipv) + dim] = DataPointF[Ndim * (ipv * nDataPointsU + ipu) + dim];
82  }
83  }
84  }
85 
86  // get S and S'u at all the knots by interpolating along the U axis
87 
88  for (int iKnotV = 0; iKnotV < nKnotsV; ++iKnotV) {
89  int ipv = mHelperV.getKnotDataPoint(iKnotV);
90  const float* DataPointFrow = &(DataPointF[Ndim * ipv * nDataPointsU]);
91  mHelperU.constructParametersGradually(Ndim, DataPointFrow, parU.get());
92 
93  for (int iKnotU = 0; iKnotU < nKnotsU; ++iKnotU) {
94  float* knotPar = &parameters[Ndim4 * (iKnotV * nKnotsU + iKnotU)];
95  for (int dim = 0; dim < Ndim; ++dim) {
96  knotPar[dim] = parU[Ndim * (2 * iKnotU) + dim]; // store S for all the knots
97  knotPar[Ndim2 + dim] = parU[Ndim * (2 * iKnotU) + Ndim + dim]; // store S'u for all the knots //SG!!!
98  }
99  }
100 
101  // recalculate F values for all ipu DataPoints at V = ipv
102  for (int ipu = 0; ipu < nDataPointsU; ipu++) {
103  float splineF[Ndim];
104  float u = mHelperU.getDataPoint(ipu).u;
105  mSpline.getGridU().interpolate(Ndim, parU.get(), u, splineF);
106  for (int dim = 0; dim < Ndim; dim++) {
107  rotDataPointF[(ipu * nDataPointsV + ipv) * Ndim + dim] = splineF[dim];
108  }
109  }
110  }
111 
112  // calculate S'v at all data points with V == V of a knot
113 
114  for (int ipu = 0; ipu < nDataPointsU; ipu++) {
115  const float* DataPointFcol = &(rotDataPointF[ipu * nDataPointsV * Ndim]);
116  mHelperV.constructParametersGradually(Ndim, DataPointFcol, parV.get());
117  for (int iKnotV = 0; iKnotV < nKnotsV; iKnotV++) {
118  for (int dim = 0; dim < Ndim; dim++) {
119  float dv = parV[(iKnotV * 2 + 1) * Ndim + dim];
120  Dv[(iKnotV * nDataPointsU + ipu) * Ndim + dim] = dv;
121  }
122  }
123  }
124 
125  // fit S'v and S''_vu at all the knots
126 
127  for (int iKnotV = 0; iKnotV < nKnotsV; ++iKnotV) {
128  const float* Dvrow = &(Dv[iKnotV * nDataPointsU * Ndim]);
129  mHelperU.constructParameters(Ndim, Dvrow, parU.get());
130  for (int iKnotU = 0; iKnotU < nKnotsU; ++iKnotU) {
131  for (int dim = 0; dim < Ndim; ++dim) {
132  parameters[Ndim4 * (iKnotV * nKnotsU + iKnotU) + Ndim + dim] = parU[Ndim * 2 * iKnotU + dim]; // store S'v for all the knots
133  parameters[Ndim4 * (iKnotV * nKnotsU + iKnotU) + Ndim3 + dim] = parU[Ndim * 2 * iKnotU + Ndim + dim]; // store S''vu for all the knots
134  }
135  }
136  }
137 }

139 std::unique_ptr<float[]> SplineHelper2D::constructParameters(int Ndim, std::function<void(float u, float v, float f[/*Ndim*/])> F, float uMin, float uMax, float vMin, float vMax)
140 {
142 
143  std::vector<float> dataPointF(getNumberOfDataPoints() * Ndim);
144 
145  double scaleU = (uMax - uMin) / ((double)mSpline.getGridU().getUmax());
146  double scaleV = (vMax - vMin) / ((double)mSpline.getGridV().getUmax());
147 
148  for (int iu = 0; iu < getNumberOfDataPointsU(); iu++) {
149  float u = uMin + mHelperU.getDataPoint(iu).u * scaleU;
150  for (int iv = 0; iv < getNumberOfDataPointsV(); iv++) {
151  float v = vMin + mHelperV.getDataPoint(iv).u * scaleV;
152  F(u, v, &dataPointF[(iv * getNumberOfDataPointsU() + iu) * Ndim]);
153  }
154  }
155  std::unique_ptr<float[]> parameters(new float[mSpline.getNumberOfParameters(Ndim)]);
156  constructParameters(Ndim, dataPointF.data(), parameters.get());
157  return parameters;
158 }
