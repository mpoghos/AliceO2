27 GPUd() void GPUTPCCompressionKernels::Thread<GPUTPCCompressionKernels::step0attached>(int nBlocks, int nThreads, int iBlock, int iThread, GPUsharedref() GPUSharedMemory& smem, processorType& processors)
28 {
29  GPUTPCGMMerger& merger = processors.tpcMerger;
30  const o2::tpc::ClusterNativeAccess* clusters = processors.tpcConverter.getClustersNative();
31  GPUTPCCompression& compressor = processors.tpcCompressor;
32  GPUParam& param = processors.param;
33 
34  char lastLeg = 0;
35  int myTrack = 0;
36  for (unsigned int i = get_global_id(0); i < (unsigned int)merger.NOutputTracks(); i += get_global_size(0)) {
37  const GPUTPCGMMergedTrack& trk = merger.OutputTracks()[i];
38  if (!trk.OK()) {
39  continue;
40  }
41  bool rejectTrk = CAMath::Abs(trk.GetParam().GetQPt()) > processors.param.rec.tpcRejectQPt;
42  int nClustersStored = 0;
43  CompressedClustersPtrsOnly& c = compressor.mPtrs;
44  unsigned int lastRow = 0, lastSlice = 0; // BUG: These should be unsigned char, but then CUDA breaks
45  GPUTPCCompressionTrackModel track;
46  for (int k = trk.NClusters() - 1; k >= 0; k--) {
47  const GPUTPCGMMergedTrackHit& hit = merger.Clusters()[trk.FirstClusterRef() + k];
48  if (hit.state & GPUTPCGMMergedTrackHit::flagReject) {
49  continue;
50  }
51 
52  int hitId = hit.num;
53  int attach = merger.ClusterAttachment()[hitId];
54  if ((attach & GPUTPCGMMerger::attachTrackMask) != i) {
55  continue; // Main attachment to different track
56  }
57  bool rejectCluster = processors.param.rec.tpcRejectionMode && (rejectTrk || ((attach & GPUTPCGMMerger::attachGoodLeg) == 0) || (attach & GPUTPCGMMerger::attachHighIncl));
58  if (rejectCluster) {
59  compressor.mClusterStatus[hitId] = 1; // Cluster rejected, do not store
60  continue;
61  }
62 
63  if (!(param.rec.tpcCompressionModes & GPUSettings::CompressionTrackModel)) {
64  continue; // No track model compression
65  }
66  const ClusterNative& orgCl = clusters->clusters[hit.slice][hit.row][hit.num - clusters->clusterOffset[hit.slice][hit.row]];
67  float x = param.tpcGeometry.Row2X(hit.row);
68  float y = param.tpcGeometry.LinearPad2Y(hit.slice, hit.row, orgCl.getPad());
69  float z = param.tpcGeometry.LinearTime2Z(hit.slice, orgCl.getTime());
70  if (nClustersStored) {
71  if (lastLeg != hit.leg && track.Mirror()) {
72  break;
73  }
74  if (track.Propagate(param.tpcGeometry.Row2X(hit.row), param.SliceParam[hit.slice].Alpha)) {
75  break;
76  }
77  }
78 
79  compressor.mClusterStatus[hitId] = 1; // Cluster compressed in track model, do not store as difference
80 
81  int cidx = trk.FirstClusterRef() + nClustersStored++;
82  if (nClustersStored == 1) {
83  unsigned char qpt = fabs(trk.GetParam().GetQPt()) < 20.f ? (trk.GetParam().GetQPt() * (127.f / 20.f) + 127.5f) : (trk.GetParam().GetQPt() > 0 ? 254 : 0);
84  track.Init(x, y, z, param.SliceParam[hit.slice].Alpha, qpt, param);
85 
86  myTrack = CAMath::AtomicAdd(&compressor.mMemory->nStoredTracks, 1);
87  compressor.mAttachedClusterFirstIndex[myTrack] = trk.FirstClusterRef();
88  lastLeg = hit.leg;
89  c.qPtA[myTrack] = qpt;
90  c.rowA[myTrack] = hit.row;
91  c.sliceA[myTrack] = hit.slice;
92  c.timeA[myTrack] = orgCl.getTimePacked();
93  c.padA[myTrack] = orgCl.padPacked;
94  } else {
95  unsigned int row = hit.row;
96  unsigned int slice = hit.slice;
97 
98  if (param.rec.tpcCompressionModes & GPUSettings::CompressionDifferences) {
99  if (lastRow > row) {
100  row += GPUCA_ROW_COUNT;
101  }
102  row -= lastRow;
103  if (lastSlice > slice) {
104  slice += compressor.NSLICES;
105  }
106  slice -= lastSlice;
107  }
108  c.rowDiffA[cidx] = row;
109  c.sliceLegDiffA[cidx] = (hit.leg == lastLeg ? 0 : compressor.NSLICES) + slice;
110  c.padResA[cidx] = orgCl.padPacked - orgCl.packPad(param.tpcGeometry.LinearY2Pad(hit.slice, hit.row, track.Y()));
111  c.timeResA[cidx] = (orgCl.getTimePacked() - orgCl.packTime(param.tpcGeometry.LinearZ2Time(hit.slice, track.Z()))) & 0xFFFFFF;
112  lastLeg = hit.leg;
113  }
114  lastRow = hit.row;
115  lastSlice = hit.slice;
116  unsigned short qtot = orgCl.qTot, qmax = orgCl.qMax;
117  unsigned char sigmapad = orgCl.sigmaPadPacked, sigmatime = orgCl.sigmaTimePacked;
118  if (param.rec.tpcCompressionModes & GPUSettings::CompressionTruncate) {
119  compressor.truncateSignificantBitsChargeMax(qmax, param);
120  compressor.truncateSignificantBitsCharge(qtot, param);
121  compressor.truncateSignificantBitsWidth(sigmapad, param);
122  compressor.truncateSignificantBitsWidth(sigmatime, param);
123  }
124  c.qTotA[cidx] = qtot;
125  c.qMaxA[cidx] = qmax;
126  c.sigmaPadA[cidx] = sigmapad;
127  c.sigmaTimeA[cidx] = sigmatime;
128  c.flagsA[cidx] = orgCl.getFlags();
129  if (k && track.Filter(y, z, hit.row)) {
130  break;
131  }
132  }
133  if (nClustersStored) {
134  CAMath::AtomicAdd(&compressor.mMemory->nStoredAttachedClusters, nClustersStored);
135  c.nTrackClusters[myTrack] = nClustersStored;
136  }
137  }
138 }

171 GPUd() void GPUTPCCompressionKernels::Thread<GPUTPCCompressionKernels::step1unattached>(int nBlocks, int nThreads, int iBlock, int iThread, GPUsharedref() GPUSharedMemory& smem, processorType& processors)
172 {
173  GPUTPCGMMerger& merger = processors.tpcMerger;
174  const o2::tpc::ClusterNativeAccess* clusters = processors.tpcConverter.getClustersNative();
175  GPUTPCCompression& compressor = processors.tpcCompressor;
176  GPUParam& param = processors.param;
177  unsigned int* sortBuffer = compressor.mClusterSortBuffer + iBlock * compressor.mNMaxClusterSliceRow;
178  for (int iSliceRow = iBlock; iSliceRow < GPUCA_NSLICES * GPUCA_ROW_COUNT; iSliceRow += nBlocks) {
179  const int iSlice = iSliceRow / GPUCA_ROW_COUNT;
180  const int iRow = iSliceRow % GPUCA_ROW_COUNT;
181  const int idOffset = clusters->clusterOffset[iSlice][iRow];
182  if (iThread == 0) {
183  smem.nCount = 0;
184  }
185  GPUbarrier();
186 
187  CompressedClustersPtrsOnly& c = compressor.mPtrs;
188  for (unsigned int i = get_local_id(0); i < clusters->nClusters[iSlice][iRow]; i += get_local_size(0)) {
189  const int idx = idOffset + i;
190  if (compressor.mClusterStatus[idx]) {
191  continue;
192  }
193  int attach = merger.ClusterAttachment()[idx];
194 
195  bool unattached = attach == 0;
196  if (unattached) {
197  if (processors.param.rec.tpcRejectionMode >= GPUSettings::RejectionStrategyB) {
198  continue;
199  }
200  } else if (processors.param.rec.tpcRejectionMode >= GPUSettings::RejectionStrategyA) {
201  if ((attach & GPUTPCGMMerger::attachGoodLeg) == 0) {
202  continue;
203  }
204  if (attach & GPUTPCGMMerger::attachHighIncl) {
205  continue;
206  }
207  int id = attach & GPUTPCGMMerger::attachTrackMask;
208  if (CAMath::Abs(merger.OutputTracks()[id].GetParam().GetQPt()) > processors.param.rec.tpcRejectQPt) {
209  continue;
210  }
211  }
212  int cidx = CAMath::AtomicAddShared(&smem.nCount, 1);
213  sortBuffer[cidx] = i;
214  }
215  GPUbarrier();
216 
217  if (param.rec.tpcCompressionModes & GPUSettings::CompressionDifferences) {
218  if (param.rec.tpcCompressionSortOrder == GPUSettings::SortZPadTime) {
219  CAAlgo::sortInBlock(sortBuffer, sortBuffer + smem.nCount, GPUTPCCompressionKernels_Compare<GPUSettings::SortZPadTime>(clusters->clusters[iSlice][iRow]));
220  } else if (param.rec.tpcCompressionSortOrder == GPUSettings::SortZTimePad) {
221  CAAlgo::sortInBlock(sortBuffer, sortBuffer + smem.nCount, GPUTPCCompressionKernels_Compare<GPUSettings::SortZTimePad>(clusters->clusters[iSlice][iRow]));
222  } else if (param.rec.tpcCompressionSortOrder == GPUSettings::SortPad) {
223  CAAlgo::sortInBlock(sortBuffer, sortBuffer + smem.nCount, GPUTPCCompressionKernels_Compare<GPUSettings::SortPad>(clusters->clusters[iSlice][iRow]));
224  } else if (param.rec.tpcCompressionSortOrder == GPUSettings::SortTime) {
225  CAAlgo::sortInBlock(sortBuffer, sortBuffer + smem.nCount, GPUTPCCompressionKernels_Compare<GPUSettings::SortTime>(clusters->clusters[iSlice][iRow]));
226  }
227  GPUbarrier();
228  }
229 
230  unsigned int lastTime = 0;
231  unsigned short lastPad = 0;
232  for (unsigned int i = 0; i < smem.nCount; i++) {
233  int cidx = idOffset + i;
234  const ClusterNative& orgCl = clusters->clusters[iSlice][iRow][sortBuffer[i]];
235  c.padDiffU[cidx] = orgCl.padPacked - lastPad;
236  c.timeDiffU[cidx] = (orgCl.getTimePacked() - lastTime) & 0xFFFFFF;
237  if (param.rec.tpcCompressionModes & GPUSettings::CompressionDifferences) {
238  lastPad = orgCl.padPacked;
239  lastTime = orgCl.getTimePacked();
240  }
241 
242  unsigned short qtot = orgCl.qTot, qmax = orgCl.qMax;
243  unsigned char sigmapad = orgCl.sigmaPadPacked, sigmatime = orgCl.sigmaTimePacked;
244  if (param.rec.tpcCompressionModes & GPUSettings::CompressionTruncate) {
245  compressor.truncateSignificantBitsChargeMax(qmax, param);
246  compressor.truncateSignificantBitsCharge(qtot, param);
247  compressor.truncateSignificantBitsWidth(sigmapad, param);
248  compressor.truncateSignificantBitsWidth(sigmatime, param);
249  }
250  c.qTotU[cidx] = qtot;
251  c.qMaxU[cidx] = qmax;
252  c.sigmaPadU[cidx] = sigmapad;
253  c.sigmaTimeU[cidx] = sigmatime;
254  c.flagsU[cidx] = orgCl.getFlags();
255  }
256  if (iThread == 0) {
257  c.nSliceRowClusters[iSlice * GPUCA_ROW_COUNT + iRow] = smem.nCount;
258  CAMath::AtomicAdd(&compressor.mMemory->nStoredUnattachedClusters, smem.nCount);
259  }
260  GPUbarrier();
261  }
262 }
