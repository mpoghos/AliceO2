28 void MergerCache::init(const framework::InputRecord& inputs)
29 {
30  // sometimes timer can be one of the inputs, but we do not want to store it in the cache. Therefore,
31  // we find its position and omit it when accessing inputs.
32  mTimerPosition = inputs.getPos("timer-publish");
33  mCache = std::vector<CacheEntryQueue>(inputs.size() - (mTimerPosition >= 0));
34 
35  mCachedInputs = 0;
36  mUpdatedInputs = 0;
37 }

39 void MergerCache::cacheInputRecord(const framework::InputRecord& inputs) //todo: rename to cacheObjects ?
40 {
41  if (uninitialized()) {
42  init(inputs);
43  }
44 
45  bool shift = false;
46  for (int i = 0; i < inputs.size(); ++i) {
47  // when we iterate besides timer input, we shift the index of cache.
48  if (mTimerPosition == i) {
49  shift = true;
50  continue;
51  }
52 
53  auto input = inputs.getByPos(i);
54  if (input.header && input.payload) {
55  auto& queue = mCache[i - shift];
56 
57  mUpdatedInputs += !queue.was_updated;
58  mCachedInputs += queue.deque.empty();
59 
60  queue.was_updated = true;
61 
62  std::unique_ptr<TObject, void (*)(TObject*)> objPtr(framework::DataRefUtils::as<TObject>(input).release(), deleteTCollections);
63 
64  if (mOverwrite && !queue.deque.empty()) {
65  assert(queue.deque.size() == 1);
66  queue.deque.front() = {std::move(objPtr)};
67  } else {
68  queue.deque.push_back({std::move(objPtr)});
69  }
70  }
71  }
72 }

84 void MergerCache::clear()
85 {
86  mCachedInputs = 0;
87  mUpdatedInputs = 0;
88  // We do not use clear() on the whole mCache vector to preserve its size. Instead
89  // we clean each queue.
90  for (auto& queue : mCache) {
91  queue.deque.clear();
92  queue.was_updated = false;
93  }
94 }
