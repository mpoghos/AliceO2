69 // When reading from a ROOT file special care must happen
70 // because uint64_t is platform specific while ULong64_t is
71 // always long long unsigned int (same for the signed version).
72 // By using this traits we make sure that any 64 bit quantity
73 // read from a root file uses the ROOT datatype, not the platform one.

98 struct RootTableBuilderHelpers {
99  template <typename... TTREEREADERVALUE>
100  static void convertTTree(TableBuilder& builder,
101  TTreeReader& reader,
102  TTREEREADERVALUE&... values)
103  {
104  std::vector<std::string> branchNames = {values.GetBranchName()...};
105  auto filler = builder.preallocatedPersist<typename TreeReaderValueTraits<std::decay_t<TTREEREADERVALUE>>::Type...>(branchNames, reader.GetEntries(true));
106  reader.Restart();
107  while (reader.Next()) {
108  filler(0, ValueExtractor::deref(values)...);
109  }
110  }
111 
112  template <typename... C>
113  static void convertASoAColumns(TableBuilder& builder, TTreeReader& reader, pack<C...>)
114  {
115  return convertTTree(builder, reader, *ColumnReaderTrait<C>::createReader(reader)...);
116  }
117 
118  template <typename T>
119  static void convertASoA(TableBuilder& builder, TTreeReader& reader)
120  {
121  return convertASoAColumns(builder, reader, typename T::persistent_columns_t{});
122  }
123 };
