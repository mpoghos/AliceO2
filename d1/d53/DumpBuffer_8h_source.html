26 template <typename T, std::enable_if_t<std::is_integral<T>::value, int> = 1>
27 void dumpBuffer(gsl::span<T> buffer)
28 {
29 
30  int i{0};
31  while (i < buffer.size()) {
32  if (i % (16 / sizeof(T)) == 0) {
33  std::cout << fmt::format("\n{:8d} : ", i * sizeof(T));
34  }
35  std::cout << fmt::format("{:0{}X} ", buffer[i], sizeof(T) * 2);
36  i++;
37  }
38  std::cout << "\n";
39 }

41 uint64_t b8to64(gsl::span<uint8_t> buffer, size_t i)
42 {
43  return (static_cast<uint64_t>(buffer[i + 0])) |
44  (static_cast<uint64_t>(buffer[i + 1]) << 8) |
45  (static_cast<uint64_t>(buffer[i + 2]) << 16) |

53 void append(std::vector<uint8_t>& buffer, uint64_t w)
54 {
55  buffer.emplace_back(static_cast<uint8_t>((w & UINT64_C(0x00000000000000FF))));
56  buffer.emplace_back(static_cast<uint8_t>((w & UINT64_C(0x000000000000FF00)) >> 8));
57  buffer.emplace_back(static_cast<uint8_t>((w & UINT64_C(0x0000000000FF0000)) >> 16));
58  buffer.emplace_back(static_cast<uint8_t>((w & UINT64_C(0x00000000FF000000)) >> 24));
59  buffer.emplace_back(static_cast<uint8_t>((w & UINT64_C(0x000000FF00000000)) >> 32));
60  buffer.emplace_back(static_cast<uint8_t>((w & UINT64_C(0x0000FF0000000000)) >> 40));
61  buffer.emplace_back(static_cast<uint8_t>((w & UINT64_C(0x00FF000000000000)) >> 48));
62  buffer.emplace_back(static_cast<uint8_t>((w & UINT64_C(0xFF00000000000000)) >> 56));
63 }

65 void dumpBuffer(gsl::span<uint8_t> buffer, std::ostream& out = std::cout, size_t maxbytes = std::numeric_limits<size_t>::max())
66 {
67  int i{0};
68  int inRDH{0};
69 
70  if (buffer.size() < 8) {
71  std::cout << "Should at least get 8 bytes to be able to dump\n";
72  return;
73  }
74  while ((i < buffer.size() - 7) && i < maxbytes) {
75  if (i % 8 == 0) {
76  out << fmt::format("\n{:8d} : ", i);
77  }
78  uint64_t w = b8to64(buffer, i);
79  out << fmt::format("{:016X} {:4d} {:4d} {:4d} {:4d} {:4d} ",
80  w,
81  (w & 0x3FF0000000000) >> 40,
82  (w & 0xFFC0000000) >> 30,
83  (w & 0x3FF00000) >> 20,
84  (w & 0xFFC00) >> 10,
85  (w & 0x3FF));
86  if ((w & 0xFFFF) == 0x4004) {
87  inRDH = 8;
88  }
89  if (inRDH) {
90  --inRDH;
91  if (inRDH == 7) {
92  out << "Begin RDH ";
93  auto const rdh = o2::mch::raw::createRDH<o2::header::RAWDataHeaderV4>(buffer.subspan(i, 64));
94  std::cout << fmt::format("ORBIT {} BX {} FEEID {}", rdhOrbit(rdh), rdhBunchCrossing(rdh),
95  rdh.feeId);
96  }
97  if (inRDH == 0) {
98  out << "End RDH ";
99  }
100  } else {
101  SampaHeader h(w & 0x3FFFFFFFFFFFF);
102  if (h.packetType() == SampaPacketType::Sync) {
103  out << "SYNC";
104  } else if (h.packetType() == SampaPacketType::Data) {
105  out << fmt::format(" n10 {:4d} chip {:2d} ch {:2d}",
106  h.nof10BitWords(), h.chipAddress(), h.channelAddress());
107  }
108  }
109  i += 8;
110  }
111  out << "\n";
112 }
113 void dumpBuffer(const std::vector<uint64_t>& buffer, std::ostream& out = std::cout, size_t maxbytes = std::numeric_limits<size_t>::max())
114 {
115  std::vector<uint8_t> b8;
116  for (auto w : buffer) {
117  append(b8, w);
118  }
119  dumpBuffer(b8, out, maxbytes);
120 }
