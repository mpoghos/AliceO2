105  static void printRDH(const o2::header::RAWDataHeader* h)
106  {
107  {

111  }
112  printf("RDH| Ver:%2u Hsz:%2u Blgt:%4u FEEId:0x%04x PBit:%u\n",

114  printf("RDH|[CRU: Offs:%5u Msz:%4u LnkId:0x%02x Packet:%3u CRUId:0x%04x]\n",
115  uint32_t(h->offsetToNext), uint32_t(h->memorySize), uint32_t(h->linkID), uint32_t(h->packetCounter), uint32_t(h->cruID));
116  printf("RDH| TrgOrb:%9u HBOrb:%9u TrgBC:%4u HBBC:%4u TrgType:%u\n",
117  uint32_t(h->triggerOrbit), uint32_t(h->heartbeatOrbit), uint32_t(h->triggerBC), uint32_t(h->heartbeatBC),
118  uint32_t(h->triggerType));
119  printf("RDH| DetField:0x%05x Par:0x%04x Stop:0x%04x PageCnt:%5u\n",
120  uint32_t(h->detectorField), uint32_t(h->par), uint32_t(h->stop), uint32_t(h->pageCnt));
121  }

123  std::vector<char> to_vector()
124  {
125  constexpr int CRUWordSize = 16;
126  const char padding[CRUWordSize] = {0};
127 
128  std::vector<char> result(size() * CRUWordSize);
129  char* out = result.data();
130  std::memcpy(out, &mEventHeader, EventHeader::PayloadSize);
131  out += EventHeader::PayloadSize;
132  LOG(DEBUG) << "write header words " << (int)mEventHeader.nGBTWords << " orbit " << int(mEventHeader.orbit) << " bc " << int(mEventHeader.bc);
133  if (mIsPadded) {
134  out += CRUWordSize - EventHeader::PayloadSize;
135  }
136  for (int i = 0; i < mEventHeader.nGBTWords; ++i) {
137  std::memcpy(out, &mEventData[2 * i], EventData::PayloadSizeFirstWord);
138  LOG(DEBUG) << " 1st word " << mEventData[2 * i].channelID << " charge " << mEventData[2 * i].charge << " time " << mEventData[2 * i].time;
139  out += EventData::PayloadSizeFirstWord;
140  std::memcpy(out, &mEventData[2 * i + 1], EventData::PayloadSizeSecondWord);
141  out += EventData::PayloadSizeSecondWord;
142  LOG(DEBUG) << " 2nd word " << mEventData[2 * i + 1].channelID << " charge " << mEventData[2 * i + 1].charge << " time " << mEventData[2 * i + 1].time;
143  if (mIsPadded) {
144  out += CRUWordSize - EventData::PayloadSizeSecondWord - EventData::PayloadSizeFirstWord;
145  }
146  }
147  return result;
148  }
149  void setIsPadded(bool isPadding128)

163 class DataPageWriter
164 {
165  std::vector<char> mBuffer;
166  int mNpacketsInBuffer = 0;
167  std::vector<std::vector<char>> mPages;
168  std::vector<int> mNpackets;
169  static constexpr int MAX_Page_size = 8192;
170 
171  public:
172  o2::header::RAWDataHeader mRDH;
173  void flush(std::ostream& str)
174  {
175  writePage();
176  mRDH.stop = 0;
177  for (int page = 0; page < int(mPages.size()); ++page) {
178  mRDH.memorySize = mPages[page].size() + mRDH.headerSize;
179  mRDH.offsetToNext = mRDH.memorySize;
180  mRDH.packetCounter = mNpackets[page];
181  RawEventData::printRDH(&mRDH);
182  str.write(reinterpret_cast<const char*>(&mRDH), sizeof(mRDH));
183  str.write(mPages[page].data(), mPages[page].size());
184  mRDH.pageCnt++;
185  }
186  if (!mPages.empty()) {
187  mRDH.memorySize = mRDH.headerSize;
188  mRDH.offsetToNext = mRDH.memorySize;
189  mRDH.stop = 1;
190  mRDH.pageCnt++;
191  RawEventData::printRDH(&mRDH);
192  str.write(reinterpret_cast<const char*>(&mRDH), sizeof(mRDH));
193  mPages.clear();
194  mNpackets.clear();
195  }
196  }
197 
198  void writePage()
199  {
200  if (mBuffer.size() == 0)
201  return;
202  mPages.emplace_back(std::move(mBuffer));
203  mNpackets.push_back(mNpacketsInBuffer);
204  mNpacketsInBuffer = 0;
205  mBuffer.clear();
206  }
207 
208  void write(std::vector<char> const& new_data)
209  {
210  if (mBuffer.size() + new_data.size() + mRDH.headerSize > MAX_Page_size)
211  writePage();
212  mBuffer.insert(mBuffer.end(), new_data.begin(), new_data.end());
213  mNpacketsInBuffer++;
214  }
