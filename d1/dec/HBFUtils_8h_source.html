26 /*
27  In the MC->Raw conversion we have to make sure that
28  1) The HB and TF starts are in sync for all detectors regardless on time (bc/orbir)
29  distribution of its signal.
30  2) All HBF and TF (RAWDataHeaders with corresponding HB and TF trigger flags) are present
31  in the emulated raw data, even if some of them had no data in particular detector.
32 
33  The HBFUtils class provides tools for interaction record -> HBF conversion and sampling
34  of IRs for which the HBF RDH should be added to the raw data from the CRU.
35 
36  See testHBFUtils.cxx for the outline of generating HBF frames for simulated data.
37  */
38

92  /*//-------------------------------------------------------------------------------------
93  Fill provided vector (cleaned) by interaction records (bc/orbit) for HBFs, considering
94  BCs between interaction records "fromIR" and "toIR" (inclusive).
95  This method provides the IRs for RDHs to add obligatory for the MC->raw conversion,
96  in order to avoid missing HBFs (or even TFs)
97  Typical use case: assume we are converting to RAW data the digits corresponding to triggers
98  for Int.records ir[0], ir[1], ... ir[N]
99  The pseudo-code:
100 
101  HBFUtils sampler;
102  uint8_t packetCounter = 0;
103  std::vector<o2::InteractionRecord> HBIRVec;
104  auto irFrom = sampler.getFirstIR();
105  for (int i=0;i<N;i++) {
106  int nHBF = sampler.fillHBIRvector(HBIRVec, irFrom, ir[i]);
107  irFrom = ir[i]+1;
108  // nHBF-1 HBframes don't have data, we need to create empty HBFs for them
109  for (int j=0;j<nHBF-1;j++) {
110  auto rdh = sampler.createRDH<RAWDataHeader>( HBIRVec[j] );
111  // dress rdh with cruID/FEE/Link ID
112  rdh.packetCounter = packetCounter++;
113  rdh.memorySize = sizeof(rdh);
114  rdh.offsetToNext = sizeof(rdh);
115  FLUSH_TO_SINK(&rdh, sizeof(rdh)); // open empty HBF
116  rdh.stop = 0x1;
117  rdh.pageCnt++;
118  FLUSH_TO_SINK(&rdh, sizeof(rdh)); // close empty HBF
119  }
120  // write RDH for the HBF with data
121  auto rdh = HBIRVec.back();
122  // dress rdh with cruID/FEE/Link ID, estimate size, offset etc. and flush
123  // flush raw data payload
124  }
125  // see tstHBFUtils for more details
126  //-------------------------------------------------------------------------------------*/
127  int fillHBIRvector(std::vector<IR>& dst, const IR& fromIR, const IR& toIR) const;

170 template <>
171 inline void HBFUtils::updateRDH<o2::header::RAWDataHeaderV5>(o2::header::RAWDataHeaderV5& rdh, const o2::InteractionRecord& rec) const
172 {
173  auto tfhb = getTFandHBinTF(rec);
174  rdh.bunchCrossing = mFirstIR.bc;
175  rdh.orbit = rec.orbit;
176  //
177  if (rec.bc == mFirstIR.bc) { // if we are starting new HB, set the HB trigger flag
178  rdh.triggerType |= o2::trigger::ORBIT | o2::trigger::HB;
179  if (tfhb.second == 0) { // if we are starting new TF, set the TF trigger flag
180  rdh.triggerType |= o2::trigger::TF;
181  }
182  }
183 }

186 template <>
187 inline void HBFUtils::updateRDH<o2::header::RAWDataHeaderV4>(o2::header::RAWDataHeaderV4& rdh, const o2::InteractionRecord& rec) const
188 {
189  auto tfhb = getTFandHBinTF(rec);
190  rdh.triggerBC = rec.bc;
191  rdh.triggerOrbit = rec.orbit;
192 
193  rdh.heartbeatBC = mFirstIR.bc;
194  rdh.heartbeatOrbit = rec.orbit;
195  //
196  if (rec.bc == mFirstIR.bc) { // if we are starting new HB, set the HB trigger flag
197  rdh.triggerType |= o2::trigger::ORBIT | o2::trigger::HB;
198  if (tfhb.second == 0) { // if we are starting new TF, set the TF trigger flag
199  rdh.triggerType |= o2::trigger::TF;
200  }
201  }

223 inline LinkSubSpec_t HBFUtils::getSubSpec(uint16_t cru, uint8_t link, uint8_t endpoint)
224 {
225  // define subspecification as in DataDistribution
226  int linkValue = (LinkSubSpec_t(link) + 1) << (endpoint == 1 ? 8 : 0);
227  return (LinkSubSpec_t(cru) << 16) | linkValue;
