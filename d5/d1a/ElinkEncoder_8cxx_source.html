21 uint16_t chipAddress(uint8_t elinkId, uint8_t chId)
22 {
23  auto opt = o2::mch::raw::indexFromElinkId(elinkId);
24  if (!opt.has_value()) {
25  throw std::invalid_argument(fmt::format("elinkId {} is not valid", elinkId));
26  }
27  return opt.value() * 2 + (chId > 31);
28 }

34 SampaHeader buildHeader(uint8_t elinkId, uint8_t chId, const std::vector<SampaCluster>& data)
35 {
36  impl::assertIsInRange("chId", chId, 0, 63);
37 
38  SampaHeader header;
39  header.packetType(SampaPacketType::Data);
40 
41  uint16_t n10{0};
42  for (const auto& s : data) {
43  n10 += s.nof10BitWords();
44  }
45  impl::assertNofBits("nof10BitWords", n10, 10);
46  header.nof10BitWords(n10);
47  header.chipAddress(chipAddress(elinkId, chId));
48  header.channelAddress(chId % 32);
49  header.hammingCode(computeHammingCode(header.uint64()));
50  header.headerParity(computeHeaderParity(header.uint64()));
51 
52  //header.bunchCrossingCounter(mLocalBunchCrossing); //FIXME: how is this one evolving ?
53  // FIXME: compute payload parity
54 
55  return header;
56 }
