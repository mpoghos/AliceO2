57 uint16_t SampaCluster::nof10BitWords() const
58 {
59  uint16_t n10{2}; // 10 bits (nsamples) + 10 bits (timestamp)
60  if (isClusterSum()) {
61  n10 += 2; // 20 bits (chargesum)
62  } else {
63  for (auto s : samples) {
64  ++n10; // 10 bits for each sample
65  }
66  }
67  return n10;
68 }
