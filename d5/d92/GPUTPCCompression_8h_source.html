111 template <class T>
112 GPUdi() void GPUTPCCompression::truncateSignificantBits(T& v, unsigned int nBits, unsigned int max)
113 {
114  if (nBits == 0) {
115  return;
116  }
117 
118  unsigned int val = v;
119  unsigned int ldz = sizeof(unsigned int) * 8 - CAMath::Clz(val);
120  if (val && ldz > nBits) {
121  if (val & (1 << (ldz - nBits - 1))) {
122  val += (1 << (ldz - nBits - 1));
123  ldz = sizeof(unsigned int) * 8 - CAMath::Clz(val);
124  }
125  val &= ((1 << ldz) - 1) ^ ((1 << (ldz - nBits)) - 1);
126  if (val >= max) {
127  val = max - 1;
128  }
129  // GPUInfo("CHANGING X %x --> %x", (unsigned int) v, val);
130  v = val;
131  }
132 }
