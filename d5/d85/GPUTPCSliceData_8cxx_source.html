28 // calculates an approximation for 1/sqrt(x)
29 // Google for 0x5f3759df :)
30 static inline float fastInvSqrt(float _x)
31 {
32  // the function calculates fast inverse sqrt
33  union {
34  float f;
35  int i;
36  } x = {_x};
37  const float xhalf = 0.5f * x.f;
38  x.i = 0x5f3759df - (x.i >> 1);
39  x.f = x.f * (1.5f - xhalf * x.f * x.f);
40  return x.f;
41 }

43 inline void GPUTPCSliceData::CreateGrid(GPUTPCRow* row, const float2* data, int ClusterDataHitNumberOffset)
44 {
45  // grid creation
46  if (row->NHits() <= 0) { // no hits or invalid data
47  // grid coordinates don't matter, since there are no hits
48  row->mGrid.CreateEmpty();
49  return;
50  }
51 
52  float yMin = 1.e3f;
53  float yMax = -1.e3f;
54  float zMin = 1.e3f;
55  float zMax = -1.e3f;
56  for (int i = ClusterDataHitNumberOffset; i < ClusterDataHitNumberOffset + row->mNHits; ++i) {
57  const float y = data[i].x;
58  const float z = data[i].y;
59  if (yMax < y) {
60  yMax = y;
61  }
62  if (yMin > y) {
63  yMin = y;
64  }
65  if (zMax < z) {
66  zMax = z;
67  }
68  if (zMin > z) {
69  zMin = z;
70  }
71  }
72 
73  float dz = zMax - zMin;
74  float tfFactor = 1.;
75  if (dz > 270.) {
76  tfFactor = dz / 250.;
77  dz = 250.;
78  }
79  const float norm = fastInvSqrt(row->mNHits / tfFactor);
80  row->mGrid.Create(yMin, yMax, zMin, zMax, CAMath::Max((yMax - yMin) * norm, 2.f), CAMath::Max(dz * norm, 2.f));
81 }

83 inline int GPUTPCSliceData::PackHitData(GPUTPCRow* const row, const GPUTPCHit* binSortedHits)
84 {
85  // hit data packing
86  static const float maxVal = (((long long int)1 << CAMath::Min((size_t)24, sizeof(cahit) * 8)) - 1); // Stay within float precision in any case!
87  static const float packingConstant = 1.f / (maxVal - 2.);
88  const float y0 = row->mGrid.YMin();
89  const float z0 = row->mGrid.ZMin();
90  const float stepY = (row->mGrid.YMax() - y0) * packingConstant;
91  const float stepZ = (row->mGrid.ZMax() - z0) * packingConstant;
92  const float stepYi = 1.f / stepY;
93  const float stepZi = 1.f / stepZ;
94 
95  row->mHy0 = y0;
96  row->mHz0 = z0;
97  row->mHstepY = stepY;
98  row->mHstepZ = stepZ;
99  row->mHstepYi = stepYi;
100  row->mHstepZi = stepZi;
101 
102  for (int hitIndex = 0; hitIndex < row->mNHits; ++hitIndex) {
103  // bin sorted index!
104  const int globalHitIndex = row->mHitNumberOffset + hitIndex;
105  const GPUTPCHit& hh = binSortedHits[hitIndex];
106  const float xx = ((hh.Y() - y0) * stepYi) + .5;
107  const float yy = ((hh.Z() - z0) * stepZi) + .5;
108  if (xx < 0 || yy < 0 || xx > maxVal || yy > maxVal) {
109  std::cout << "!!!! hit packing error!!! " << xx << " " << yy << " (" << maxVal << ")" << std::endl;
110  return 1;
111  }
112  // HitData is bin sorted
113  mHitData[globalHitIndex].x = (cahit)xx;
114  mHitData[globalHitIndex].y = (cahit)yy;
115  }
116  return 0;
117 }

178 int GPUTPCSliceData::InitFromClusterData(GPUconstantref() const MEM_CONSTANT(GPUConstantMem) * mem, int iSlice)
179 {
181  // 0. sort rows
183 
184  mMaxZ = 0.f;
185 
186  std::unique_ptr<float2[]> YZData_p(new float2[mNumberOfHits]);
187  std::unique_ptr<int[]> tmpHitIndex_p(new int[mNumberOfHits]);
188  float2* YZData = YZData_p.get();
189  int* tmpHitIndex = tmpHitIndex_p.get();
190 
191  int RowOffset[GPUCA_ROW_COUNT];
192  const unsigned int* NumberOfClustersInRow;
193  unsigned int NumberOfClustersInRowA[GPUCA_ROW_COUNT];
194  if (mem->param.earlyTpcTransform) {
195  memset(NumberOfClustersInRowA, 0, GPUCA_ROW_COUNT * sizeof(NumberOfClustersInRowA[0]));
196  for (int i = 0; i < mNumberOfHits; i++) {
197  const int tmpRow = mClusterData[i].row;
198  NumberOfClustersInRowA[tmpRow]++;
199  }
200  NumberOfClustersInRow = NumberOfClustersInRowA;
201  } else {
202  NumberOfClustersInRow = &mem->ioPtrs.clustersNative->nClusters[iSlice][0];
203  }
204 
205  int tmpOffset = 0;
206  for (int i = 0; i < GPUCA_ROW_COUNT; i++) {
207  if ((long long int)NumberOfClustersInRow[i] >= ((long long int)1 << (sizeof(calink) * 8))) {
208  GPUError("Too many clusters in row %d for row indexing (%d >= %lld), indexing insufficient", i, NumberOfClustersInRow[i], ((long long int)1 << (sizeof(calink) * 8)));
209  return (1);
210  }
211  if (NumberOfClustersInRow[i] >= (1 << 24)) {
212  GPUError("Too many clusters in row %d for hit id indexing (%d >= %d), indexing insufficient", i, NumberOfClustersInRow[i], 1 << 24);
213  return (1);
214  }
215  RowOffset[i] = tmpOffset;
216  tmpOffset += NumberOfClustersInRow[i];
217  }
218 
219  if (mem->param.earlyTpcTransform) {
220  int RowsFilled[GPUCA_ROW_COUNT];
221  memset(RowsFilled, 0, GPUCA_ROW_COUNT * sizeof(int));
222  for (int i = 0; i < mNumberOfHits; i++) {
223  float2 tmp;
224  tmp.x = mClusterData[i].y;
225  tmp.y = mClusterData[i].z;
226  if (fabsf(tmp.y) > mMaxZ) {
227  mMaxZ = fabsf(tmp.y);
228  }
229  int tmpRow = mClusterData[i].row;
230  int newIndex = RowOffset[tmpRow] + (RowsFilled[tmpRow])++;
231  YZData[newIndex] = tmp;
232  tmpHitIndex[newIndex] = i;
233  }
234  } else {
235  size_t k = 0;
236  for (int i = 0; i < GPUCA_ROW_COUNT; i++) {
237  for (unsigned int j = 0; j < NumberOfClustersInRow[i]; j++) {
238  float2 tmp;
239  float x;
240  GPUTPCConvertImpl::convert(*mem, iSlice, i, mem->ioPtrs.clustersNative->clusters[iSlice][i][j].getPad(), mem->ioPtrs.clustersNative->clusters[iSlice][i][j].getTime(), x, tmp.x, tmp.y);
241  tmpHitIndex[k] = k;
242  YZData[k++] = tmp;
243  }
244  }
245  }
246 
248  // 2. fill HitData and FirstHitInBin
250 
251  vecpod<GPUTPCHit> binSortedHits(mNumberOfHits + sizeof(GPUCA_ROWALIGNMENT));
252 
253  int gridContentOffset = 0;
254  int hitOffset = 0;
255 
256  int binCreationMemorySize = 103 * 2 + mNumberOfHits;
257  vecpod<calink> binCreationMemory(binCreationMemorySize);
258 
259  for (int rowIndex = 0; rowIndex < GPUCA_ROW_COUNT; ++rowIndex) {
260  GPUTPCRow& row = mRows[rowIndex];
261  if (NumberOfClustersInRow[rowIndex] == 0) {
262  row.mGrid.CreateEmpty();
263  row.mNHits = 0;
264  row.mFullSize = 0;
265  row.mHitNumberOffset = 0;
266  row.mFirstHitInBinOffset = 0;
267  row.mHy0 = 0.f;
268  row.mHz0 = 0.f;
269  row.mHstepY = 1.f;
270  row.mHstepZ = 1.f;
271  row.mHstepYi = 1.f;
272  row.mHstepZi = 1.f;
273  continue;
274  }
275  row.mNHits = NumberOfClustersInRow[rowIndex];
276  row.mHitNumberOffset = hitOffset;
277  hitOffset += GPUProcessor::nextMultipleOf<sizeof(GPUCA_ROWALIGNMENT) / sizeof(unsigned short)>(NumberOfClustersInRow[rowIndex]);
278 
279  row.mFirstHitInBinOffset = gridContentOffset;
280 
281  CreateGrid(&row, YZData, RowOffset[rowIndex]);
282  const GPUTPCGrid& grid = row.mGrid;
283  const int numberOfBins = grid.N();
284  if ((long long int)numberOfBins >= ((long long int)1 << (sizeof(calink) * 8))) {
285  GPUError("Too many bins in row %d for grid (%d >= %lld), indexing insufficient", rowIndex, numberOfBins, ((long long int)1 << (sizeof(calink) * 8)));
286  return (1);
287  }
288 
289  int binCreationMemorySizeNew = numberOfBins * 2 + 6 + row.mNHits + sizeof(GPUCA_ROWALIGNMENT) / sizeof(unsigned short) * (GPUCA_ROW_COUNT + 1) + 1;
290  if (binCreationMemorySizeNew > binCreationMemorySize) {
291  binCreationMemorySize = binCreationMemorySizeNew;
292  binCreationMemory.resize(binCreationMemorySize);
293  }
294 
295  calink* c = binCreationMemory.data(); // number of hits in all previous bins
296  calink* bins = c + numberOfBins + 3; // cache for the bin index for every hit in this row, 3 extra empty bins at the end!!!
297  calink* filled = bins + row.mNHits; // counts how many hits there are per bin
298 
299  for (unsigned int bin = 0; bin < row.mGrid.N() + 3; ++bin) {
300  filled[bin] = 0; // initialize filled[] to 0
301  }
302  for (int hitIndex = 0; hitIndex < row.mNHits; ++hitIndex) {
303  const int globalHitIndex = RowOffset[rowIndex] + hitIndex;
304  const calink bin = row.mGrid.GetBin(YZData[globalHitIndex].x, YZData[globalHitIndex].y);
305 
306  bins[hitIndex] = bin;
307  ++filled[bin];
308  }
309 
310  calink n = 0;
311  for (int bin = 0; bin < numberOfBins + 3; ++bin) {
312  c[bin] = n;
313  n += filled[bin];
314  }
315 
316  for (int hitIndex = 0; hitIndex < row.mNHits; ++hitIndex) {
317  const calink bin = bins[hitIndex];
318  --filled[bin];
319  const calink ind = c[bin] + filled[bin]; // generate an index for this hit that is >= c[bin] and < c[bin + 1]
320  const int globalBinsortedIndex = row.mHitNumberOffset + ind;
321  const int globalHitIndex = RowOffset[rowIndex] + hitIndex;
322 
323  // allows to find the global hit index / coordinates from a global bin sorted hit index
324  mClusterDataIndex[globalBinsortedIndex] = tmpHitIndex[globalHitIndex];
325  binSortedHits[ind].SetY(YZData[globalHitIndex].x);
326  binSortedHits[ind].SetZ(YZData[globalHitIndex].y);
327  }
328 
329  if (PackHitData(&row, binSortedHits.data())) {
330  return (1);
331  }
332 
333  for (int i = 0; i < numberOfBins; ++i) {
334  mFirstHitInBin[row.mFirstHitInBinOffset + i] = c[i]; // global bin-sorted hit index
335  }
336  const calink a = c[numberOfBins];
337  // grid.N is <= row.mNHits
338  const int nn = numberOfBins + grid.Ny() + 3;
339  for (int i = numberOfBins; i < nn; ++i) {
340  mFirstHitInBin[row.mFirstHitInBinOffset + i] = a;
341  }
342 
343  row.mFullSize = nn;
344  gridContentOffset += nn;
345 
346  // Make pointer aligned
347  gridContentOffset = GPUProcessor::nextMultipleOf<sizeof(GPUCA_ROWALIGNMENT) / sizeof(calink)>(gridContentOffset);
348  }
349 
350  return (0);
351 }
