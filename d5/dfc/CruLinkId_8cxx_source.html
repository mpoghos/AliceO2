36 CruLinkId decodeCruLinkId(uint32_t x)
37 {
38  uint16_t cruId = static_cast<uint16_t>((x & 0x3FFE0) >> 5);
39  uint16_t linkId = static_cast<uint8_t>(x & 0x1F);
40  uint16_t deId = static_cast<uint16_t>((x & 0xFFFC0000) >> 17);
41  return CruLinkId(cruId, linkId, deId);
42 }
