262  // this creates the workflow spec
263  struct ProcessAttributes {
264  // worker instance
265  std::shared_ptr<WriterType> writer;
266  // keys for branch name options
267  std::vector<std::pair<std::string, std::string>> branchNameOptions;
268  // number of events to be processed
269  int nEvents = -1;
270  // starting with all inputs, every input which has been indicated 'ready' is removed
271  std::unordered_set<std::string> activeInputs;
272  // event counter
273  int counter = 0;
274  // indicate what to terminate upon ready: process or workflow
275  TerminationPolicy terminationPolicy = TerminationPolicy::Process;
276  // custom termination condition
277  TerminationCondition terminationCondition;
278  // the total number of served branches on the n inputs
279  size_t nofBranches;
280  };
281  auto processAttributes = std::make_shared<ProcessAttributes>();
282  processAttributes->writer = mWriter;
283  processAttributes->branchNameOptions = mBranchNameOptions;
284  processAttributes->terminationCondition = mTerminationCondition;
285 
286  // set the list of active inputs, every input which is indecated as 'ready' will be removed from the list
287  // the process is ready if the list is empty
288  for (auto const& input : mInputs) {
289  processAttributes->activeInputs.emplace(input.binding);
290  }
291  processAttributes->nofBranches = mNofBranches;
292 
293  // the init function is returned to the DPL in order to init the process
294  auto initFct = [processAttributes, TerminationPolicyMap = TerminationPolicyMap](InitContext& ic) {
295  auto& branchNameOptions = processAttributes->branchNameOptions;
296  auto filename = ic.options().get<std::string>("outfile");
297  auto treename = ic.options().get<std::string>("treename");
298  processAttributes->nEvents = ic.options().get<int>("nevents");
299  if (processAttributes->nEvents > 0 && processAttributes->activeInputs.size() != processAttributes->nofBranches) {
300  LOG(WARNING) << "the n inputs serve in total m branches with n != m, this means that there will be data for\n"
301  << "different branches on the same input. Be aware that the --nevents option might lead to incomplete\n"
302  << "data in the output file as the number of processed input sets is counted";
303  }
304  try {
305  processAttributes->terminationPolicy = TerminationPolicyMap.at(ic.options().get<std::string>("terminate"));
306  } catch (std::out_of_range&) {
307  throw std::invalid_argument(std::string("invalid termination policy: ") + ic.options().get<std::string>("terminate"));
308  }
309  if (filename.empty() || treename.empty()) {
310  throw std::invalid_argument("output file name and tree name are mandatory options");
311  }
312  for (size_t branchIndex = 0; branchIndex < branchNameOptions.size(); branchIndex++) {
313  // pair of key (first) - value (second)
314  if (branchNameOptions[branchIndex].first.empty()) {
315  continue;
316  }
317  auto branchName = ic.options().get<std::string>(branchNameOptions[branchIndex].first.c_str());
318  processAttributes->writer->setBranchName(branchIndex, branchName.c_str());
319  }
320  processAttributes->writer->init(filename.c_str(), treename.c_str());
321 
322  // the callback to be set as hook at stop of processing for the framework
323  auto finishWriting = [processAttributes]() {
324  processAttributes->writer->close();
325  };
326  ic.services().get<CallbackService>().set(CallbackService::Id::Stop, finishWriting);
327 
328  auto processingFct = [processAttributes](ProcessingContext& pc) {
329  auto& writer = processAttributes->writer;
330  auto& terminationPolicy = processAttributes->terminationPolicy;
331  auto& terminationCondition = processAttributes->terminationCondition;
332  auto& activeInputs = processAttributes->activeInputs;
333  auto& counter = processAttributes->counter;
334  auto& nEvents = processAttributes->nEvents;
335  if (writer->isClosed()) {
336  return;
337  }
338 
339  // if the termination condition contains function of type CheckProcessing, this is checked
340  // before the processing, currently implemented logic:
341  // - processing is skipped if this is indicated by at least one input
342  // - if input is 'ready' it is removed from the list of active inputs
343  auto checkProcessing = [&terminationCondition, &activeInputs](auto const& inputs) {
344  bool doProcessing = true;
345  if (std::holds_alternative<TerminationCondition::CheckProcessing>(terminationCondition.check)) {
346  auto& check = std::get<TerminationCondition::CheckProcessing>(terminationCondition.check);
347  for (auto const& ref : inputs) {
348  auto iter = activeInputs.find(ref.spec->binding);
349  auto [action, ready] = check(ref);
350  if (action == TerminationCondition::Action::SkipProcessing) {
351  // this condition tells us not to process the data any further
352  doProcessing = false;
353  }
354  if (iter != activeInputs.end() && ready) {
355  // this input is ready, remove from active inputs
356  activeInputs.erase(iter);
357  }
358  }
359  }
360  return doProcessing;
361  };
362  // if the termination condition contains function of type CheckReady, this is checked
363  // after the processing. All inputs marked 'ready' are removed from the list of active inputs,
364  // process treated as 'ready' if no active inputs
365  auto checkReady = [&terminationCondition, &activeInputs](auto const& inputs) {
366  if (std::holds_alternative<TerminationCondition::CheckReady>(terminationCondition.check)) {
367  auto& check = std::get<TerminationCondition::CheckReady>(terminationCondition.check);
368  for (auto const& ref : inputs) {
369  auto iter = activeInputs.find(ref.spec->binding);
370  if (iter != activeInputs.end() && check(ref)) {
371  // this input is ready, remove from active inputs
372  activeInputs.erase(iter);
373  }
374  }
375  }
376  return activeInputs.size() == 0;
377  };
378 
379  if (checkProcessing(pc.inputs())) {
380  (*writer)(pc.inputs());
381  counter = counter + 1;
382  }
383 
384  if ((nEvents >= 0 && counter == nEvents) || checkReady(pc.inputs())) {
385  writer->close();
386  pc.services().get<ControlService>().readyToQuit(terminationPolicy == TerminationPolicy::Workflow ? QuitRequest::All : QuitRequest::Me);
387  }
388  };
389 
390  return processingFct;
391  };

408  {"configurable branch name"} // help message

452  template <size_t N, typename... Args>
453  void parseConstructorArgs(TerminationPolicy arg, Args&&... args)
454  {
455  static_assert(N == 0, "wrong argument order, default file and tree, and all options must come before branch specs");
456  for (const auto& policy : TerminationPolicyMap) {
457  if (policy.second == arg) {
458  mDefaultTerminationPolicy = policy.first;
459  return parseConstructorArgs<N>(std::forward<Args>(args)...);
460  }
461  }
462  // here we only get if the enum and map definitions are inconsistent
463  throw std::logic_error("Internal mismatch of policy ids and keys");
464  }
