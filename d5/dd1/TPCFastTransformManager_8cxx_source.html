28 int TPCFastTransformManager::create(TPCFastTransform& fastTransform, AliTPCTransform* transform, Long_t TimeStamp)
29 {
31 
32  AliTPCcalibDB* pCalib = AliTPCcalibDB::Instance();
33  if (!pCalib) {
34  return storeError(-1, "TPCFastTransformManager::Init: No TPC calibration instance found");
35  }
36 
37  AliTPCParam* tpcParam = pCalib->GetParameters();
38  if (!tpcParam) {
39  return storeError(-2, "TPCFastTransformManager::Init: No TPCParam object found");
40  }
41 
42  if (!transform) {
43  transform = pCalib->GetTransform();
44  }
45  if (!transform) {
46  return storeError(-3, "TPCFastTransformManager::Init: No TPC transformation found");
47  }
48 
49  mOrigTransform = transform;
50 
51  tpcParam->Update();
52  tpcParam->ReadGeoMatrices();
53 
54  const AliTPCRecoParam* rec = transform->GetCurrentRecoParam();
55  if (!rec) {
56  return storeError(-5, "TPCFastTransformManager::Init: No TPC Reco Param set in transformation");
57  }
58 
59  bool useCorrectionMap = rec->GetUseCorrectionMap();
60 
61  if (useCorrectionMap) {
62  transform->SetCorrectionMapMode(kTRUE); // If the simulation set this to false to simulate distortions, we need to reverse it for the transformation
63  }
64  // find last calibrated time bin
65 
66  fLastTimeBin = rec->GetLastBin();
67 
68  const int nRows = tpcParam->GetNRowLow() + tpcParam->GetNRowUp();
69 
70  TPCFastTransformGeo geo;
71 
72  { // construct the geometry
73  geo.startConstruction(nRows);
74 
75  float tpcZlengthSideA = tpcParam->GetZLength(0);
76  float tpcZlengthSideC = tpcParam->GetZLength(TPCFastTransformGeo::getNumberOfSlices() / 2);
77 
78  geo.setTPCzLength(tpcZlengthSideA, tpcZlengthSideC);
79  geo.setTPCalignmentZ(-mOrigTransform->GetDeltaZCorrTime());
80 
81  for (int row = 0; row < geo.getNumberOfRows(); row++) {
82  int slice = 0, sector = 0, secrow = 0;
83  AliHLTTPCGeometry::Slice2Sector(slice, row, sector, secrow);
84  Int_t nPads = tpcParam->GetNPads(sector, secrow);
85  float xRow = tpcParam->GetPadRowRadii(sector, secrow);
86  float padWidth = tpcParam->GetInnerPadPitchWidth();
87  if (row >= tpcParam->GetNRowLow()) {
88  padWidth = tpcParam->GetOuterPadPitchWidth();
89  }
90  geo.setTPCrow(row, xRow, nPads, padWidth);
91  }
92  geo.finishConstruction();
93  }
94 
95  TPCDistortionIRS distortion;
96 
97  { // create the distortion map
98 
99  const int nDistortionScenarios = 1;
100 
101  distortion.startConstruction(geo, nDistortionScenarios);
102 
103  IrregularSpline2D3D spline;
104 
105  { // adjust the number of knots and the knot positions for the TPC distortion splines
106  // make bining similar to old HLT transformation
107  // TODO: use IrregularSpline2D3DCaibrator similar to O2 code
108 
109  int nKnotsU = 15;
110  int nAxisTicksU = tpcParam->GetNPads(0, 10);
111  int nKnotsV = 20;
112  int nAxisTicksV = fLastTimeBin + 1;
113  float knotsU[nKnotsU];
114  float knotsV[nKnotsV];
115  for (int i = 0; i < nKnotsU; i++) {
116  knotsU[i] = 1. / (nKnotsU - 1) * i;
117  }
118  for (int i = 0; i < nKnotsV; i++) {
119  knotsV[i] = 1. / (nKnotsV - 1) * i;
120  }
121 
122  double d1 = 0.6;
123  double d2 = 0.9 - d1;
124  double d3 = 1. - d2 - d1;
125 
126  for (int i = 0; i < 5; i++) { // 5 bins in first 6% of drift
127  knotsV[i] = i / 4. * d1;
128  }
129  for (int i = 0; i < 10; i++) { // 10 bins for 6% <-> 90%
130  knotsV[4 + i] = d1 + i / 9. * d2;
131  }
132  for (int i = 0; i < 5; i++) { // 5 bins for last 90% <-> 100%
133  knotsV[13 + i] = d1 + d2 + i / 4. * d3;
134  }
135 
136  spline.construct(nKnotsU, knotsU, nAxisTicksU, nKnotsV, knotsV, nAxisTicksV);
137  } // spline
138 
139  int scenario = 0;
140  distortion.setSplineScenario(scenario, spline);
141 
142  for (int row = 0; row < geo.getNumberOfRows(); row++) {
143  distortion.setRowScenarioID(row, scenario);
144  }
145 
146  distortion.finishConstruction();
147  } // .. create the distortion map
148 
149  { // create the fast transform object
150 
151  fastTransform.startConstruction(distortion);
152 
153  // tell the transformation to apply the space charge distortions
154  fastTransform.setApplyDistortionOn();
155 
156  // set some initial calibration values, will be reinitialised later int updateCalibration()
157  const float t0 = 0.;
158  const float vDrift = 0.f;
159  const float vdCorrY = 0.;
160  const float ldCorr = 0.;
161  const float tofCorr = 0.;
162  const float primVtxZ = 0.;
163  const long int initTimeStamp = -1;
164  fastTransform.setCalibration(initTimeStamp, t0, vDrift, vdCorrY, ldCorr, tofCorr, primVtxZ);
165 
166  fastTransform.finishConstruction();
167  }
168 
169  return updateCalibration(fastTransform, TimeStamp);
170 }

172 int TPCFastTransformManager::updateCalibration(TPCFastTransform& fastTransform, Long_t TimeStamp)
173 {
174  // Update the calibration with the new time stamp
175 
176  Long_t lastTS = fastTransform.getTimeStamp();
177 
178  // deinitialize
179 
180  fastTransform.setTimeStamp(-1);
181 
182  if (TimeStamp < 0) {
183  return 0;
184  }
185 
186  // search for the calibration database
187 
188  if (!mOrigTransform) {
189  return storeError(-1, "TPCFastTransformManager::SetCurrentTimeStamp: TPC transformation has not been set properly");
190  }
191 
192  AliTPCcalibDB* pCalib = AliTPCcalibDB::Instance();
193  if (!pCalib) {
194  return storeError(-2, "TPCFastTransformManager::SetCurrentTimeStamp: No TPC calibration found");
195  }
196 
197  AliTPCParam* tpcParam = pCalib->GetParameters();
198  if (!tpcParam) {
199  return storeError(-3, "TPCFastTransformManager::SetCurrentTimeStamp: No TPCParam object found");
200  }
201 
202  AliTPCRecoParam* recoParam = mOrigTransform->GetCurrentRecoParamNonConst();
203  if (!recoParam) {
204  return storeError(-5, "TPCFastTransformManager::Init: No TPC Reco Param set in transformation");
205  }
206 
207  // calibration found, set the initialized status back
208 
209  fastTransform.setTimeStamp(lastTS);
210 
211  // less than 60 seconds from the previois time stamp, don't do anything
212 
213  if (lastTS >= 0 && TMath::Abs(lastTS - TimeStamp) < 60) {
214  return 0;
215  }
216 
217  // start the initialization
218 
219  bool useCorrectionMap = recoParam->GetUseCorrectionMap();
220 
221  if (useCorrectionMap) {
222  // If the simulation set this to false to simulate distortions, we need to reverse it for the transformation
223  // This is a design feature. Historically HLT code runs as a part of simulation, not reconstruction.
224  mOrigTransform->SetCorrectionMapMode(kTRUE);
225  }
226 
227  // set the current time stamp
228 
229  mOrigTransform->SetCurrentTimeStamp(static_cast<UInt_t>(TimeStamp));
230  fastTransform.setTimeStamp(TimeStamp);
231 
232  // find last calibrated time bin
233 
234  fLastTimeBin = recoParam->GetLastBin();
235 
236  double t0 = mOrigTransform->GetTBinOffset();
237  double driftCorrPT = mOrigTransform->GetDriftCorrPT();
238  double vdCorrectionTime = mOrigTransform->GetVDCorrectionTime();
239  double vdCorrectionTimeGY = mOrigTransform->GetVDCorrectionTimeGY();
240  double time0CorrTime = mOrigTransform->GetTime0CorrTime();
241 
242  // original formula:
243  // L = (t-t0)*ZWidth*driftCorrPT*vdCorrectionTime*( 1 + yLab*vdCorrectionTimeGY ) - time0CorrTime + 3.*tpcParam->GetZSigma();
244  // Z = Z(L) - fDeltaZCorrTime
245  // chebyshev distortions for xyz
246  // Time-of-flight correction: ldrift += dist-to-vtx*tofCorr
247 
248  // fast transform formula:
249  // L = (t-t0)*(mVdrift + mVdriftCorrY*yLab ) + mLdriftCorr
250  // Z = Z(L) + tpcAlignmentZ
251  // spline distortions for xyz
252  // Time-of-flight correction: ldrift += dist-to-vtx*tofCorr
253 
254  double vDrift = tpcParam->GetZWidth() * driftCorrPT * vdCorrectionTime;
255  double vdCorrY = vDrift * vdCorrectionTimeGY;
256  double ldCorr = -time0CorrTime + 3 * tpcParam->GetZSigma();
257 
258  double tofCorr = (0.01 * tpcParam->GetDriftV()) / TMath::C();
259  double primVtxZ = mOrigTransform->GetPrimVertex()[2];
260 
261  bool useTOFcorrection = recoParam->GetUseTOFCorrection();
262 
263  if (!useTOFcorrection) {
264  tofCorr = 0;
265  }
266 
267  fastTransform.setCalibration(TimeStamp, t0, vDrift, vdCorrY, ldCorr, tofCorr, primVtxZ);
268 
269  // now calculate the distortion map: dx,du,dv = ( origTransform() -> x,u,v) - fastTransformNominal:x,u,v
270 
271  const TPCFastTransformGeo& geo = fastTransform.getGeometry();
272 
273  TPCDistortionIRS& distortion = fastTransform.getDistortionNonConst();
274 
275  // switch TOF correction off for a while
276 
277  recoParam->SetUseTOFCorrection(kFALSE);
278 
279  for (int slice = 0; slice < geo.getNumberOfSlices(); slice++) {
280 
281  for (int row = 0; row < geo.getNumberOfRows(); row++) {
282 
283  const TPCFastTransformGeo::RowInfo& rowInfo = geo.getRowInfo(row);
284 
285  const IrregularSpline2D3D& spline = distortion.getSpline(slice, row);
286 
287  float* data = distortion.getSplineDataNonConst(slice, row);
288 
289  for (int knot = 0; knot < spline.getNumberOfKnots(); knot++) {
290 
291  data[3 * knot + 0] = 0.f;
292  data[3 * knot + 1] = 0.f;
293  data[3 * knot + 2] = 0.f;
294 
295  // x cordinate of the knot
296  float x = rowInfo.x;
297 
298  // spline (su,sv) cordinates of the knot (su,sv) in (0,1)x(0,1)
299  float su = 0, sv = 0;
300  spline.getKnotUV(knot, su, sv);
301 
302  // x, u, v cordinates of the knot (local cartesian coord. of slice towards central electrode )
303  float u = 0, v = 0;
304  geo.convScaledUVtoUV(slice, row, su, sv, u, v);
305 
306  // row, pad, time coordinates of the knot
307  float vertexTime = 0.f;
308  float pad = 0.f, time = 0.f;
309  fastTransform.convUVtoPadTime(slice, row, u, v, pad, time, vertexTime);
310 
311  // nominal x,y,z coordinates of the knot (without distortions and time-of-flight correction)
312  float y = 0, z = 0;
313  geo.convUVtoLocal(slice, u, v, y, z);
314 
315  // original TPC transformation (row,pad,time) -> (x,y,z) without time-of-flight correction
316  float ox = 0, oy = 0, oz = 0;
317  {
318  int sector = 0, secrow = 0;
319  AliHLTTPCGeometry::Slice2Sector(slice, row, sector, secrow);
320  int is[] = {sector};
321  double xx[] = {static_cast<double>(secrow), pad, time};
322  mOrigTransform->Transform(xx, is, 0, 1);
323  ox = xx[0];
324  oy = xx[1];
325  oz = xx[2];
326  }
327  // convert to u,v
328  float ou = 0, ov = 0;
329  geo.convLocalToUV(slice, oy, oz, ou, ov);
330 
331  // distortions in x,u,v:
332  float dx = ox - x;
333  float du = ou - u;
334  float dv = ov - v;
335  data[3 * knot + 0] = dx;
336  data[3 * knot + 1] = du;
337  data[3 * knot + 2] = dv;
338  } // knots
339 
340  spline.correctEdges(data);
341  } // row
342  } // slice
343 
344  // set back the time-of-flight correction;
345 
346  recoParam->SetUseTOFCorrection(useTOFcorrection);
347 
348  return 0;
349 }
