31 SampaChannelHandler handlePacket(std::string& result)
32 {
33  return [&result](DsElecId dsId, uint8_t channel, SampaCluster sc) {
34  result += fmt::format("{}-ch-{}-ts-{}-q", asString(dsId), channel, sc.timestamp);
35  if (sc.isClusterSum()) {
36  result += fmt::format("-{}", sc.chargeSum);
37  } else {
38  for (auto s : sc.samples) {
39  result += fmt::format("-{}", s);
40  }
41  }
42  result += "\n";
43  };
44 }

64 SampaHeader createHeader(std::vector<SampaCluster> clusters)
65 {
66  uint16_t n10{0};
67  for (auto c : clusters) {
68  n10 += c.nof10BitWords();
69  }
70  SampaHeader sh;
71  sh.nof10BitWords(n10);
72  sh.packetType(SampaPacketType::Data);
73  sh.hammingCode(computeHammingCode(sh.uint64()));
74  sh.headerParity(computeHeaderParity(sh.uint64()));
75  return sh;
76 }

78 void append(uint64_t prefix, std::vector<uint64_t>& buffer, uint8_t& index, uint64_t& word, int data)
79 {
80  word |= static_cast<uint64_t>(data) << (index * 10);
81  if (index == 0) {
82  buffer.emplace_back(prefix | word);
83  index = 4;
84  word = 0;
85  } else {
86  --index;
87  }
88 }

90 void bufferizeClusters(const std::vector<SampaCluster>& clusters,
91  std::vector<uint64_t>& b64,
92  const uint64_t prefix = 0)
93 {
94  uint64_t word{0};
95  uint8_t index{4};
96  for (auto& c : clusters) {
97  std::cout << "c=" << c << "\n";
98  append(prefix, b64, index, word, c.nofSamples());
99  append(prefix, b64, index, word, c.timestamp);
100  if (c.isClusterSum()) {
101  append(prefix, b64, index, word, c.chargeSum & 0x3FF);
102  append(prefix, b64, index, word, (c.chargeSum & 0xFFC00) >> 10);
103  } else {
104  for (auto& s : c.samples) {
105  append(prefix, b64, index, word, s);
106  }
107  }
108  }
109  while (index != 4) {
110  append(prefix, b64, index, word, 0);
111  }
112 }

114 std::vector<uint64_t> createBuffer(const std::vector<SampaCluster>& clusters,
115  uint8_t chip,
116  uint8_t ch,
117  uint64_t prefix,
118  bool sync)
119 {
120  auto sh = createHeader(clusters);
121  sh.chipAddress(chip);
122  sh.channelAddress(ch);
123  std::vector<uint64_t> b64;
124  if (sync) {
125  b64.emplace_back(sampaSyncWord | prefix);
126  }
127  b64.emplace_back(sh.uint64() | prefix);
128  bufferizeClusters(clusters, b64, prefix);
129  return b64;
130 }

143 template <typename CHARGESUM>
144 std::string testDecode(const std::vector<SampaCluster>& clustersFirstChannel,
145  const std::vector<SampaCluster>& clustersSecondChannel = {})
146 {
147  std::string results;
148  uint64_t prefix{22}; // 14-bits value.
149  // exact value not relevant as long as it is non-zero.
150  // Idea being to populate bits 50-63 with some data to ensure
151  // the decoder is only using the lower 50 bits to get the sync and
152  // header values, for instance.
153  prefix <<= 50;
154  bool sync{true};
155  uint16_t dummySolarId{0};
156  uint8_t dummyGroup{0};
157  uint8_t chip = 5;
158  uint8_t ch = 31;
159  uint8_t index = (chip - (ch > 32)) / 2;
160  DsElecId dsId{dummySolarId, dummyGroup, index};
161  UserLogicElinkDecoder<CHARGESUM> dec(dsId, handlePacket(results));
162  auto b64 = createBuffer(clustersFirstChannel, chip, ch, prefix, sync);
163  if (clustersSecondChannel.size()) {
164  auto b64_2 = createBuffer(clustersSecondChannel, chip, ch / 2, prefix, !sync);
165  std::copy(b64_2.begin(), b64_2.end(), std::back_inserter(b64));
166  }
167  decodeBuffer(dec, b64);
168  return results;
169 }
