28 bool GPUTPCGMSliceTrack::FilterErrors(const GPUTPCGMMerger* merger, int iSlice, float maxSinPhi, float sinPhiMargin)
29 {
30  float lastX;
31  if (merger->Param().earlyTpcTransform) {
32  lastX = mOrigTrack->Cluster(mOrigTrack->NClusters() - 1).GetX(); // TODO: Why is this needed, Row2X should work, but looses some tracks
33  } else {
34  //float lastX = merger->Param().tpcGeometry.Row2X(mOrigTrack->Cluster(mOrigTrack->NClusters() - 1).GetRow()); // TODO: again, why does this reduce efficiency?
35  float y, z;
36  const GPUTPCSliceOutCluster& clo = mOrigTrack->Cluster(mOrigTrack->NClusters() - 1);
37  const ClusterNative& cl = merger->GetConstantMem()->ioPtrs.clustersNative->clustersLinear[clo.GetId()];
38  GPUTPCConvertImpl::convert(*merger->GetConstantMem(), iSlice, clo.GetRow(), cl.getPad(), cl.getTime(), lastX, y, z);
39  }
40 
41  const int N = 3;
42 
43  float bz = -merger->Param().ConstBz;
44 
45  float k = mQPt * bz;
46  float dx = (1.f / N) * (lastX - mX);
47  float kdx = k * dx;
48  float dxBz = dx * bz;
49  float kdx205 = 2.f + kdx * kdx * 0.5f;
50 
51  {
52  merger->Param().GetClusterErrors2(0, mZ, mSinPhi, mDzDs, mC0, mC2);
53  float C0a, C2a;
54  merger->Param().GetClusterRMS2(0, mZ, mSinPhi, mDzDs, C0a, C2a);
55  if (C0a > mC0) {
56  mC0 = C0a;
57  }
58  if (C2a > mC2) {
59  mC2 = C2a;
60  }
61 
62  mC3 = 0;
63  mC5 = 1;
64  mC7 = 0;
65  mC9 = 1;
66  mC10 = 0;
67  mC12 = 0;
68  mC14 = 10;
69  }
70 
71  for (int iStep = 0; iStep < N; iStep++) {
72  float err2Y, err2Z;
73 
74  { // transport block
75  float ex = mCosPhi;
76  float ey = mSinPhi;
77  float ey1 = kdx + ey;
78  if (fabsf(ey1) > maxSinPhi) {
79  if (ey1 > maxSinPhi && ey1 < maxSinPhi + sinPhiMargin) {
80  ey1 = maxSinPhi - 0.01;
81  } else if (ey1 > -maxSinPhi - sinPhiMargin) {
82  ey1 = -maxSinPhi + 0.01;
83  } else {
84  return 0;
85  }
86  }
87 
88  float ss = ey + ey1;
89  float ex1 = sqrt(1.f - ey1 * ey1);
90 
91  float cc = ex + ex1;
92  float dxcci = dx / cc;
93 
94  float dy = dxcci * ss;
95  float norm2 = 1.f + ey * ey1 + ex * ex1;
96  float dl = dxcci * sqrt(norm2 + norm2);
97 
98  float dS;
99  {
100  float dSin = 0.5f * k * dl;
101  float a = dSin * dSin;
102  const float k2 = 1.f / 6.f;
103  const float k4 = 3.f / 40.f;
104  dS = dl + dl * a * (k2 + a * (k4)); //+ k6*a) );
105  }
106 
107  float dz = dS * mDzDs;
108  float ex1i = 1.f / ex1;
109  {
110  merger->Param().GetClusterErrors2(0, mZ, mSinPhi, mDzDs, err2Y, err2Z);
111  float C0a, C2a;
112  merger->Param().GetClusterRMS2(0, mZ, mSinPhi, mDzDs, C0a, C2a);
113  if (C0a > err2Y) {
114  err2Y = C0a;
115  }
116  if (C2a > err2Z) {
117  err2Z = C2a;
118  }
119  }
120 
121  float hh = kdx205 * dxcci * ex1i;
122  float h2 = hh * mSecPhi;
123 
124  mX += dx;
125  mY += dy;
126  mZ += dz;
127  mSinPhi = ey1;
128  mCosPhi = ex1;
129  mSecPhi = ex1i;
130 
131  float h4 = bz * dxcci * hh;
132 
133  float c20 = mC3;
134  float c22 = mC5;
135  float c31 = mC7;
136  float c33 = mC9;
137  float c40 = mC10;
138  float c42 = mC12;
139  float c44 = mC14;
140 
141  float c20ph4c42 = c20 + h4 * c42;
142  float h2c22 = h2 * c22;
143  float h4c44 = h4 * c44;
144  float n7 = c31 + dS * c33;
145  float n10 = c40 + h2 * c42 + h4c44;
146  float n12 = c42 + dxBz * c44;
147 
148  mC0 += h2 * h2c22 + h4 * h4c44 + 2.f * (h2 * c20ph4c42 + h4 * c40);
149 
150  mC3 = c20ph4c42 + h2c22 + dxBz * n10;
151  mC10 = n10;
152 
153  mC5 = c22 + dxBz * (c42 + n12);
154  mC12 = n12;
155 
156  mC2 += dS * (c31 + n7);
157  mC7 = n7;
158  } // end transport block
159 
160  // Filter block
161 
162  float c00 = mC0, c11 = mC2, c20 = mC3, c31 = mC7, c40 = mC10;
163 
164  float mS0 = 1.f / (err2Y + c00);
165  float mS2 = 1.f / (err2Z + c11);
166 
167  // K = CHtS
168 
169  float k00, k11, k20, k31, k40;
170 
171  k00 = c00 * mS0;
172  k20 = c20 * mS0;
173  k40 = c40 * mS0;
174 
175  mC0 -= k00 * c00;
176  mC5 -= k20 * c20;
177  mC10 -= k00 * c40;
178  mC12 -= k40 * c20;
179  mC3 -= k20 * c00;
180  mC14 -= k40 * c40;
181 
182  k11 = c11 * mS2;
183  k31 = c31 * mS2;
184 
185  mC7 -= k31 * c11;
186  mC2 -= k11 * c11;
187  mC9 -= k31 * c31;
188  }
189 
190  //* Check that the track parameters and covariance matrix are reasonable
191 
192  bool ok = CAMath::Finite(mX) && CAMath::Finite(mY) && CAMath::Finite(mZ) && CAMath::Finite(mSinPhi) && CAMath::Finite(mDzDs) && CAMath::Finite(mQPt) && CAMath::Finite(mCosPhi) && CAMath::Finite(mSecPhi) && CAMath::Finite(mZOffset) && CAMath::Finite(mC0) && CAMath::Finite(mC2) &&
193  CAMath::Finite(mC3) && CAMath::Finite(mC5) && CAMath::Finite(mC7) && CAMath::Finite(mC9) && CAMath::Finite(mC10) && CAMath::Finite(mC12) && CAMath::Finite(mC14);
194 
195  if (mC0 <= 0.f || mC2 <= 0.f || mC5 <= 0.f || mC9 <= 0.f || mC14 <= 0.f || mC0 > 5.f || mC2 > 5.f || mC5 > 2.f || mC9 > 2.f) {
196  ok = 0;
197  }
198 
199  if (ok) {
200  ok = ok && (mC3 * mC3 <= mC5 * mC0) && (mC7 * mC7 <= mC9 * mC2) && (mC10 * mC10 <= mC14 * mC0) && (mC12 * mC12 <= mC14 * mC5);
201  }
202 
203  return ok;

206 bool GPUTPCGMSliceTrack::TransportToX(float x, float Bz, GPUTPCGMBorderTrack& b, float maxSinPhi, bool doCov) const
207 {
208  Bz = -Bz;
209  float ex = mCosPhi;
210  float ey = mSinPhi;
211  float k = mQPt * Bz;
212  float dx = x - mX;
213  float ey1 = k * dx + ey;
214 
215  if (fabsf(ey1) > maxSinPhi) {
216  return 0;
217  }
218 
219  float ex1 = sqrt(1.f - ey1 * ey1);
220  float dxBz = dx * Bz;
221 
222  float ss = ey + ey1;
223  float cc = ex + ex1;
224  float dxcci = dx / cc;
225  float norm2 = 1.f + ey * ey1 + ex * ex1;
226 
227  float dy = dxcci * ss;
228 
229  float dS;
230  {
231  float dl = dxcci * sqrt(norm2 + norm2);
232  float dSin = 0.5f * k * dl;
233  float a = dSin * dSin;
234  const float k2 = 1.f / 6.f;
235  const float k4 = 3.f / 40.f;
236  // const float k6 = 5.f/112.f;
237  dS = dl + dl * a * (k2 + a * (k4)); //+ k6*a) );
238  }
239 
240  float dz = dS * mDzDs;
241 
242  b.SetPar(0, mY + dy);
243  b.SetPar(1, mZ + dz);
244  b.SetPar(2, ey1);
245  b.SetPar(3, mDzDs);
246  b.SetPar(4, mQPt);
247  b.SetZOffset(mZOffset);
248 
249  if (!doCov) {
250  return (1);
251  }
252 
253  float ex1i = 1.f / ex1;
254  float hh = dxcci * ex1i * norm2;
255  float h2 = hh * mSecPhi;
256  float h4 = Bz * dxcci * hh;
257 
258  float c20 = mC3;
259  float c22 = mC5;
260  float c31 = mC7;
261  float c33 = mC9;
262  float c40 = mC10;
263  float c42 = mC12;
264  float c44 = mC14;
265 
266  float c20ph4c42 = c20 + h4 * c42;
267  float h2c22 = h2 * c22;
268  float h4c44 = h4 * c44;
269  float n7 = c31 + dS * c33;
270 
271  if (fabsf(mQPt) > 6.66) // Special treatment for low Pt
272  {
273  b.SetCov(0, CAMath::Max(mC0, mC0 + h2 * h2c22 + h4 * h4c44 + 2.f * (h2 * c20ph4c42 + h4 * c40))); // Do not decrease Y cov for matching!
274  float C2tmp = dS * 2.f * c31;
275  if (C2tmp < 0) {
276  C2tmp = 0;
277  }
278  b.SetCov(1, mC2 + C2tmp + dS * dS * c33); // Incorrect formula, correct would be "dS * (c31 + n7)", but we need to make sure cov(Z) increases regardless of the direction of the propagation
279  } else {
280  b.SetCov(0, mC0 + h2 * h2c22 + h4 * h4c44 + 2.f * (h2 * c20ph4c42 + h4 * c40));
281  b.SetCov(1, mC2 + dS * (c31 + n7));
282  }
283  b.SetCov(2, c22 + dxBz * (c42 + c42 + dxBz * c44));
284  b.SetCov(3, c33);
285  b.SetCov(4, c44);
286  b.SetCovD(0, c20ph4c42 + h2c22 + dxBz * (c40 + h2 * c42 + h4c44));
287  b.SetCovD(1, n7);
288 
289  b.LimitCov();
290 
291  return 1;
292 }

294 bool GPUTPCGMSliceTrack::TransportToXAlpha(float newX, float sinAlpha, float cosAlpha, float Bz, GPUTPCGMBorderTrack& b, float maxSinPhi) const
295 {
296  //*
297 
298  float c00 = mC0;
299  float c11 = mC2;
300  float c20 = mC3;
301  float c22 = mC5;
302  float c31 = mC7;
303  float c33 = mC9;
304  float c40 = mC10;
305  float c42 = mC12;
306  float c44 = mC14;
307 
308  float x, y;
309  float z = mZ;
310  float sinPhi = mSinPhi;
311  float cosPhi = mCosPhi;
312  float secPhi = mSecPhi;
313  float dzds = mDzDs;
314  float qpt = mQPt;
315 
316  // Rotate the coordinate system in XY on the angle alpha
317  {
318  float sP = sinPhi, cP = cosPhi;
319  cosPhi = cP * cosAlpha + sP * sinAlpha;
320  sinPhi = -cP * sinAlpha + sP * cosAlpha;
321 
322  if (CAMath::Abs(sinPhi) > GPUCA_MAX_SIN_PHI || CAMath::Abs(cP) < 1.e-2) {
323  return 0;
324  }
325 
326  secPhi = 1. / cosPhi;
327  float j0 = cP * secPhi;
328  float j2 = cosPhi / cP;
329  x = mX * cosAlpha + mY * sinAlpha;
330  y = -mX * sinAlpha + mY * cosAlpha;
331 
332  c00 *= j0 * j0;
333  c40 *= j0;
334 
335  c22 *= j2 * j2;
336  c42 *= j2;
337  if (cosPhi < 0.f) { // rotate to 180'
338  cosPhi = -cosPhi;
339  secPhi = -secPhi;
340  sinPhi = -sinPhi;
341  dzds = -dzds;
342  qpt = -qpt;
343  c20 = -c20;
344  c31 = -c31;
345  c40 = -c40;
346  }
347  }
348 
349  Bz = -Bz;
350  float ex = cosPhi;
351  float ey = sinPhi;
352  float k = qpt * Bz;
353  float dx = newX - x;
354  float ey1 = k * dx + ey;
355 
356  if (fabsf(ey1) > maxSinPhi) {
357  return 0;
358  }
359 
360  float ex1 = sqrt(1.f - ey1 * ey1);
361 
362  float dxBz = dx * Bz;
363 
364  float ss = ey + ey1;
365  float cc = ex + ex1;
366  float dxcci = dx / cc;
367  float norm2 = 1.f + ey * ey1 + ex * ex1;
368 
369  float dy = dxcci * ss;
370 
371  float dS;
372  {
373  float dl = dxcci * sqrt(norm2 + norm2);
374  float dSin = 0.5f * k * dl;
375  float a = dSin * dSin;
376  const float k2 = 1.f / 6.f;
377  const float k4 = 3.f / 40.f;
378  // const float k6 = 5.f/112.f;
379  dS = dl + dl * a * (k2 + a * (k4)); //+ k6*a) );
380  }
381 
382  float ex1i = 1.f / ex1;
383  float dz = dS * dzds;
384 
385  float hh = dxcci * ex1i * norm2;
386  float h2 = hh * secPhi;
387  float h4 = Bz * dxcci * hh;
388 
389  float c20ph4c42 = c20 + h4 * c42;
390  float h2c22 = h2 * c22;
391  float h4c44 = h4 * c44;
392  float n7 = c31 + dS * c33;
393 
394  b.SetPar(0, y + dy);
395  b.SetPar(1, z + dz);
396  b.SetPar(2, ey1);
397  b.SetPar(3, dzds);
398  b.SetPar(4, qpt);
399  b.SetZOffset(mZOffset);
400 
401  b.SetCov(0, c00 + h2 * h2c22 + h4 * h4c44 + 2.f * (h2 * c20ph4c42 + h4 * c40));
402  b.SetCov(1, c11 + dS * (c31 + n7));
403  b.SetCov(2, c22 + dxBz * (c42 + c42 + dxBz * c44));
404  b.SetCov(3, c33);
405  b.SetCov(4, c44);
406  b.SetCovD(0, c20ph4c42 + h2c22 + dxBz * (c40 + h2 * c42 + h4c44));
407  b.SetCovD(1, n7);
408 
409  b.LimitCov();
410 
411  return 1;
412 }
