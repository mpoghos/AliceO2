57 void ClusterizerTask::process(const std::string inputFileName, const std::string outputFileName)
58 {
59  LOG(DEBUG) << "Running clusterization on new event";
60 
61  // Create reader, initilize clusterizer geometry
62  init();
63 
64  // Load output file
65  std::unique_ptr<TFile> outFile(TFile::Open(outputFileName.data(), "recreate"));
66  if (!outFile || outFile->IsZombie()) {
67  LOG(FATAL) << "Failed to open output file " << outputFileName;
68  }
69 
70  // Create output tree
71  std::unique_ptr<TTree> outTree = std::make_unique<TTree>("o2sim", "EMCAL clusters");
72  outTree->Branch("EMCALCluster", &mClustersArray);
73  outTree->Branch("EMCALClusterDigitIndices", &mClustersDigitIndices);
74 
75  // Loop over entries of the input tree
76  mDigitReader->openInput(inputFileName);
77  while (mDigitReader->readNextEntry()) {
78  mClusterizer.findClusters(*mDigitReader->getDigitArray()); // Find clusters on digits given in reader::mDigitArray (pass by ref)
79 
80  // Get found clusters + digit indices for output
81  mClustersArray = mClusterizer.getFoundClusters();
82  mClustersDigitIndices = mClusterizer.getFoundClustersDigitIndices();
83  outTree->Fill();
84  }
85 
86  // Write, close, and destroy tree/file
87  outTree->Write();
88  outTree.reset(); // here we reset the unique ptr, not the tree!
89  outFile->Close();
90 }
