98 GPUhdi() int GPUCommonMath::Nint(float x)
99 {
100  int i;
101  if (x >= 0) {
102  i = int(x + 0.5f);
103  if (x + 0.5f == float(i) && i & 1)
104  i--;
105  } else {
106  i = int(x - 0.5f);
107  if (x - 0.5f == float(i) && i & 1)
108  i++;
109  }
110  return i;
111 }

125 GPUhdi() unsigned int GPUCommonMath::Clz(unsigned int x)
126 {
127 #if (defined(__GNUC__) || defined(__clang__) || defined(__CUDACC__) || defined(__HIPCC__)) && (!defined(__OPENCL__) || defined(__OPENCLCPP__))
128  return x == 0 ? 32 : CHOICE(__builtin_clz(x), __clz(x), __builtin_clz(x)); // use builtin if available
129 #else
130  for (int i = 31; i >= 0; i--) {
131  if (x & (1 << i)) {
132  return (31 - i);
133  }
134  }
135  return 32;
136 #endif
137 }
