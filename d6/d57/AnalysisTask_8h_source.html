275  template <typename T, size_t At>
276  static auto extractSomethingFromRecord(InputRecord& record, std::vector<ExpressionInfo> const infos)
277  {
278  using decayed = std::decay_t<T>;
279  if constexpr (is_specialization<decayed, soa::Filtered>::value) {
280  for (auto& info : infos) {
281  if (info.index == At)
282  return extractFilteredFromRecord<decayed>(record, info, soa::make_originals_from_type<decayed>());
283  }
284  } else if constexpr (soa::is_type_with_policy_v<decayed>) {
285  if constexpr (std::is_same_v<typename decayed::policy_t, soa::FilteredIndexPolicy>) {
286  for (auto& info : infos) {
287  if (info.index == At)
288  return extractFilteredFromRecord<decayed>(record, info, soa::make_originals_from_type<decayed>());
289  }
290  } else {
291  return extractFromRecord<decayed>(record, soa::make_originals_from_type<decayed>());
292  }
293  } else {
294  return extractFromRecord<decayed>(record, soa::make_originals_from_type<decayed>());
295  }
296  O2_BUILTIN_UNREACHABLE();
297  }

312  template <typename Task, typename R, typename C, typename Grouping, typename... Associated>
313  static void invokeProcess(Task& task, InputRecord& inputs, R (C::*)(Grouping, Associated...), std::vector<ExpressionInfo> const infos)
314  {
315  auto groupingTable = AnalysisDataProcessorBuilder::bindGroupingTable(inputs, &C::process, infos);
316  auto associatedTables = AnalysisDataProcessorBuilder::bindAssociatedTables(inputs, &C::process, infos);
317 
318  if constexpr (sizeof...(Associated) == 0) {
319  // No extra tables: we need to either iterate over the contents of
320  // grouping or pass the whole grouping table, depending on whether Grouping
321  // is a o2::soa::Table or a o2::soa::RowView
322  if constexpr (is_specialization<std::decay_t<Grouping>, o2::soa::Table>::value) {
323  task.process(groupingTable);
324  } else if constexpr (is_base_of_template<o2::soa::RowViewBase, std::decay_t<Grouping>>::value) {
325  for (auto& groupedElement : groupingTable) {
326  task.process(groupedElement);
327  }
328  } else if constexpr (is_specialization<std::decay_t<Grouping>, o2::soa::Join>::value) {
329  task.process(groupingTable);
330  } else if constexpr (is_specialization<std::decay_t<Grouping>, o2::soa::Filtered>::value) {
331  task.process(groupingTable);
332  } else {
333  static_assert(always_static_assert_v<Grouping>,
334  "The first argument of the process method of your task must be either"
335  " a o2::soa::Table or a o2::soa::RowView");
336  }
337  } else if constexpr (sizeof...(Associated) == 1) {
338  // One extra table provided: if the first argument itself is a table,
339  // then we simply pass it over to the process method and let the user do the
340  // double looping (e.g. if they want to do some association between different
341  // physics quantities.
342  //
343  // If the first argument is a single element, we consider that the user
344  // wants to do a loop first on the table associated to the first element,
345  // then to the subgroup of the second table which is associated to the
346  // first one. E.g.:
347  //
348  // MyTask::process(Collision const& collision, Tracks const& tracks)
349  //
350  // Will iterate on all the tracks for the provided collision.
351  if constexpr (is_specialization<std::decay_t<Grouping>, o2::soa::Table>::value) {
352  static_assert((is_specialization<std::decay_t<Associated>, o2::soa::Table>::value && ...),
353  "You cannot have a soa::RowView iterator as an argument after the "
354  " first argument of type soa::Table which is found as in the "
355  " prototype of the task process method.");
356  task.process(groupingTable, std::get<0>(associatedTables));
357  } else if constexpr (is_specialization<std::decay_t<Grouping>, o2::soa::RowViewBase>::value) {
358  using AssociatedType = std::tuple_element_t<0, std::tuple<Associated...>>;
359  if constexpr (is_specialization<std::decay_t<AssociatedType>, o2::soa::RowViewBase>::value) {
360  auto groupedTable = std::get<0>(associatedTables);
361  size_t currentGrouping = 0;
362  Grouping groupingElement = groupingTable.begin();
363  for (auto& groupedElement : groupedTable) {
364  // FIXME: this only works for collisions for now...
365  auto groupingIndex = groupedElement.collisionId(); // Fine for the moment.
366  // We find the associated collision, assuming they are sorted.
367  while (groupingIndex > currentGrouping) {
368  // This const_cast is done because I do not want people to be
369  // able to move the iterator in the user code.
370  ++const_cast<std::decay_t<Grouping>&>(groupingElement);
371  ++const_cast<std::decay_t<AssociatedType>&>(groupedElement);
372  }
373  task.process(groupingElement, groupedElement);
374  }
375  } else if constexpr (is_base_of_template<o2::soa::Table, std::decay_t<AssociatedType>>::value) {
376  auto allGroupedTable = std::get<0>(associatedTables);
377  using groupingMetadata = typename aod::MetadataTrait<std::decay_t<Grouping>>::metadata;
378  arrow::compute::FunctionContext ctx;
379  std::vector<arrow::compute::Datum> groupsCollection;
380  auto indexColumnName = std::string("f") + groupingMetadata::label() + "ID";
381  std::vector<uint64_t> offsets;
382  auto result = o2::framework::sliceByColumn(&ctx, indexColumnName,
383  allGroupedTable.asArrowTable(), &groupsCollection, &offsets);
384  if (result.ok() == false) {
385  LOGF(ERROR, "Error while splitting second collection");
386  return;
387  }
388  size_t currentGrouping = 0;
389  auto groupingElement = groupingTable.begin();
390 
391  // FIXME: this assumes every groupingElement has a group associated,
392  // which migh not be the case.
393  size_t oi = 0;
394  if constexpr (is_specialization<std::decay_t<AssociatedType>, o2::soa::Table>::value) {
395  for (auto& groupedDatum : groupsCollection) {
396  auto groupedElementsTable = arrow::util::get<std::shared_ptr<arrow::Table>>(groupedDatum.value);
397  task.process(groupingElement, AssociatedType{groupedElementsTable, offsets[oi]});
398  ++const_cast<std::decay_t<Grouping>&>(groupingElement);
399  ++oi;
400  }
401  }
402  if constexpr (is_specialization<std::decay_t<AssociatedType>, o2::soa::Filtered>::value) {
403  // FIXME: we need to implement the case for the grouped filtered case.
404  static_assert(always_static_assert_v<std::decay_t<AssociatedType>>, "Grouping Filtered is not yet supported");
405  } else if constexpr (is_specialization<std::decay_t<AssociatedType>, o2::soa::Join>::value || is_specialization<std::decay_t<AssociatedType>, o2::soa::Concat>::value) {
406  for (auto& groupedDatum : groupsCollection) {
407  auto groupedElementsTable = arrow::util::get<std::shared_ptr<arrow::Table>>(groupedDatum.value);
408  task.process(groupingElement, AssociatedType{{groupedElementsTable}, offsets[oi]});
409  ++const_cast<std::decay_t<Grouping>&>(groupingElement);
410  ++oi;
411  }
412  }
413  } else {
414  static_assert(always_static_assert_v<AssociatedType>, "I do not know how to iterate on this");
415  }
416  } else {
417  static_assert(always_static_assert_v<Grouping>,
418  "Only grouping by Collision is supported for now");
419  }
420  } else {
421  static_assert(always_static_assert_v<Grouping, Associated...>,
422  "Unable to find a way to iterate on the provided set of arguments. Probably unimplemented");
423  }
424  }
425 };

595 template <typename T, typename... Args>
596 DataProcessorSpec adaptAnalysisTask(std::string name, Args&&... args)
597 {
598  auto task = std::make_shared<T>(std::forward<Args>(args)...);
599 
600  std::vector<OutputSpec> outputs;
601  auto tupledTask = o2::framework::to_tuple_refs(*task.get());
602  static_assert(has_process<T>::value || has_run<T>::value || has_init<T>::value,
603  "At least one of process(...), T::run(...), init(...) must be defined");
604 
605  std::vector<InputSpec> inputs;
606  std::vector<ExpressionInfo> expressionInfos;
607 
608  if constexpr (has_process<T>::value) {
609  // this pushes (I,schemaPtr,nullptr) into expressionInfos for arguments that are Filtered/filtered_iterators
610  AnalysisDataProcessorBuilder::inputsFromArgs(&T::process, inputs, expressionInfos);
611  // here the FilterManager will prepare the gandiva trees matched to schemas and put the pointers into expressionInfos
612  std::apply([&expressionInfos](auto&... x) {
613  return (FilterManager<std::decay_t<decltype(x)>>::createExpressionTrees(x, expressionInfos), ...);
614  },
615  tupledTask);
616  }
617 
618  std::apply([&outputs](auto&... x) { return (OutputManager<std::decay_t<decltype(x)>>::appendOutput(outputs, x), ...); }, tupledTask);
619 
620  auto algo = AlgorithmSpec::InitCallback{[task, expressionInfos](InitContext& ic) {
621  auto& callbacks = ic.services().get<CallbackService>();
622  auto endofdatacb = [task](EndOfStreamContext& eosContext) {
623  auto tupledTask = o2::framework::to_tuple_refs(*task.get());
624  std::apply([&eosContext](auto&&... x) { return (OutputManager<std::decay_t<decltype(x)>>::postRun(eosContext, x), ...); }, tupledTask);
625  eosContext.services().get<ControlService>().readyToQuit(QuitRequest::Me);
626  };
627  callbacks.set(CallbackService::Id::EndOfStream, endofdatacb);
628 
629  if constexpr (has_init<T>::value) {
630  task->init(ic);
631  }
632  return [task, expressionInfos](ProcessingContext& pc) {
633  auto tupledTask = o2::framework::to_tuple_refs(*task.get());
634  std::apply([&pc](auto&&... x) { return (OutputManager<std::decay_t<decltype(x)>>::prepare(pc, x), ...); }, tupledTask);
635  if constexpr (has_run<T>::value) {
636  task->run(pc);
637  }
638  if constexpr (has_process<T>::value) {
639  AnalysisDataProcessorBuilder::invokeProcess(*(task.get()), pc.inputs(), &T::process, expressionInfos);
640  }
