113  void addNewPrecluster(UInt_t ip, UShort_t row)
114  {
115  mPreClusterHeads.push_back(mPixels.size());
116  // new head does not point yet (-1) on other pixels, store just the entry of the pixel in the ChipPixelData
117  mPixels.emplace_back(-1, ip);
118  int lastIndex = mPreClusterIndices.size();
119  mPreClusterIndices.push_back(lastIndex);
120  mCurr[row] = lastIndex; // store index of the new precluster in the current column buffer
121  }
122 
124  void expandPreCluster(UInt_t ip, UShort_t row, int preClusIndex)
125  {
126  auto& firstIndex = mPreClusterHeads[mPreClusterIndices[preClusIndex]];
127  mPixels.emplace_back(firstIndex, ip);
128  firstIndex = mPixels.size() - 1;
129  mCurr[row] = preClusIndex;
130  }
