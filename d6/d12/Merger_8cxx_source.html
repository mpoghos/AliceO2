54 void Merger::run(framework::ProcessingContext& ctx)
55 {
56  mCache.cacheInputRecord(ctx.inputs());
57 
58  if (shouldMergeCache(ctx)) {
59 
60  mergeCache();
61 
62  cleanCacheAfterMerging();
63  }
64 
65  if (shouldPublish(ctx)) {
66 
67  publish(ctx.outputs());
68 
69  cleanCacheAfterPublishing();
70 
71  if (mConfig.timespan.value == Timespan::LastDifference) {
72  mMergedObjects.reset();
73  }
74  }

77 std::function<void()> Merger::prepareTimerCallback(InitContext& ictx) const
78 {
79  return [&timesliceIndex = ictx.services().get<TimesliceIndex>(),
80  timeLast = std::make_shared<steady_clock::time_point>(steady_clock::now()),
81  periodMs = this->mConfig.publicationDecision.param * 1000,
82  timesliceID = uint64_t(0)]() mutable {
83  auto timeNow = steady_clock::now();
84 
85  if (duration_cast<milliseconds>(timeNow - *timeLast).count() > periodMs) {
86 
87  data_matcher::VariableContext context;
88  context.put(data_matcher::ContextUpdate{0, timesliceID});
89  context.commit();
90 
91  timesliceIndex.replaceLRUWith(context);
92 
93  timesliceID++;
94  *timeLast = timeNow;
95  }
96  };
97 }

116 void Merger::mergeCache()
117 {
118  switch (mConfig.mergingMode.value) {
119  case MergingMode::Binwise: {
120 
121  size_t i = 0;
122  if (!mMergedObjects) {
123  for (; i < mCache.size(); i++) {
124  if (!mCache[i].deque.empty()) {
125  mMergedObjects.reset(mCache[i].deque[0].obj->Clone());
126  mCache.setMerged(i, 0);
127  break;
128  }
129  }
130  }
131 
132  if (!mMergedObjects) {
133  LOG(INFO) << "mergeCache(): The cache is empty, nothing to merge.";
134  return;
135  }
136 
137  auto unpackedMergedObjects = unpackObjects(mMergedObjects.get());
138  std::vector<TObjArray> unpackedCollectionsOfObjects(unpackedMergedObjects.size());
139  // todo: unpack straight to TCollection?
140  for (; i < mCache.size(); i++) {
141  for (const auto& entry : mCache[i].deque) {
142  if (!entry.is_merged) {
143  auto unpackedCachedObjects = unpackObjects(entry.obj.get());
144  assert(unpackedMergedObjects.size() == unpackedCachedObjects.size());
145 
146  for (int j = 0; j < unpackedCachedObjects.size(); j++) {
147  unpackedCollectionsOfObjects[j].Add(unpackedCachedObjects[j]);
148  }
149  }
150  }
151  }
152 
153  for (int k = 0; k < unpackedMergedObjects.size(); k++) {
154 
155  TObject* mergedObject = unpackedMergedObjects[k];
156  const char* className = mergedObject->ClassName();
157  Long64_t errorCode = 0;
158 
159  //todo: investigate -NOCHECK flag for histogram merging
160  auto objectMergeInterface = dynamic_cast<MergeInterface*>(mergedObject);
161  if (objectMergeInterface) {
162  errorCode = objectMergeInterface->merge(&unpackedCollectionsOfObjects[k]);
163  } else if (strncmp(className, "TH1", 3) == 0) {
164  errorCode = reinterpret_cast<TH1*>(mergedObject)->Merge(&unpackedCollectionsOfObjects[k]);
165  } else if (strncmp(className, "TH2", 3) == 0) {
166  errorCode = reinterpret_cast<TH2*>(mergedObject)->Merge(&unpackedCollectionsOfObjects[k]);
167  } else if (strncmp(className, "TH3", 3) == 0) {
168  errorCode = reinterpret_cast<TH3*>(mergedObject)->Merge(&unpackedCollectionsOfObjects[k]);
169  } else if (strncmp(className, "THn", 3) == 0) {
170  errorCode = reinterpret_cast<THn*>(mergedObject)->Merge(&unpackedCollectionsOfObjects[k]);
171  } else if (strncmp(className, "THnSparse", 8) == 0) {
172  errorCode = reinterpret_cast<THnSparse*>(mergedObject)->Merge(&unpackedCollectionsOfObjects[k]);
173  } else if (strcmp(className, "TTree") == 0) {
174  errorCode = reinterpret_cast<TTree*>(mergedObject)->Merge(&unpackedCollectionsOfObjects[k]);
175  } else {
176  // LOG(ERROR) << "Object with type " << className << " is not one of mergeable type.";
177  throw std::runtime_error("Object with type '" + std::string(className) + "' is not one of mergeable type.");
178  // todo: maybe it is fine to just overwrite?
179  }
180 
181  if (errorCode == -1) {
182  throw std::runtime_error("Binwise merging object of type '" + std::string(className) + "' failed.");
183  // LOG(ERROR) << "Merging object of type " << className << " failed";
184  return;
185  }
186  }
187 
188  break;
189  }
190  case MergingMode::Concatenate: {
191 
192  if (!mMergedObjects) {
193  mMergedObjects = std::make_unique<TObjArray>();
194  }
195  for (const auto& queue : mCache) {
196  for (const auto& entry : queue.deque) {
197  if (!entry.is_merged) {
198 
199  //todo: optimisations - check only once if this is a collection
200  //todo: optimisations - TCollection::Merge() falls back to slow TMethodCall! (from mikolaj's presentation)
201  TCollection* entryAsCollection = dynamic_cast<TCollection*>(entry.obj.get());
202  assert(mMergedObjects);
203  if (entryAsCollection) {
204  reinterpret_cast<TCollection*>(mMergedObjects.get())->AddAll(entryAsCollection);
205  } else {
206  reinterpret_cast<TCollection*>(mMergedObjects.get())->Add(entry.obj.get());
207  }
208  }
209  }
210  }
211 
212  break;
213  }
214  case MergingMode::Timewise:
215  throw std::runtime_error("MergingMode::Timewise not supported yet");
216  default:
217  break;
218  }
219 }

221 void Merger::publish(framework::DataAllocator& allocator)
222 {
223  if (mMergedObjects) {
224  allocator.snapshot(framework::OutputRef{MergerBuilder::mergerOutputBinding(), mSubSpec}, *mMergedObjects);
225  if (mConfig.ownershipMode.value == OwnershipMode::Full) {
226  // in full mode the ownership is passed to the framework, the framework however is
227  // simply serializing and dispatching the object and will delete it after that,
228  // so we do here.
229  mMergedObjects.reset();
230  }
231  }

234 std::vector<TObject*> Merger::unpackObjects(TObject* obj)
235 {
236  if (auto objMergeInterface = dynamic_cast<MergeInterface*>(obj)) {
237  return objMergeInterface->unpack();
238  } else if (mConfig.unpackingMethod.value == UnpackingMethod::NoUnpackingNeeded) {
239  return std::vector<TObject*>{obj};
240  } else if (mConfig.unpackingMethod.value == UnpackingMethod::TCollection) {
241  // todo: this could be also checked by casting
242  return {};
243  } else {
244  return {};
245  }
246 }

248 bool Merger::shouldMergeCache(framework::ProcessingContext& ctx)
249 {
250  switch (mConfig.mergingTime.value) {
251  case MergingTime::AfterArrival:
252  return true;
253  case MergingTime::WhenXInputsCached:
254  return double(mCache.cachedInputs()) / mCache.size() >= mConfig.mergingTime.param;
255  case MergingTime::BeforePublication:
256  return shouldPublish(ctx);
257  default:
258  return false;
259  }
260 }
261 
262 bool Merger::shouldPublish(framework::ProcessingContext& ctx)
263 {
264  switch (mConfig.publicationDecision.value) {
265  case PublicationDecision::EachNSeconds:
266  return ctx.inputs().isValid("timer-publish");
267  case PublicationDecision::WhenXInputsUpdated:
268  return double(mCache.updatedInputs()) / mCache.size() >= mConfig.publicationDecision.param;
269  default:
270  return false;
271  }
272 }
