31 // This is a very simple example showing how to iterate over tracks
32 // and create a new collection for them.
33 // FIXME: this should really inherit from AnalysisTask but
34 // we need GCC 7.4+ for that
35 struct ATask {
36  Produces<aod::EtaPhi> etaphi;
37  Produces<aod::EtaPhiPt> etaphipt;
38 
39  void process(aod::Tracks const& tracks)
40  {
41  for (auto& track : tracks) {
42  float phi = asin(track.snp()) + track.alpha() + static_cast<float>(M_PI);
43  float eta = log(tan(0.25f * static_cast<float>(M_PI) - 0.5f * atan(track.tgl())));
44  float pt = 1.f / track.signed1Pt();
45  etaphi(eta, phi);
46  etaphipt(eta, phi, pt);
47  }
48  }
49 };
