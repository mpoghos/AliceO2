48  void init(framework::InitContext& ic)
49  {
50  std::string detStrL = mID.getName();
51  std::transform(detStrL.begin(), detStrL.end(), detStrL.begin(), ::tolower);
52 
53  auto filename = ic.options().get<std::string>((detStrL + "-digit-outfile").c_str());
54  auto treename = ic.options().get<std::string>("treename");
55 
56  mOutFile = std::make_unique<TFile>(filename.c_str(), "RECREATE");
57  if (!mOutFile || mOutFile->IsZombie()) {
58  LOG(ERROR) << "Failed to open " << filename << " output file";
59  } else {
60  LOG(INFO) << "Opened " << filename << " output file";
61  }
62  mOutTree = std::make_unique<TTree>(treename.c_str(), treename.c_str());
63  }
64 
65  void run(framework::ProcessingContext& pc)
66  {
67  if (mFinished) {
68  return;
69  }
70  std::string detStr = mID.getName();
71  std::string detStrL = mID.getName();
72  std::transform(detStrL.begin(), detStrL.end(), detStrL.begin(), ::tolower);
73 
74  // retrieve the digits from the input
75  auto inDigits = pc.inputs().get<std::vector<o2::ft0::Digit>>((detStr + "digits").c_str());
76 
77  auto inLabels = pc.inputs().get<MCCont*>((detStr + "digitsMCTR").c_str());
78  LOG(INFO) << "RECEIVED DIGITS SIZE " << inDigits.size();
79 
80  auto digitsP = &inDigits;
81  auto labelsRaw = inLabels.get();
82  // connect this to a particular branch
83 
84  auto brDig = getOrMakeBranch(*mOutTree.get(), (detStr + "Digit").c_str(), &digitsP);
85  auto brLbl = getOrMakeBranch(*mOutTree.get(), (detStr + "DigitMCTruth").c_str(), &labelsRaw);
86  mOutTree->Fill();
87 
88  mOutFile->cd();
89  mOutTree->Write();
90  mOutTree.reset(); // delete the tree before closing the file
91  mOutFile->Close();
92  mFinished = true;
93  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
94  }
95  FT0DPLDigitWriter() = default;
96 
97  protected:
98  template <typename T>
99  TBranch* getOrMakeBranch(TTree& tree, std::string brname, T* ptr)
100  {
101  if (auto br = tree.GetBranch(brname.c_str())) {
102  br->SetAddress(static_cast<void*>(ptr));
103  return br;
104  }
105  // otherwise make it
106  return tree.Branch(brname.c_str(), ptr);
107  }
108 
109  bool mFinished = false;
110  o2::detectors::DetID mID = DETID;
111  o2::header::DataOrigin mOrigin = DETOR;
112  std::vector<o2::ft0::Digit> mDigits; // input digits
113  std::unique_ptr<TFile> mOutFile;
114  std::unique_ptr<TTree> mOutTree;
