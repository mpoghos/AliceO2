59 void TPCITSMatchingDPL::run(ProcessingContext& pc)
60 {
61  LOG(INFO) << "TPCITSMatchingDPL " << mFinished << " NLanes " << mTPCClusLanes.size();
62  if (mFinished) {
63  return;
64  }
65 
66  const auto tracksITS = pc.inputs().get<gsl::span<o2::its::TrackITS>>("trackITS");
67  const auto trackClIdxITS = pc.inputs().get<gsl::span<int>>("trackClIdx");
68  const auto tracksITSROF = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("trackITSROF");
69  const auto clusITS = pc.inputs().get<gsl::span<o2::itsmft::Cluster>>("clusITS");
70  const auto clusITSROF = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("clusITSROF");
71  const auto tracksTPC = pc.inputs().get<gsl::span<o2::tpc::TrackTPC>>("trackTPC");
72  const auto tracksTPCClRefs = pc.inputs().get<gsl::span<o2::tpc::TPCClRefElem>>("trackTPCClRefs");
73 
74  //---------------------------->> TPC Clusters loading >>------------------------------------------
75  int operation = 0;
76  uint64_t activeSectors = 0;
77  std::bitset<o2::tpc::Constants::MAXSECTOR> validSectors = 0;
78  std::map<int, DataRef> datarefs;
79  for (auto const& lane : mTPCClusLanes) {
80  std::string inputLabel = "clusTPC" + std::to_string(lane);
81  LOG(INFO) << "Reading lane " << lane << " " << inputLabel;
82  auto ref = pc.inputs().get(inputLabel);
83  auto const* sectorHeader = DataRefUtils::getHeader<o2::tpc::TPCSectorHeader*>(ref);
84  if (sectorHeader == nullptr) {
85  // FIXME: think about error policy
86  LOG(ERROR) << "sector header missing on header stack";
87  return;
88  }
89  const int& sector = sectorHeader->sector;
90  // check the current operation, this is used to either signal eod or noop
91  // FIXME: the noop is not needed any more once the lane configuration with one
92  // channel per sector is used
93  if (sector < 0) {
94  if (operation < 0 && operation != sector) {
95  // we expect the same operation on all inputs
96  LOG(ERROR) << "inconsistent lane operation, got " << sector << ", expecting " << operation;
97  } else if (operation == 0) {
98  // store the operation
99  operation = sector;
100  }
101  continue;
102  }
103  if (validSectors.test(sector)) {
104  // have already data for this sector, this should not happen in the current
105  // sequential implementation, for parallel path merged at the tracker stage
106  // multiple buffers need to be handled
107  throw std::runtime_error("can only have one data set per sector");
108  }
109  activeSectors |= sectorHeader->activeSectors;
110  validSectors.set(sector);
111  datarefs[sector] = ref;
112  }
113 
114  if (operation == -1) {
115  // EOD is transmitted in the sectorHeader with sector number equal to -1
116  LOG(WARNING) << "operation = " << operation;
117  }
118 
119  auto printInputLog = [&validSectors, &activeSectors](auto& r, const char* comment, auto& s) {
120  LOG(INFO) << comment << " " << *(r.spec) << ", size " << DataRefUtils::getPayloadSize(r) //
121  << " for sector " << s //
122  << std::endl //
123  << " input status: " << validSectors //
124  << std::endl //
125  << " active sectors: " << std::bitset<o2::tpc::Constants::MAXSECTOR>(activeSectors);
126  };
127 
128  if (activeSectors == 0 || (activeSectors & validSectors.to_ulong()) != activeSectors) {
129  // not all sectors available
130  // Since we expect complete input, this should not happen (why does the bufferization considered for TPC CA tracker? Ask Matthias)
131  throw std::runtime_error("Did not receive TPC clusters data for all sectors");
132  /*
133  for (auto const& refentry : datarefs) {
134  auto& sector = refentry.first;
135  auto& ref = refentry.second;
136  auto payploadSize = DataRefUtils::getPayloadSize(ref);
137  mBufferedTPCClusters[sector].resize(payploadSize);
138  std::copy(ref.payload, ref.payload + payploadSize, mBufferedTPCClusters[sector].begin());
139 
140  printInputLog(ref, "buffering", sector);
141  }
142  // not needed to send something, DPL will simply drop this timeslice, whenever the
143  // data for all sectors is available, the output is sent in that time slice
144  return;
145  */
146  }
147  //------------------------------------------------------------------------------
148  std::array<std::vector<MCLabelContainer>, o2::tpc::Constants::MAXSECTOR> mcInputs; // DUMMY
149  std::array<gsl::span<const char>, o2::tpc::Constants::MAXSECTOR> clustersTPC;
150  auto sectorStatus = validSectors;
151 
152  for (auto const& refentry : datarefs) {
153  auto& sector = refentry.first;
154  auto& ref = refentry.second;
155  clustersTPC[sector] = gsl::span(ref.payload, DataRefUtils::getPayloadSize(ref));
156  sectorStatus.reset(sector);
157  printInputLog(ref, "received", sector);
158  }
159  if (sectorStatus.any()) {
160  LOG(ERROR) << "Reading bufferized TPC clusters, this should not happen";
161  // some of the inputs have been buffered
162  for (size_t sector = 0; sector < sectorStatus.size(); ++sector) {
163  if (sectorStatus.test(sector)) {
164  clustersTPC[sector] = gsl::span(&mBufferedTPCClusters[sector].front(), mBufferedTPCClusters[sector].size());
165  }
166  }
167  }
168 
169  // Just print TPC clusters status
170  {
171  // make human readable information from the bitfield
172  std::string bitInfo;
173  auto nActiveBits = validSectors.count();
174  if (((uint64_t)0x1 << nActiveBits) == validSectors.to_ulong() + 1) {
175  // sectors 0 to some upper bound are active
176  bitInfo = "0-" + std::to_string(nActiveBits - 1);
177  } else {
178  int rangeStart = -1;
179  int rangeEnd = -1;
180  for (size_t sector = 0; sector < validSectors.size(); sector++) {
181  if (validSectors.test(sector)) {
182  if (rangeStart < 0) {
183  if (rangeEnd >= 0) {
184  bitInfo += ",";
185  }
186  bitInfo += std::to_string(sector);
187  if (nActiveBits == 1) {
188  break;
189  }
190  rangeStart = sector;
191  }
192  rangeEnd = sector;
193  } else {
194  if (rangeStart >= 0 && rangeEnd > rangeStart) {
195  bitInfo += "-" + std::to_string(rangeEnd);
196  }
197  rangeStart = -1;
198  }
199  }
200  if (rangeStart >= 0 && rangeEnd > rangeStart) {
201  bitInfo += "-" + std::to_string(rangeEnd);
202  }
203  }
204  LOG(INFO) << "running matching for sector(s) " << bitInfo;
205  }
206 
207  o2::tpc::ClusterNativeAccess clusterIndex;
208  std::unique_ptr<o2::tpc::ClusterNative[]> clusterBuffer;
209  o2::tpc::MCLabelContainer clusterMCBuffer;
210  memset(&clusterIndex, 0, sizeof(clusterIndex));
211  o2::tpc::ClusterNativeHelper::Reader::fillIndex(clusterIndex, clusterBuffer, clusterMCBuffer, clustersTPC, mcInputs, [&validSectors](auto& index) { return validSectors.test(index); });
212  //----------------------------<< TPC Clusters loading <<------------------------------------------
213 
214  //
215  const o2::dataformats::MCTruthContainer<o2::MCCompLabel>* lblITSPtr = nullptr;
216  std::unique_ptr<const o2::dataformats::MCTruthContainer<o2::MCCompLabel>> lblITS;
217 
218  const o2::dataformats::MCTruthContainer<o2::MCCompLabel>* lblClusITSPtr = nullptr;
219  std::unique_ptr<const o2::dataformats::MCTruthContainer<o2::MCCompLabel>> lblClusITS;
220 
221  const o2::dataformats::MCTruthContainer<o2::MCCompLabel>* lblTPCPtr = nullptr;
222  std::unique_ptr<const o2::dataformats::MCTruthContainer<o2::MCCompLabel>> lblTPC;
223 
224  if (mUseMC) {
225  lblITS = pc.inputs().get<const o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>("trackITSMCTR");
226  lblITSPtr = lblITS.get();
227 
228  lblClusITS = pc.inputs().get<const o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>("clusITSMCTR");
229  lblClusITSPtr = lblClusITS.get();
230 
231  lblTPC = pc.inputs().get<const o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>("trackTPCMCTR");
232  lblTPCPtr = lblTPC.get();
233  }
234  //
235  // pass input data to MatchTPCITS object
236  mMatching.setITSTracksInp(tracksITS);
237  mMatching.setITSTrackClusIdxInp(trackClIdxITS);
238  mMatching.setITSTrackROFRecInp(tracksITSROF);
239  mMatching.setITSClustersInp(clusITS);
240  mMatching.setITSClusterROFRecInp(clusITSROF);
241  mMatching.setTPCTracksInp(tracksTPC);
242  mMatching.setTPCTrackClusIdxInp(tracksTPCClRefs);
243  mMatching.setTPCClustersInp(&clusterIndex);
244 
245  if (mUseMC) {
246  mMatching.setITSTrkLabelsInp(lblITSPtr);
247  mMatching.setITSClsLabelsInp(lblClusITSPtr);
248  mMatching.setTPCTrkLabelsInp(lblTPCPtr);
249  }
250 
251  // const gsl::span<o2::ft0::RecPoints> fitInfo; // not yet POD
252  std::vector<o2::ft0::RecPoints> fitInfo;
253  if (o2::globaltracking::MatchITSTPCParams::Instance().runAfterBurner) {
254  // fitInfo = pc.inputs().get<gsl::span<o2::ft0::RecPoints>>("fitInfo");
255  fitInfo = pc.inputs().get<std::vector<o2::ft0::RecPoints>>("fitInfo");
256  mMatching.setFITInfoInp(&fitInfo);
257  }
258 
259  /*
260  const std::vector<o2::ft0::RecPoints>* rpFIT = nullptr;
261  std::unique_ptr<const std::vector<o2::ft0::RecPoints>> rpFITU;
262  if (o2::globaltracking::MatchITSTPCParams::Instance().runAfterBurner) {
263  rpFITU = pc.inputs().get<const std::vector<o2::ft0::RecPoints>*>("fitInfo");
264  rpFIT = rpFITU.get();
265  mMatching.setFITInfoInp(rpFIT);
266  }
267  */
268  mMatching.run();
269 
270  /* // at the moment we don't assume need for bufferization, no nead to clear
271  for (auto& secClBuf : mBufferedTPCClusters) {
272  secClBuf.clear();
273  }
274  */
275 
276  pc.outputs().snapshot(Output{"GLO", "TPCITS", 0, Lifetime::Timeframe}, mMatching.getMatchedTracks());
277  if (mUseMC) {
278  pc.outputs().snapshot(Output{"GLO", "TPCITS_ITSMC", 0, Lifetime::Timeframe}, mMatching.getMatchedITSLabels());
279  pc.outputs().snapshot(Output{"GLO", "TPCITS_TPCMC", 0, Lifetime::Timeframe}, mMatching.getMatchedTPCLabels());
280  }
281  mFinished = true;
282  pc.services().get<ControlService>().endOfStream();
283  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
284 }

286 DataProcessorSpec getTPCITSMatchingSpec(bool useMC, const std::vector<int>& tpcClusLanes)
287 {
288 
289  std::vector<InputSpec> inputs;
290  std::vector<OutputSpec> outputs;
291  inputs.emplace_back("trackITS", "ITS", "TRACKS", 0, Lifetime::Timeframe);
292  inputs.emplace_back("trackClIdx", "ITS", "TRACKCLSID", 0, Lifetime::Timeframe);
293  inputs.emplace_back("trackITSROF", "ITS", "ITSTrackROF", 0, Lifetime::Timeframe);
294  inputs.emplace_back("clusITS", "ITS", "CLUSTERS", 0, Lifetime::Timeframe);
295  inputs.emplace_back("clusITSROF", "ITS", "ITSClusterROF", 0, Lifetime::Timeframe);
296  inputs.emplace_back("trackTPC", "TPC", "TRACKS", 0, Lifetime::Timeframe);
297  inputs.emplace_back("trackTPCClRefs", "TPC", "CLUSREFS", 0, Lifetime::Timeframe);
298 
299  for (auto lane : tpcClusLanes) {
300  std::string clusBind = "clusTPC" + std::to_string(lane);
301  inputs.emplace_back(clusBind.c_str(), "TPC", "CLUSTERNATIVE", lane, Lifetime::Timeframe);
302  }
303 
304  if (o2::globaltracking::MatchITSTPCParams::Instance().runAfterBurner) {
305  inputs.emplace_back("fitInfo", "FT0", "RECPOINTS", 0, Lifetime::Timeframe);
306  }
307 
308  outputs.emplace_back("GLO", "TPCITS", 0, Lifetime::Timeframe);
309 
310  if (useMC) {
311  inputs.emplace_back("trackITSMCTR", "ITS", "TRACKSMCTR", 0, Lifetime::Timeframe);
312  inputs.emplace_back("trackTPCMCTR", "TPC", "TRACKSMCLBL", 0, Lifetime::Timeframe);
313  inputs.emplace_back("clusITSMCTR", "ITS", "CLUSTERSMCTR", 0, Lifetime::Timeframe);
314  //
315  outputs.emplace_back("GLO", "TPCITS_ITSMC", 0, Lifetime::Timeframe);
316  outputs.emplace_back("GLO", "TPCITS_TPCMC", 0, Lifetime::Timeframe);
317  }
318 
319  return DataProcessorSpec{
320  "itstpc-track-matcher",
321  inputs,
322  outputs,
323  AlgorithmSpec{adaptFromTask<TPCITSMatchingDPL>(useMC, tpcClusLanes)},
324  Options{}};
325 }
