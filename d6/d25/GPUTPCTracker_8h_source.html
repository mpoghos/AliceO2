198  GPUd() static int CalculateHitWeight(int NHits, float chi2, int)
199  {
200  const float chi2_suppress = 6.f;
201  float weight = (((float)NHits * (chi2_suppress - chi2 / 500.f)) * (1e9f / chi2_suppress / 160.f));
202  if (weight < 0.f || weight > 2e9f) {
203  return 0;
204  }
205  return ((int)weight);
206  // return( (NHits << 16) + num);
