41 void retrieveHits(std::vector<TChain*> const& chains,
42  const char* brname,
43  int sourceID,
44  int entryID,
45  std::vector<T>* hits)
46 {
47  auto br = chains[sourceID]->GetBranch(brname);
48  if (!br) {
49  LOG(ERROR) << "No branch found";
50  return;
51  }
52  br->SetAddress(&hits);
53  br->GetEntry(entryID);
54 }

56 DataProcessorSpec getTOFDigitizerSpec(int channel, bool useCCDB)
57 {
58  // setup of some data structures shared between init and processing functions
59  // (a shared pointer is used since then automatic cleanup is guaranteed with a lifetime beyond
60  // one process call)
61  auto simChains = std::make_shared<std::vector<TChain*>>();
62 
63  // the instance of the actual digitizer
64  auto digitizer = std::make_shared<o2::tof::Digitizer>();
65 
66  // containers for digits and labels
67  auto digits = std::make_shared<std::vector<o2::tof::Digit>>();
68  // auto digitsAccum = std::make_shared<o2::tof::Digit>(); // accumulator for all digits
69  auto labels = std::make_shared<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>();
70 
71  // the actual processing function which get called whenever new data is incoming
72  auto process = [simChains, digitizer, digits, /*digitsAccum,*/ labels, channel, useCCDB](ProcessingContext& pc) {
73  static bool finished = false;
74  if (finished) {
75  return;
76  }
77  // RS: at the moment using hardcoded flag for continuos readout
78  static o2::parameters::GRPObject::ROMode roMode = o2::parameters::GRPObject::CONTINUOUS;
79 
80  // read collision context from input
81  auto context = pc.inputs().get<o2::steer::RunContext*>("collisioncontext");
82  auto& timesview = context->getEventRecords();
83  LOG(DEBUG) << "GOT " << timesview.size() << " COLLISSION TIMES";
84 
85  // if there is nothing to do ... return
86  if (timesview.size() == 0) {
87  return;
88  }
89 
90  TStopwatch timer;
91  timer.Start();
92 
93  LOG(INFO) << " CALLING TOF DIGITIZATION ";
94  o2::dataformats::CalibLHCphaseTOF lhcPhaseObj;
95  o2::dataformats::CalibTimeSlewingParamTOF channelCalibObj;
96 
97  if (useCCDB) { // read calibration objects from ccdb
98  // check LHC phase
99  auto lhcPhase = pc.inputs().get<o2::dataformats::CalibLHCphaseTOF*>("tofccdbLHCphase");
100  auto channelCalib = pc.inputs().get<o2::dataformats::CalibTimeSlewingParamTOF*>("tofccdbChannelCalib");
101 
102  o2::dataformats::CalibLHCphaseTOF lhcPhaseObjTmp = std::move(*lhcPhase);
103  o2::dataformats::CalibTimeSlewingParamTOF channelCalibObjTmp = std::move(*channelCalib);
104 
105  // make a copy in global scope
106  lhcPhaseObj = lhcPhaseObjTmp;
107  channelCalibObj = channelCalibObjTmp;
108  } else { // calibration objects set to zero
109  lhcPhaseObj.addLHCphase(0, 0);
110  lhcPhaseObj.addLHCphase(2000000000, 0);
111 
112  for (int ich = 0; ich < o2::dataformats::CalibTimeSlewingParamTOF::NCHANNELS; ich++) {
113  channelCalibObj.addTimeSlewingInfo(ich, 0, 0);
114  int sector = ich / o2::dataformats::CalibTimeSlewingParamTOF::NCHANNELXSECTOR;
115  int channelInSector = ich % o2::dataformats::CalibTimeSlewingParamTOF::NCHANNELXSECTOR;
116  channelCalibObj.setFractionUnderPeak(sector, channelInSector, 1);
117  }
118  }
119 
120  o2::tof::CalibTOFapi calibapi(long(0), &lhcPhaseObj, &channelCalibObj);
121  digitizer->setCalibApi(&calibapi);
122 
123  static std::vector<o2::tof::HitType> hits;
124  // o2::dataformats::MCTruthContainer<o2::MCCompLabel> labelAccum;
125 
126  auto& eventParts = context->getEventParts();
127  // loop over all composite collisions given from context
128  // (aka loop over all the interaction records)
129  for (int collID = 0; collID < timesview.size(); ++collID) {
130  digitizer->setEventTime(timesview[collID].timeNS);
131 
132  // for each collision, loop over the constituents event and source IDs
133  // (background signal merging is basically taking place here)
134  for (auto& part : eventParts[collID]) {
135  digitizer->setEventID(part.entryID);
136  digitizer->setSrcID(part.sourceID);
137 
138  // get the hits for this event and this source
139  hits.clear();
140  retrieveHits(*simChains.get(), "TOFHit", part.sourceID, part.entryID, &hits);
141 
142  // LOG(INFO) << "For collision " << collID << " eventID " << part.entryID << " found " << hits.size() << " hits ";
143 
144  // call actual digitization procedure
145  labels->clear();
146  digits->clear();
147  digitizer->process(&hits, digits.get());
148  // copy digits into accumulator
149  //std::copy(digits->begin(), digits->end(), std::back_inserter(*digitsAccum.get()));
150  //labelAccum.mergeAtBack(*labels);
151  // LOG(INFO) << "Have " << digits->size() << " digits ";
152  }
153  }
154  if (digitizer->isContinuous()) {
155  digits->clear();
156  labels->clear();
157  digitizer->flushOutputContainer(*digits.get());
158  LOG(INFO) << "FLUSHING LEFTOVER STUFF " << digits->size();
159  }
160 
161  std::vector<Digit>* digitsVector = digitizer->getDigitPerTimeFrame();
162  std::vector<ReadoutWindowData>* readoutwindow = digitizer->getReadoutWindowData();
163  std::vector<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>* mcLabVecOfVec = digitizer->getMCTruthPerTimeFrame();
164 
165  // here we have all digits and we can send them to consumer (aka snapshot it onto output)
166  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "DIGITS", 0, Lifetime::Timeframe}, *digitsVector);
167  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "DIGITSMCTR", 0, Lifetime::Timeframe}, *mcLabVecOfVec);
168  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "READOUTWINDOW", 0, Lifetime::Timeframe}, *readoutwindow);
169  LOG(INFO) << "TOF: Sending ROMode= " << roMode << " to GRPUpdater";
170  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "ROMode", 0, Lifetime::Timeframe}, roMode);
171 
172  timer.Stop();
173  LOG(INFO) << "Digitization took " << timer.CpuTime() << "s";
174 
175  // we should be only called once; tell DPL that this process is ready to exit
176  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
177  finished = true;
178  };
179 
180  // init function returning the lambda taking a ProcessingContext
181  auto initIt = [simChains, process, digitizer, labels](InitContext& ctx) {
182  // setup the input chain for the hits
183  simChains->emplace_back(new TChain("o2sim"));
184 
185  // add the main (background) file
186  simChains->back()->AddFile(ctx.options().get<std::string>("simFile").c_str());
187 
188  // maybe add a particular signal file
189  auto signalfilename = ctx.options().get<std::string>("simFileS");
190  if (signalfilename.size() > 0) {
191  simChains->emplace_back(new TChain("o2sim"));
192  simChains->back()->AddFile(signalfilename.c_str());
193  }
194 
195  // make sure that the geometry is loaded (TODO will this be done centrally?)
196  if (!gGeoManager) {
197  o2::base::GeometryManager::loadGeometry();
198  }
199 
200  // init digitizer
201  digitizer->init();
202  const bool isContinuous = ctx.options().get<int>("pileup");
203  LOG(INFO) << "CONTINUOUS " << isContinuous;
204  digitizer->setContinuous(isContinuous);
205  digitizer->setMCTruthContainer(labels.get());
206 
207  // return the actual processing function which is now setup/configured
208  return process;
209  };
210 
211  // create the full data processor spec using
212  // a name identifier
213  // input description
214  // algorithmic description (here a lambda getting called once to setup the actual processing function)
215  // options that can be used for this processor (here: input file names where to take the hits)
216  std::vector<InputSpec> inputs;
217  inputs.emplace_back("collisioncontext", "SIM", "COLLISIONCONTEXT", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe);
218  if (useCCDB) {
219  inputs.emplace_back("tofccdbLHCphase", o2::header::gDataOriginTOF, "LHCphase");
220  inputs.emplace_back("tofccdbChannelCalib", o2::header::gDataOriginTOF, "ChannelCalib");
221  }
222  return DataProcessorSpec{
223  "TOFDigitizer",
224  inputs,
225  Outputs{OutputSpec{o2::header::gDataOriginTOF, "DIGITS", 0, Lifetime::Timeframe},
226  OutputSpec{o2::header::gDataOriginTOF, "READOUTWINDOW", 0, Lifetime::Timeframe},
227  OutputSpec{o2::header::gDataOriginTOF, "DIGITSMCTR", 0, Lifetime::Timeframe},
228  OutputSpec{o2::header::gDataOriginTOF, "ROMode", 0, Lifetime::Timeframe}},
229  AlgorithmSpec{initIt},
230  Options{{"simFile", VariantType::String, "o2sim.root", {"Sim (background) input filename"}},
231  {"simFileS", VariantType::String, "", {"Sim (signal) input filename"}},
232  {"pileup", VariantType::Int, 1, {"whether to run in continuous time mode"}}}
233  // I can't use VariantType::Bool as it seems to have a problem
234  };
235 }
