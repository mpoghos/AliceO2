21 int HBFUtils::getHBF(const IR& rec) const
22 {
24  auto diff = rec.differenceInBC(mFirstIR);
25  if (diff < 0) {
26  LOG(ERROR) << "IR " << rec.bc << '/' << rec.orbit << " is ahead of the reference IR "
27  << mFirstIR.bc << '/' << mFirstIR.orbit;
28  throw std::runtime_error("Requested IR is ahead of the reference IR");
29  }
30  return diff / o2::constants::lhc::LHCMaxBunches;
31 }

34 int HBFUtils::fillHBIRvector(std::vector<IR>& dst, const IR& fromIR, const IR& toIR) const
35 {
36  // Fill provided vector (cleaned) by interaction records (bc/orbit) for HBFs, considering
37  // BCs between interaction records "fromIR" and "toIR" (inclusive).
38  dst.clear();
39  int hb0 = getHBF(fromIR), hb1 = getHBF(toIR);
40  if (fromIR.bc != mFirstIR.bc) { // unless we are just starting the HBF of fromIR, it was already counted
41  hb0++;
42  }
43  for (int ihb = hb0; ihb <= hb1; ihb++) {
44  dst.emplace_back(getIRHBF(ihb));
45  }
46  return dst.size();
47 }

81 void HBFUtils::printRDH(const void* rdhP)
82 {
83  int version = (reinterpret_cast<const char*>(rdhP))[0];
84  switch (version) {
85  case 4:
86  printRDH(*reinterpret_cast<const o2::header::RAWDataHeaderV4*>(rdhP));
87  break;
88  case 5:
89  printRDH(*reinterpret_cast<const o2::header::RAWDataHeaderV5*>(rdhP));
90  break;
91  default:
92  LOG(ERROR) << "Unexpected RDH version " << version << " from";
93  dumpRDH(rdhP);
94  throw std::runtime_error("invalid RDH provided");
95  break;
96  };
97 }

124 bool HBFUtils::checkRDH(const void* rdhP, bool verbose)
125 {
126  int version = (reinterpret_cast<const char*>(rdhP))[0];
127  bool ok = true;
128  switch (version) {
129  case 4:
130  ok = checkRDH(*reinterpret_cast<const o2::header::RAWDataHeaderV4*>(rdhP), verbose);
131  break;
132  case 5:
133  ok = checkRDH(*reinterpret_cast<const o2::header::RAWDataHeaderV5*>(rdhP), verbose);
134  break;
135  default:
136  ok = false;
137  if (verbose) {
138  LOG(ERROR) << "Unexpected RDH version " << version << " from";
139  }
140  break;
141  };
142  if (!ok && verbose) {
143  dumpRDH(rdhP);
144  }
145  return ok;
146 }

177 bool HBFUtils::checkRDH(const o2::header::RAWDataHeaderV4& rdh, bool verbose)
178 {
179  // check if rdh conforms with RDH4 fields
180  bool ok = true;
181  if (rdh.version != 4) {
182  if (verbose) {
183  LOG(ERROR) << "RDH version 4 is expected instead of " << int(rdh.version);
184  }
185  ok = false;
186  }
187  if (rdh.headerSize != 64) {
188  if (verbose) {
189  LOG(ERROR) << "RDH with header size of 64 B is expected instead of " << int(rdh.headerSize);
190  }
191  ok = false;
192  }
193  if (rdh.memorySize < 64 || rdh.offsetToNext < 64 || rdh.memorySize > MAXCRUPage || rdh.offsetToNext > MAXCRUPage) {
194  if (verbose) {
195  LOG(ERROR) << "RDH expected to have memorySize/offsetToNext in 64 : 8192 bytes range instead of "
196  << int(rdh.memorySize) << '/' << int(rdh.offsetToNext);
197  }
198  ok = false;
199  }
200  if (rdh.zero0 || rdh.word3 || rdh.zero41 || rdh.zero42 || rdh.word5 || rdh.zero6 || rdh.word7) {
201  if (verbose) {
202  LOG(ERROR) << "Some reserved fields of RDH v4 are not empty";
203  }
204  ok = false;
205  }
206  if (!ok && verbose) {
207  dumpRDH(rdh);
208  }
209  return ok;
210 }
