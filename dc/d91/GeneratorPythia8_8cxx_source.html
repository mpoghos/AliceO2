83 Bool_t
84  GeneratorPythia8::importParticles()
85 {
88  TClonesArray& clonesParticles = *mParticles;
89  clonesParticles.Clear();
90 
91  /* loop over particles */
92  // auto weight = mPythia.info.weight(); // TBD: use weights
93  auto nParticles = mPythia.event.size();
94  for (Int_t iparticle = 1; iparticle < nParticles; iparticle++) { // first particle is system
95  auto particle = mPythia.event[iparticle];
96  auto pdg = particle.id();
97  auto st = particle.statusHepMC();
98  auto px = particle.px();
99  auto py = particle.py();
100  auto pz = particle.pz();
101  auto et = particle.e();
102  auto vx = particle.xProd();
103  auto vy = particle.yProd();
104  auto vz = particle.zProd();
105  auto vt = particle.tProd();
106  auto m1 = particle.mother1();
107  auto m2 = particle.mother2();
108  auto d1 = particle.daughter1();
109  auto d2 = particle.daughter2();
110  new (clonesParticles[iparticle]) TParticle(pdg, st, m1, m2, d1, d2, px, py, pz, et, vx, vy, vz, vt);
111  }
112 
114  return kTRUE;
115 }
