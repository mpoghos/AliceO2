56 WorkflowSpec defineDataProcessing(ConfigContext const& config)
57 {
58  int objectsBins = config.options().get<int>("obj-bins");
59  double objectsRate = config.options().get<double>("obj-rate");
60  int objectsProducers = config.options().get<int>("obj-producers");
61 
62  int mergersLayers = config.options().get<int>("mergers-layers");
63  MergingTime mergersMergeDecision =
64  config.options().get<std::string>("mergers-merge-decision") == "publication" ? MergingTime::BeforePublication : MergingTime::AfterArrival;
65  PublicationDecision mergersPublicationDecision =
66  config.options().get<std::string>("mergers-publication-decision") == "all-updated" ? PublicationDecision::WhenXInputsUpdated : PublicationDecision::EachNSeconds;
67  double mergersPublicationInterval = config.options().get<double>("mergers-publication-interval");
68  OwnershipMode mergersOwnershipMode =
69  config.options().get<std::string>("mergers-ownership-mode") == "full" ? OwnershipMode::Full : OwnershipMode::Integral;
70 
71  WorkflowSpec specs;
72  // clang-format off
73  // one 1D histo, binwise
74  {
75  Inputs mergersInputs;
76  for (size_t p = 0; p < objectsProducers; p++) {
77  mergersInputs.push_back({ "mo", "TST",
78  "HISTO", static_cast<o2::header::DataHeader::SubSpecificationType>(p + 1),
79  Lifetime::Timeframe });
80  DataProcessorSpec producer{
81  "producer-histo" + std::to_string(p), Inputs{},
82  Outputs{ { { "mo" },
83  "TST",
84  "HISTO",
85  static_cast<o2::header::DataHeader::SubSpecificationType>(p + 1),
86  Lifetime::Timeframe } },
87  AlgorithmSpec{
88  (AlgorithmSpec::ProcessCallback)[ p, periodus = int(1000000 / objectsRate), objectsBins, objectsProducers ](
89  ProcessingContext& processingContext) mutable { static auto lastTime = steady_clock::now();
90  auto now = steady_clock::now();
91 
92  if (duration_cast<microseconds>(now - lastTime).count() > periodus) {
93 
94  lastTime += microseconds(periodus);
95 
96  auto subspec = static_cast<o2::header::DataHeader::SubSpecificationType>(p + 1);
97  TH1F& histo = processingContext.outputs().make<TH1F>(Output{ "TST", "HISTO", subspec }, "gauss", "gauss", objectsBins, -3, 3);
98  histo.FillRandom("gaus", 1000);
99  }
100  }
101  }
102  };
103  specs.push_back(producer);
104  }
105 
106  MergerInfrastructureBuilder mergersBuilder;
107  mergersBuilder.setInfrastructureName("histos");
108  mergersBuilder.setInputSpecs(mergersInputs);
109  mergersBuilder.setOutputSpec({{ "main" }, "TST", "HISTO", 0 });
110  MergerConfig mergerConfig;
111  mergerConfig.ownershipMode = { mergersOwnershipMode };
112  mergerConfig.publicationDecision = { mergersPublicationDecision, mergersPublicationDecision == PublicationDecision::EachNSeconds ? mergersPublicationInterval : 1.0 };
113  mergerConfig.mergingTime = { mergersMergeDecision };
114  mergerConfig.timespan = { Timespan::FullHistory };
115  mergerConfig.topologySize = { TopologySize::NumberOfLayers, mergersLayers };
116  mergersBuilder.setConfig(mergerConfig);
117 
118  mergersBuilder.generateInfrastructure(specs);
119 
120  DataProcessorSpec printer{
121  "printer-bins",
122  Inputs{
123  { "histo", "TST", "HISTO", 0 }
124  },
125  Outputs{},
126  AlgorithmSpec{
127  (AlgorithmSpec::InitCallback) [](InitContext&) {
128  return (AlgorithmSpec::ProcessCallback) [](ProcessingContext& processingContext) mutable {
129 // LOG(INFO) << "printer invoked";
130  auto histo = processingContext.inputs().get<TH1F*>("histo");
131  std::string bins = "BINS:";
132  for (int i = 1; i <= histo->GetNbinsX(); i++) {
133  bins += " " + std::to_string((int) histo->GetBinContent(i));
134  }
135  LOG(INFO) << bins;
136  };
137  }
138  }
139  };
140  specs.push_back(printer);
141  }
142 
143  return specs;
144 }
