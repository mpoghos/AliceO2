73 Bool_t PrimaryGenerator::GenerateEvent(FairGenericStack* pStack)
74 {
78  if (!mEmbedTree)
79  return FairPrimaryGenerator::GenerateEvent(pStack);
80 
84  mEmbedTree->GetEntry(mEmbedIndex);
85  setInteractionVertex(mEmbedEvent);
86 
88  if (!FairPrimaryGenerator::GenerateEvent(pStack))
89  return kFALSE;
90 
92  auto o2event = dynamic_cast<MCEventHeader*>(fEvent);
93  if (o2event) {
94  o2event->setEmbeddingFileName(mEmbedFile->GetName());
95  o2event->setEmbeddingEventIndex(mEmbedIndex);
96  }
97 
99  mEmbedIndex++;
100  mEmbedIndex %= mEmbedEntries;
101 
103  return kTRUE;
104 }

128 void PrimaryGenerator::setInteractionDiamond(const Double_t* xyz, const Double_t* sigmaxyz)
129 {
132  LOG(INFO) << "Setting interaction diamond: position = {"
133  << xyz[0] << "," << xyz[1] << "," << xyz[2] << "} cm";
134  LOG(INFO) << "Setting interaction diamond: width = {"
135  << sigmaxyz[0] << "," << sigmaxyz[1] << "," << sigmaxyz[2] << "} cm";
136  SetBeam(xyz[0], xyz[1], sigmaxyz[0], sigmaxyz[1]);
137  SetTarget(xyz[2], sigmaxyz[2]);
138 
139  auto const& param = InteractionDiamondParam::Instance();
140  SmearVertexXY(false);
141  SmearVertexZ(false);
142  SmearGausVertexXY(false);
143  SmearGausVertexZ(false);
144  if (param.distribution == o2::eventgen::EVertexDistribution::kFlat) {
145  SmearVertexXY(true);
146  SmearVertexZ(true);
147  } else if (param.distribution == o2::eventgen::EVertexDistribution::kGaus) {
148  SmearGausVertexXY(true);
149  SmearGausVertexZ(true);
150  } else {
151  LOG(ERROR) << "PrimaryGenerator: Unsupported vertex distribution";
152  }
153 }

172 Bool_t PrimaryGenerator::embedInto(TString fname)
173 {
177  if (mEmbedFile && mEmbedFile->IsOpen()) {
178  LOG(ERROR) << "Another embedding file is currently open";
179  return kFALSE;
180  }
181 
183  mEmbedFile = TFile::Open(fname);
184  if (!mEmbedFile || !mEmbedFile->IsOpen()) {
185  LOG(ERROR) << "Cannot open file for embedding: " << fname;
186  return kFALSE;
187  }
188 
190  mEmbedTree = (TTree*)mEmbedFile->Get("o2sim");
191  if (!mEmbedTree) {
192  LOG(ERROR) << R"(Cannot find "o2sim" tree for embedding in )" << fname;
193  return kFALSE;
194  }
195 
197  mEmbedEntries = mEmbedTree->GetEntries();
198  if (mEmbedEntries <= 0) {
199  LOG(ERROR) << "Invalid number of entries found in tree for embedding: " << mEmbedEntries;
200  return kFALSE;
201  }
202 
204  mEmbedEvent = new MCEventHeader;
205  mEmbedTree->SetBranchAddress("MCEventHeader.", &mEmbedEvent);
206 
208  return kTRUE;
209 }
