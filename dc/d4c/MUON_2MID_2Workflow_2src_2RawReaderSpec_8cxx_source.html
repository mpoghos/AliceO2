37 class RawReaderDeviceDPL
38 {
39  public:
40  void init(of::InitContext& ic)
41  {
42  auto filename = ic.options().get<std::string>("mid-raw-infile");
43  auto readContinuous = ic.options().get<bool>("mid-read-continuous");
44  if (!mRawFileReader.init(filename.c_str(), readContinuous)) {
45  return;
46  }
47 
48  mHBperTimeframe = ic.options().get<int>("mid-hb-per-timeframe");
49 
50  auto stop = [this]() {
51  LOG(INFO) << "Capacities: ROFRecords: " << mDecoder.getROFRecords().capacity() << " data: " << mDecoder.getData().capacity();
52  LOG(INFO) << "Read " << mHBCounter << " HBs in " << mTimer.count() << " s";
53  double scaleFactor = 1.e6 / mNROFs;
54  LOG(INFO) << "Processing time / " << mNROFs << " ROFs: full: " << mTimer.count() * scaleFactor << " us decoding: " << mTimerAlgo.count() * scaleFactor << " us";
55  };
56  ic.services().get<of::CallbackService>().set(of::CallbackService::Id::Stop, stop);
57  }
58 
59  void run(of::ProcessingContext& pc)
60  {
61  if (mRawFileReader.getState() != 0) {
62  return;
63  }
64 
65  auto tStart = std::chrono::high_resolution_clock::now();
66 
67  size_t nHBs = 0;
68 
69  while (nHBs < mHBperTimeframe && mRawFileReader.readAllGBTs(false)) {
70  nHBs += mRawFileReader.getNumberOfHBs(0);
71  }
72  mHBCounter += nHBs;
73 
74  auto tAlgoStart = std::chrono::high_resolution_clock::now();
75  mDecoder.process(mRawFileReader.getData());
76  mTimerAlgo += std::chrono::high_resolution_clock::now() - tAlgoStart;
77 
78  mRawFileReader.clear();
79 
80  pc.outputs().snapshot(of::Output{"MID", "DATA", 0, of::Lifetime::Timeframe}, mDecoder.getData());
81  pc.outputs().snapshot(of::Output{"MID", "DATAROF", 0, of::Lifetime::Timeframe}, mDecoder.getROFRecords());
82 
83  mTimer += std::chrono::high_resolution_clock::now() - tStart;
84  mNROFs += mDecoder.getROFRecords().size();
85 
86  if (mRawFileReader.getState() == 1) {
87  pc.services().get<of::ControlService>().endOfStream();
88  }
89  }
90 
91  private:
92  Decoder mDecoder{};
93  RawFileReader<raw::RawUnit> mRawFileReader{};
94  int mHBperTimeframe{256};
95  unsigned long int mHBCounter{0};
96  std::chrono::duration<double> mTimer{0};
97  std::chrono::duration<double> mTimerAlgo{0};
98  unsigned int mNROFs{0};
99 };
