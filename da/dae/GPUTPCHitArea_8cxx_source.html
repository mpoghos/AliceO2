25 GPUd() void GPUTPCHitArea::Init(const MEM_TYPE(GPUTPCRow) & row, GPUconstantref() const MEM_GLOBAL(GPUTPCSliceData) & slice, float y, float z, float dy, float dz)
26 {
27  // initialisation
28  mHitOffset = row.HitNumberOffset();
29  mY = y;
30  mZ = z;
31  mMinZ = z - dz;
32  mMaxZ = z + dz;
33  mMinY = y - dy;
34  mMaxY = y + dy;
35  int bYmin, bZmin, bYmax; // boundary bin indexes
36  row.Grid().GetBin(mMinY, mMinZ, &bYmin, &bZmin);
37  row.Grid().GetBin(mMaxY, mMaxZ, &bYmax, &mBZmax);
38  mBDY = bYmax - bYmin + 1; // bin index span in y direction
39  mNy = row.Grid().Ny();
40  mIndYmin = bZmin * mNy + bYmin; // same as grid.GetBin(mMinY, mMinZ), i.e. the smallest bin index of interest
41  // mIndYmin + mBDY then is the largest bin index of interest with the same Z
42  mIz = bZmin;
43 
44 // for given mIz (which is min atm.) get
45 #ifdef GPUCA_TEXTURE_FETCH_NEIGHBORS
46  mHitYfst = tex1Dfetch(gAliTexRefu, ((char*)slice.FirstHitInBin(row) - slice.GPUTextureBaseConst()) / sizeof(calink) + mIndYmin);
47  mHitYlst = tex1Dfetch(gAliTexRefu, ((char*)slice.FirstHitInBin(row) - slice.GPUTextureBaseConst()) / sizeof(calink) + mIndYmin + mBDY);
48 #else
49  mHitYfst = slice.FirstHitInBin(row, mIndYmin); // first and
50  mHitYlst = slice.FirstHitInBin(row, mIndYmin + mBDY); // last hit index in the bin
51 #endif // GPUCA_TEXTURE_FETCH_NEIGHBORS
52  mIh = mHitYfst;
53 }

56 GPUd() int GPUTPCHitArea::GetNext(GPUconstantref() const MEM_GLOBAL(GPUTPCTracker) & tracker, const MEM_TYPE(GPUTPCRow) & row, GPUconstantref() const MEM_GLOBAL(GPUTPCSliceData) & slice, GPUTPCHit* h)
57 {
58  // get next hit index
59 
60  // min coordinate
61  const float y0 = row.Grid().YMin();
62  const float z0 = row.Grid().ZMin();
63 
64  // step vector
65  const float stepY = row.HstepY();
66  const float stepZ = row.HstepZ();
67 
68  int ret = -1;
69  do {
70  while (mIh >= mHitYlst) {
71  if (mIz >= mBZmax) {
72  return -1;
73  }
74  // go to next z and start y from the min again
75  ++mIz;
76  mIndYmin += mNy;
77 #ifdef GPUCA_TEXTURE_FETCH_NEIGHBORS
78  mHitYfst = tex1Dfetch(gAliTexRefu, ((char*)slice.FirstHitInBin(row) - slice.GPUTextureBaseConst()) / sizeof(calink) + mIndYmin);
79  mHitYlst = tex1Dfetch(gAliTexRefu, ((char*)slice.FirstHitInBin(row) - slice.GPUTextureBaseConst()) / sizeof(calink) + mIndYmin + mBDY);
80 #else
81  mHitYfst = slice.FirstHitInBin(row, mIndYmin);
82  mHitYlst = slice.FirstHitInBin(row, mIndYmin + mBDY);
83 #endif
84  mIh = mHitYfst;
85  }
86 
87 #ifdef GPUCA_TEXTURE_FETCH_NEIGHBORS
88  cahit2 tmpval = tex1Dfetch(gAliTexRefu2, ((char*)slice.HitData(row) - slice.GPUTextureBaseConst()) / sizeof(cahit2) + mIh);
89  h->SetY(y0 + tmpval.x * stepY);
90  h->SetZ(z0 + tmpval.y * stepZ);
91 #else
92  h->SetY(y0 + tracker.HitDataY(row, mIh) * stepY);
93  h->SetZ(z0 + tracker.HitDataZ(row, mIh) * stepZ);
94 #endif
95 
96  if (h->Z() > mMaxZ || h->Z() < mMinZ || h->Y() < mMinY || h->Y() > mMaxY) {
97  mIh++;
98  continue;
99  }
100  ret = mIh;
101  mIh++;
102  break;
103  } while (1);
104  return ret;
105 }
