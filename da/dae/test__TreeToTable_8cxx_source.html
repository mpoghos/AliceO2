26 BOOST_AUTO_TEST_CASE(TreeToTableConversion)
27 {
28  using namespace o2::framework;
30  Int_t ndp = 10000000;
31 
32  TFile f1("tree2table.root", "RECREATE");
33  TTree t1("t1", "a simple Tree with simple variables");
34  Float_t px, py, pz;
35  Double_t random;
36  Int_t ev;
37  t1.Branch("px", &px, "px/F");
38  t1.Branch("py", &py, "py/F");
39  t1.Branch("pz", &pz, "pz/F");
40  t1.Branch("random", &random, "random/D");
41  t1.Branch("ev", &ev, "ev/I");
42 
43  //fill the tree
44  for (Int_t i = 0; i < ndp; i++) {
45  gRandom->Rannor(px, py);
46  pz = px * px + py * py;
47  random = gRandom->Rndm();
48  ev = i + 1;
49  t1.Fill();
50  }
51 
52  // Create an arrow table from this.
53  TreeToTable tr2ta(&t1);
54  auto stat = tr2ta.AddAllColumns();
55  if (!stat) {
56  LOG(INFO) << "Table was not created!";
57  return;
58  }
59  auto table = tr2ta.Process();
60 
61  // test result
62  BOOST_REQUIRE_EQUAL(table.get()->Validate().ok(), true);
63  BOOST_REQUIRE_EQUAL(table.get()->num_rows(), ndp);
64  BOOST_REQUIRE_EQUAL(table.get()->num_columns(), 5);
65  BOOST_REQUIRE_EQUAL(table.get()->column(0)->type()->id(), arrow::float32()->id());
66  BOOST_REQUIRE_EQUAL(table.get()->column(1)->type()->id(), arrow::float32()->id());
67  BOOST_REQUIRE_EQUAL(table.get()->column(2)->type()->id(), arrow::float32()->id());
68  BOOST_REQUIRE_EQUAL(table.get()->column(3)->type()->id(), arrow::float64()->id());
69  BOOST_REQUIRE_EQUAL(table.get()->column(4)->type()->id(), arrow::int32()->id());
70 
71  f1.Close();
72 }
