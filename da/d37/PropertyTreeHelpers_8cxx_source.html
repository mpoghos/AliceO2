26 void PropertyTreeHelpers::populate(std::vector<ConfigParamSpec> const& schema, boost::property_tree::ptree& pt, boost::program_options::variables_map const& vmap)
27 {
28  for (auto& spec : schema) {
29  // strip short version to get the correct key
30  std::string key = spec.name.substr(0, spec.name.find(","));
31  try {
32  switch (spec.type) {
33  case VariantType::Int:
34  pt.put(key, vmap[key].as<int>());
35  break;
36  case VariantType::Int64:
37  pt.put(key, vmap[key].as<int64_t>());
38  break;
39  case VariantType::Float:
40  pt.put(key, vmap[key].as<float>());
41  break;
42  case VariantType::Double:
43  pt.put(key, vmap[key].as<double>());
44  break;
45  case VariantType::String:
46  if (auto v = boost::any_cast<std::string>(&vmap[key].value())) {
47  pt.put(key, *v);
48  }
49  break;
50  case VariantType::Bool:
51  pt.put(key, vmap[key].as<bool>());
52  break;
53  case VariantType::Unknown:
54  case VariantType::Empty:
55  default:
56  throw std::runtime_error("Unknown variant type");
57  }
58  } catch (std::runtime_error& re) {
59  throw re;
60  } catch (std::exception& e) {
61  throw std::invalid_argument(std::string("missing option: ") + key + " (" + e.what() + ")");
62  } catch (...) {
63  throw std::invalid_argument(std::string("missing option: ") + key);
64  }
65  }
66 }
