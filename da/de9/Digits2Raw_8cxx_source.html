10 /*
11  Digits to RAW data coding. RAW data format - class DataFormat/Detectors/FIT/FT0/RawEventData
12  18 PMs (GBT links) 12 MCPs each and 1 TCM, each stream transmit separately
13 Event header - 80bits
14  uint startDescriptor : 4;
15  uint nGBTWords : 4;
16  uint reservedField : 32;
17  uint orbit : 32;
18  uint bc : 12;
19 
20  Event data 40bits
21  short int time : 12;
22  short int charge : 12;
23  unsigned short int numberADC : 1;
24  bool isDoubleEvent : 1;
25  bool is1TimeLostEvent : 1;
26  bool is2TimeLostEvent : 1;
27  bool isADCinGate : 1;
28  bool isTimeInfoLate : 1;
29  bool isAmpHigh : 1;
30  bool isEventInTVDC : 1;
31  bool isTimeInfoLost : 1;
32  uint channelID : 4;
33 GBT packet:
34 RDH + Event header + event data, 2 channels per 1 GBT word;
35 if no data for this PM - only headers.
36 
37 Trigger mode : detector sends data to FLP at each trigger;
38 Continueous mode : for only bunches with data at least in 1 channel.
39 */

70 void Digits2Raw::readDigits(const std::string fileDigitsName)
71 {
72  LOG(INFO) << "**********Digits2Raw::convertDigits" << std::endl;
73 
74  o2::ft0::LookUpTable lut{o2::ft0::Digits2Raw::linear()};
75  LOG(INFO) << " ##### LookUp set ";
76 
77  TFile* fdig = TFile::Open(fileDigitsName.data());
78  assert(fdig != nullptr);
79  LOG(INFO) << " Open digits file " << fileDigitsName.data();
80  TTree* digTree = (TTree*)fdig->Get("o2sim");
81  std::vector<o2::ft0::Digit>* digArr = new std::vector<o2::ft0::Digit>;
82  digTree->SetBranchAddress("FT0Digit", &digArr);
83  Int_t nevD = digTree->GetEntries(); // digits in cont. readout may be grouped as few events per entry
84  uint32_t old_orbit = ~0;
85  o2::InteractionRecord intRecord;
86 
87  for (Int_t iev = 0; iev < nevD; iev++) {
88  digTree->GetEvent(iev);
89  for (const auto& digit : *digArr) {
90  intRecord = digit.getInteractionRecord();
91  uint32_t current_orbit = intRecord.orbit;
92  LOG(DEBUG) << "old orbit " << old_orbit << " new orbit " << current_orbit;
93  if (old_orbit != current_orbit) {
94  for (DataPageWriter& writer : mPages)
95  writer.flush(mFileDest);
96  for (int nlink = 0; nlink < NPMs; ++nlink)
97  setRDH(mPages[nlink].mRDH, nlink, intRecord);
98  old_orbit = current_orbit;
99  }
100  convertDigits(digit, lut);
101  }
102  }
103  for (DataPageWriter& writer : mPages)
104  writer.flush(mFileDest);
105  for (int nlink = 0; nlink < NPMs; ++nlink)
106  setRDH(mPages[nlink].mRDH, nlink, intRecord);
107 }

110 void Digits2Raw::convertDigits(const o2::ft0::Digit& digit, const o2::ft0::LookUpTable& lut)
111 {
112  auto intRecord = digit.getInteractionRecord();
113  std::vector<o2::ft0::ChannelData> mTimeAmp = digit.getChDgData();
114  // check empty event
115  if (mTimeAmp.size() != 0) {
116  bool is0TVX = digit.getisVrtx();
117  int oldlink = -1;
118  int nchannels = 0;
119  for (auto& d : mTimeAmp) {
120  int nlink = lut.getLink(d.ChId);
121  if (nlink != oldlink) {
122  if (oldlink >= 0) {
123  uint nGBTWords = uint((nchannels + 1) / 2);
124  if ((nchannels % 2) == 1)
125  mRawEventData.mEventData[nchannels] = {};
126  mRawEventData.mEventHeader.nGBTWords = nGBTWords;
127  mPages[oldlink].write(mRawEventData.to_vector());
128  }
129  oldlink = nlink;
130  mRawEventData.mEventHeader = makeGBTHeader(nlink, intRecord);
131  nchannels = 0;
132  }
133  auto& newData = mRawEventData.mEventData[nchannels];
134  newData.charge = d.QTCAmpl;
135  newData.time = d.CFDTime;
136  newData.is1TimeLostEvent = 0;
137  newData.is2TimeLostEvent = 0;
138  newData.isADCinGate = 1;
139  newData.isAmpHigh = 0;
140  newData.isDoubleEvent = 0;
141  newData.isEventInTVDC = is0TVX ? 1 : 0;
142  newData.isTimeInfoLate = 0;
143  newData.isTimeInfoLost = 0;
144  int chain = std::rand() % 2;
145  newData.numberADC = chain ? 1 : 0;
146  newData.channelID = lut.getMCP(d.ChId);
147  LOG(DEBUG) << "packed GBT " << nlink << " channelID " << (int)newData.channelID << " charge " << newData.charge << " time " << newData.time << " chain " << int(newData.numberADC) << " size " << sizeof(newData);
148  nchannels++;
149  }
150  // fill mEventData[nchannels] with 0s to flag that this is a dummy data
151  uint nGBTWords = uint((nchannels + 1) / 2);
152  if ((nchannels % 2) == 1)
153  mRawEventData.mEventData[nchannels] = {};
154  mRawEventData.mEventHeader.nGBTWords = nGBTWords;
155  mPages[oldlink].write(mRawEventData.to_vector());
156  }
157 }
