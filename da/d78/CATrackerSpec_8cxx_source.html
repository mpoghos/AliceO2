95  // Default Settings

119  // Parse the config string
120  const char* opt = options.c_str();
121  if (opt && *opt) {
122  printf("Received options %s\n", opt);
123  const char* optPtr = opt;
124  while (optPtr && *optPtr) {
125  while (*optPtr == ' ') {
126  optPtr++;

128  const char* nextPtr = strstr(optPtr, " ");
129  const int optLen = nextPtr ? nextPtr - optPtr : strlen(optPtr);

137 #ifdef GPUCA_BUILD_EVENT_DISPLAY

148  sscanf(optPtr + 3, "%f", &solenoidBz);
149  printf("Using solenoid field %f\n", solenoidBz);

151  sscanf(optPtr + 5, "%f", &refX);
152  printf("Propagating to reference X %f\n", refX);
153  } else if (optLen > 5 && strncmp(optPtr, "debug=", 6) == 0) {
154  sscanf(optPtr + 6, "%d", &debugLevel);
155  printf("Debug level set to %d\n", debugLevel);
156  } else if (optLen > 8 && strncmp(optPtr, "threads=", 8) == 0) {
157  sscanf(optPtr + 8, "%d", &nThreads);
158  printf("Using %d threads\n", nThreads);
159  } else if (optLen > 8 && strncmp(optPtr, "gpuType=", 8) == 0) {
160  int len = std::min(optLen - 8, 1023);
161  memcpy(gpuType, optPtr + 8, len);
162  gpuType[len] = 0;
163  useGPU = true;
164  printf("Using GPU Type %s\n", gpuType);
165  } else if (optLen > 15 && strncmp(optPtr, "transformation=", 15) == 0) {
166  int len = std::min(optLen - 15, 1023);
167  memcpy(tpcTransformationFileName, optPtr + 15, len);

202  config.configReconstruction.tpcRejectionMode = GPUSettings::RejectionStrategyA; // Implement TPC Strategy A
203  config.configReconstruction.tpcRejectQPt = 1.f / 0.05f; // Reject clusters of tracks < 50 MeV
204  config.configReconstruction.tpcCompressionModes = GPUSettings::CompressionFull; // Activate all compression steps
205  config.configReconstruction.tpcCompressionSortOrder = GPUSettings::SortPad; // Sort order for differences compression
206  config.configReconstruction.tpcSigBitsCharge = 4; // Number of significant bits in TPC cluster chargs
207  config.configReconstruction.tpcSigBitsWidth = 3; // Number of significant bits in TPC cluster width

212  config.configWorkflow.steps.set(GPUDataTypes::RecoStep::TPCConversion,

216  GPUDataTypes::RecoStep::TPCdEdx);
217  // Alternative steps: TRDTracking | ITSTracking
218  config.configWorkflow.inputs.set(GPUDataTypes::InOutType::TPCClusters);
219  // Alternative inputs: GPUDataTypes::InOutType::TRDTracklets
220  config.configWorkflow.outputs.set(GPUDataTypes::InOutType::TPCMergedTracks, GPUDataTypes::InOutType::TPCCompressedClusters);
221  // Alternative outputs: GPUDataTypes::InOutType::TPCSectorTracks, GPUDataTypes::InOutType::TRDTracks
222  if (caClusterer) { // Override some settings if we have raw data as input
223  config.configWorkflow.inputs.set(GPUDataTypes::InOutType::TPCRaw);

244  gm.createPadPlaneArray();
245  gm.createClusterMatrixArray();
246  std::unique_ptr<o2::trd::TRDGeometryFlat> gf(gm);
247  config.trdGeometry = gf.get();*/

257  auto processingFct = [processAttributes, processMC, caClusterer](ProcessingContext& pc) {
258  if (processAttributes->readyToQuit) {
259  return;
260  }
261  auto& parser = processAttributes->parser;
262  auto& tracker = processAttributes->tracker;
263  uint64_t activeSectors = 0;
264  auto& verbosity = processAttributes->verbosity;
265 
266  // FIXME cleanup almost duplicated code
267  auto& validMcInputs = processAttributes->validMcInputs;
268  auto& mcInputs = processAttributes->mcInputs;
269  std::array<std::vector<o2::tpc::Digit>, NSectors> inputDigits;
270  std::array<std::unique_ptr<const MCLabelContainer>, NSectors> inputDigitsMC;
271  if (processMC) {
272  // we can later extend this to multiple inputs
273  for (auto const& inputId : processAttributes->inputIds) {
274  std::string inputLabel = "mclblin" + std::to_string(inputId);
275  auto ref = pc.inputs().get(inputLabel);
276  auto const* sectorHeader = DataRefUtils::getHeader<o2::tpc::TPCSectorHeader*>(ref);
277  if (sectorHeader == nullptr) {
278  // FIXME: think about error policy
279  LOG(ERROR) << "sector header missing on header stack";
280  return;
281  }
282  const int& sector = sectorHeader->sector;
283  if (sector < 0) {
284  continue;
285  }
286  if (validMcInputs.test(sector)) {
287  // have already data for this sector, this should not happen in the current
288  // sequential implementation, for parallel path merged at the tracker stage
289  // multiple buffers need to be handled
290  throw std::runtime_error("can only have one MC data set per sector");
291  }
292  if (caClusterer) {
293  inputDigitsMC[sector] = std::move(pc.inputs().get<const MCLabelContainer*>(inputLabel.c_str()));
294  } else {
295  mcInputs[sector] = std::move(pc.inputs().get<std::vector<MCLabelContainer>>(inputLabel.c_str()));
296  }
297  validMcInputs.set(sector);
298  activeSectors |= sectorHeader->activeSectors;
299  if (verbosity > 1) {
300  LOG(INFO) << "received " << *(ref.spec) << " MC label containers"
301  << " for sector " << sector //
302  << std::endl //
303  << " mc input status: " << validMcInputs //
304  << std::endl //
305  << " active sectors: " << std::bitset<NSectors>(activeSectors); //
306  }
307  }
308  }
309 
310  auto& validInputs = processAttributes->validInputs;
311  int operation = 0;
312  std::map<int, DataRef> datarefs;
313  for (auto const& inputId : processAttributes->inputIds) {
314  std::string inputLabel = "input" + std::to_string(inputId);
315  auto ref = pc.inputs().get(inputLabel);
316  auto const* sectorHeader = DataRefUtils::getHeader<o2::tpc::TPCSectorHeader*>(ref);
317  if (sectorHeader == nullptr) {
318  // FIXME: think about error policy
319  LOG(ERROR) << "sector header missing on header stack";
320  return;
321  }
322  const int& sector = sectorHeader->sector;
323  // check the current operation, this is used to either signal eod or noop
324  // FIXME: the noop is not needed any more once the lane configuration with one
325  // channel per sector is used
326  if (sector < 0) {
327  if (operation < 0 && operation != sector) {
328  // we expect the same operation on all inputs
329  LOG(ERROR) << "inconsistent lane operation, got " << sector << ", expecting " << operation;
330  } else if (operation == 0) {
331  // store the operation
332  operation = sector;
333  }
334  continue;
335  }
336  if (validInputs.test(sector)) {
337  // have already data for this sector, this should not happen in the current
338  // sequential implementation, for parallel path merged at the tracker stage
339  // multiple buffers need to be handled
340  throw std::runtime_error("can only have one cluster data set per sector");
341  }
342  activeSectors |= sectorHeader->activeSectors;
343  validInputs.set(sector);
344  datarefs[sector] = ref;
345  if (caClusterer) {
346  inputDigits[sector] = std::move(pc.inputs().get<const std::vector<o2::tpc::Digit>>(inputLabel));
347  }
348  }
349 
350  if (operation == -1) {
351  // EOD is transmitted in the sectorHeader with sector number equal to -1
352  o2::tpc::TPCSectorHeader sh{-1};
353  sh.activeSectors = activeSectors;
354  pc.outputs().snapshot(OutputRef{"outTracks", 0, {sh}}, -1);
355  pc.outputs().snapshot(OutputRef{"outClusRefs", 0, {sh}}, -1);
356  if (processMC) {
357  pc.outputs().snapshot(OutputRef{"mclblout", 0, {sh}}, -1);
358  }
359  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
360  processAttributes->readyToQuit = true;
361  return;
362  }
363  auto printInputLog = [&verbosity, &validInputs, &activeSectors](auto& r, const char* comment, auto& s) {
364  if (verbosity > 1) {
365  LOG(INFO) << comment << " " << *(r.spec) << ", size " << DataRefUtils::getPayloadSize(r) //
366  << " for sector " << s //
367  << std::endl //
368  << " input status: " << validInputs //
369  << std::endl //
370  << " active sectors: " << std::bitset<NSectors>(activeSectors); //
371  }
372  };
373  auto& bufferedInputs = processAttributes->bufferedInputs;
374  if (activeSectors == 0 || (activeSectors & validInputs.to_ulong()) != activeSectors ||
375  (processMC && (activeSectors & validMcInputs.to_ulong()) != activeSectors)) {
376  // not all sectors available, we have to buffer the inputs
377  if (caClusterer) {
378  throw std::runtime_error("Buffering not possible with digits");
379  }
380  for (auto const& refentry : datarefs) {
381  auto& sector = refentry.first;
382  auto& ref = refentry.second;
383  auto payloadSize = DataRefUtils::getPayloadSize(ref);
384  bufferedInputs[sector].resize(payloadSize);
385  std::copy(ref.payload, ref.payload + payloadSize, bufferedInputs[sector].begin());
386  printInputLog(ref, "buffering", sector);
387  }
388 
389  // not needed to send something, DPL will simply drop this timeslice, whenever the
390  // data for all sectors is available, the output is sent in that time slice
391  return;
392  }
393  assert(processMC == false || validMcInputs == validInputs);
394  std::array<gsl::span<const char>, NSectors> inputs;
395  auto inputStatus = validInputs;
396  for (auto const& refentry : datarefs) {
397  auto& sector = refentry.first;
398  auto& ref = refentry.second;
399  inputs[sector] = gsl::span(ref.payload, DataRefUtils::getPayloadSize(ref));
400  inputStatus.reset(sector);
401  printInputLog(ref, "received", sector);
402  }
403  if (inputStatus.any()) {
404  // some of the inputs have been buffered
405  for (size_t sector = 0; sector < inputStatus.size(); ++sector) {
406  if (inputStatus.test(sector)) {
407  inputs[sector] = gsl::span(&bufferedInputs[sector].front(), bufferedInputs[sector].size());
408  }
409  }
410  }
411  if (verbosity > 0) {
412  if (inputStatus.any()) {
413  LOG(INFO) << "using buffered data for " << inputStatus.count() << " sector(s)";
414  }
415  // make human readable information from the bitfield
416  std::string bitInfo;
417  auto nActiveBits = validInputs.count();
418  if (((uint64_t)0x1 << nActiveBits) == validInputs.to_ulong() + 1) {
419  // sectors 0 to some upper bound are active
420  bitInfo = "0-" + std::to_string(nActiveBits - 1);
421  } else {
422  int rangeStart = -1;
423  int rangeEnd = -1;
424  for (size_t sector = 0; sector < validInputs.size(); sector++) {
425  if (validInputs.test(sector)) {
426  if (rangeStart < 0) {
427  if (rangeEnd >= 0) {
428  bitInfo += ",";
429  }
430  bitInfo += std::to_string(sector);
431  if (nActiveBits == 1) {
432  break;
433  }
434  rangeStart = sector;
435  }
436  rangeEnd = sector;
437  } else {
438  if (rangeStart >= 0 && rangeEnd > rangeStart) {
439  bitInfo += "-" + std::to_string(rangeEnd);
440  }
441  rangeStart = -1;
442  }
443  }
444  if (rangeStart >= 0 && rangeEnd > rangeStart) {
445  bitInfo += "-" + std::to_string(rangeEnd);
446  }
447  }
448  LOG(INFO) << "running tracking for sector(s) " << bitInfo;
449  }
450 
451  std::vector<TrackTPC> tracks;
452  std::vector<uint32_t> clusRefs;
453  MCLabelContainer tracksMCTruth;
454  GPUO2InterfaceIOPtrs ptrs;
455  ClusterNativeAccess clusterIndex;
456  std::unique_ptr<ClusterNative[]> clusterBuffer;
457  MCLabelContainer clustersMCBuffer;
458  ptrs.outputTracks = &tracks;
459  ptrs.outputClusRefs = &clusRefs;
460  ptrs.outputTracksMCTruth = (processMC ? &tracksMCTruth : nullptr);
461  if (caClusterer) {
462  // Todo: If we have zero-suppressed input, we have to fill this pointer instead: ptrs.tpcZS
463  ptrs.o2Digits = &inputDigits; // TODO: We will also create ClusterNative as output stored in ptrs. Should be added to the output
464  if (processMC) {
465  ptrs.o2DigitsMC = &inputDigitsMC;
466  }
467  } else {
468  memset(&clusterIndex, 0, sizeof(clusterIndex));
469  ClusterNativeHelper::Reader::fillIndex(clusterIndex, clusterBuffer, clustersMCBuffer, inputs, mcInputs, [&validInputs](auto& index) { return validInputs.test(index); });
470  ptrs.clusters = &clusterIndex;
471  }
472  int retVal = tracker->runTracking(&ptrs);
473  if (retVal != 0) {
474  // FIXME: error policy
475  LOG(ERROR) << "tracker returned error code " << retVal;
476  }
477  LOG(INFO) << "found " << tracks.size() << " track(s)";
478  pc.outputs().snapshot(OutputRef{"outTracks"}, tracks);
479  pc.outputs().snapshot(OutputRef{"outClusRefs"}, clusRefs);
480  if (processMC) {
481  LOG(INFO) << "sending " << tracksMCTruth.getIndexedSize() << " track label(s)";
482  pc.outputs().snapshot(OutputRef{"mclblout"}, tracksMCTruth);
483  }
484 
485  // TODO - Process Compressed Clusters Output
486  const o2::tpc::CompressedClusters* compressedClusters = ptrs.compressedClusters; // This is a ROOT-serializable container with compressed TPC clusters
487  // Example to decompress clusters
488  //#include "TPCClusterDecompressor.cxx"
489  //o2::tpc::ClusterNativeAccess clustersNativeDecoded; // Cluster native access structure as used by the tracker
490  //std::vector<o2::tpc::ClusterNative> clusterBuffer; // std::vector that will hold the actual clusters, clustersNativeDecoded will point inside here
491  //mDecoder.decompress(clustersCompressed, clustersNativeDecoded, clusterBuffer, param); // Run decompressor
492 
493  validInputs.reset();
494  if (processMC) {
495  validMcInputs.reset();
496  for (auto& mcInput : mcInputs) {
497  mcInput.clear();
498  }
499  }
500  };
501 
502  return processingFct;
503  };

505  // FIXME: find out how to handle merge inputs in a simple and intuitive way
506  // changing the binding name of the input in order to identify inputs by unique labels
507  // in the processing. Think about how the processing can be made agnostic of input size,
508  // e.g. by providing a span of inputs under a certain label
509  auto createInputSpecs = [inputIds](bool makeMcInput, bool caClusterer) {
510  Inputs inputs;
511  if (caClusterer) {
512  inputs.emplace_back(InputSpec{"input", gDataOriginTPC, "DIGITS", 0, Lifetime::Timeframe});
513  } else {
514  inputs.emplace_back(InputSpec{"input", gDataOriginTPC, "CLUSTERNATIVE", 0, Lifetime::Timeframe});
515  }
516  if (makeMcInput) {
517  if (caClusterer) {
518  constexpr o2::header::DataDescription datadesc("DIGITSMCTR");
519  inputs.emplace_back(InputSpec{"mclblin", gDataOriginTPC, datadesc, 0, Lifetime::Timeframe});
520  } else {
521  inputs.emplace_back(InputSpec{"mclblin", gDataOriginTPC, "CLNATIVEMCLBL", 0, Lifetime::Timeframe});
522  }
523  }
524

526  [inputIds](InputSpec& input, size_t index) {
