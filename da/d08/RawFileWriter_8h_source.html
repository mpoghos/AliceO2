87  // expand buffer by positive increment and return old size

95  // append to the end of the buffer and return the point where appended to
96  size_t pushBack(const char* ptr, size_t sz, bool keepLastOnFlash = true)
97  {
98  if (!sz) {
99  return buffer.size();
100  }
101  nBytesWritten += sz;
102  // do we have a space one this superpage?
103  if ((writer->mSuperPageSize - int(buffer.size())) < 0) { // need to flush
104  flushSuperPage(keepLastOnFlash);
105  }
106  auto offs = expandBufferBy(sz);
107  memmove(&buffer[offs], ptr, sz);
108  return offs;
109  }
110  // add RDH to buffer. In case this requires flushing of the superpage
111  // do not keep the previous page
112  size_t pushBack(const RDH& rdh)
113  {
114  nRDHWritten++;
115  return pushBack(reinterpret_cast<const char*>(&rdh), sizeof(RDH), false);
116  }

187  // This is a placeholder for the function responsible to split large payload to pieces
188  // fitting 8kB CRU pages.
189  // The RawFileWriter receives from the encoder the payload to format according to the CRU format
190  // In case this payload size does not fit into the CRU page (this may happen even if it is
191  // less than 8KB, since it might be added to already partially populated CRU page of the HBF)
192  // it will write on the page only part of the payload and carry over the rest on extra page(s).
193  // By default the RawFileWriter will simply chunk payload as is considers necessary, but some
194  // detectors want their CRU pages to be self-consistent and in case of payload splitting they
195  // add in the end of page to be closed and beginning of the new page to be opened
196  // (right after the RDH) detector-specific trailer and header respectively.
197  //
198  // The role of this method is to suggest to writer how to split the payload:
199  // If this method was set to the RawFileWriter using
200  // RawFileWriter::setCarryOverCallBack(pointer_on_the_converter_class);
201  // then the RawFileWriter will call it before splitting.
202  //
203  // It provides to the carryOverMethod method the following info:
204  // rdh : RDH of the CRU page being written
205  // data : original payload received by the RawFileWriter
206  // ptr : pointer on the data in the payload which was not yet added to the link CRU pages
207  // maxSize : maximum size (multiple of 16 bytes) of the bloc starting at ptr which it can
208  // accomodate at the current CRU page (i.e. what it would write by default)
209  // splitID : number of times this payload was already split, 0 at 1st call
210  // trailer : what it wants to add in the end of the CRU page where the data starting from ptr
211  // will be added. The trailer is supplied as an empy vector, which carryOverMethod
212  // may populate, but its size must be multiple of 16 bytes.
213  // header : what it wants to add right after the RDH of the new CRU page before the rest of
214  // the payload (starting at ptr+actualSize) will be written
215  //
216  // The method mast return actual size of the bloc which can be written (<=maxSize).
217  // If this method populates the trailer, it must ensure that it returns the actual size such that
218  // actualSize + trailer.size() <= maxSize
219  // In case returned actualSize == 0, current CRU page will be closed w/o adding anything, and new
220  // query of this method will be done on the new CRU page
221 
222  int carryOverMethod(const RDH& rdh, const gsl::span<char> data, const char* ptr, int maxSize, int splitID,
223  std::vector<char>& trailer, std::vector<char>& header) const
224  {
225  return maxSize; // do nothing
226  }
